7#define BLOCK_GROWTH_SIZE (1024 * 1024 - sizeof(struct mp_block))
8
9/*
10 * The inner union is an approximation for C11's max_align_t, and the
11 * struct + offsetof computes _Alignof. This can all just be replaced
12 * with _Alignof(max_align_t) if/when C11 is part of the baseline.
13 * Note that _Alignof(X) need not be the same as sizeof(X); it's only
14 * required to be a (possibly trivial) factor. They are the same for
15 * most architectures, but m68k for example has only 2-byte alignment
16 * for its 4-byte and 8-byte types, so using sizeof would waste space.
17 *
18 * Add more types to the union if the current set is insufficient.
19 */
20struct git_max_alignment {
21  char unalign;
22  union {
23    uintmax_t max_align_uintmax;
24    void *max_align_pointer;
25  } aligned;
26};
27#define GIT_MAX_ALIGNMENT offsetof(struct git_max_alignment, aligned)
28
29/*
30 * Allocate a new mp_block and insert it after the block specified in
31 * `insert_after`. If `insert_after` is NULL, then insert block at the
32 * head of the linked list.
33 */
34static struct mp_block *mem_pool_alloc_block(struct mem_pool *pool,
35               size_t block_alloc,
36               struct mp_block *insert_after)
37{
38  struct mp_block *p;
39
40  pool->pool_alloc += sizeof(struct mp_block) + block_alloc;
41  p = xmalloc(st_add(sizeof(struct mp_block), block_alloc));
42
43  p->next_free = (char *)p->space;
44  p->end = p->next_free + block_alloc;
45
46  if (insert_after) {
47    p->next_block = insert_after->next_block;
48    insert_after->next_block = p;
49  } else {
50    p->next_block = pool->mp_block;
51    pool->mp_block = p;
52  }
53
54  return p;
55}

57void mem_pool_init(struct mem_pool *pool, size_t initial_size)
58{
59  memset(pool, 0, sizeof(*pool));
60  pool->block_alloc = BLOCK_GROWTH_SIZE;
61
62  if (initial_size > 0)
63    mem_pool_alloc_block(pool, initial_size, NULL);
64}

66void mem_pool_discard(struct mem_pool *pool, int invalidate_memory)
67{
68  struct mp_block *block, *block_to_free;
69
70  block = pool->mp_block;
71  while (block)
72  {
73    block_to_free = block;
74    block = block->next_block;
75
76    if (invalidate_memory)
77      memset(block_to_free->space, 0xDD, ((char *)block_to_free->end) - ((char *)block_to_free->space));
78
79    free(block_to_free);
80  }
81
82  pool->mp_block = NULL;
83  pool->pool_alloc = 0;
84}

86void *mem_pool_alloc(struct mem_pool *pool, size_t len)
87{
88  struct mp_block *p = NULL;
89  void *r;
90
91  /* round up to a 'GIT_MAX_ALIGNMENT' alignment */
92  if (len & (GIT_MAX_ALIGNMENT - 1))
93    len += GIT_MAX_ALIGNMENT - (len & (GIT_MAX_ALIGNMENT - 1));
94
95  if (pool->mp_block &&
96      pool->mp_block->end - pool->mp_block->next_free >= len)
97    p = pool->mp_block;
98
99  if (!p) {
100    if (len >= (pool->block_alloc / 2))
101      return mem_pool_alloc_block(pool, len, pool->mp_block);
102
103    p = mem_pool_alloc_block(pool, pool->block_alloc, NULL);
104  }
105
106  r = p->next_free;
107  p->next_free += len;
108  return r;
109}

111void *mem_pool_calloc(struct mem_pool *pool, size_t count, size_t size)
112{
113  size_t len = st_mult(count, size);
114  void *r = mem_pool_alloc(pool, len);
115  memset(r, 0, len);
116  return r;
117}

119char *mem_pool_strdup(struct mem_pool *pool, const char *str)
120{
121  size_t len = strlen(str) + 1;
122  char *ret = mem_pool_alloc(pool, len);
123
124  return memcpy(ret, str, len);
125}

127char *mem_pool_strndup(struct mem_pool *pool, const char *str, size_t len)
128{
129  char *p = memchr(str, '\0', len);
130  size_t actual_len = (p ? p - str : len);
131  char *ret = mem_pool_alloc(pool, actual_len+1);
132
133  ret[actual_len] = '\0';
134  return memcpy(ret, str, actual_len);
135}

137int mem_pool_contains(struct mem_pool *pool, void *mem)
138{
139  struct mp_block *p;
140
141  /* Check if memory is allocated in a block */
142  for (p = pool->mp_block; p; p = p->next_block)
143    if ((mem >= ((void *)p->space)) &&
144        (mem < ((void *)p->end)))
145      return 1;
146
147  return 0;
148}

150void mem_pool_combine(struct mem_pool *dst, struct mem_pool *src)
151{
152  struct mp_block *p;
153
154  /* Append the blocks from src to dst */
155  if (dst->mp_block && src->mp_block) {
156    /*
157     * src and dst have blocks, append
158     * blocks from src to dst.
159     */
160    p = dst->mp_block;
161    while (p->next_block)
162      p = p->next_block;
163
164    p->next_block = src->mp_block;
165  } else if (src->mp_block) {
166    /*
167     * src has blocks, dst is empty.
168     */
169    dst->mp_block = src->mp_block;
170  } else {
171    /* src is empty, nothing to do. */
172  }
173
174  dst->pool_alloc += src->pool_alloc;
175  src->pool_alloc = 0;
176  src->mp_block = NULL;
177}
