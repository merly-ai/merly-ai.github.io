309  if (!ce)
310    die(_("make_cache_entry failed for path '%s'"), path);

312  return status;

316           char *ps_matched,
317           const struct checkout_opts *opts)
318{
319  ce->ce_flags &= ~CE_MATCHED;
320  if (!opts->ignore_skipworktree && ce_skip_worktree(ce))
321    return;
322  if (opts->source_tree && !(ce->ce_flags & CE_UPDATE))
323    /*
324     * "git checkout tree-ish -- path", but this entry
325     * is in the original index but is not in tree-ish

343   * opts->source_tree != NULL.

345  if (ce_path_match(&the_index, ce, &opts->pathspec, ps_matched))
346    ce->ce_flags |= CE_MATCHED;

368static int checkout_worktree(const struct checkout_opts *opts,
369           const struct branch_info *info)
370{
371  struct checkout state = CHECKOUT_INIT;
372  int nr_checkouts = 0, nr_unmerged = 0;
373  int errs = 0;
374  int pos;
375  int pc_workers, pc_threshold;
376  struct mem_pool ce_mem_pool;
377
378  state.force = 1;
379  state.refresh_cache = 1;
380  state.istate = &the_index;
381
382  mem_pool_init(&ce_mem_pool, 0);
383  get_parallel_checkout_configs(&pc_workers, &pc_threshold);
384  init_checkout_metadata(&state.meta, info->refname,
385             info->commit ? &info->commit->object.oid : &info->oid,
386             NULL);
387
388  enable_delayed_checkout(&state);
389
390  if (pc_workers > 1)
391    init_parallel_checkout();
392
393  for (pos = 0; pos < active_nr; pos++) {
394    struct cache_entry *ce = active_cache[pos];
395    if (ce->ce_flags & CE_MATCHED) {
396      if (!ce_stage(ce)) {
397        errs |= checkout_entry(ce, &state,
398                   NULL, &nr_checkouts);
399        continue;
400      }
401      if (opts->writeout_stage)
402        errs |= checkout_stage(opts->writeout_stage,
403                   ce, pos,
404                   &state,
405                   &nr_checkouts, opts->overlay_mode);
406      else if (opts->merge)
407        errs |= checkout_merged(pos, &state,
408              &nr_unmerged,
409              &ce_mem_pool);
410      pos = skip_same_name(ce, pos) - 1;
411    }
412  }
413  if (pc_workers > 1)
414    errs |= run_parallel_checkout(&state, pc_workers, pc_threshold,
415                NULL, NULL);
416  mem_pool_discard(&ce_mem_pool, should_validate_cache_entries());
417  remove_marked_cache_entries(&the_index, 1);
418  remove_scheduled_dirs();
419  errs |= finish_delayed_checkout(&state, &nr_checkouts, opts->show_progress);
420
421  if (opts->count_checkout_paths) {
422    if (nr_unmerged)
423      fprintf_ln(stderr, Q_("Recreated %d merge conflict",
424                "Recreated %d merge conflicts",
425                nr_unmerged),
426           nr_unmerged);
427    if (opts->source_tree)
428      fprintf_ln(stderr, Q_("Updated %d path from %s",
429                "Updated %d paths from %s",
430                nr_checkouts),
431           nr_checkouts,
432           find_unique_abbrev(&opts->source_tree->object.oid,
433                  DEFAULT_ABBREV));
434    else if (!nr_unmerged || nr_checkouts)
435      fprintf_ln(stderr, Q_("Updated %d path from the index",
436                "Updated %d paths from the index",
437                nr_checkouts),
438           nr_checkouts);
439  }
440
441  return errs;
442}

513     */
514    if (rev && new_branch_info->commit && strcmp(rev, "HEAD"))
515      rev = oid_to_hex_r(rev_oid, &new_branch_info->commit->object.oid);
516
517    if (opts->checkout_index && opts->checkout_worktree)
518      patch_mode = "--patch=checkout";
519    else if (opts->checkout_index && !opts->checkout_worktree)
520      patch_mode = "--patch=reset";
521    else if (!opts->checkout_index && opts->checkout_worktree)
522      patch_mode = "--patch=worktree";
523    else
524      BUG("either flag must have been set, worktree=%d, index=%d",
525          opts->checkout_worktree, opts->checkout_index);
526    return run_add_interactive(rev, patch_mode, &opts->pathspec);
527  }
528
529  repo_hold_locked_index(the_repository, &lock_file, LOCK_DIE_ON_ERROR);
530  if (read_cache_preload(&opts->pathspec) < 0)
531    return error(_("index file corrupt"));
532
533  if (opts->source_tree)
534    read_tree_some(opts->source_tree, &opts->pathspec);
535
536  ps_matched = xcalloc(opts->pathspec.nr, 1);
537
538  /*
539   * Make sure all pathspecs participated in locating the paths
540   * to be checked out.
541   */
542  for (pos = 0; pos < active_nr; pos++)
543    if (opts->overlay_mode)
544      mark_ce_for_checkout_overlay(active_cache[pos],
545                 ps_matched,
546                 opts);
547    else
548      mark_ce_for_checkout_no_overlay(active_cache[pos],
549              ps_matched,
550              opts);
551
552  if (report_path_error(ps_matched, &opts->pathspec)) {
553    free(ps_matched);
554    return 1;
555  }
556  free(ps_matched);
557
558  /* "checkout -m path" to recreate conflicted state */
559  if (opts->merge)
560    unmerge_marked_index(&the_index);
561
562  /* Any unmerged paths? */
563  for (pos = 0; pos < active_nr; pos++) {
564    const struct cache_entry *ce = active_cache[pos];
565    if (ce->ce_flags & CE_MATCHED) {
566      if (!ce_stage(ce))
567        continue;
568      if (opts->ignore_unmerged) {
569        if (!opts->quiet)
570          warning(_("path '%s' is unmerged"), ce->name);
571      } else if (opts->writeout_stage) {
572        errs |= check_stage(opts->writeout_stage, ce, pos, opts->overlay_mode);
573      } else if (opts->merge) {
574        errs |= check_stages((1<<2) | (1<<3), ce, pos);
575      } else {
576        errs = 1;
577        error(_("path '%s' is unmerged"), ce->name);
578      }
579      pos = skip_same_name(ce, pos) - 1;
580    }
581  }
582  if (errs)
583    return 1;
584
585  /* Now we are committed to check them out */
586  if (opts->checkout_worktree)
587    errs |= checkout_worktree(opts, new_branch_info);
588  else
589    remove_marked_cache_entries(&the_index, 1);
590
591  /*
592   * Allow updating the index when checking out from the index.
593   * This is to save new stat info.
594   */
595  if (opts->checkout_worktree && !opts->checkout_index && !opts->source_tree)
596    checkout_index = 1;
597  else
598    checkout_index = opts->checkout_index;
599
600  if (checkout_index) {
601    if (write_locked_index(&the_index, &lock_file, COMMIT_LOCK))
602      die(_("unable to write new index file"));
603  } else {
604    /*
605     * NEEDSWORK: if --worktree is not specified, we
606     * should save stat info of checked out files in the
607     * index to avoid the next (potentially costly)
608     * refresh. But it's a bit tricker to do...
609     */
610    rollback_lock_file(&lock_file);
611  }
612
613  read_ref_full("HEAD", 0, &rev, NULL);
614  head = lookup_commit_reference_gently(the_repository, &rev, 1);
615
616  errs |= post_checkout_hook(head, head, 0);
617  return errs;
618}

650static int reset_tree(struct tree *tree, const struct checkout_opts *o,
651          int worktree, int *writeout_error,
652          struct branch_info *info)
653{
654  struct unpack_trees_options opts;
655  struct tree_desc tree_desc;
656
657  memset(&opts, 0, sizeof(opts));
658  opts.head_idx = -1;
659  opts.update = worktree;
660  opts.skip_unmerged = !worktree;
661  opts.reset = o->force ? UNPACK_RESET_OVERWRITE_UNTRACKED :
662        UNPACK_RESET_PROTECT_UNTRACKED;
663  opts.preserve_ignored = (!o->force && !o->overwrite_ignore);
664  opts.merge = 1;
665  opts.fn = oneway_merge;
666  opts.verbose_update = o->show_progress;
667  opts.src_index = &the_index;
668  opts.dst_index = &the_index;
669  init_checkout_metadata(&opts.meta, info->refname,
670             info->commit ? &info->commit->object.oid : null_oid(),
671             NULL);
672  parse_tree(tree);
673  init_tree_desc(&tree_desc, tree->buffer, tree->size);
674  switch (unpack_trees(1, &tree_desc, &opts)) {
675  case -2:
676    *writeout_error = 1;
677    /*
678     * We return 0 nevertheless, as the index is all right
679     * and more importantly we have made best efforts to
680     * update paths in the work tree, and we cannot revert
681     * them.
682     */
683    /* fallthrough */
684  case 0:
685    return 0;
686  default:
687    return 128;
688  }
689}

712static int merge_working_tree(const struct checkout_opts *opts,
713            struct branch_info *old_branch_info,
714            struct branch_info *new_branch_info,
715            int *writeout_error)
716{
717  int ret;
718  struct lock_file lock_file = LOCK_INIT;
719  struct tree *new_tree;
720
721  hold_locked_index(&lock_file, LOCK_DIE_ON_ERROR);
722  if (read_cache_preload(NULL) < 0)
723    return error(_("index file corrupt"));
724
725  resolve_undo_clear();
726  if (opts->new_orphan_branch && opts->orphan_from_empty_tree) {
727    if (new_branch_info->commit)
728      BUG("'switch --orphan' should never accept a commit as starting point");
729    new_tree = parse_tree_indirect(the_hash_algo->empty_tree);
730  } else
731    new_tree = get_commit_tree(new_branch_info->commit);
732  if (opts->discard_changes) {
733    ret = reset_tree(new_tree, opts, 1, writeout_error, new_branch_info);
734    if (ret)
735      return ret;
736  } else {
737    struct tree_desc trees[2];
738    struct tree *tree;
739    struct unpack_trees_options topts;
740    const struct object_id *old_commit_oid;
741
742    memset(&topts, 0, sizeof(topts));
743    topts.head_idx = -1;
744    topts.src_index = &the_index;
745    topts.dst_index = &the_index;
746
747    setup_unpack_trees_porcelain(&topts, "checkout");
748
749    refresh_cache(REFRESH_QUIET);
750
751    if (unmerged_cache()) {
752      error(_("you need to resolve your current index first"));
753      return 1;
754    }
755
756    /* 2-way merge to the new branch */
757    topts.initial_checkout = is_cache_unborn();
758    topts.update = 1;
759    topts.merge = 1;
760    topts.quiet = opts->merge && old_branch_info->commit;
761    topts.verbose_update = opts->show_progress;
762    topts.fn = twoway_merge;
763    init_checkout_metadata(&topts.meta, new_branch_info->refname,
764               new_branch_info->commit ?
765               &new_branch_info->commit->object.oid :
766               &new_branch_info->oid, NULL);
767    topts.preserve_ignored = !opts->overwrite_ignore;
768
769    old_commit_oid = old_branch_info->commit ?
770      &old_branch_info->commit->object.oid :
771      the_hash_algo->empty_tree;
772    tree = parse_tree_indirect(old_commit_oid);
773    if (!tree)
774      die(_("unable to parse commit %s"),
775        oid_to_hex(old_commit_oid));
776
777    init_tree_desc(&trees[0], tree->buffer, tree->size);
778    parse_tree(new_tree);
779    tree = new_tree;
780    init_tree_desc(&trees[1], tree->buffer, tree->size);
781
782    ret = unpack_trees(2, trees, &topts);
783    clear_unpack_trees_porcelain(&topts);
784    if (ret == -1) {
785      /*
786       * Unpack couldn't do a trivial merge; either
787       * give up or do a real merge, depending on
788       * whether the merge flag was used.
789       */
790      struct tree *work;
791      struct tree *old_tree;
792      struct merge_options o;
793      struct strbuf sb = STRBUF_INIT;
794      struct strbuf old_commit_shortname = STRBUF_INIT;
795
796      if (!opts->merge)
797        return 1;
798
799      /*
800       * Without old_branch_info->commit, the below is the same as
801       * the two-tree unpack we already tried and failed.
802       */
803      if (!old_branch_info->commit)
804        return 1;
805      old_tree = get_commit_tree(old_branch_info->commit);
806
807      if (repo_index_has_changes(the_repository, old_tree, &sb))
808        die(_("cannot continue with staged changes in "
809              "the following files:\n%s"), sb.buf);
810      strbuf_release(&sb);
811
812      /* Do more real merge */
813
814      /*
815       * We update the index fully, then write the
816       * tree from the index, then merge the new
817       * branch with the current tree, with the old
818       * branch as the base. Then we reset the index
819       * (but not the working tree) to the new
820       * branch, leaving the working tree as the
821       * merged version, but skipping unmerged
822       * entries in the index.
823       */
824
825      add_files_to_cache(NULL, NULL, 0);
826      init_merge_options(&o, the_repository);
827      o.verbosity = 0;
828      work = write_in_core_index_as_tree(the_repository);
829
830      ret = reset_tree(new_tree,
831           opts, 1,
832           writeout_error, new_branch_info);
833      if (ret)
834        return ret;
835      o.ancestor = old_branch_info->name;
836      if (!old_branch_info->name) {
837        strbuf_add_unique_abbrev(&old_commit_shortname,
838               &old_branch_info->commit->object.oid,
839               DEFAULT_ABBREV);
840        o.ancestor = old_commit_shortname.buf;
841      }
842      o.branch1 = new_branch_info->name;
843      o.branch2 = "local";
844      ret = merge_trees(&o,
845            new_tree,
846            work,
847            old_tree);
848      if (ret < 0)
849        exit(128);
850      ret = reset_tree(new_tree,
851           opts, 0,
852           writeout_error, new_branch_info);
853      strbuf_release(&o.obuf);
854      strbuf_release(&old_commit_shortname);
855      if (ret)
856        return ret;
857    }
858  }
859
860  if (!cache_tree_fully_valid(active_cache_tree))
861    cache_tree_update(&the_index, WRITE_TREE_SILENT | WRITE_TREE_REPAIR);
862
863  if (write_locked_index(&the_index, &lock_file, COMMIT_LOCK))
864    die(_("unable to write new index file"));
865
866  if (!opts->discard_changes && !opts->quiet && new_branch_info->commit)
867    show_local_changes(&new_branch_info->commit->object, &opts->diff_options);
868
869  return 0;
870}

883static void update_refs_for_switch(const struct checkout_opts *opts,
884           struct branch_info *old_branch_info,
885           struct branch_info *new_branch_info)
886{
887  struct strbuf msg = STRBUF_INIT;
888  const char *old_desc, *reflog_msg;
889  if (opts->new_branch) {
890    if (opts->new_orphan_branch) {
891      char *refname;
892
893      refname = mkpathdup("refs/heads/%s", opts->new_orphan_branch);
894      if (opts->new_branch_log &&
895          !should_autocreate_reflog(refname)) {
896        int ret;
897        struct strbuf err = STRBUF_INIT;
898
899        ret = safe_create_reflog(refname, &err);
900        if (ret) {
901          fprintf(stderr, _("Can not do reflog for '%s': %s\n"),
902            opts->new_orphan_branch, err.buf);
903          strbuf_release(&err);
904          free(refname);
905          return;
906        }
907        strbuf_release(&err);
908      }
909      free(refname);
910    }
911    else
912      create_branch(the_repository,
913              opts->new_branch, new_branch_info->name,
914              opts->new_branch_force ? 1 : 0,
915              opts->new_branch_force ? 1 : 0,
916              opts->new_branch_log,
917              opts->quiet,
918              opts->track,
919              0);
920    free(new_branch_info->name);
921    free(new_branch_info->refname);
922    new_branch_info->name = xstrdup(opts->new_branch);
923    setup_branch_path(new_branch_info);
924  }
925
926  old_desc = old_branch_info->name;
927  if (!old_desc && old_branch_info->commit)
928    old_desc = oid_to_hex(&old_branch_info->commit->object.oid);
929
930  reflog_msg = getenv("GIT_REFLOG_ACTION");
931  if (!reflog_msg)
932    strbuf_addf(&msg, "checkout: moving from %s to %s",
933      old_desc ? old_desc : "(invalid)", new_branch_info->name);
934  else
935    strbuf_insertstr(&msg, 0, reflog_msg);
936
937  if (!strcmp(new_branch_info->name, "HEAD") && !new_branch_info->path && !opts->force_detach) {
938    /* Nothing to do. */
939  } else if (opts->force_detach || !new_branch_info->path) {  /* No longer on any branch. */
940    update_ref(msg.buf, "HEAD", &new_branch_info->commit->object.oid, NULL,
941         REF_NO_DEREF, UPDATE_REFS_DIE_ON_ERR);
942    if (!opts->quiet) {
943      if (old_branch_info->path &&
944          advice_enabled(ADVICE_DETACHED_HEAD) && !opts->force_detach)
945        detach_advice(new_branch_info->name);
946      describe_detached_head(_("HEAD is now at"), new_branch_info->commit);
947    }
948  } else if (new_branch_info->path) {  /* Switch branches. */
949    if (create_symref("HEAD", new_branch_info->path, msg.buf) < 0)
950      die(_("unable to update HEAD"));
951    if (!opts->quiet) {
952      if (old_branch_info->path && !strcmp(new_branch_info->path, old_branch_info->path)) {
953        if (opts->new_branch_force)
954          fprintf(stderr, _("Reset branch '%s'\n"),
955            new_branch_info->name);
956        else
957          fprintf(stderr, _("Already on '%s'\n"),
958            new_branch_info->name);
959      } else if (opts->new_branch) {
960        if (opts->branch_exists)
961          fprintf(stderr, _("Switched to and reset branch '%s'\n"), new_branch_info->name);
962        else
963          fprintf(stderr, _("Switched to a new branch '%s'\n"), new_branch_info->name);
964      } else {
965        fprintf(stderr, _("Switched to branch '%s'\n"),
966          new_branch_info->name);
967      }
968    }
969    if (old_branch_info->path && old_branch_info->name) {
970      if (!ref_exists(old_branch_info->path) && reflog_exists(old_branch_info->path))
971        delete_reflog(old_branch_info->path);
972    }
973  }
974  remove_branch_state(the_repository, !opts->quiet);
975  strbuf_release(&msg);
976  if (!opts->quiet &&
977      (new_branch_info->path || (!opts->force_detach && !strcmp(new_branch_info->name, "HEAD"))))
978    report_tracking(new_branch_info);
979}

1000static void suggest_reattach(struct commit *commit, struct rev_info *revs)

1230static int parse_branchname_arg(int argc, const char **argv,
1231        int dwim_new_local_branch_ok,
1232        struct branch_info *new_branch_info,
1233        struct checkout_opts *opts,
1234        struct object_id *rev)
1235{
1236  const char **new_branch = &opts->new_branch;
1237  int argcount = 0;
1238  const char *arg;
1239  int dash_dash_pos;
1240  int has_dash_dash = 0;
1241  int i;
1242
1243  /*
1244   * case 1: git checkout <ref> -- [<paths>]
1245   *
1246   *   <ref> must be a valid tree, everything after the '--' must be
1247   *   a path.
1248   *
1249   * case 2: git checkout -- [<paths>]
1250   *
1251   *   everything after the '--' must be paths.
1252   *
1253   * case 3: git checkout <something> [--]
1254   *
1255   *   (a) If <something> is a commit, that is to
1256   *       switch to the branch or detach HEAD at it.  As a special case,
1257   *       if <something> is A...B (missing A or B means HEAD but you can
1258   *       omit at most one side), and if there is a unique merge base
1259   *       between A and B, A...B names that merge base.
1260   *
1261   *   (b) If <something> is _not_ a commit, either "--" is present
1262   *       or <something> is not a path, no -t or -b was given, and
1263   *       and there is a tracking branch whose name is <something>
1264   *       in one and only one remote (or if the branch exists on the
1265   *       remote named in checkout.defaultRemote), then this is a
1266   *       short-hand to fork local <something> from that
1267   *       remote-tracking branch.
1268   *
1269   *   (c) Otherwise, if "--" is present, treat it like case (1).
1270   *
1271   *   (d) Otherwise :
1272   *       - if it's a reference, treat it like case (1)
1273   *       - else if it's a path, treat it like case (2)
1274   *       - else: fail.
1275   *
1276   * case 4: git checkout <something> <paths>
1277   *
1278   *   The first argument must not be ambiguous.
1279   *   - If it's *only* a reference, treat it like case (1).
1280   *   - If it's only a path, treat it like case (2).
1281   *   - else: fail.
1282   *
1283   */
1284  if (!argc)
1285    return 0;
1286
1287  if (!opts->accept_pathspec) {
1288    if (argc > 1)
1289      die(_("only one reference expected"));
1290    has_dash_dash = 1; /* helps disambiguate */
1291  }
1292
1293  arg = argv[0];
1294  dash_dash_pos = -1;
1295  for (i = 0; i < argc; i++) {
1296    if (opts->accept_pathspec && !strcmp(argv[i], "--")) {
1297      dash_dash_pos = i;
1298      break;
1299    }
1300  }
1301  if (dash_dash_pos == 0)
1302    return 1; /* case (2) */
1303  else if (dash_dash_pos == 1)
1304    has_dash_dash = 1; /* case (3) or (1) */
1305  else if (dash_dash_pos >= 2)
1306    die(_("only one reference expected, %d given."), dash_dash_pos);
1307  opts->count_checkout_paths = !opts->quiet && !has_dash_dash;
1308
1309  if (!strcmp(arg, "-"))
1310    arg = "@{-1}";
1311
1312  if (get_oid_mb(arg, rev)) {
1313    /*
1314     * Either case (3) or (4), with <something> not being
1315     * a commit, or an attempt to use case (1) with an
1316     * invalid ref.
1317     *
1318     * It's likely an error, but we need to find out if
1319     * we should auto-create the branch, case (3).(b).
1320     */
1321    int recover_with_dwim = dwim_new_local_branch_ok;
1322
1323    int could_be_checkout_paths = !has_dash_dash &&
1324      check_filename(opts->prefix, arg);
1325
1326    if (!has_dash_dash && !no_wildcard(arg))
1327      recover_with_dwim = 0;
1328
1329    /*
1330     * Accept "git checkout foo", "git checkout foo --"
1331     * and "git switch foo" as candidates for dwim.
1332     */
1333    if (!(argc == 1 && !has_dash_dash) &&
1334        !(argc == 2 && has_dash_dash) &&
1335        opts->accept_pathspec)
1336      recover_with_dwim = 0;
1337
1338    if (recover_with_dwim) {
1339      const char *remote = parse_remote_branch(arg, rev,
1340                 could_be_checkout_paths);
1341      if (remote) {
1342        *new_branch = arg;
1343        arg = remote;
1344        /* DWIMmed to create local branch, case (3).(b) */
1345      } else {
1346        recover_with_dwim = 0;
1347      }
1348    }
1349
1350    if (!recover_with_dwim) {
1351      if (has_dash_dash)
1352        die(_("invalid reference: %s"), arg);
1353      return argcount;
1354    }
1355  }
1356
1357  /* we can't end up being in (2) anymore, eat the argument */
1358  argcount++;
1359  argv++;
1360  argc--;
1361
1362  setup_new_branch_info_and_source_tree(new_branch_info, opts, rev, arg);
1363
1364  if (!opts->source_tree)                   /* case (1): want a tree */
1365    die(_("reference is not a tree: %s"), arg);
1366
1367  if (!has_dash_dash) {  /* case (3).(d) -> (1) */
1368    /*
1369     * Do not complain the most common case
1370     *  git checkout branch
1371     * even if there happen to be a file called 'branch';
1372     * it would be extremely annoying.
1373     */
1374    if (argc)
1375      verify_non_filename(opts->prefix, arg);
1376  } else if (opts->accept_pathspec) {
1377    argcount++;
1378    argv++;
1379    argc--;
1380  }
1381
1382  return argcount;
1383}

1515      !opts->force_detach &&

1517      !opts->new_branch_force &&
1518      new_branch_info->name &&
1519      !new_branch_info->path)
1520    die_expecting_a_branch(new_branch_info);
1521
1522  if (!opts->can_switch_when_in_progress)
1523    die_if_some_operation_in_progress();
1524
1525  if (new_branch_info->path && !opts->force_detach && !opts->new_branch &&
1526      !opts->ignore_other_worktrees) {
1527    int flag;
1528    char *head_ref = resolve_refdup("HEAD", 0, NULL, &flag);
1529    if (head_ref &&
1530        (!(flag & REF_ISSYMREF) || strcmp(head_ref, new_branch_info->path)))
1531      die_if_checked_out(new_branch_info->path, 1);
1532    free(head_ref);
1533  }
1534
1535  if (!new_branch_info->commit && opts->new_branch) {
1536    struct object_id rev;
1537    int flag;
1538
1539    if (!read_ref_full("HEAD", 0, &rev, &flag) &&
1540        (flag & REF_ISSYMREF) && is_null_oid(&rev))
1541      return switch_unborn_to_new_branch(opts);
1542  }
1543  return switch_branches(opts, new_branch_info);
1544}

1615       struct checkout_opts *opts, struct option *options,
1616       const char * const usagestr[],

1620
1621  opts->overwrite_ignore = 1;
1622  opts->prefix = prefix;

1625  git_config(git_checkout_config, opts);
1626  if (the_repository->gitdir) {
1627    prepare_repo_settings(the_repository);
1628    the_repository->settings.command_requires_full_index = 0;

1631  opts->track = BRANCH_TRACK_UNSPECIFIED;

1633  if (!opts->accept_pathspec && !opts->accept_ref)

1643      opts->show_progress = 0;

1645      opts->show_progress = isatty(2);

1648  if (opts->conflict_style) {
1649    opts->merge = 1; /* implied */

1672      opts->checkout_index = -opts->checkout_index - 1;
1673    if (opts->checkout_worktree < 0)
1674      opts->checkout_worktree = -opts->checkout_worktree - 1;

1680   * "git restore --staged [--worktree] --source HEAD"

1735

1738
1739    setup_new_branch_info_and_source_tree(new_branch_info,
1740                  opts, &rev,
1741                  opts->from_treeish);
1742
1743    if (!opts->source_tree)
1744      die(_("reference is not a tree: %s"), opts->from_treeish);
1745  }
1746

1749             opts->patch_mode ? PATHSPEC_PREFIX_ORIGIN : 0,
1750             prefix, argv);

1761        argv[0], opts->new_branch);
1762
1763    if (opts->force_detach)
1764      die(_("git checkout: --detach does not take a path argument '%s'"),
1765          argv[0]);
1766  }
1767
1768  if (opts->pathspec_from_file) {
1769    if (opts->pathspec.nr)
1770      die(_("'%s' and pathspec arguments cannot be used together"), "--pathspec-from-file");

1773      die(_("options '%s' and '%s' cannot be used together"), "--pathspec-from-file", "--detach");

1776      die(_("options '%s' and '%s' cannot be used together"), "--pathspec-from-file", "--patch");
1777
1778    parse_pathspec_file(&opts->pathspec, 0,
1779            0,
1780            prefix, opts->pathspec_from_file, opts->pathspec_file_nul);
1781  } else if (opts->pathspec_file_nul) {
1782    die(_("the option '%s' requires '%s'"), "--pathspec-file-nul", "--pathspec-from-file");
1783  }
1784
1785  opts->pathspec.recursive = 1;
1786
1787  if (opts->pathspec.nr) {
1788    if (1 < !!opts->writeout_stage + !!opts->force + !!opts->merge)
1789      die(_("git checkout: --ours/--theirs, --force and --merge are incompatible when\n"
1790            "checking out of the index."));
1791  } else {
1792    if (opts->accept_pathspec && !opts->empty_pathspec_ok &&
1793        !opts->patch_mode)  /* patch mode is special */
1794      die(_("you must specify path(s) to restore"));
1795  }
1796
1797  if (opts->new_branch) {
1798    struct strbuf buf = STRBUF_INIT;

1804        validate_new_branchname(opts->new_branch, &buf, 0);
1805    strbuf_release(&buf);
1806  }
1807
1808  if (opts->patch_mode || opts->pathspec.nr)
1809    return checkout_paths(opts, new_branch_info);
1810  else
1811    return checkout_branch(opts, new_branch_info);
1812}
