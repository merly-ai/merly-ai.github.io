52__attribute__((format (printf, 1, 2)))
53static void report(const char *fmt, ...)
54{
55  va_list vp;
56
57  if (!verbose)
58    return;
59
60  /*
61   * It is possible, though unlikely, that a caller could use the verbose
62   * output to synchronize with addition of objects to the object
63   * database. The current implementation of ODB transactions leaves
64   * objects invisible while a transaction is active, so flush the
65   * transaction here before reporting a change made by update-index.
66   */
67  flush_odb_transaction();
68  va_start(vp, fmt);
69  vprintf(fmt, vp);
70  putchar('\n');
71  va_end(vp);
72}

277static int add_one_path(const struct cache_entry *old, const char *path, int len, struct stat *st)
278{
279  int option;
280  struct cache_entry *ce;
281
282  /* Was the old index entry already up-to-date? */
283  if (old && !ce_stage(old) && !ce_match_stat(old, st, 0))
284    return 0;
285
286  ce = make_empty_cache_entry(&the_index, len);
287  memcpy(ce->name, path, len);
288  ce->ce_flags = create_ce_flags(0);
289  ce->ce_namelen = len;
290  fill_stat_cache_info(&the_index, ce, st);
291  ce->ce_mode = ce_mode_from_stat(old, st->st_mode);
292
293  if (index_path(&the_index, &ce->oid, path, st,
294           info_only ? 0 : HASH_WRITE_OBJECT)) {
295    discard_cache_entry(ce);
296    return -1;
297  }
298  option = allow_add ? ADD_CACHE_OK_TO_ADD : 0;
299  option |= allow_replace ? ADD_CACHE_OK_TO_REPLACE : 0;
300  if (add_cache_entry(ce, option)) {
301    discard_cache_entry(ce);
302    return error("%s: cannot add to the index - missing --add option?", path);
303  }
304  return 0;
305}

307/*
308 * Handle a path that was a directory. Four cases:
309 *
310 *  - it's already a gitlink in the index, and we keep it that
311 *    way, and update it if we can (if we cannot find the HEAD,
312 *    we're going to keep it unchanged in the index!)
313 *
314 *  - it's a *file* in the index, in which case it should be
315 *    removed as a file if removal is allowed, since it doesn't
316 *    exist as such any more. If removal isn't allowed, it's
317 *    an error.
318 *
319 *    (NOTE! This is old and arguably fairly strange behaviour.
320 *    We might want to make this an error unconditionally, and
321 *    use "--force-remove" if you actually want to force removal).
322 *
323 *  - it used to exist as a subdirectory (ie multiple files with
324 *    this particular prefix) in the index, in which case it's wrong
325 *    to try to update it as a directory.
326 *
327 *  - it doesn't exist at all in the index, but it is a valid
328 *    git directory, and it should be *added* as a gitlink.
329 */
330static int process_directory(const char *path, int len, struct stat *st)
331{
332  struct object_id oid;
333  int pos = cache_name_pos(path, len);
334
335  /* Exact match: file or existing gitlink */
336  if (pos >= 0) {
337    const struct cache_entry *ce = active_cache[pos];
338    if (S_ISGITLINK(ce->ce_mode)) {
339
340      /* Do nothing to the index if there is no HEAD! */
341      if (resolve_gitlink_ref(path, "HEAD", &oid) < 0)
342        return 0;
343
344      return add_one_path(ce, path, len, st);
345    }
346    /* Should this be an unconditional error? */
347    return remove_one_path(path);
348  }
349
350  /* Inexact match: is there perhaps a subdirectory match? */
351  pos = -pos-1;
352  while (pos < active_nr) {
353    const struct cache_entry *ce = active_cache[pos++];
354
355    if (strncmp(ce->name, path, len))
356      break;
357    if (ce->name[len] > '/')
358      break;
359    if (ce->name[len] < '/')
360      continue;
361
362    /* Subdirectory match - error out */
363    return error("%s: is a directory - add individual files instead", path);
364  }
365
366  /* No match - should we add it as a gitlink? */
367  if (!resolve_gitlink_ref(path, "HEAD", &oid))
368    return add_one_path(NULL, path, len, st);
369
370  /* Error out. */
371  return error("%s: is a directory - add files inside instead", path);
372}

374static int process_path(const char *path, struct stat *st, int stat_errno)
375{
376  int pos, len;
377  const struct cache_entry *ce;
378
379  len = strlen(path);
380  if (has_symlink_leading_path(path, len))
381    return error("'%s' is beyond a symbolic link", path);
382
383  pos = cache_name_pos(path, len);
384  ce = pos < 0 ? NULL : active_cache[pos];
385  if (ce && ce_skip_worktree(ce)) {
386    /*
387     * working directory version is assumed "good"
388     * so updating it does not make sense.
389     * On the other hand, removing it from index should work
390     */
391    if (!ignore_skip_worktree_entries && allow_remove &&
392        remove_file_from_cache(path))
393      return error("%s: cannot remove from the index", path);
394    return 0;
395  }
396
397  /*
398   * First things first: get the stat information, to decide
399   * what to do about the pathname!
400   */
401  if (stat_errno)
402    return process_lstat_error(path, stat_errno);
403
404  if (S_ISDIR(st->st_mode))
405    return process_directory(path, len, st);
406
407  return add_one_path(ce, path, len, st);
408}

410static int add_cacheinfo(unsigned int mode, const struct object_id *oid,
411       const char *path, int stage)
412{
413  int len, option;
414  struct cache_entry *ce;
415
416  if (!verify_path(path, mode))
417    return error("Invalid path '%s'", path);
418
419  len = strlen(path);
420  ce = make_empty_cache_entry(&the_index, len);
421
422  oidcpy(&ce->oid, oid);
423  memcpy(ce->name, path, len);
424  ce->ce_flags = create_ce_flags(stage);
425  ce->ce_namelen = len;
426  ce->ce_mode = create_ce_mode(mode);
427  if (assume_unchanged)
428    ce->ce_flags |= CE_VALID;
429  option = allow_add ? ADD_CACHE_OK_TO_ADD : 0;
430  option |= allow_replace ? ADD_CACHE_OK_TO_REPLACE : 0;
431  if (add_cache_entry(ce, option))
432    return error("%s: cannot add to the index - missing --add option?",
433           path);
434  report("add '%s'", path);
435  return 0;
436}

456static void update_one(const char *path)
457{
458  int stat_errno = 0;
459  struct stat st;
460
461  if (mark_valid_only || mark_skip_worktree_only || force_remove ||
462      mark_fsmonitor_only)
463    st.st_mode = 0;
464  else if (lstat(path, &st) < 0) {
465    st.st_mode = 0;
466    stat_errno = errno;
467  } /* else stat is valid */
468
469  if (!verify_path(path, st.st_mode)) {
470    fprintf(stderr, "Ignoring path %s\n", path);
471    return;
472  }
473  if (mark_valid_only) {
474    if (mark_ce_flags(path, CE_VALID, mark_valid_only == MARK_FLAG))
475      die("Unable to mark file %s", path);
476    return;
477  }
478  if (mark_skip_worktree_only) {
479    if (mark_ce_flags(path, CE_SKIP_WORKTREE, mark_skip_worktree_only == MARK_FLAG))
480      die("Unable to mark file %s", path);
481    return;
482  }
483  if (mark_fsmonitor_only) {
484    if (mark_ce_flags(path, CE_FSMONITOR_VALID, mark_fsmonitor_only == MARK_FLAG))
485      die("Unable to mark file %s", path);
486    return;
487  }
488
489  if (force_remove) {
490    if (remove_file_from_cache(path))
491      die("git update-index: unable to remove %s", path);
492    report("remove '%s'", path);
493    return;
494  }
495  if (process_path(path, &st, stat_errno))
496    die("Unable to process path %s", path);
497  report("add '%s'", path);
498}

500static void read_index_info(int nul_term_line)
501{
502  const int hexsz = the_hash_algo->hexsz;
503  struct strbuf buf = STRBUF_INIT;
504  struct strbuf uq = STRBUF_INIT;
505  strbuf_getline_fn getline_fn;
506
507  getline_fn = nul_term_line ? strbuf_getline_nul : strbuf_getline_lf;
508  while (getline_fn(&buf, stdin) != EOF) {
509    char *ptr, *tab;
510    char *path_name;
511    struct object_id oid;
512    unsigned int mode;
513    unsigned long ul;
514    int stage;
515
516    /* This reads lines formatted in one of three formats:
517     *
518     * (1) mode         SP sha1          TAB path
519     * The first format is what "git apply --index-info"
520     * reports, and used to reconstruct a partial tree
521     * that is used for phony merge base tree when falling
522     * back on 3-way merge.
523     *
524     * (2) mode SP type SP sha1          TAB path
525     * The second format is to stuff "git ls-tree" output
526     * into the index file.
527     *
528     * (3) mode         SP sha1 SP stage TAB path
529     * This format is to put higher order stages into the
530     * index file and matches "git ls-files --stage" output.
531     */
532    errno = 0;
533    ul = strtoul(buf.buf, &ptr, 8);
534    if (ptr == buf.buf || *ptr != ' '
535        || errno || (unsigned int) ul != ul)
536      goto bad_line;
537    mode = ul;
538
539    tab = strchr(ptr, '\t');
540    if (!tab || tab - ptr < hexsz + 1)
541      goto bad_line;
542
543    if (tab[-2] == ' ' && '0' <= tab[-1] && tab[-1] <= '3') {
544      stage = tab[-1] - '0';
545      ptr = tab + 1; /* point at the head of path */
546      tab = tab - 2; /* point at tail of sha1 */
547    }
548    else {
549      stage = 0;
550      ptr = tab + 1; /* point at the head of path */
551    }
552
553    if (get_oid_hex(tab - hexsz, &oid) ||
554      tab[-(hexsz + 1)] != ' ')
555      goto bad_line;
556
557    path_name = ptr;
558    if (!nul_term_line && path_name[0] == '"') {
559      strbuf_reset(&uq);
560      if (unquote_c_style(&uq, path_name, NULL)) {
561        die("git update-index: bad quoting of path name");
562      }
563      path_name = uq.buf;
564    }
565
566    if (!verify_path(path_name, mode)) {
567      fprintf(stderr, "Ignoring path %s\n", path_name);
568      continue;
569    }
570
571    if (!mode) {
572      /* mode == 0 means there is no such path -- remove */
573      if (remove_file_from_cache(path_name))
574        die("git update-index: unable to remove %s",
575            ptr);
576    }
577    else {
578      /* mode ' ' sha1 '\t' name
579       * ptr[-1] points at tab,
580       * ptr[-41] is at the beginning of sha1
581       */
582      ptr[-(hexsz + 2)] = ptr[-1] = 0;
583      if (add_cacheinfo(mode, &oid, path_name, stage))
584        die("git update-index: unable to update %s",
585            path_name);
586    }
587    continue;
588
589  bad_line:
590    die("malformed index info %s", buf.buf);
591  }
592  strbuf_release(&buf);
593  strbuf_release(&uq);
594}

604static struct cache_entry *read_one_ent(const char *which,
605          struct object_id *ent, const char *path,
606          int namelen, int stage)
607{
608  unsigned short mode;
609  struct object_id oid;
610  struct cache_entry *ce;
611
612  if (get_tree_entry(the_repository, ent, path, &oid, &mode)) {
613    if (which)
614      error("%s: not in %s branch.", path, which);
615    return NULL;
616  }
617  if (!the_index.sparse_index && mode == S_IFDIR) {
618    if (which)
619      error("%s: not a blob in %s branch.", path, which);
620    return NULL;
621  }
622  ce = make_empty_cache_entry(&the_index, namelen);
623
624  oidcpy(&ce->oid, &oid);
625  memcpy(ce->name, path, namelen);
626  ce->ce_flags = create_ce_flags(stage);
627  ce->ce_namelen = namelen;
628  ce->ce_mode = create_ce_mode(mode);
629  return ce;
630}

632static int unresolve_one(const char *path)
633{
634  int namelen = strlen(path);
635  int pos;
636  int ret = 0;
637  struct cache_entry *ce_2 = NULL, *ce_3 = NULL;
638
639  /* See if there is such entry in the index. */
640  pos = cache_name_pos(path, namelen);
641  if (0 <= pos) {
642    /* already merged */
643    pos = unmerge_cache_entry_at(pos);
644    if (pos < active_nr) {
645      const struct cache_entry *ce = active_cache[pos];
646      if (ce_stage(ce) &&
647          ce_namelen(ce) == namelen &&
648          !memcmp(ce->name, path, namelen))
649        return 0;
650    }
651    /* no resolve-undo information; fall back */
652  } else {
653    /* If there isn't, either it is unmerged, or
654     * resolved as "removed" by mistake.  We do not
655     * want to do anything in the former case.
656     */
657    pos = -pos-1;
658    if (pos < active_nr) {
659      const struct cache_entry *ce = active_cache[pos];
660      if (ce_namelen(ce) == namelen &&
661          !memcmp(ce->name, path, namelen)) {
662        fprintf(stderr,
663          "%s: skipping still unmerged path.\n",
664          path);
665        goto free_return;
666      }
667    }
668  }
669
670  /* Grab blobs from given path from HEAD and MERGE_HEAD,
671   * stuff HEAD version in stage #2,
672   * stuff MERGE_HEAD version in stage #3.
673   */
674  ce_2 = read_one_ent("our", &head_oid, path, namelen, 2);
675  ce_3 = read_one_ent("their", &merge_head_oid, path, namelen, 3);
676
677  if (!ce_2 || !ce_3) {
678    ret = -1;
679    goto free_return;
680  }
681  if (oideq(&ce_2->oid, &ce_3->oid) &&
682      ce_2->ce_mode == ce_3->ce_mode) {
683    fprintf(stderr, "%s: identical in both, skipping.\n",
684      path);
685    goto free_return;
686  }
687
688  remove_file_from_cache(path);
689  if (add_cache_entry(ce_2, ADD_CACHE_OK_TO_ADD)) {
690    error("%s: cannot add our version to the index.", path);
691    ret = -1;
692    goto free_return;
693  }
694  if (!add_cache_entry(ce_3, ADD_CACHE_OK_TO_ADD))
695    return 0;
696  error("%s: cannot add their version to the index.", path);
697  ret = -1;
698 free_return:
699  discard_cache_entry(ce_2);
700  discard_cache_entry(ce_3);
701  return ret;
702}

704static void read_head_pointers(void)
705{
706  if (read_ref("HEAD", &head_oid))
707    die("No HEAD -- no initial commit yet?");
708  if (read_ref("MERGE_HEAD", &merge_head_oid)) {
709    fprintf(stderr, "Not in the middle of a merge.\n");
710    exit(0);
711  }
712}

714static int do_unresolve(int ac, const char **av,
715      const char *prefix, int prefix_length)
716{
717  int i;
718  int err = 0;
719
720  /* Read HEAD and MERGE_HEAD; if MERGE_HEAD does not exist, we
721   * are not doing a merge, so exit with success status.
722   */
723  read_head_pointers();
724
725  for (i = 1; i < ac; i++) {
726    const char *arg = av[i];
727    char *p = prefix_path(prefix, prefix_length, arg);
728    err |= unresolve_one(p);
729    free(p);
730  }
731  return err;
732}
733
734static int do_reupdate(int ac, const char **av,
735           const char *prefix)
736{
737  /* Read HEAD and run update-index on paths that are
738   * merged and already different between index and HEAD.
739   */
740  int pos;
741  int has_head = 1;
742  struct pathspec pathspec;
743
744  parse_pathspec(&pathspec, 0,
745           PATHSPEC_PREFER_CWD,
746           prefix, av + 1);
747
748  if (read_ref("HEAD", &head_oid))
749    /* If there is no HEAD, that means it is an initial
750     * commit.  Update everything in the index.
751     */
752    has_head = 0;
753 redo:
754  for (pos = 0; pos < active_nr; pos++) {
755    const struct cache_entry *ce = active_cache[pos];
756    struct cache_entry *old = NULL;
757    int save_nr;
758    char *path;
759
760    if (ce_stage(ce) || !ce_path_match(&the_index, ce, &pathspec, NULL))
761      continue;
762    if (has_head)
763      old = read_one_ent(NULL, &head_oid,
764             ce->name, ce_namelen(ce), 0);
765    if (old && ce->ce_mode == old->ce_mode &&
766        oideq(&ce->oid, &old->oid)) {
767      discard_cache_entry(old);
768      continue; /* unchanged */
769    }
770
771    /* At this point, we know the contents of the sparse directory are
772     * modified with respect to HEAD, so we expand the index and restart
773     * to process each path individually
774     */
775    if (S_ISSPARSEDIR(ce->ce_mode)) {
776      ensure_full_index(&the_index);
777      goto redo;
778    }
779
780    /* Be careful.  The working tree may not have the
781     * path anymore, in which case, under 'allow_remove',
782     * or worse yet 'allow_replace', active_nr may decrease.
783     */
784    save_nr = active_nr;
785    path = xstrdup(ce->name);
786    update_one(path);
787    free(path);
788    discard_cache_entry(old);
789    if (save_nr != active_nr)
790      goto redo;
791  }
792  clear_pathspec(&pathspec);
793  return 0;
794}

801static int refresh(struct refresh_params *o, unsigned int flag)
802{

818}

856static int parse_new_style_cacheinfo(const char *arg,
857             unsigned int *mode,
858             struct object_id *oid,
859             const char **path)
860{
861  unsigned long ul;
862  char *endp;
863  const char *p;
864
865  if (!arg)
866    return -1;
867
868  errno = 0;
869  ul = strtoul(arg, &endp, 8);
870  if (errno || endp == arg || *endp != ',' || (unsigned int) ul != ul)
871    return -1; /* not a new-style cacheinfo */
872  *mode = ul;
873  endp++;
874  if (parse_oid_hex(endp, oid, &p) || *p != ',')
875    return -1;
876  *path = p + 1;
877  return 0;
878}

1052    OPT_BOOL('z', NULL, &nul_term_line,
1053       N_("with --stdin: input lines are terminated by null bytes")),
1054    {OPTION_LOWLEVEL_CALLBACK, 0, "stdin", &read_from_stdin, NULL,
1055      N_("read list of paths to be updated from standard input"),
1056      PARSE_OPT_NONEG | PARSE_OPT_NOARG,
1057      NULL, 0, stdin_callback},
1058    {OPTION_LOWLEVEL_CALLBACK, 0, "index-info", &nul_term_line, NULL,
1059      N_("add entries from standard input to the index"),
1060      PARSE_OPT_NONEG | PARSE_OPT_NOARG,
1061      NULL, 0, stdin_cacheinfo_callback},
1062    {OPTION_LOWLEVEL_CALLBACK, 0, "unresolve", &has_errors, NULL,
1063      N_("repopulate stages #2 and #3 for the listed paths"),
1064      PARSE_OPT_NONEG | PARSE_OPT_NOARG,
1065      NULL, 0, unresolve_callback},
1066    {OPTION_LOWLEVEL_CALLBACK, 'g', "again", &has_errors, NULL,
1067      N_("only update entries that differ from HEAD"),
1068      PARSE_OPT_NONEG | PARSE_OPT_NOARG,
1069      NULL, 0, reupdate_callback},
1070    OPT_BIT(0, "ignore-missing", &refresh_args.flags,
1071      N_("ignore files missing from worktree"),
1072      REFRESH_IGNORE_MISSING),
1073    OPT_SET_INT(0, "verbose", &verbose,
1074      N_("report actions to standard output"), 1),
1075    OPT_CALLBACK_F(0, "clear-resolve-undo", NULL, NULL,
1076      N_("(for porcelains) forget saved unresolved conflicts"),
1077      PARSE_OPT_NOARG | PARSE_OPT_NONEG,
1078      resolve_undo_clear_callback),
1079    OPT_INTEGER(0, "index-version", &preferred_index_format,
1080      N_("write index in this format")),
1081    OPT_BOOL(0, "split-index", &split_index,
1082      N_("enable or disable split index")),
1083    OPT_BOOL(0, "untracked-cache", &untracked_cache,
1084      N_("enable/disable untracked cache")),
1085    OPT_SET_INT(0, "test-untracked-cache", &untracked_cache,
1086          N_("test if the filesystem supports untracked cache"), UC_TEST),
1087    OPT_SET_INT(0, "force-untracked-cache", &untracked_cache,
1088          N_("enable untracked cache without testing the filesystem"), UC_FORCE),
1089    OPT_SET_INT(0, "force-write-index", &force_write,
1090      N_("write out the index even if is not flagged as changed"), 1),
1091    OPT_BOOL(0, "fsmonitor", &fsmonitor,
1092      N_("enable or disable file system monitor")),
1093    {OPTION_SET_INT, 0, "fsmonitor-valid", &mark_fsmonitor_only, NULL,
1094      N_("mark files as fsmonitor valid"),
1095      PARSE_OPT_NOARG | PARSE_OPT_NONEG, NULL, MARK_FLAG},
1096    {OPTION_SET_INT, 0, "no-fsmonitor-valid", &mark_fsmonitor_only, NULL,
1097      N_("clear fsmonitor valid bit"),
1098      PARSE_OPT_NOARG | PARSE_OPT_NONEG, NULL, UNMARK_FLAG},
1099    OPT_END()
1100  };
1101
1102  if (argc == 2 && !strcmp(argv[1], "-h"))
1103    usage_with_options(update_index_usage, options);
1104
1105  git_config(git_default_config, NULL);
1106
1107  prepare_repo_settings(r);
1108  the_repository->settings.command_requires_full_index = 0;
1109
1110  /* we will diagnose later if it turns out that we need to update it */
1111  newfd = hold_locked_index(&lock_file, 0);
1112  if (newfd < 0)
1113    lock_error = errno;
1114
1115  entries = read_cache();
1116  if (entries < 0)
1117    die("cache corrupted");
1118
1119  the_index.updated_skipworktree = 1;
1120
1121  /*
1122   * Custom copy of parse_options() because we want to handle
1123   * filename arguments as they come.
1124   */
1125  parse_options_start(&ctx, argc, argv, prefix,
1126          options, PARSE_OPT_STOP_AT_NON_OPTION);
1127
1128  /*
1129   * Allow the object layer to optimize adding multiple objects in
1130   * a batch.
1131   */
1132  begin_odb_transaction();
1133  while (ctx.argc) {
1134    if (parseopt_state != PARSE_OPT_DONE)
1135      parseopt_state = parse_options_step(&ctx, options,
1136                  update_index_usage);
1137    if (!ctx.argc)
1138      break;
1139    switch (parseopt_state) {
1140    case PARSE_OPT_HELP:
1141    case PARSE_OPT_ERROR:
1142      exit(129);
1143    case PARSE_OPT_COMPLETE:
1144      exit(0);
1145    case PARSE_OPT_NON_OPTION:
1146    case PARSE_OPT_DONE:
1147    {
1148      const char *path = ctx.argv[0];
1149      char *p;
1150
1151      setup_work_tree();
1152      p = prefix_path(prefix, prefix_length, path);
1153      update_one(p);
1154      if (set_executable_bit)
1155        chmod_path(set_executable_bit, p);
1156      free(p);
1157      ctx.argc--;
1158      ctx.argv++;
1159      break;
1160    }
1161    case PARSE_OPT_UNKNOWN:
1162      if (ctx.argv[0][1] == '-')
1163        error("unknown option '%s'", ctx.argv[0] + 2);
1164      else
1165        error("unknown switch '%c'", *ctx.opt);
1166      usage_with_options(update_index_usage, options);
1167    }
1168  }
1169  argc = parse_options_end(&ctx);
1170
1171  getline_fn = nul_term_line ? strbuf_getline_nul : strbuf_getline_lf;
1172  if (preferred_index_format) {
1173    if (preferred_index_format < INDEX_FORMAT_LB ||
1174        INDEX_FORMAT_UB < preferred_index_format)
1175      die("index-version %d not in range: %d..%d",
1176          preferred_index_format,
1177          INDEX_FORMAT_LB, INDEX_FORMAT_UB);
1178
1179    if (the_index.version != preferred_index_format)
1180      active_cache_changed |= SOMETHING_CHANGED;
1181    the_index.version = preferred_index_format;
1182  }
1183
1184  if (read_from_stdin) {
1185    struct strbuf buf = STRBUF_INIT;
1186    struct strbuf unquoted = STRBUF_INIT;
1187
1188    setup_work_tree();
1189    while (getline_fn(&buf, stdin) != EOF) {
1190      char *p;
1191      if (!nul_term_line && buf.buf[0] == '"') {
1192        strbuf_reset(&unquoted);
1193        if (unquote_c_style(&unquoted, buf.buf, NULL))
1194          die("line is badly quoted");
1195        strbuf_swap(&buf, &unquoted);
1196      }
1197      p = prefix_path(prefix, prefix_length, buf.buf);
1198      update_one(p);
1199      if (set_executable_bit)
1200        chmod_path(set_executable_bit, p);
1201      free(p);
1202    }
1203    strbuf_release(&unquoted);
1204    strbuf_release(&buf);
1205  }
1206
1207  /*
1208   * By now we have added all of the new objects
1209   */
1210  end_odb_transaction();
1211
1212  if (split_index > 0) {
1213    if (git_config_get_split_index() == 0)
1214      warning(_("core.splitIndex is set to false; "
1215          "remove or change it, if you really want to "
1216          "enable split index"));
1217    if (the_index.split_index)
1218      the_index.cache_changed |= SPLIT_INDEX_ORDERED;
1219    else
1220      add_split_index(&the_index);
1221  } else if (!split_index) {
1222    if (git_config_get_split_index() == 1)
1223      warning(_("core.splitIndex is set to true; "
1224          "remove or change it, if you really want to "
1225          "disable split index"));
1226    remove_split_index(&the_index);
1227  }
1228
1229  prepare_repo_settings(r);
1230  switch (untracked_cache) {
1231  case UC_UNSPECIFIED:
1232    break;
1233  case UC_DISABLE:
1234    if (r->settings.core_untracked_cache == UNTRACKED_CACHE_WRITE)
1235      warning(_("core.untrackedCache is set to true; "
1236          "remove or change it, if you really want to "
1237          "disable the untracked cache"));
1238    remove_untracked_cache(&the_index);
1239    report(_("Untracked cache disabled"));
1240    break;
1241  case UC_TEST:
1242    setup_work_tree();
1243    return !test_if_untracked_cache_is_supported();
1244  case UC_ENABLE:
1245  case UC_FORCE:
1246    if (r->settings.core_untracked_cache == UNTRACKED_CACHE_REMOVE)
1247      warning(_("core.untrackedCache is set to false; "
1248          "remove or change it, if you really want to "
1249          "enable the untracked cache"));
1250    add_untracked_cache(&the_index);
1251    report(_("Untracked cache enabled for '%s'"), get_git_work_tree());
1252    break;
1253  default:
1254    BUG("bad untracked_cache value: %d", untracked_cache);
1255  }
1256
1257  if (fsmonitor > 0) {
1258    enum fsmonitor_mode fsm_mode = fsm_settings__get_mode(r);
1259    enum fsmonitor_reason reason = fsm_settings__get_reason(r);
1260
1261    /*
1262     * The user wants to turn on FSMonitor using the command
1263     * line argument.  (We don't know (or care) whether that
1264     * is the IPC or HOOK version.)
1265     *
1266     * Use one of the __get routines to force load the FSMonitor
1267     * config settings into the repo-settings.  That will detect
1268     * whether the file system is compatible so that we can stop
1269     * here with a nice error message.
1270     */
1271    if (reason > FSMONITOR_REASON_OK)
1272      die("%s",
1273          fsm_settings__get_incompatible_msg(r, reason));
1274
1275    if (fsm_mode == FSMONITOR_MODE_DISABLED) {
1276      warning(_("core.fsmonitor is unset; "
1277        "set it if you really want to "
1278        "enable fsmonitor"));
1279    }
1280    add_fsmonitor(&the_index);
1281    report(_("fsmonitor enabled"));
1282  } else if (!fsmonitor) {
1283    enum fsmonitor_mode fsm_mode = fsm_settings__get_mode(r);
1284    if (fsm_mode > FSMONITOR_MODE_DISABLED)
1285      warning(_("core.fsmonitor is set; "
1286        "remove it if you really want to "
1287        "disable fsmonitor"));
1288    remove_fsmonitor(&the_index);
1289    report(_("fsmonitor disabled"));
1290  }
1291
1292  if (active_cache_changed || force_write) {
1293    if (newfd < 0) {
1294      if (refresh_args.flags & REFRESH_QUIET)
1295        exit(128);
1296      unable_to_lock_die(get_index_file(), lock_error);
1297    }
1298    if (write_locked_index(&the_index, &lock_file, COMMIT_LOCK))
1299      die("Unable to write new index file");
1300  }
1301
1302  rollback_lock_file(&lock_file);
1303
1304  return has_errors ? 1 : 0;
1305}
