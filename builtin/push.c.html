64static void refspec_append_mapped(struct refspec *refspec, const char *ref,
65          struct remote *remote, struct ref *local_refs)
66{
67  const char *branch_name;
68  struct ref *matched = NULL;
69
70  /* Does "ref" uniquely name our ref? */
71  if (count_refspec_match(ref, local_refs, &matched) != 1) {
72    refspec_append(refspec, ref);
73    return;
74  }
75
76  if (remote->push.nr) {
77    struct refspec_item query;
78    memset(&query, 0, sizeof(struct refspec_item));
79    query.src = matched->name;
80    if (!query_refspecs(&remote->push, &query) && query.dst) {
81      refspec_appendf(refspec, "%s%s:%s",
82          query.force ? "+" : "",
83          query.src, query.dst);
84      return;
85    }
86  }
87
88  if (push_default == PUSH_DEFAULT_UPSTREAM &&
89      skip_prefix(matched->name, "refs/heads/", &branch_name)) {
90    struct branch *branch = branch_get(branch_name);
91    if (branch->merge_nr == 1 && branch->merge[0]->src) {
92      refspec_appendf(refspec, "%s:%s",
93          ref, branch->merge[0]->src);
94      return;
95    }
96  }
97
98  refspec_append(refspec, ref);
99}

101static void set_refspecs(const char **refs, int nr, const char *repo)
102{
103  struct remote *remote = NULL;
104  struct ref *local_refs = NULL;
105  int i;
106
107  for (i = 0; i < nr; i++) {
108    const char *ref = refs[i];
109    if (!strcmp("tag", ref)) {
110      if (nr <= ++i)
111        die(_("tag shorthand without <tag>"));
112      ref = refs[i];
113      if (deleterefs)
114        refspec_appendf(&rs, ":refs/tags/%s", ref);
115      else
116        refspec_appendf(&rs, "refs/tags/%s", ref);
117    } else if (deleterefs) {
118      if (strchr(ref, ':') || !*ref)
119        die(_("--delete only accepts plain target ref names"));
120      refspec_appendf(&rs, ":%s", ref);
121    } else if (!strchr(ref, ':')) {
122      if (!remote) {
123        /* lazily grab remote and local_refs */
124        remote = remote_get(repo);
125        local_refs = get_local_heads();
126      }
127      refspec_append_mapped(&rs, ref, remote, local_refs);
128    } else
129      refspec_append(&rs, ref);
130  }
131}

143static NORETURN void die_push_simple(struct branch *branch,
144             struct remote *remote)
145{
146  /*
147   * There's no point in using shorten_unambiguous_ref here,
148   * as the ambiguity would be on the remote side, not what
149   * we have locally. Plus, this is supposed to be the simple
150   * mode. If the user is doing something crazy like setting
151   * upstream to a non-branch, we should probably be showing
152   * them the big ugly fully qualified ref.
153   */
154  const char *advice_pushdefault_maybe = "";
155  const char *advice_automergesimple_maybe = "";
156  const char *short_upstream = branch->merge[0]->src;
157
158  skip_prefix(short_upstream, "refs/heads/", &short_upstream);
159
160  /*
161   * Don't show advice for people who explicitly set
162   * push.default.
163   */
164  if (push_default == PUSH_DEFAULT_UNSPECIFIED)
165    advice_pushdefault_maybe = _("\n"
166         "To choose either option permanently, "
167         "see push.default in 'git help config'.\n");
168  if (git_branch_track != BRANCH_TRACK_SIMPLE)
169    advice_automergesimple_maybe = _("\n"
170         "To avoid automatically configuring "
171         "upstream branches when their name\n"
172         "doesn't match the local branch, see option "
173         "'simple' of branch.autoSetupMerge\n"
174         "in 'git help config'.\n");
175  die(_("The upstream branch of your current branch does not match\n"
176        "the name of your current branch.  To push to the upstream branch\n"
177        "on the remote, use\n"
178        "\n"
179        "    git push %s HEAD:%s\n"
180        "\n"
181        "To push to the branch of the same name on the remote, use\n"
182        "\n"
183        "    git push %s HEAD\n"
184        "%s%s"),
185      remote->name, short_upstream,
186      remote->name, advice_pushdefault_maybe,
187      advice_automergesimple_maybe);
188}

197static const char *get_upstream_ref(int flags, struct branch *branch, const char *remote_name)
198{
199  if (branch->merge_nr == 0 && (flags & TRANSPORT_PUSH_AUTO_UPSTREAM)) {
200    /* if missing, assume same; set_upstream will be defined later */
201    return branch->refname;
202  }
203
204  if (!branch->merge_nr || !branch->merge || !branch->remote_name) {
205    const char *advice_autosetup_maybe = "";
206    if (!(flags & TRANSPORT_PUSH_AUTO_UPSTREAM)) {
207      advice_autosetup_maybe = _("\n"
208             "To have this happen automatically for "
209             "branches without a tracking\n"
210             "upstream, see 'push.autoSetupRemote' "
211             "in 'git help config'.\n");
212    }
213    die(_("The current branch %s has no upstream branch.\n"
214        "To push the current branch and set the remote as upstream, use\n"
215        "\n"
216        "    git push --set-upstream %s %s\n"
217        "%s"),
218        branch->name,
219        remote_name,
220        branch->name,
221        advice_autosetup_maybe);
222  }
223  if (branch->merge_nr != 1)
224    die(_("The current branch %s has multiple upstream branches, "
225        "refusing to push."), branch->name);
226
227  return branch->merge[0]->src;
228}

230static void setup_default_push_refspecs(int *flags, struct remote *remote)
231{
232  struct branch *branch;
233  const char *dst;
234  int same_remote;
235
236  switch (push_default) {
237  case PUSH_DEFAULT_MATCHING:
238    refspec_append(&rs, ":");
239    return;
240
241  case PUSH_DEFAULT_NOTHING:
242    die(_("You didn't specify any refspecs to push, and "
243        "push.default is \"nothing\"."));
244    return;
245  default:
246    break;
247  }
248
249  branch = branch_get(NULL);
250  if (!branch)
251    die(_(message_detached_head_die), remote->name);
252
253  dst = branch->refname;
254  same_remote = !strcmp(remote->name, remote_for_branch(branch, NULL));
255
256  switch (push_default) {
257  default:
258  case PUSH_DEFAULT_UNSPECIFIED:
259  case PUSH_DEFAULT_SIMPLE:
260    if (!same_remote)
261      break;
262    if (strcmp(branch->refname, get_upstream_ref(*flags, branch, remote->name)))
263      die_push_simple(branch, remote);
264    break;
265
266  case PUSH_DEFAULT_UPSTREAM:
267    if (!same_remote)
268      die(_("You are pushing to remote '%s', which is not the upstream of\n"
269            "your current branch '%s', without telling me what to push\n"
270            "to update which remote branch."),
271          remote->name, branch->name);
272    dst = get_upstream_ref(*flags, branch, remote->name);
273    break;
274
275  case PUSH_DEFAULT_CURRENT:
276    break;
277  }
278
279  /*
280   * this is a default push - if auto-upstream is enabled and there is
281   * no upstream defined, then set it (with options 'simple', 'upstream',
282   * and 'current').
283   */
284  if ((*flags & TRANSPORT_PUSH_AUTO_UPSTREAM) && branch->merge_nr == 0)
285    *flags |= TRANSPORT_PUSH_SET_UPSTREAM;
286
287  refspec_appendf(&rs, "%s:%s", branch->refname, dst);
288}

365static int push_with_options(struct transport *transport, struct refspec *rs,
366           int flags)
367{
368  int err;
369  unsigned int reject_reasons;
370  char *anon_url = transport_anonymize_url(transport->url);
371
372  transport_set_verbosity(transport, verbosity, progress);
373  transport->family = family;
374
375  if (receivepack)
376    transport_set_option(transport,
377             TRANS_OPT_RECEIVEPACK, receivepack);
378  transport_set_option(transport, TRANS_OPT_THIN, thin ? "yes" : NULL);
379
380  if (!is_empty_cas(&cas)) {
381    if (!transport->smart_options)
382      die("underlying transport does not support --%s option",
383          CAS_OPT_NAME);
384    transport->smart_options->cas = &cas;
385  }
386
387  if (verbosity > 0)
388    fprintf(stderr, _("Pushing to %s\n"), anon_url);
389  trace2_region_enter("push", "transport_push", the_repository);
390  err = transport_push(the_repository, transport,
391           rs, flags, &reject_reasons);
392  trace2_region_leave("push", "transport_push", the_repository);
393  if (err != 0) {
394    fprintf(stderr, "%s", push_get_color(PUSH_COLOR_ERROR));
395    error(_("failed to push some refs to '%s'"), anon_url);
396    fprintf(stderr, "%s", push_get_color(PUSH_COLOR_RESET));
397  }
398
399  err |= transport_disconnect(transport);
400  free(anon_url);
401  if (!err)
402    return 0;
403
404  if (reject_reasons & REJECT_NON_FF_HEAD) {
405    advise_pull_before_push();
406  } else if (reject_reasons & REJECT_NON_FF_OTHER) {
407    advise_checkout_pull_push();
408  } else if (reject_reasons & REJECT_ALREADY_EXISTS) {
409    advise_ref_already_exists();
410  } else if (reject_reasons & REJECT_FETCH_FIRST) {
411    advise_ref_fetch_first();
412  } else if (reject_reasons & REJECT_NEEDS_FORCE) {
413    advise_ref_needs_force();
414  } else if (reject_reasons & REJECT_REF_NEEDS_UPDATE) {
415    advise_ref_needs_update();
416  }
417
418  return 1;
419}

421static int do_push(int flags,
422       const struct string_list *push_options,
423       struct remote *remote)
424{
425  int i, errs;
426  const char **url;
427  int url_nr;
428  struct refspec *push_refspec = &rs;
429
430  if (push_options->nr)
431    flags |= TRANSPORT_PUSH_OPTIONS;
432
433  if (!push_refspec->nr && !(flags & TRANSPORT_PUSH_ALL)) {
434    if (remote->push.nr) {
435      push_refspec = &remote->push;
436    } else if (!(flags & TRANSPORT_PUSH_MIRROR))
437      setup_default_push_refspecs(&flags, remote);
438  }
439  errs = 0;
440  url_nr = push_url_of_remote(remote, &url);
441  if (url_nr) {
442    for (i = 0; i < url_nr; i++) {
443      struct transport *transport =
444        transport_get(remote, url[i]);
445      if (flags & TRANSPORT_PUSH_OPTIONS)
446        transport->push_options = push_options;
447      if (push_with_options(transport, push_refspec, flags))
448        errs++;
449    }
450  } else {
451    struct transport *transport =
452      transport_get(remote, NULL);
453    if (flags & TRANSPORT_PUSH_OPTIONS)
454      transport->push_options = push_options;
455    if (push_with_options(transport, push_refspec, flags))
456      errs++;
457  }
458  return !!errs;
459}

492static int git_push_config(const char *k, const char *v, void *cb)
493{
494  const char *slot_name;
495  int *flags = cb;
496  int status;
497
498  status = git_gpg_config(k, v, NULL);
499  if (status)
500    return status;
501
502  if (!strcmp(k, "push.followtags")) {
503    if (git_config_bool(k, v))
504      *flags |= TRANSPORT_PUSH_FOLLOW_TAGS;
505    else
506      *flags &= ~TRANSPORT_PUSH_FOLLOW_TAGS;
507    return 0;
508  } else if (!strcmp(k, "push.autosetupremote")) {
509    if (git_config_bool(k, v))
510      *flags |= TRANSPORT_PUSH_AUTO_UPSTREAM;
511    return 0;
512  } else if (!strcmp(k, "push.gpgsign")) {
513    const char *value;
514    if (!git_config_get_value("push.gpgsign", &value)) {
515      switch (git_parse_maybe_bool(value)) {
516      case 0:
517        set_push_cert_flags(flags, SEND_PACK_PUSH_CERT_NEVER);
518        break;
519      case 1:
520        set_push_cert_flags(flags, SEND_PACK_PUSH_CERT_ALWAYS);
521        break;
522      default:
523        if (value && !strcasecmp(value, "if-asked"))
524          set_push_cert_flags(flags, SEND_PACK_PUSH_CERT_IF_ASKED);
525        else
526          return error(_("invalid value for '%s'"), k);
527      }
528    }
529  } else if (!strcmp(k, "push.recursesubmodules")) {
530    const char *value;
531    if (!git_config_get_value("push.recursesubmodules", &value))
532      recurse_submodules = parse_push_recurse_submodules_arg(k, value);
533  } else if (!strcmp(k, "submodule.recurse")) {
534    int val = git_config_bool(k, v) ?
535      RECURSE_SUBMODULES_ON_DEMAND : RECURSE_SUBMODULES_OFF;
536    recurse_submodules = val;
537  } else if (!strcmp(k, "push.pushoption")) {
538    if (!v)
539      return config_error_nonbool(k);
540    else
541      if (!*v)
542        string_list_clear(&push_options_config, 0);
543      else
544        string_list_append(&push_options_config, v);
545    return 0;
546  } else if (!strcmp(k, "color.push")) {
547    push_use_color = git_config_colorbool(k, v);
548    return 0;
549  } else if (skip_prefix(k, "color.push.", &slot_name)) {
550    int slot = parse_push_color_slot(slot_name);
551    if (slot < 0)
552      return 0;
553    if (!v)
554      return config_error_nonbool(k);
555    return color_parse(v, push_colors[slot]);
556  } else if (!strcmp(k, "push.useforceifincludes")) {
557    if (git_config_bool(k, v))
558      *flags |= TRANSPORT_PUSH_FORCE_IF_INCLUDES;
559    else
560      *flags &= ~TRANSPORT_PUSH_FORCE_IF_INCLUDES;
561    return 0;
562  }
563
564  return git_default_config(k, v, NULL);
565}
566
567int cmd_push(int argc, const char **argv, const char *prefix)
568{
569  int flags = 0;
570  int tags = 0;
571  int push_cert = -1;
572  int rc;
573  const char *repo = NULL;  /* default repository */
574  struct string_list push_options_cmdline = STRING_LIST_INIT_DUP;
575  struct string_list *push_options;
576  const struct string_list_item *item;
577  struct remote *remote;
578
579  struct option options[] = {
580    OPT__VERBOSITY(&verbosity),
581    OPT_STRING( 0 , "repo", &repo, N_("repository"), N_("repository")),
582    OPT_BIT( 0 , "all", &flags, N_("push all refs"), TRANSPORT_PUSH_ALL),
583    OPT_BIT( 0 , "mirror", &flags, N_("mirror all refs"),
584          (TRANSPORT_PUSH_MIRROR|TRANSPORT_PUSH_FORCE)),
585    OPT_BOOL('d', "delete", &deleterefs, N_("delete refs")),
586    OPT_BOOL( 0 , "tags", &tags, N_("push tags (can't be used with --all or --mirror)")),
587    OPT_BIT('n' , "dry-run", &flags, N_("dry run"), TRANSPORT_PUSH_DRY_RUN),
588    OPT_BIT( 0,  "porcelain", &flags, N_("machine-readable output"), TRANSPORT_PUSH_PORCELAIN),
589    OPT_BIT('f', "force", &flags, N_("force updates"), TRANSPORT_PUSH_FORCE),
590    OPT_CALLBACK_F(0, CAS_OPT_NAME, &cas, N_("<refname>:<expect>"),
591             N_("require old value of ref to be at this value"),
592             PARSE_OPT_OPTARG | PARSE_OPT_LITERAL_ARGHELP, parseopt_push_cas_option),
593    OPT_BIT(0, TRANS_OPT_FORCE_IF_INCLUDES, &flags,
594      N_("require remote updates to be integrated locally"),
595      TRANSPORT_PUSH_FORCE_IF_INCLUDES),
596    OPT_CALLBACK(0, "recurse-submodules", &recurse_submodules, "(check|on-demand|no)",
597           N_("control recursive pushing of submodules"), option_parse_recurse_submodules),
598    OPT_BOOL_F( 0 , "thin", &thin, N_("use thin pack"), PARSE_OPT_NOCOMPLETE),
599    OPT_STRING( 0 , "receive-pack", &receivepack, "receive-pack", N_("receive pack program")),
600    OPT_STRING( 0 , "exec", &receivepack, "receive-pack", N_("receive pack program")),
601    OPT_BIT('u', "set-upstream", &flags, N_("set upstream for git pull/status"),
602      TRANSPORT_PUSH_SET_UPSTREAM),
603    OPT_BOOL(0, "progress", &progress, N_("force progress reporting")),
604    OPT_BIT(0, "prune", &flags, N_("prune locally removed refs"),
605      TRANSPORT_PUSH_PRUNE),
606    OPT_BIT(0, "no-verify", &flags, N_("bypass pre-push hook"), TRANSPORT_PUSH_NO_HOOK),
607    OPT_BIT(0, "follow-tags", &flags, N_("push missing but relevant tags"),
608      TRANSPORT_PUSH_FOLLOW_TAGS),
609    OPT_CALLBACK_F(0, "signed", &push_cert, "(yes|no|if-asked)", N_("GPG sign the push"),
610        PARSE_OPT_OPTARG, option_parse_push_signed),
611    OPT_BIT(0, "atomic", &flags, N_("request atomic transaction on remote side"), TRANSPORT_PUSH_ATOMIC),
612    OPT_STRING_LIST('o', "push-option", &push_options_cmdline, N_("server-specific"), N_("option to transmit")),
613    OPT_SET_INT('4', "ipv4", &family, N_("use IPv4 addresses only"),
614        TRANSPORT_FAMILY_IPV4),
615    OPT_SET_INT('6', "ipv6", &family, N_("use IPv6 addresses only"),
616        TRANSPORT_FAMILY_IPV6),
617    OPT_END()
618  };
619
620  packet_trace_identity("push");
621  git_config(git_push_config, &flags);
622  argc = parse_options(argc, argv, prefix, options, push_usage, 0);
623  push_options = (push_options_cmdline.nr
624    ? &push_options_cmdline
625    : &push_options_config);
626  set_push_cert_flags(&flags, push_cert);
627
628  if (deleterefs && (tags || (flags & (TRANSPORT_PUSH_ALL | TRANSPORT_PUSH_MIRROR))))
629    die(_("options '%s' and '%s' cannot be used together"), "--delete", "--all/--mirror/--tags");
630  if (deleterefs && argc < 2)
631    die(_("--delete doesn't make sense without any refs"));
632
633  if (recurse_submodules == RECURSE_SUBMODULES_CHECK)
634    flags |= TRANSPORT_RECURSE_SUBMODULES_CHECK;
635  else if (recurse_submodules == RECURSE_SUBMODULES_ON_DEMAND)
636    flags |= TRANSPORT_RECURSE_SUBMODULES_ON_DEMAND;
637  else if (recurse_submodules == RECURSE_SUBMODULES_ONLY)
638    flags |= TRANSPORT_RECURSE_SUBMODULES_ONLY;
639
640  if (tags)
641    refspec_append(&rs, "refs/tags/*");
642
643  if (argc > 0) {
644    repo = argv[0];
645    set_refspecs(argv + 1, argc - 1, repo);
646  }
647
648  remote = pushremote_get(repo);
649  if (!remote) {
650    if (repo)
651      die(_("bad repository '%s'"), repo);
652    die(_("No configured push destination.\n"
653        "Either specify the URL from the command-line or configure a remote repository using\n"
654        "\n"
655        "    git remote add <name> <url>\n"
656        "\n"
657        "and then push using the remote name\n"
658        "\n"
659        "    git push <name>\n"));
660  }
661
662  if (remote->mirror)
663    flags |= (TRANSPORT_PUSH_MIRROR|TRANSPORT_PUSH_FORCE);
664
665  if (flags & TRANSPORT_PUSH_ALL) {
666    if (tags)
667      die(_("options '%s' and '%s' cannot be used together"), "--all", "--tags");
668    if (argc >= 2)
669      die(_("--all can't be combined with refspecs"));
670  }
671  if (flags & TRANSPORT_PUSH_MIRROR) {
672    if (tags)
673      die(_("options '%s' and '%s' cannot be used together"), "--mirror", "--tags");
674    if (argc >= 2)
675      die(_("--mirror can't be combined with refspecs"));
676  }
677  if ((flags & TRANSPORT_PUSH_ALL) && (flags & TRANSPORT_PUSH_MIRROR))
678    die(_("options '%s' and '%s' cannot be used together"), "--all", "--mirror");
679
680  if (!is_empty_cas(&cas) && (flags & TRANSPORT_PUSH_FORCE_IF_INCLUDES))
681    cas.use_force_if_includes = 1;
682
683  for_each_string_list_item(item, push_options)
684    if (strchr(item->string, '\n'))
685      die(_("push options must not have new line characters"));
686
687  rc = do_push(flags, push_options, remote);
688  string_list_clear(&push_options_cmdline, 0);
689  string_list_clear(&push_options_config, 0);
690  if (rc == -1)
691    usage_with_options(push_usage, options);
692  else
693    return rc;
694}
