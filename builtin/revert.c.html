44static int option_parse_x(const struct option *opt,
45        const char *arg, int unset)
46{
47  struct replay_opts **opts_ptr = opt->value;
48  struct replay_opts *opts = *opts_ptr;
49
50  if (unset)
51    return 0;
52
53  ALLOC_GROW(opts->xopts, opts->xopts_nr + 1, opts->xopts_alloc);

58static int option_parse_m(const struct option *opt,
59        const char *arg, int unset)
60{
61  struct replay_opts *replay = opt->value;
62  char *end;
63
64  if (unset) {
65    replay->mainline = 0;
66    return 0;
67  }
68
69  replay->mainline = strtol(arg, &end, 10);
70  if (*end || replay->mainline <= 0)
71    return error(_("option `%s' expects a number greater than zero"),
72           opt->long_name);
73
74  return 0;
75}

78static void verify_opt_compatible(const char *me, const char *base_opt, ...)
79{
80  const char *this_opt;
81  va_list ap;
82
83  va_start(ap, base_opt);
84  while ((this_opt = va_arg(ap, const char *))) {
85    if (va_arg(ap, int))
86      break;
87  }
88  va_end(ap);
89
90  if (this_opt)
91    die(_("%s: %s cannot be used with %s"), me, this_opt, base_opt);
92}

94static int run_sequencer(int argc, const char **argv, struct replay_opts *opts)
95{
96  const char * const * usage_str = revert_or_cherry_pick_usage(opts);
97  const char *me = action_name(opts);
98  const char *cleanup_arg = NULL;
99  int cmd = 0;
100  struct option base_options[] = {
101    OPT_CMDMODE(0, "quit", &cmd, N_("end revert or cherry-pick sequence"), 'q'),
102    OPT_CMDMODE(0, "continue", &cmd, N_("resume revert or cherry-pick sequence"), 'c'),
103    OPT_CMDMODE(0, "abort", &cmd, N_("cancel revert or cherry-pick sequence"), 'a'),
104    OPT_CMDMODE(0, "skip", &cmd, N_("skip current commit and continue"), 's'),
105    OPT_CLEANUP(&cleanup_arg),
106    OPT_BOOL('n', "no-commit", &opts->no_commit, N_("don't automatically commit")),
107    OPT_BOOL('e', "edit", &opts->edit, N_("edit the commit message")),
108    OPT_NOOP_NOARG('r', NULL),
109    OPT_BOOL('s', "signoff", &opts->signoff, N_("add a Signed-off-by trailer")),
110    OPT_CALLBACK('m', "mainline", opts, N_("parent-number"),
111           N_("select mainline parent"), option_parse_m),
112    OPT_RERERE_AUTOUPDATE(&opts->allow_rerere_auto),
113    OPT_STRING(0, "strategy", &opts->strategy, N_("strategy"), N_("merge strategy")),
114    OPT_CALLBACK('X', "strategy-option", &opts, N_("option"),
115      N_("option for merge strategy"), option_parse_x),
116    { OPTION_STRING, 'S', "gpg-sign", &opts->gpg_sign, N_("key-id"),
117      N_("GPG sign commit"), PARSE_OPT_OPTARG, NULL, (intptr_t) "" },
118    OPT_END()
119  };
120  struct option *options = base_options;
121
122  if (opts->action == REPLAY_PICK) {
123    struct option cp_extra[] = {
124      OPT_BOOL('x', NULL, &opts->record_origin, N_("append commit name")),
125      OPT_BOOL(0, "ff", &opts->allow_ff, N_("allow fast-forward")),
126      OPT_BOOL(0, "allow-empty", &opts->allow_empty, N_("preserve initially empty commits")),
127      OPT_BOOL(0, "allow-empty-message", &opts->allow_empty_message, N_("allow commits with empty messages")),
128      OPT_BOOL(0, "keep-redundant-commits", &opts->keep_redundant_commits, N_("keep redundant, empty commits")),
129      OPT_END(),
130    };
131    options = parse_options_concat(options, cp_extra);
132  } else if (opts->action == REPLAY_REVERT) {
133    struct option cp_extra[] = {
134      OPT_BOOL(0, "reference", &opts->commit_use_reference,
135         N_("use the 'reference' format to refer to commits")),
136      OPT_END(),
137    };
138    options = parse_options_concat(options, cp_extra);
139  }
140
141  argc = parse_options(argc, argv, NULL, options, usage_str,
142      PARSE_OPT_KEEP_ARGV0 |
143      PARSE_OPT_KEEP_UNKNOWN);
144
145  prepare_repo_settings(the_repository);
146  the_repository->settings.command_requires_full_index = 0;
147
148  /* implies allow_empty */
149  if (opts->keep_redundant_commits)
150    opts->allow_empty = 1;
151
152  if (cleanup_arg) {
153    opts->default_msg_cleanup = get_cleanup_mode(cleanup_arg, 1);
154    opts->explicit_cleanup = 1;
155  }
156
157  /* Check for incompatible command line arguments */
158  if (cmd) {
159    char *this_operation;
160    if (cmd == 'q')
161      this_operation = "--quit";
162    else if (cmd == 'c')
163      this_operation = "--continue";
164    else if (cmd == 's')
165      this_operation = "--skip";
166    else {
167      assert(cmd == 'a');
168      this_operation = "--abort";
169    }
170
171    verify_opt_compatible(me, this_operation,
172        "--no-commit", opts->no_commit,
173        "--signoff", opts->signoff,
174        "--mainline", opts->mainline,
175        "--strategy", opts->strategy ? 1 : 0,
176        "--strategy-option", opts->xopts ? 1 : 0,
177        "-x", opts->record_origin,
178        "--ff", opts->allow_ff,
179        "--rerere-autoupdate", opts->allow_rerere_auto == RERERE_AUTOUPDATE,
180        "--no-rerere-autoupdate", opts->allow_rerere_auto == RERERE_NOAUTOUPDATE,
181        NULL);
182  }
183
184  if (!opts->strategy && opts->default_strategy) {
185    opts->strategy = opts->default_strategy;
186    opts->default_strategy = NULL;
187  }
188
189  if (opts->allow_ff)
190    verify_opt_compatible(me, "--ff",
191        "--signoff", opts->signoff,
192        "--no-commit", opts->no_commit,
193        "-x", opts->record_origin,
194        "--edit", opts->edit > 0,
195        NULL);
196
197  if (cmd) {
198    opts->revs = NULL;
199  } else {
200    struct setup_revision_opt s_r_opt;
201    opts->revs = xmalloc(sizeof(*opts->revs));
202    repo_init_revisions(the_repository, opts->revs, NULL);
203    opts->revs->no_walk = 1;
204    opts->revs->unsorted_input = 1;
205    if (argc < 2)
206      usage_with_options(usage_str, options);
207    if (!strcmp(argv[1], "-"))
208      argv[1] = "@{-1}";
209    memset(&s_r_opt, 0, sizeof(s_r_opt));
210    s_r_opt.assume_dashdash = 1;
211    argc = setup_revisions(argc, argv, opts->revs, &s_r_opt);
212  }
213
214  if (argc > 1)
215    usage_with_options(usage_str, options);
216
217  /* These option values will be free()d */
218  opts->gpg_sign = xstrdup_or_null(opts->gpg_sign);
219  opts->strategy = xstrdup_or_null(opts->strategy);
220  if (!opts->strategy && getenv("GIT_TEST_MERGE_ALGORITHM"))
221    opts->strategy = xstrdup(getenv("GIT_TEST_MERGE_ALGORITHM"));
222
223  if (cmd == 'q') {
224    int ret = sequencer_remove_state(opts);
225    if (!ret)
226      remove_branch_state(the_repository, 0);
227    return ret;
228  }
229  if (cmd == 'c')
230    return sequencer_continue(the_repository, opts);
231  if (cmd == 'a')
232    return sequencer_rollback(the_repository, opts);
233  if (cmd == 's')
234    return sequencer_skip(the_repository, opts);
235  return sequencer_pick_revisions(the_repository, opts);
236}

238int cmd_revert(int argc, const char **argv, const char *prefix)
239{
240  struct replay_opts opts = REPLAY_OPTS_INIT;
241  int res;
242
243  opts.action = REPLAY_REVERT;
244  sequencer_init_config(&opts);
245  res = run_sequencer(argc, argv, &opts);
246  if (res < 0)
247    die(_("revert failed"));
248  return res;
