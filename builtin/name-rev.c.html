40/* Cutoff searching any commits older than this one */
41static void set_commit_cutoff(struct commit *commit)
42{
43
44  if (cutoff > commit->date)
45    cutoff = commit->date;
46
47  if (generation_cutoff) {
48    timestamp_t generation = commit_graph_generation(commit);
49
50    if (generation_cutoff > generation)
51      generation_cutoff = generation;

55/* adjust the commit date cutoff with a slop to allow for slightly incorrect
56 * commit timestamps in case of clock skew.

58static void adjust_cutoff_timestamp_for_slop(void)
59{

101static int is_better_name(struct rev_name *name,
102        timestamp_t taggerdate,
103        int generation,
104        int distance,
105        int from_tag)
106{
107  int name_distance = effective_distance(name->distance, name->generation);
108  int new_distance = effective_distance(distance, generation);
109
110  /*
111   * When comparing names based on tags, prefer names
112   * based on the older tag, even if it is farther away.
113   */
114  if (from_tag && name->from_tag)
115    return (name->taggerdate > taggerdate ||
116      (name->taggerdate == taggerdate &&
117       name_distance > new_distance));
118
119  /*
120   * We know that at least one of them is a non-tag at this point.
121   * favor a tag over a non-tag.
122   */
123  if (name->from_tag != from_tag)
124    return from_tag;
125
126  /*
127   * We are now looking at two non-tags.  Tiebreak to favor
128   * shorter hops.
129   */
130  if (name_distance != new_distance)
131    return name_distance > new_distance;
132
133  /* ... or tiebreak to favor older date */
134  if (name->taggerdate != taggerdate)
135    return name->taggerdate > taggerdate;
136
137  /* keep the current one if we cannot decide */
138  return 0;
139}

181static void name_rev(struct commit *start_commit,
182    const char *tip_name, timestamp_t taggerdate,
183    int from_tag, int deref)
184{
185  struct prio_queue queue;
186  struct commit *commit;
187  struct commit **parents_to_queue = NULL;
188  size_t parents_to_queue_nr, parents_to_queue_alloc = 0;
189  struct rev_name *start_name;
190
191  parse_commit(start_commit);
192  if (commit_is_before_cutoff(start_commit))
193    return;
194
195  start_name = create_or_update_name(start_commit, taggerdate, 0, 0,
196             from_tag);
197  if (!start_name)
198    return;
199  if (deref)
200    start_name->tip_name = xstrfmt("%s^0", tip_name);
201  else
202    start_name->tip_name = xstrdup(tip_name);
203
204  memset(&queue, 0, sizeof(queue)); /* Use the prio_queue as LIFO */
205  prio_queue_put(&queue, start_commit);
206
207  while ((commit = prio_queue_get(&queue))) {
208    struct rev_name *name = get_commit_rev_name(commit);
209    struct commit_list *parents;
210    int parent_number = 1;
211
212    parents_to_queue_nr = 0;
213
214    for (parents = commit->parents;
215        parents;
216        parents = parents->next, parent_number++) {
217      struct commit *parent = parents->item;
218      struct rev_name *parent_name;
219      int generation, distance;
220
221      parse_commit(parent);
222      if (commit_is_before_cutoff(parent))
223        continue;
224
225      if (parent_number > 1) {
226        generation = 0;
227        distance = name->distance + MERGE_TRAVERSAL_WEIGHT;
228      } else {
229        generation = name->generation + 1;
230        distance = name->distance + 1;
231      }
232
233      parent_name = create_or_update_name(parent, taggerdate,
234                  generation,
235                  distance, from_tag);
236      if (parent_name) {
237        if (parent_number > 1)
238          parent_name->tip_name =
239            get_parent_name(name,
240                parent_number);
241        else
242          parent_name->tip_name = name->tip_name;
243        ALLOC_GROW(parents_to_queue,
244             parents_to_queue_nr + 1,
245             parents_to_queue_alloc);
246        parents_to_queue[parents_to_queue_nr] = parent;
247        parents_to_queue_nr++;
248      }
249    }
250
251    /* The first parent must come out first from the prio_queue */
252    while (parents_to_queue_nr)
253      prio_queue_put(&queue,
254               parents_to_queue[--parents_to_queue_nr]);
255  }
256
257  clear_prio_queue(&queue);
258  free(parents_to_queue);
259}

346static int name_ref(const char *path, const struct object_id *oid, int flags, void *cb_data)
347{
348  struct object *o = parse_object(the_repository, oid);
349  struct name_ref_data *data = cb_data;
350  int can_abbreviate_output = data->tags_only && data->name_only;
351  int deref = 0;
352  int from_tag = 0;
353  struct commit *commit = NULL;
354  timestamp_t taggerdate = TIME_MAX;
355
356  if (data->tags_only && !starts_with(path, "refs/tags/"))
357    return 0;
358
359  if (data->exclude_filters.nr) {
360    struct string_list_item *item;
361
362    for_each_string_list_item(item, &data->exclude_filters) {
363      if (subpath_matches(path, item->string) >= 0)
364        return 0;
365    }
366  }
367
368  if (data->ref_filters.nr) {
369    struct string_list_item *item;
370    int matched = 0;
371
372    /* See if any of the patterns match. */
373    for_each_string_list_item(item, &data->ref_filters) {
374      /*
375       * Check all patterns even after finding a match, so
376       * that we can see if a match with a subpath exists.
377       * When a user asked for 'refs/tags/v*' and 'v1.*',
378       * both of which match, the user is showing her
379       * willingness to accept a shortened output by having
380       * the 'v1.*' in the acceptable refnames, so we
381       * shouldn't stop when seeing 'refs/tags/v1.4' matches
382       * 'refs/tags/v*'.  We should show it as 'v1.4'.
383       */
384      switch (subpath_matches(path, item->string)) {
385      case -1: /* did not match */
386        break;
387      case 0: /* matched fully */
388        matched = 1;
389        break;
390      default: /* matched subpath */
391        matched = 1;
392        can_abbreviate_output = 1;
393        break;
394      }
395    }
396
397    /* If none of the patterns matched, stop now */
398    if (!matched)
399      return 0;
400  }
401
402  while (o && o->type == OBJ_TAG) {
403    struct tag *t = (struct tag *) o;
404    if (!t->tagged)
405      break; /* broken repository */
406    o = parse_object(the_repository, &t->tagged->oid);
407    deref = 1;
408    taggerdate = t->date;
409  }
410  if (o && o->type == OBJ_COMMIT) {
411    commit = (struct commit *)o;
412    from_tag = starts_with(path, "refs/tags/");
413    if (taggerdate == TIME_MAX)
414      taggerdate = commit->date;
415  }
416
417  add_to_tip_table(oid, path, can_abbreviate_output, commit, taggerdate,
418       from_tag, deref);
419  return 0;
420}

518static void name_rev_line(char *p, struct name_ref_data *data)
519{
520  struct strbuf buf = STRBUF_INIT;
521  int counter = 0;
522  char *p_start;
523  const unsigned hexsz = the_hash_algo->hexsz;
524
525  for (p_start = p; *p; p++) {
526#define ishex(x) (isdigit((x)) || ((x) >= 'a' && (x) <= 'f'))
527    if (!ishex(*p))
528      counter = 0;
529    else if (++counter == hexsz &&
530       !ishex(*(p+1))) {
531      struct object_id oid;
532      const char *name = NULL;
533      char c = *(p+1);
534      int p_len = p - p_start + 1;
535
536      counter = 0;
537
538      *(p+1) = 0;
539      if (!get_oid(p - (hexsz - 1), &oid)) {
540        struct object *o =
541          lookup_object(the_repository, &oid);
542        if (o)
543          name = get_rev_name(o, &buf);
544      }
545      *(p+1) = c;
546
547      if (!name)
548        continue;
549
550      if (data->name_only)
551        printf("%.*s%s", p_len - hexsz, p_start, name);
552      else
553        printf("%.*s (%s)", p_len, p_start, name);
554      p_start = p + 1;
555    }
556  }
557
558  /* flush */
559  if (p_start != p)
560    fwrite(p_start, p - p_start, 1, stdout);
561
562  strbuf_release(&buf);
563}

567  struct object_array revs = OBJECT_ARRAY_INIT;
568  int all = 0, annotate_stdin = 0, transform_stdin = 0, allow_undefined = 1, always = 0, peel_tag = 0;
569  struct name_ref_data data = { 0, 0, STRING_LIST_INIT_NODUP, STRING_LIST_INIT_NODUP };
570  struct option opts[] = {
571    OPT_BOOL(0, "name-only", &data.name_only, N_("print only ref-based names (no object names)")),
572    OPT_BOOL(0, "tags", &data.tags_only, N_("only use tags to name the commits")),
573    OPT_STRING_LIST(0, "refs", &data.ref_filters, N_("pattern"),
574           N_("only use refs matching <pattern>")),
575    OPT_STRING_LIST(0, "exclude", &data.exclude_filters, N_("pattern"),
576           N_("ignore refs matching <pattern>")),
577    OPT_GROUP(""),
578    OPT_BOOL(0, "all", &all, N_("list all commits reachable from all refs")),
579    OPT_BOOL(0, "stdin", &transform_stdin, N_("deprecated: use --annotate-stdin instead")),
580    OPT_BOOL(0, "annotate-stdin", &annotate_stdin, N_("annotate text from stdin")),
581    OPT_BOOL(0, "undefined", &allow_undefined, N_("allow to print `undefined` names (default)")),
582    OPT_BOOL(0, "always",     &always,
583         N_("show abbreviated commit object as fallback")),
584    {
585      /* A Hidden OPT_BOOL */
586      OPTION_SET_INT, 0, "peel-tag", &peel_tag, NULL,
587      N_("dereference tags in the input (internal use)"),
588      PARSE_OPT_NOARG | PARSE_OPT_HIDDEN, NULL, 1,
589    },
590    OPT_END(),
591  };
592
593  init_commit_rev_name(&rev_names);
594  git_config(git_default_config, NULL);
595  argc = parse_options(argc, argv, prefix, opts, name_rev_usage, 0);
596
597  if (transform_stdin) {
598    warning("--stdin is deprecated. Please use --annotate-stdin instead, "
599          "which is functionally equivalent.\n"
600          "This option will be removed in a future release.");
601    annotate_stdin = 1;
602  }
603
604  if (all + annotate_stdin + !!argc > 1) {
605    error("Specify either a list, or --all, not both!");
606    usage_with_options(name_rev_usage, opts);
607  }
608  if (all || annotate_stdin)
609    disable_cutoff();
610
611  for (; argc; argc--, argv++) {
612    struct object_id oid;
613    struct object *object;
614    struct commit *commit;
615
616    if (get_oid(*argv, &oid)) {
617      fprintf(stderr, "Could not get sha1 for %s. Skipping.\n",
618          *argv);
619      continue;
620    }
621
622    commit = NULL;
623    object = parse_object(the_repository, &oid);
624    if (object) {
625      struct object *peeled = deref_tag(the_repository,
626                object, *argv, 0);
627      if (peeled && peeled->type == OBJ_COMMIT)
628        commit = (struct commit *)peeled;
629    }
630
631    if (!object) {
632      fprintf(stderr, "Could not get object for %s. Skipping.\n",
633          *argv);
634      continue;
635    }
636
637    if (commit)
638      set_commit_cutoff(commit);
639
640    if (peel_tag) {
641      if (!commit) {
642        fprintf(stderr, "Could not get commit for %s. Skipping.\n",
643          *argv);
644        continue;
645      }
646      object = (struct object *)commit;
647    }
648    add_object_array(object, *argv, &revs);
649  }
650
651  adjust_cutoff_timestamp_for_slop();
652
653  for_each_ref(name_ref, &data);
654  name_tips();
655
656  if (annotate_stdin) {
657    struct strbuf sb = STRBUF_INIT;
658
659    while (strbuf_getline(&sb, stdin) != EOF) {
660      strbuf_addch(&sb, '\n');
661      name_rev_line(sb.buf, &data);
662    }
663    strbuf_release(&sb);
664  } else if (all) {
665    int i, max;
666
667    max = get_max_object_index();
668    for (i = 0; i < max; i++) {
669      struct object *obj = get_indexed_object(i);
670      if (!obj || obj->type != OBJ_COMMIT)
671        continue;
672      show_name(obj, NULL,
673          always, allow_undefined, data.name_only);
674    }
675  } else {
676    int i;
677    for (i = 0; i < revs.nr; i++)
678      show_name(revs.objects[i].item, revs.objects[i].name,
679          always, allow_undefined, data.name_only);
680  }
681
682  UNLEAK(revs);
683  return 0;
684}
