49static int reset_index(const char *ref, const struct object_id *oid, int reset_type, int quiet)
50{
51  int i, nr = 0;
52  struct tree_desc desc[2];
53  struct tree *tree;
54  struct unpack_trees_options opts;
55  int ret = -1;
56
57  memset(&opts, 0, sizeof(opts));
58  opts.head_idx = 1;
59  opts.src_index = &the_index;
60  opts.dst_index = &the_index;
61  opts.fn = oneway_merge;
62  opts.merge = 1;
63  init_checkout_metadata(&opts.meta, ref, oid, NULL);
64  if (!quiet)
65    opts.verbose_update = 1;
66  switch (reset_type) {
67  case KEEP:
68  case MERGE:
69    opts.update = 1;
70    opts.preserve_ignored = 0; /* FIXME: !overwrite_ignore */
71    break;
72  case HARD:
73    opts.update = 1;
74    opts.reset = UNPACK_RESET_OVERWRITE_UNTRACKED;
75    break;
76  case MIXED:
77    opts.reset = UNPACK_RESET_PROTECT_UNTRACKED;
78    /* but opts.update=0, so working tree not updated */
79    break;
80  default:
81    BUG("invalid reset_type passed to reset_index");
82  }
83
84  read_cache_unmerged();
85
86  if (reset_type == KEEP) {
87    struct object_id head_oid;
88    if (get_oid("HEAD", &head_oid))
89      return error(_("You do not have a valid HEAD."));
90    if (!fill_tree_descriptor(the_repository, desc + nr, &head_oid))
91      return error(_("Failed to find tree of HEAD."));
92    nr++;
93    opts.fn = twoway_merge;
94  }
95
96  if (!fill_tree_descriptor(the_repository, desc + nr, oid)) {
97    error(_("Failed to find tree of %s."), oid_to_hex(oid));
98    goto out;
99  }
100  nr++;
101
102  if (unpack_trees(nr, desc, &opts))
103    goto out;
104
105  if (reset_type == MIXED || reset_type == HARD) {
106    tree = parse_tree_indirect(oid);
107    prime_cache_tree(the_repository, the_repository->index, tree);
108  }
109
110  ret = 0;
111
112out:
113  for (i = 0; i < nr; i++)
114    free((void *)desc[i].buffer);
115  return ret;
116}

132static void update_index_from_diff(struct diff_queue_struct *q,
133    struct diff_options *opt, void *data)
134{
135  int i;
136  int intent_to_add = *(int *)data;
137
138  for (i = 0; i < q->nr; i++) {
139    int pos;
140    struct diff_filespec *one = q->queue[i]->one;
141    int is_in_reset_tree = one->mode && !is_null_oid(&one->oid);
142    struct cache_entry *ce;
143
144    if (!is_in_reset_tree && !intent_to_add) {
145      remove_file_from_cache(one->path);
146      continue;
147    }
148
149    ce = make_cache_entry(&the_index, one->mode, &one->oid, one->path,
150              0, 0);
151
152    /*
153     * If the file 1) corresponds to an existing index entry with
154     * skip-worktree set, or 2) does not exist in the index but is
155     * outside the sparse checkout definition, add a skip-worktree bit
156     * to the new index entry. Note that a sparse index will be expanded
157     * if this entry is outside the sparse cone - this is necessary
158     * to properly construct the reset sparse directory.
159     */
160    pos = cache_name_pos(one->path, strlen(one->path));
161    if ((pos >= 0 && ce_skip_worktree(active_cache[pos])) ||
162        (pos < 0 && !path_in_sparse_checkout(one->path, &the_index)))
163      ce->ce_flags |= CE_SKIP_WORKTREE;
164
165    if (!ce)
166      die(_("make_cache_entry failed for path '%s'"),
167          one->path);
168    if (!is_in_reset_tree) {
169      ce->ce_flags |= CE_INTENT_TO_ADD;
170      set_object_name_for_intent_to_add_entry(ce);
171    }
172    add_cache_entry(ce, ADD_CACHE_OK_TO_ADD | ADD_CACHE_OK_TO_REPLACE);
173  }
174}

176static int pathspec_needs_expanded_index(const struct pathspec *pathspec)
177{
178  unsigned int i, pos;
179  int res = 0;
180  char *skip_worktree_seen = NULL;
181
182  /*
183   * When using a magic pathspec, assume for the sake of simplicity that
184   * the index needs to be expanded to match all matchable files.
185   */
186  if (pathspec->magic)
187    return 1;
188
189  for (i = 0; i < pathspec->nr; i++) {
190    struct pathspec_item item = pathspec->items[i];
191
192    /*
193     * If the pathspec item has a wildcard, the index should be expanded
194     * if the pathspec has the possibility of matching a subset of entries inside
195     * of a sparse directory (but not the entire directory).
196     *
197     * If the pathspec item is a literal path, the index only needs to be expanded
198     * if a) the pathspec isn't in the sparse checkout cone (to make sure we don't
199     * expand for in-cone files) and b) it doesn't match any sparse directories
200     * (since we can reset whole sparse directories without expanding them).
201     */
202    if (item.nowildcard_len < item.len) {
203      /*
204       * Special case: if the pattern is a path inside the cone
205       * followed by only wildcards, the pattern cannot match
206       * partial sparse directories, so we know we don't need to
207       * expand the index.
208       *
209       * Examples:
210       * - in-cone/foo***: doesn't need expanded index
211       * - not-in-cone/bar*: may need expanded index
212       * - **.c: may need expanded index
213       */
214      if (strspn(item.original + item.nowildcard_len, "*") == item.len - item.nowildcard_len &&
215          path_in_cone_mode_sparse_checkout(item.original, &the_index))
216        continue;
217
218      for (pos = 0; pos < active_nr; pos++) {
219        struct cache_entry *ce = active_cache[pos];
220
221        if (!S_ISSPARSEDIR(ce->ce_mode))
222          continue;
223
224        /*
225         * If the pre-wildcard length is longer than the sparse
226         * directory name and the sparse directory is the first
227         * component of the pathspec, need to expand the index.
228         */
229        if (item.nowildcard_len > ce_namelen(ce) &&
230            !strncmp(item.original, ce->name, ce_namelen(ce))) {
231          res = 1;
232          break;
233        }
234
235        /*
236         * If the pre-wildcard length is shorter than the sparse
237         * directory and the pathspec does not match the whole
238         * directory, need to expand the index.
239         */
240        if (!strncmp(item.original, ce->name, item.nowildcard_len) &&
241            wildmatch(item.original, ce->name, 0)) {
242          res = 1;
243          break;
244        }
245      }
246    } else if (!path_in_cone_mode_sparse_checkout(item.original, &the_index) &&
247         !matches_skip_worktree(pathspec, i, &skip_worktree_seen))
248      res = 1;
249
250    if (res > 0)
251      break;
252  }
253
254  free(skip_worktree_seen);
255  return res;
256}

258static int read_from_tree(const struct pathspec *pathspec,
259        struct object_id *tree_oid,
260        int intent_to_add)
261{
262  struct diff_options opt;
263
264  memset(&opt, 0, sizeof(opt));
265  copy_pathspec(&opt.pathspec, pathspec);
266  opt.output_format = DIFF_FORMAT_CALLBACK;
267  opt.format_callback = update_index_from_diff;
268  opt.format_callback_data = &intent_to_add;
269  opt.flags.override_submodule_config = 1;
270  opt.flags.recursive = 1;
271  opt.repo = the_repository;
272  opt.change = diff_change;
273  opt.add_remove = diff_addremove;
274
275  if (pathspec->nr && the_index.sparse_index && pathspec_needs_expanded_index(pathspec))
276    ensure_full_index(&the_index);
277
278  if (do_diff_cache(tree_oid, &opt))
279    return 1;
280  diffcore_std(&opt);
281  diff_flush(&opt);
282
283  return 0;
284}

308static void parse_args(struct pathspec *pathspec,
309           const char **argv, const char *prefix,
310           int patch_mode,
311           const char **rev_ret)
312{
313  const char *rev = "HEAD";
314  struct object_id unused;
315  /*
316   * Possible arguments are:
317   *
318   * git reset [-opts] [<rev>]
319   * git reset [-opts] <tree> [<paths>...]
320   * git reset [-opts] <tree> -- [<paths>...]
321   * git reset [-opts] -- [<paths>...]
322   * git reset [-opts] <paths>...
323   *
324   * At this point, argv points immediately after [-opts].
325   */
326
327  if (argv[0]) {
328    if (!strcmp(argv[0], "--")) {
329      argv++; /* reset to HEAD, possibly with paths */
330    } else if (argv[1] && !strcmp(argv[1], "--")) {
331      rev = argv[0];
332      argv += 2;
333    }
334    /*
335     * Otherwise, argv[0] could be either <rev> or <paths> and
336     * has to be unambiguous. If there is a single argument, it
337     * can not be a tree
338     */
339    else if ((!argv[1] && !get_oid_committish(argv[0], &unused)) ||
340       (argv[1] && !get_oid_treeish(argv[0], &unused))) {
341      /*
342       * Ok, argv[0] looks like a commit/tree; it should not
343       * be a filename.
344       */
345      verify_non_filename(prefix, argv[0]);
346      rev = *argv++;
347    } else {
348      /* Otherwise we treat this as a filename */
349      verify_filename(prefix, argv[0], 1);
350    }
351  }
352  *rev_ret = rev;
353
354  parse_pathspec(pathspec, 0,
355           PATHSPEC_PREFER_FULL |
356           (patch_mode ? PATHSPEC_PREFIX_ORIGIN : 0),
357           prefix, argv);
358}

360static int reset_refs(const char *rev, const struct object_id *oid)
361{
362  int update_ref_status;
363  struct strbuf msg = STRBUF_INIT;
364  struct object_id *orig = NULL, oid_orig,
365    *old_orig = NULL, oid_old_orig;
366
367  if (!get_oid("ORIG_HEAD", &oid_old_orig))
368    old_orig = &oid_old_orig;
369  if (!get_oid("HEAD", &oid_orig)) {
370    orig = &oid_orig;
371    set_reflog_message(&msg, "updating ORIG_HEAD", NULL);
372    update_ref(msg.buf, "ORIG_HEAD", orig, old_orig, 0,
373         UPDATE_REFS_MSG_ON_ERR);
374  } else if (old_orig)
375    delete_ref(NULL, "ORIG_HEAD", old_orig, 0);
376  set_reflog_message(&msg, "updating HEAD", rev);
377  update_ref_status = update_ref(msg.buf, "HEAD", oid, orig, 0,
378               UPDATE_REFS_MSG_ON_ERR);
379  strbuf_release(&msg);
380  return update_ref_status;
381}

391int cmd_reset(int argc, const char **argv, const char *prefix)
392{
393  int reset_type = NONE, update_ref_status = 0, quiet = 0;
394  int no_refresh = 0;
395  int patch_mode = 0, pathspec_file_nul = 0, unborn;
396  const char *rev, *pathspec_from_file = NULL;
397  struct object_id oid;
398  struct pathspec pathspec;
399  int intent_to_add = 0;
400  const struct option options[] = {
401    OPT__QUIET(&quiet, N_("be quiet, only report errors")),
402    OPT_BOOL(0, "no-refresh", &no_refresh,
403        N_("skip refreshing the index after reset")),
404    OPT_SET_INT(0, "mixed", &reset_type,
405            N_("reset HEAD and index"), MIXED),
406    OPT_SET_INT(0, "soft", &reset_type, N_("reset only HEAD"), SOFT),
407    OPT_SET_INT(0, "hard", &reset_type,
408        N_("reset HEAD, index and working tree"), HARD),
409    OPT_SET_INT(0, "merge", &reset_type,
410        N_("reset HEAD, index and working tree"), MERGE),
411    OPT_SET_INT(0, "keep", &reset_type,
412        N_("reset HEAD but keep local changes"), KEEP),
413    OPT_CALLBACK_F(0, "recurse-submodules", NULL,
414          "reset", "control recursive updating of submodules",
415          PARSE_OPT_OPTARG, option_parse_recurse_submodules_worktree_updater),
416    OPT_BOOL('p', "patch", &patch_mode, N_("select hunks interactively")),
417    OPT_BOOL('N', "intent-to-add", &intent_to_add,
418        N_("record only the fact that removed paths will be added later")),
419    OPT_PATHSPEC_FROM_FILE(&pathspec_from_file),
420    OPT_PATHSPEC_FILE_NUL(&pathspec_file_nul),
421    OPT_END()
422  };
423
424  git_config(git_reset_config, NULL);
425
426  argc = parse_options(argc, argv, prefix, options, git_reset_usage,
427            PARSE_OPT_KEEP_DASHDASH);
428  parse_args(&pathspec, argv, prefix, patch_mode, &rev);
429
430  if (pathspec_from_file) {
431    if (patch_mode)
432      die(_("options '%s' and '%s' cannot be used together"), "--pathspec-from-file", "--patch");
433
434    if (pathspec.nr)
435      die(_("'%s' and pathspec arguments cannot be used together"), "--pathspec-from-file");
436
437    parse_pathspec_file(&pathspec, 0,
438            PATHSPEC_PREFER_FULL,
439            prefix, pathspec_from_file, pathspec_file_nul);
440  } else if (pathspec_file_nul) {
441    die(_("the option '%s' requires '%s'"), "--pathspec-file-nul", "--pathspec-from-file");
442  }
443
444  unborn = !strcmp(rev, "HEAD") && get_oid("HEAD", &oid);
445  if (unborn) {
446    /* reset on unborn branch: treat as reset to empty tree */
447    oidcpy(&oid, the_hash_algo->empty_tree);
448  } else if (!pathspec.nr && !patch_mode) {
449    struct commit *commit;
450    if (get_oid_committish(rev, &oid))
451      die(_("Failed to resolve '%s' as a valid revision."), rev);
452    commit = lookup_commit_reference(the_repository, &oid);
453    if (!commit)
454      die(_("Could not parse object '%s'."), rev);
455    oidcpy(&oid, &commit->object.oid);
456  } else {
457    struct tree *tree;
458    if (get_oid_treeish(rev, &oid))
459      die(_("Failed to resolve '%s' as a valid tree."), rev);
460    tree = parse_tree_indirect(&oid);
461    if (!tree)
462      die(_("Could not parse object '%s'."), rev);
463    oidcpy(&oid, &tree->object.oid);
464  }
465
466  if (patch_mode) {
467    if (reset_type != NONE)
468      die(_("options '%s' and '%s' cannot be used together"), "--patch", "--{hard,mixed,soft}");
469    trace2_cmd_mode("patch-interactive");
470    return run_add_interactive(rev, "--patch=reset", &pathspec);
471  }
472
473  /* git reset tree [--] paths... can be used to
474   * load chosen paths from the tree into the index without
475   * affecting the working tree nor HEAD. */
476  if (pathspec.nr) {
477    if (reset_type == MIXED)
478      warning(_("--mixed with paths is deprecated; use 'git reset -- <paths>' instead."));
479    else if (reset_type != NONE)
480      die(_("Cannot do %s reset with paths."),
481          _(reset_type_names[reset_type]));
482  }
483  if (reset_type == NONE)
484    reset_type = MIXED; /* by default */
485
486  if (pathspec.nr)
487    trace2_cmd_mode("path");
488  else
489    trace2_cmd_mode(reset_type_names[reset_type]);
490
491  if (reset_type != SOFT && (reset_type != MIXED || get_git_work_tree()))
492    setup_work_tree();
493
494  if (reset_type == MIXED && is_bare_repository())
495    die(_("%s reset is not allowed in a bare repository"),
496        _(reset_type_names[reset_type]));
497
498  if (intent_to_add && reset_type != MIXED)
499    die(_("the option '%s' requires '%s'"), "-N", "--mixed");
500
501  prepare_repo_settings(the_repository);
502  the_repository->settings.command_requires_full_index = 0;
503
504  if (read_cache() < 0)
505    die(_("index file corrupt"));
506
507  /* Soft reset does not touch the index file nor the working tree
508   * at all, but requires them in a good order.  Other resets reset
509   * the index file to the tree object we are switching to. */
510  if (reset_type == SOFT || reset_type == KEEP)
511    die_if_unmerged_cache(reset_type);
512
513  if (reset_type != SOFT) {
514    struct lock_file lock = LOCK_INIT;
515    hold_locked_index(&lock, LOCK_DIE_ON_ERROR);
516    if (reset_type == MIXED) {
517      int flags = quiet ? REFRESH_QUIET : REFRESH_IN_PORCELAIN;
518      if (read_from_tree(&pathspec, &oid, intent_to_add))
519        return 1;
520      the_index.updated_skipworktree = 1;
521      if (!no_refresh && get_git_work_tree()) {
522        uint64_t t_begin, t_delta_in_ms;
523
524        t_begin = getnanotime();
525        refresh_index(&the_index, flags, NULL, NULL,
526                _("Unstaged changes after reset:"));
527        t_delta_in_ms = (getnanotime() - t_begin) / 1000000;
528        if (!quiet && advice_enabled(ADVICE_RESET_NO_REFRESH_WARNING) && t_delta_in_ms > REFRESH_INDEX_DELAY_WARNING_IN_MS) {
529          advise(_("It took %.2f seconds to refresh the index after reset.  You can use\n"
530             "'--no-refresh' to avoid this."), t_delta_in_ms / 1000.0);
531        }
532      }
533    } else {
534      struct object_id dummy;
535      char *ref = NULL;
536      int err;
537
538      dwim_ref(rev, strlen(rev), &dummy, &ref, 0);
539      if (ref && !starts_with(ref, "refs/"))
540        FREE_AND_NULL(ref);
541
542      err = reset_index(ref, &oid, reset_type, quiet);
543      if (reset_type == KEEP && !err)
544        err = reset_index(ref, &oid, MIXED, quiet);
545      if (err)
546        die(_("Could not reset index file to revision '%s'."), rev);
547      free(ref);
548    }
549
550    if (write_locked_index(&the_index, &lock, COMMIT_LOCK))
551      die(_("Could not write new index file."));
552  }
553
554  if (!pathspec.nr && !unborn) {
555    /* Any resets without paths update HEAD to the head being
556     * switched to, saving the previous head in ORIG_HEAD before. */
557    update_ref_status = reset_refs(rev, &oid);
558
559    if (reset_type == HARD && !update_ref_status && !quiet)
560      print_new_head_line(lookup_commit_reference(the_repository, &oid));
561  }
562  if (!pathspec.nr)
563    remove_branch_state(the_repository, 0);
564
565  return update_ref_status;
566}
