15static const char diff_files_usage[] =
16"git diff-files [-q] [-0 | -1 | -2 | -3 | -c | --cc] [<common-diff-options>] [<path>...]"
17COMMON_DIFF_OPTIONS_HELP;
18
19int cmd_diff_files(int argc, const char **argv, const char *prefix)
20{
21  struct rev_info rev;
22  int result;
23  unsigned options = 0;
24
25  if (argc == 2 && !strcmp(argv[1], "-h"))
26    usage(diff_files_usage);
27
28  git_config(git_diff_basic_config, NULL); /* no "diff" UI options */
29  repo_init_revisions(the_repository, &rev, prefix);
30  rev.abbrev = 0;
31
32  /*
33   * Consider "intent-to-add" files as new by default, unless
34   * explicitly specified in the command line or anywhere else.
35   */
36  rev.diffopt.ita_invisible_in_index = 1;
37
38  prefix = precompose_argv_prefix(argc, argv, prefix);
39
40  argc = setup_revisions(argc, argv, &rev, NULL);
41  while (1 < argc && argv[1][0] == '-') {
42    if (!strcmp(argv[1], "--base"))
43      rev.max_count = 1;
44    else if (!strcmp(argv[1], "--ours"))
45      rev.max_count = 2;
46    else if (!strcmp(argv[1], "--theirs"))
47      rev.max_count = 3;
48    else if (!strcmp(argv[1], "-q"))
49      options |= DIFF_SILENT_ON_REMOVED;
50    else
51      usage(diff_files_usage);
52    argv++; argc--;
53  }
54  if (!rev.diffopt.output_format)
55    rev.diffopt.output_format = DIFF_FORMAT_RAW;
56  rev.diffopt.rotate_to_strict = 1;
57
58  /*
59   * Make sure there are NO revision (i.e. pending object) parameter,
60   * rev.max_count is reasonable (0 <= n <= 3), and
61   * there is no other revision filtering parameters.
62   */
63  if (rev.pending.nr ||
64      rev.min_age != -1 || rev.max_age != -1 ||
65      3 < rev.max_count)
66    usage(diff_files_usage);
67
68  /*
69   * "diff-files --base -p" should not combine merges because it
70   * was not asked to.  "diff-files -c -p" should not densify
71   * (the user should ask with "diff-files --cc" explicitly).
72   */
73  if (rev.max_count == -1 &&
74      (rev.diffopt.output_format & DIFF_FORMAT_PATCH))
75    diff_merges_set_dense_combined_if_unset(&rev);
76
77  if (read_cache_preload(&rev.diffopt.pathspec) < 0) {
78    perror("read_cache_preload");
79    result = -1;
80    goto cleanup;
81  }
82cleanup:
83  result = run_diff_files(&rev, options);
84  result = diff_result_code(&rev.diffopt, result);
85  release_revisions(&rev);
86  return result;
87}
