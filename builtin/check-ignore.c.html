34static void output_pattern(const char *path, struct path_pattern *pattern)
35{
36  char *bang  = (pattern && pattern->flags & PATTERN_FLAG_NEGATIVE)  ? "!" : "";
37  char *slash = (pattern && pattern->flags & PATTERN_FLAG_MUSTBEDIR) ? "/" : "";
38  if (!nul_term_line) {
39    if (!verbose) {
40      write_name_quoted(path, stdout, '\n');
41    } else {
42      if (pattern) {
43        quote_c_style(pattern->pl->src, NULL, stdout, 0);
44        printf(":%d:%s%s%s\t",
45               pattern->srcpos,
46               bang, pattern->pattern, slash);
47      }
48      else {
49        printf("::\t");
50      }
51      quote_c_style(path, NULL, stdout, 0);
52      fputc('\n', stdout);
53    }
54  } else {
55    if (!verbose) {
56      printf("%s%c", path, '\0');
57    } else {
58      if (pattern)
59        printf("%s%c%d%c%s%s%s%c%s%c",
60               pattern->pl->src, '\0',
61               pattern->srcpos, '\0',
62               bang, pattern->pattern, slash, '\0',
63               path, '\0');
64      else
65        printf("%c%c%c%s%c", '\0', '\0', '\0', path, '\0');
66    }
67  }
68}

70static int check_ignore(struct dir_struct *dir,
71      const char *prefix, int argc, const char **argv)
72{
73  const char *full_path;
74  char *seen;
75  int num_ignored = 0, i;
76  struct path_pattern *pattern;
77  struct pathspec pathspec;
78
79  if (!argc) {
80    if (!quiet)
81      fprintf(stderr, "no pathspec given.\n");
82    return 0;
83  }
84
85  /*
86   * check-ignore just needs paths. Magic beyond :/ is really
87   * irrelevant.
88   */
89  parse_pathspec(&pathspec,
90           PATHSPEC_ALL_MAGIC & ~PATHSPEC_FROMTOP,
91           PATHSPEC_SYMLINK_LEADING_PATH |
92           PATHSPEC_KEEP_ORDER,
93           prefix, argv);
94
95  die_path_inside_submodule(&the_index, &pathspec);
96
97  /*
98   * look for pathspecs matching entries in the index, since these
99   * should not be ignored, in order to be consistent with
100   * 'git status', 'git add' etc.
101   */
102  seen = find_pathspecs_matching_against_index(&pathspec, &the_index,
103                 PS_HEED_SKIP_WORKTREE);
104  for (i = 0; i < pathspec.nr; i++) {
105    full_path = pathspec.items[i].match;
106    pattern = NULL;
107    if (!seen[i]) {
108      int dtype = DT_UNKNOWN;
109      pattern = last_matching_pattern(dir, &the_index,
110              full_path, &dtype);
111      if (!verbose && pattern &&
112          pattern->flags & PATTERN_FLAG_NEGATIVE)
113        pattern = NULL;
114    }
115    if (!quiet && (pattern || show_non_matching))
116      output_pattern(pathspec.items[i].original, pattern);
117    if (pattern)
118      num_ignored++;
119  }
120  free(seen);
121  clear_pathspec(&pathspec);
122
123  return num_ignored;
124}

126static int check_ignore_stdin_paths(struct dir_struct *dir, const char *prefix)
127{
128  struct strbuf buf = STRBUF_INIT;
129  struct strbuf unquoted = STRBUF_INIT;
130  char *pathspec[2] = { NULL, NULL };
131  strbuf_getline_fn getline_fn;
132  int num_ignored = 0;
133
134  getline_fn = nul_term_line ? strbuf_getline_nul : strbuf_getline_lf;
135  while (getline_fn(&buf, stdin) != EOF) {
136    if (!nul_term_line && buf.buf[0] == '"') {
137      strbuf_reset(&unquoted);
138      if (unquote_c_style(&unquoted, buf.buf, NULL))
139        die("line is badly quoted");
140      strbuf_swap(&buf, &unquoted);
141    }
142    pathspec[0] = buf.buf;
143    num_ignored += check_ignore(dir, prefix,
144              1, (const char **)pathspec);
145    maybe_flush_or_die(stdout, "check-ignore to stdout");
146  }
147  strbuf_release(&buf);
148  strbuf_release(&unquoted);
149  return num_ignored;
150}

152int cmd_check_ignore(int argc, const char **argv, const char *prefix)
153{
154  int num_ignored;
155  struct dir_struct dir = DIR_INIT;
156
157  git_config(git_default_config, NULL);
158
159  argc = parse_options(argc, argv, prefix, check_ignore_options,
160           check_ignore_usage, 0);
161
162  if (stdin_paths) {
163    if (argc > 0)
164      die(_("cannot specify pathnames with --stdin"));
165  } else {
166    if (nul_term_line)
167      die(_("-z only makes sense with --stdin"));
168    if (argc == 0)
169      die(_("no path specified"));
170  }
171  if (quiet) {
172    if (argc > 1)
173      die(_("--quiet is only valid with a single pathname"));
174    if (verbose)
175      die(_("cannot have both --quiet and --verbose"));
176  }
177  if (show_non_matching && !verbose)
178    die(_("--non-matching is only valid with --verbose"));
179
180  /* read_cache() is only necessary so we can watch out for submodules. */
181  if (!no_index && read_cache() < 0)
182    die(_("index file corrupt"));
183
184  setup_standard_excludes(&dir);
185
186  if (stdin_paths) {
187    num_ignored = check_ignore_stdin_paths(&dir, prefix);
188  } else {
189    num_ignored = check_ignore(&dir, prefix, argc, argv);
190    maybe_flush_or_die(stdout, "ignore to stdout");
191  }
192
193  dir_clear(&dir);
194
195  return !num_ignored;
196}
