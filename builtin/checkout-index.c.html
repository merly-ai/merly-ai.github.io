17#define CHECKOUT_ALL 4
18static int nul_term_line;
19static int checkout_stage; /* default to checkout stage0 */
20static int ignore_skip_worktree; /* default to 0 */
21static int to_tempfile;
22static char topath[4][TEMPORARY_FILENAME_LENGTH + 1];
23
24static struct checkout state = CHECKOUT_INIT;

26static void write_tempfile_record(const char *name, const char *prefix)
27{
28  int i;
29  int have_tempname = 0;
30
31  if (CHECKOUT_ALL == checkout_stage) {
32    for (i = 1; i < 4; i++)
33      if (topath[i][0]) {
34        have_tempname = 1;
35        break;
36      }
37
38    if (have_tempname) {
39      for (i = 1; i < 4; i++) {
40        if (i > 1)
41          putchar(' ');
42        if (topath[i][0])
43          fputs(topath[i], stdout);
44        else
45          putchar('.');
46      }
47    }
48  } else if (topath[checkout_stage][0]) {
49    have_tempname = 1;
50    fputs(topath[checkout_stage], stdout);
51  }
52
53  if (have_tempname) {
54    putchar('\t');
55    write_name_quoted_relative(name, prefix, stdout,
56             nul_term_line ? '\0' : '\n');
57  }
58
59  for (i = 0; i < 4; i++) {
60    topath[i][0] = 0;
61  }
62}

64static int checkout_file(const char *name, const char *prefix)
65{
66  int namelen = strlen(name);
67  int pos = cache_name_pos(name, namelen);
68  int has_same_name = 0;
69  int is_file = 0;
70  int is_skipped = 1;
71  int did_checkout = 0;
72  int errs = 0;
73
74  if (pos < 0)
75    pos = -pos - 1;
76
77  while (pos < active_nr) {
78    struct cache_entry *ce = active_cache[pos];
79    if (ce_namelen(ce) != namelen ||
80        memcmp(ce->name, name, namelen))
81      break;
82    has_same_name = 1;
83    pos++;
84    if (S_ISSPARSEDIR(ce->ce_mode))
85      break;
86    is_file = 1;
87    if (!ignore_skip_worktree && ce_skip_worktree(ce))
88      break;
89    is_skipped = 0;
90    if (ce_stage(ce) != checkout_stage
91        && (CHECKOUT_ALL != checkout_stage || !ce_stage(ce)))
92      continue;
93    did_checkout = 1;
94    if (checkout_entry(ce, &state,
95           to_tempfile ? topath[ce_stage(ce)] : NULL,
96           NULL) < 0)
97      errs++;
98  }
99
100  if (did_checkout) {
101    if (to_tempfile)
102      write_tempfile_record(name, prefix);
103    return errs > 0 ? -1 : 0;
104  }
105
106  /*
107   * At this point we know we didn't try to check anything out. If it was
108   * because we did find an entry but it was stage 0, that's not an
109   * error.
110   */
111  if (has_same_name && checkout_stage == CHECKOUT_ALL)
112    return 0;
113
114  if (!state.quiet) {
115    fprintf(stderr, "git checkout-index: %s ", name);
116    if (!has_same_name)
117      fprintf(stderr, "is not in the cache");
118    else if (!is_file)
119      fprintf(stderr, "is a sparse directory");
120    else if (is_skipped)
121      fprintf(stderr, "has skip-worktree enabled; "
122          "use '--ignore-skip-worktree-bits' to checkout");
123    else if (checkout_stage)
124      fprintf(stderr, "does not exist at stage %d",
125        checkout_stage);
126    else
127      fprintf(stderr, "is unmerged");
128    fputc('\n', stderr);
129  }
130  return -1;
131}

133static int checkout_all(const char *prefix, int prefix_length)
134{
135  int i, errs = 0;
136  struct cache_entry *last_ce = NULL;
137
138  for (i = 0; i < active_nr ; i++) {
139    struct cache_entry *ce = active_cache[i];
140
141    if (S_ISSPARSEDIR(ce->ce_mode)) {
142      if (!ce_skip_worktree(ce))
143        BUG("sparse directory '%s' does not have skip-worktree set", ce->name);
144
145      /*
146       * If the current entry is a sparse directory and skip-worktree
147       * entries are being checked out, expand the index and continue
148       * the loop on the current index position (now pointing to the
149       * first entry inside the expanded sparse directory).
150       */
151      if (ignore_skip_worktree) {
152        ensure_full_index(&the_index);
153        ce = active_cache[i];
154      }
155    }
156
157    if (!ignore_skip_worktree && ce_skip_worktree(ce))
158      continue;
159    if (ce_stage(ce) != checkout_stage
160        && (CHECKOUT_ALL != checkout_stage || !ce_stage(ce)))
161      continue;
162    if (prefix && *prefix &&
163        (ce_namelen(ce) <= prefix_length ||
164         memcmp(prefix, ce->name, prefix_length)))
165      continue;
166    if (last_ce && to_tempfile) {
167      if (ce_namelen(last_ce) != ce_namelen(ce)
168          || memcmp(last_ce->name, ce->name, ce_namelen(ce)))
169        write_tempfile_record(last_ce->name, prefix);
170    }
171    if (checkout_entry(ce, &state,
172           to_tempfile ? topath[ce_stage(ce)] : NULL,
173           NULL) < 0)
174      errs++;
175    last_ce = ce;
176  }
177  if (last_ce && to_tempfile)
178    write_tempfile_record(last_ce->name, prefix);
179  return !!errs;
180}

187static int option_parse_stage(const struct option *opt,
188            const char *arg, int unset)
189{
190  BUG_ON_OPT_NEG(unset);
191
192  if (!strcmp(arg, "all")) {
193    to_tempfile = 1;
194    checkout_stage = CHECKOUT_ALL;
195  } else {
196    int ch = arg[0];
197    if ('1' <= ch && ch <= '3')
198      checkout_stage = arg[0] - '0';
199    else
200      die(_("stage should be between 1 and 3 or all"));
201  }
202  return 0;
203}

205int cmd_checkout_index(int argc, const char **argv, const char *prefix)
206{
207  int i;
208  struct lock_file lock_file = LOCK_INIT;
209  int all = 0;
210  int read_from_stdin = 0;
211  int prefix_length;
212  int force = 0, quiet = 0, not_new = 0;
213  int index_opt = 0;
214  int err = 0;
215  int pc_workers, pc_threshold;
216  struct option builtin_checkout_index_options[] = {
217    OPT_BOOL('a', "all", &all,
218      N_("check out all files in the index")),
219    OPT_BOOL(0, "ignore-skip-worktree-bits", &ignore_skip_worktree,
220      N_("do not skip files with skip-worktree set")),
221    OPT__FORCE(&force, N_("force overwrite of existing files"), 0),
222    OPT__QUIET(&quiet,
223      N_("no warning for existing files and files not in index")),
224    OPT_BOOL('n', "no-create", &not_new,
225      N_("don't checkout new files")),
226    OPT_BOOL('u', "index", &index_opt,
227       N_("update stat information in the index file")),
228    OPT_BOOL('z', NULL, &nul_term_line,
229      N_("paths are separated with NUL character")),
230    OPT_BOOL(0, "stdin", &read_from_stdin,
231      N_("read list of paths from the standard input")),
232    OPT_BOOL(0, "temp", &to_tempfile,
233      N_("write the content to temporary files")),
234    OPT_STRING(0, "prefix", &state.base_dir, N_("string"),
235      N_("when creating files, prepend <string>")),
236    OPT_CALLBACK_F(0, "stage", NULL, "(1|2|3|all)",
237      N_("copy out the files from named stage"),
238      PARSE_OPT_NONEG, option_parse_stage),
239    OPT_END()
240  };
241
242  if (argc == 2 && !strcmp(argv[1], "-h"))
243    usage_with_options(builtin_checkout_index_usage,
244           builtin_checkout_index_options);
245  git_config(git_default_config, NULL);
246  prefix_length = prefix ? strlen(prefix) : 0;
247
248  prepare_repo_settings(the_repository);
249  the_repository->settings.command_requires_full_index = 0;
250
251  if (read_cache() < 0) {
252    die("invalid cache");
253  }
254
255  argc = parse_options(argc, argv, prefix, builtin_checkout_index_options,
256      builtin_checkout_index_usage, 0);
257  state.istate = &the_index;
258  state.force = force;
259  state.quiet = quiet;
260  state.not_new = not_new;
261
262  if (!state.base_dir)
263    state.base_dir = "";
264  state.base_dir_len = strlen(state.base_dir);
265
266  /*
267   * when --prefix is specified we do not want to update cache.
268   */
269  if (index_opt && !state.base_dir_len && !to_tempfile) {
270    state.refresh_cache = 1;
271    state.istate = &the_index;
272    hold_locked_index(&lock_file, LOCK_DIE_ON_ERROR);
273  }
274
275  get_parallel_checkout_configs(&pc_workers, &pc_threshold);
276  if (pc_workers > 1)
277    init_parallel_checkout();
278
279  /* Check out named files first */
280  for (i = 0; i < argc; i++) {
281    const char *arg = argv[i];
282    char *p;
283
284    if (all)
285      die("git checkout-index: don't mix '--all' and explicit filenames");
286    if (read_from_stdin)
287      die("git checkout-index: don't mix '--stdin' and explicit filenames");
288    p = prefix_path(prefix, prefix_length, arg);
289    err |= checkout_file(p, prefix);
290    free(p);
291  }
292
293  if (read_from_stdin) {
294    struct strbuf buf = STRBUF_INIT;
295    struct strbuf unquoted = STRBUF_INIT;
296    strbuf_getline_fn getline_fn;
297
298    if (all)
299      die("git checkout-index: don't mix '--all' and '--stdin'");
300
301    getline_fn = nul_term_line ? strbuf_getline_nul : strbuf_getline_lf;
302    while (getline_fn(&buf, stdin) != EOF) {
303      char *p;
304      if (!nul_term_line && buf.buf[0] == '"') {
305        strbuf_reset(&unquoted);
306        if (unquote_c_style(&unquoted, buf.buf, NULL))
307          die("line is badly quoted");
308        strbuf_swap(&buf, &unquoted);
309      }
310      p = prefix_path(prefix, prefix_length, buf.buf);
311      err |= checkout_file(p, prefix);
312      free(p);
313    }
314    strbuf_release(&unquoted);
315    strbuf_release(&buf);
316  }
317
318  if (all)
319    err |= checkout_all(prefix, prefix_length);
320
321  if (pc_workers > 1)
322    err |= run_parallel_checkout(&state, pc_workers, pc_threshold,
323               NULL, NULL);
324
325  if (err)
326    return 1;
327
328  if (is_lock_file_locked(&lock_file) &&
329      write_locked_index(&the_index, &lock_file, COMMIT_LOCK))
330    die("Unable to write new index file");
331  return 0;
332}
