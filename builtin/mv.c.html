22enum update_mode {
23  BOTH = 0,
24  WORKING_DIRECTORY = (1 << 1),
25  INDEX = (1 << 2),
26  SPARSE = (1 << 3),
27  SKIP_WORKTREE_DIR = (1 << 4),
28};

30#define DUP_BASENAME 1
31#define KEEP_TRAILING_SLASH 2
32
33static const char **internal_prefix_pathspec(const char *prefix,
34               const char **pathspec,
35               int count, unsigned flags)
36{
37  int i;
38  const char **result;
39  int prefixlen = prefix ? strlen(prefix) : 0;
40  ALLOC_ARRAY(result, count + 1);
41
42  /* Create an intermediate copy of the pathspec based on the flags */
43  for (i = 0; i < count; i++) {
44    int length = strlen(pathspec[i]);
45    int to_copy = length;
46    char *it;
47    while (!(flags & KEEP_TRAILING_SLASH) &&
48           to_copy > 0 && is_dir_sep(pathspec[i][to_copy - 1]))
49      to_copy--;
50
51    it = xmemdupz(pathspec[i], to_copy);
52    if (flags & DUP_BASENAME) {
53      result[i] = xstrdup(basename(it));
54      free(it);
55    } else {
56      result[i] = it;
57    }
58  }
59  result[count] = NULL;
60
61  /* Prefix the pathspec and free the old intermediate strings */
62  for (i = 0; i < count; i++) {
63    const char *match = prefix_path(prefix, prefixlen, result[i]);
64    free((char *) result[i]);
65    result[i] = match;
66  }
67
68  return result;
69}

71static const char *add_slash(const char *path)
72{
73  size_t len = strlen(path);
74  if (path[len - 1] != '/') {
75    char *with_slash = xmalloc(st_add(len, 2));
76    memcpy(with_slash, path, len);
77    with_slash[len++] = '/';
78    with_slash[len] = 0;
79    return with_slash;
80  }
81  return path;
82}

84#define SUBMODULE_WITH_GITDIR ((const char *)1)

86static void prepare_move_submodule(const char *src, int first,
87           const char **submodule_gitfile)
88{
89  struct strbuf submodule_dotgit = STRBUF_INIT;
90  if (!S_ISGITLINK(active_cache[first]->ce_mode))
91    die(_("Directory %s is in index and no submodule?"), src);
92  if (!is_staging_gitmodules_ok(&the_index))
93    die(_("Please stage your changes to .gitmodules or stash them to proceed"));
94  strbuf_addf(&submodule_dotgit, "%s/.git", src);
95  *submodule_gitfile = read_gitfile(submodule_dotgit.buf);
96  if (*submodule_gitfile)
97    *submodule_gitfile = xstrdup(*submodule_gitfile);
98  else
99    *submodule_gitfile = SUBMODULE_WITH_GITDIR;
100  strbuf_release(&submodule_dotgit);
101}

103static int index_range_of_same_dir(const char *src, int length,
104           int *first_p, int *last_p)
105{
106  const char *src_w_slash = add_slash(src);
107  int first, last, len_w_slash = length + 1;
108
109  first = cache_name_pos(src_w_slash, len_w_slash);
110  if (first >= 0)
111    die(_("%.*s is in index"), len_w_slash, src_w_slash);
112
113  first = -1 - first;
114  for (last = first; last < active_nr; last++) {
115    const char *path = active_cache[last]->name;
116    if (strncmp(path, src_w_slash, len_w_slash))
117      break;
118  }
119  if (src_w_slash != src)
120    free((char *)src_w_slash);
121  *first_p = first;
122  *last_p = last;
123  return last - first;
124}

126/*
127 * Check if an out-of-cone directory should be in the index. Imagine this case
128 * that all the files under a directory are marked with 'CE_SKIP_WORKTREE' bit
129 * and thus the directory is sparsified.
130 *
131 * Return 0 if such directory exist (i.e. with any of its contained files not
132 * marked with CE_SKIP_WORKTREE, the directory would be present in working tree).
133 * Return 1 otherwise.
134 */
135static int check_dir_in_index(const char *name)
136{
137  const char *with_slash = add_slash(name);
138  int length = strlen(with_slash);
139
140  int pos = cache_name_pos(with_slash, length);
141  const struct cache_entry *ce;
142
143  if (pos < 0) {
144    pos = -pos - 1;
145    if (pos >= the_index.cache_nr)
146      return 1;
147    ce = active_cache[pos];
148    if (strncmp(with_slash, ce->name, length))
149      return 1;
150    if (ce_skip_worktree(ce))
151      return 0;
152  }
153  return 1;
154}

156int cmd_mv(int argc, const char **argv, const char *prefix)
157{
158  int i, flags, gitmodules_modified = 0;
159  int verbose = 0, show_only = 0, force = 0, ignore_errors = 0, ignore_sparse = 0;
160  struct option builtin_mv_options[] = {
161    OPT__VERBOSE(&verbose, N_("be verbose")),
162    OPT__DRY_RUN(&show_only, N_("dry run")),
163    OPT__FORCE(&force, N_("force move/rename even if target exists"),
164         PARSE_OPT_NOCOMPLETE),
165    OPT_BOOL('k', NULL, &ignore_errors, N_("skip move/rename errors")),
166    OPT_BOOL(0, "sparse", &ignore_sparse, N_("allow updating entries outside of the sparse-checkout cone")),
167    OPT_END(),
168  };
169  const char **source, **destination, **dest_path, **submodule_gitfile;
170  enum update_mode *modes;
171  struct stat st;
172  struct string_list src_for_dst = STRING_LIST_INIT_NODUP;
173  struct lock_file lock_file = LOCK_INIT;
174  struct cache_entry *ce;
175  struct string_list only_match_skip_worktree = STRING_LIST_INIT_NODUP;
176
177  git_config(git_default_config, NULL);
178
179  argc = parse_options(argc, argv, prefix, builtin_mv_options,
180           builtin_mv_usage, 0);
181  if (--argc < 1)
182    usage_with_options(builtin_mv_usage, builtin_mv_options);
183
184  hold_locked_index(&lock_file, LOCK_DIE_ON_ERROR);
185  if (read_cache() < 0)
186    die(_("index file corrupt"));
187
188  source = internal_prefix_pathspec(prefix, argv, argc, 0);
189  modes = xcalloc(argc, sizeof(enum update_mode));
190  /*
191   * Keep trailing slash, needed to let
192   * "git mv file no-such-dir/" error out, except in the case
193   * "git mv directory no-such-dir/".
194   */
195  flags = KEEP_TRAILING_SLASH;
196  if (argc == 1 && is_directory(argv[0]) && !is_directory(argv[1]))
197    flags = 0;
198  dest_path = internal_prefix_pathspec(prefix, argv + argc, 1, flags);
199  submodule_gitfile = xcalloc(argc, sizeof(char *));
200
201  if (dest_path[0][0] == '\0')
202    /* special case: "." was normalized to "" */
203    destination = internal_prefix_pathspec(dest_path[0], argv, argc, DUP_BASENAME);
204  else if (!lstat(dest_path[0], &st) &&
205      S_ISDIR(st.st_mode)) {
206    dest_path[0] = add_slash(dest_path[0]);
207    destination = internal_prefix_pathspec(dest_path[0], argv, argc, DUP_BASENAME);
208  } else {
209    if (argc != 1)
210      die(_("destination '%s' is not a directory"), dest_path[0]);
211    destination = dest_path;
212  }
213
214  /* Checking */
215  for (i = 0; i < argc; i++) {
216    const char *src = source[i], *dst = destination[i];
217    int length;
218    const char *bad = NULL;
219    int skip_sparse = 0;
220
221    if (show_only)
222      printf(_("Checking rename of '%s' to '%s'\n"), src, dst);
223
224    length = strlen(src);
225    if (lstat(src, &st) < 0) {
226      int pos;
227      const struct cache_entry *ce;
228
229      pos = cache_name_pos(src, length);
230      if (pos < 0) {
231        const char *src_w_slash = add_slash(src);
232        if (!path_in_sparse_checkout(src_w_slash, &the_index) &&
233            !check_dir_in_index(src)) {
234          modes[i] |= SKIP_WORKTREE_DIR;
235          goto dir_check;
236        }
237        /* only error if existence is expected. */
238        if (!(modes[i] & SPARSE))
239          bad = _("bad source");
240        goto act_on_entry;
241      }
242      ce = active_cache[pos];
243      if (!ce_skip_worktree(ce)) {
244        bad = _("bad source");
245        goto act_on_entry;
246      }
247      if (!ignore_sparse) {
248        string_list_append(&only_match_skip_worktree, src);
249        goto act_on_entry;
250      }
251      /* Check if dst exists in index */
252      if (cache_name_pos(dst, strlen(dst)) < 0) {
253        modes[i] |= SPARSE;
254        goto act_on_entry;
255      }
256      if (!force) {
257        bad = _("destination exists");
258        goto act_on_entry;
259      }
260      modes[i] |= SPARSE;
261      goto act_on_entry;
262    }
263    if (!strncmp(src, dst, length) &&
264        (dst[length] == 0 || dst[length] == '/')) {
265      bad = _("can not move directory into itself");
266      goto act_on_entry;
267    }
268    if (S_ISDIR(st.st_mode)
269        && lstat(dst, &st) == 0) {
270      bad = _("cannot move directory over file");
271      goto act_on_entry;
272    }
273
274dir_check:
275    if (S_ISDIR(st.st_mode)) {
276      int j, dst_len, n;
277      int first = cache_name_pos(src, length), last;
278
279      if (first >= 0) {
280        prepare_move_submodule(src, first,
281                   submodule_gitfile + i);
282        goto act_on_entry;
283      } else if (index_range_of_same_dir(src, length,
284                 &first, &last) < 1) {
285        bad = _("source directory is empty");
286        goto act_on_entry;
287      }
288
289      /* last - first >= 1 */
290      modes[i] |= WORKING_DIRECTORY;
291      n = argc + last - first;
292      REALLOC_ARRAY(source, n);
293      REALLOC_ARRAY(destination, n);
294      REALLOC_ARRAY(modes, n);
295      REALLOC_ARRAY(submodule_gitfile, n);
296
297      dst = add_slash(dst);
298      dst_len = strlen(dst);
299
300      for (j = 0; j < last - first; j++) {
301        const struct cache_entry *ce = active_cache[first + j];
302        const char *path = ce->name;
303        source[argc + j] = path;
304        destination[argc + j] =
305          prefix_path(dst, dst_len, path + length + 1);
306        memset(modes + argc + j, 0, sizeof(enum update_mode));
307        modes[argc + j] |= ce_skip_worktree(ce) ? SPARSE : INDEX;
308        submodule_gitfile[argc + j] = NULL;
309      }
310      argc += last - first;
311      goto act_on_entry;
312    }
313    if (!(ce = cache_file_exists(src, length, 0))) {
314      bad = _("not under version control");
315      goto act_on_entry;
316    }
317    if (ce_stage(ce)) {
318      bad = _("conflicted");
319      goto act_on_entry;
320    }
321    if (lstat(dst, &st) == 0 &&
322        (!ignore_case || strcasecmp(src, dst))) {
323      bad = _("destination exists");
324      if (force) {
325        /*
326         * only files can overwrite each other:
327         * check both source and destination
328         */
329        if (S_ISREG(st.st_mode) || S_ISLNK(st.st_mode)) {
330          if (verbose)
331            warning(_("overwriting '%s'"), dst);
332          bad = NULL;
333        } else
334          bad = _("Cannot overwrite");
335      }
336      goto act_on_entry;
337    }
338    if (string_list_has_string(&src_for_dst, dst)) {
339      bad = _("multiple sources for the same target");
340      goto act_on_entry;
341    }
342    if (is_dir_sep(dst[strlen(dst) - 1])) {
343      bad = _("destination directory does not exist");
344      goto act_on_entry;
345    }
346
347    /*
348     * We check if the paths are in the sparse-checkout
349     * definition as a very final check, since that
350     * allows us to point the user to the --sparse
351     * option as a way to have a successful run.
352     */
353    if (!ignore_sparse &&
354        !path_in_sparse_checkout(src, &the_index)) {
355      string_list_append(&only_match_skip_worktree, src);
356      skip_sparse = 1;
357    }
358    if (!ignore_sparse &&
359        !path_in_sparse_checkout(dst, &the_index)) {
360      string_list_append(&only_match_skip_worktree, dst);
361      skip_sparse = 1;
362    }
363
364    if (skip_sparse)
365      goto remove_entry;
366
367    string_list_insert(&src_for_dst, dst);
368
369act_on_entry:
370    if (!bad)
371      continue;
372    if (!ignore_errors)
373      die(_("%s, source=%s, destination=%s"),
374           bad, src, dst);
375remove_entry:
376    if (--argc > 0) {
377      int n = argc - i;
378      memmove(source + i, source + i + 1,
379        n * sizeof(char *));
380      memmove(destination + i, destination + i + 1,
381        n * sizeof(char *));
382      memmove(modes + i, modes + i + 1,
383        n * sizeof(enum update_mode));
384      memmove(submodule_gitfile + i, submodule_gitfile + i + 1,
385        n * sizeof(char *));
386      i--;
387    }
388  }
389
390  if (only_match_skip_worktree.nr) {
391    advise_on_updating_sparse_paths(&only_match_skip_worktree);
392    if (!ignore_errors)
393      return 1;
394  }
395
396  for (i = 0; i < argc; i++) {
397    const char *src = source[i], *dst = destination[i];
398    enum update_mode mode = modes[i];
399    int pos;
400    struct checkout state = CHECKOUT_INIT;
401    state.istate = &the_index;
402
403    if (force)
404      state.force = 1;
405    if (show_only || verbose)
406      printf(_("Renaming %s to %s\n"), src, dst);
407    if (show_only)
408      continue;
409    if (!(mode & (INDEX | SPARSE | SKIP_WORKTREE_DIR)) &&
410        rename(src, dst) < 0) {
411      if (ignore_errors)
412        continue;
413      die_errno(_("renaming '%s' failed"), src);
414    }
415    if (submodule_gitfile[i]) {
416      if (!update_path_in_gitmodules(src, dst))
417        gitmodules_modified = 1;
418      if (submodule_gitfile[i] != SUBMODULE_WITH_GITDIR)
419        connect_work_tree_and_git_dir(dst,
420                    submodule_gitfile[i],
421                    1);
422    }
423
424    if (mode & (WORKING_DIRECTORY | SKIP_WORKTREE_DIR))
425      continue;
426
427    pos = cache_name_pos(src, strlen(src));
428    assert(pos >= 0);
429    rename_cache_entry_at(pos, dst);
430
431    if ((mode & SPARSE) &&
432        (path_in_sparse_checkout(dst, &the_index))) {
433      int dst_pos;
434
435      dst_pos = cache_name_pos(dst, strlen(dst));
436      active_cache[dst_pos]->ce_flags &= ~CE_SKIP_WORKTREE;
437
438      if (checkout_entry(active_cache[dst_pos], &state, NULL, NULL))
439        die(_("cannot checkout %s"), active_cache[dst_pos]->name);
440    }
441  }
442
443  if (gitmodules_modified)
444    stage_updated_gitmodules(&the_index);
445
446  if (write_locked_index(&the_index, &lock_file,
447             COMMIT_LOCK | SKIP_IF_UNCHANGED))
448    die(_("Unable to write new index file"));
449
450  string_list_clear(&src_for_dst, 0);
451  UNLEAK(source);
452  UNLEAK(dest_path);
453  free(submodule_gitfile);
454  free(modes);
455  return 0;
456}
