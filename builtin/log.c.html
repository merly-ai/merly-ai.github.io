80static int parse_decoration_style(const char *value)
81{
82  switch (git_parse_maybe_bool(value)) {
83  case 1:
84    return DECORATE_SHORT_REFS;
85  case 0:
86    return 0;
87  default:
88    break;
89  }
90  if (!strcmp(value, "full"))
91    return DECORATE_FULL_REFS;
92  else if (!strcmp(value, "short"))
93    return DECORATE_SHORT_REFS;
94  else if (!strcmp(value, "auto"))
95    return auto_decoration_style();
96  /*
97   * Please update _git_log() in git-completion.bash when you
98   * add new decoration styles.
99   */
100  return -1;
101}

164static void cmd_log_init_finish(int argc, const char **argv, const char *prefix,
165       struct rev_info *rev, struct setup_revision_opt *opt)
166{
167  struct userformat_want w;
168  int quiet = 0, source = 0, mailmap;
169  static struct line_opt_callback_data line_cb = {NULL, NULL, STRING_LIST_INIT_DUP};
170  static struct string_list decorate_refs_exclude = STRING_LIST_INIT_NODUP;
171  static struct string_list decorate_refs_exclude_config = STRING_LIST_INIT_NODUP;
172  static struct string_list decorate_refs_include = STRING_LIST_INIT_NODUP;
173  struct decoration_filter decoration_filter = {&decorate_refs_include,
174                  &decorate_refs_exclude,
175                  &decorate_refs_exclude_config};
176  static struct revision_sources revision_sources;
177
178  const struct option builtin_log_options[] = {
179    OPT__QUIET(&quiet, N_("suppress diff output")),
180    OPT_BOOL(0, "source", &source, N_("show source")),
181    OPT_BOOL(0, "use-mailmap", &mailmap, N_("use mail map file")),
182    OPT_ALIAS(0, "mailmap", "use-mailmap"),
183    OPT_STRING_LIST(0, "decorate-refs", &decorate_refs_include,
184        N_("pattern"), N_("only decorate refs that match <pattern>")),
185    OPT_STRING_LIST(0, "decorate-refs-exclude", &decorate_refs_exclude,
186        N_("pattern"), N_("do not decorate refs that match <pattern>")),
187    OPT_CALLBACK_F(0, "decorate", NULL, NULL, N_("decorate options"),
188             PARSE_OPT_OPTARG, decorate_callback),
189    OPT_CALLBACK('L', NULL, &line_cb, "range:file",
190           N_("trace the evolution of line range <start>,<end> or function :<funcname> in <file>"),
191           log_line_range_callback),
192    OPT_END()
193  };
194
195  line_cb.rev = rev;
196  line_cb.prefix = prefix;
197
198  mailmap = use_mailmap_config;
199  argc = parse_options(argc, argv, prefix,
200           builtin_log_options, builtin_log_usage,
201           PARSE_OPT_KEEP_ARGV0 | PARSE_OPT_KEEP_UNKNOWN |
202           PARSE_OPT_KEEP_DASHDASH);
203
204  if (quiet)
205    rev->diffopt.output_format |= DIFF_FORMAT_NO_OUTPUT;
206  argc = setup_revisions(argc, argv, rev, opt);
207
208  /* Any arguments at this point are not recognized */
209  if (argc > 1)
210    die(_("unrecognized argument: %s"), argv[1]);
211
212  if (rev->line_level_traverse && rev->prune_data.nr)
213    die(_("-L<range>:<file> cannot be used with pathspec"));
214
215  memset(&w, 0, sizeof(w));
216  userformat_find_requirements(NULL, &w);
217
218  if (!rev->show_notes_given && (!rev->pretty_given || w.notes))
219    rev->show_notes = 1;
220  if (rev->show_notes)
221    load_display_notes(&rev->notes_opt);
222
223  if ((rev->diffopt.pickaxe_opts & DIFF_PICKAXE_KINDS_MASK) ||
224      rev->diffopt.filter || rev->diffopt.flags.follow_renames)
225    rev->always_show_header = 0;
226
227  if (source || w.source) {
228    init_revision_sources(&revision_sources);
229    rev->sources = &revision_sources;
230  }
231
232  if (mailmap) {
233    rev->mailmap = xmalloc(sizeof(struct string_list));
234    string_list_init_nodup(rev->mailmap);
235    read_mailmap(rev->mailmap);
236  }
237
238  if (rev->pretty_given && rev->commit_format == CMIT_FMT_RAW) {
239    /*
240     * "log --pretty=raw" is special; ignore UI oriented
241     * configuration variables such as decoration.
242     */
243    if (!decoration_given)
244      decoration_style = 0;
245    if (!rev->abbrev_commit_given)
246      rev->abbrev_commit = 0;
247  }
248
249  if (rev->commit_format == CMIT_FMT_USERFORMAT) {
250    if (!w.decorate) {
251      /*
252       * Disable decoration loading if the format will not
253       * show them anyway.
254       */
255      decoration_style = 0;
256    } else if (!decoration_style) {
257      /*
258       * If we are going to show them, make sure we do load
259       * them here, but taking care not to override a
260       * specific style set by config or --decorate.
261       */
262      decoration_style = DECORATE_SHORT_REFS;
263    }
264  }
265
266  if (decoration_style || rev->simplify_by_decoration) {
267    const struct string_list *config_exclude =
268      repo_config_get_value_multi(the_repository,
269                "log.excludeDecoration");
270
271    if (config_exclude) {
272      struct string_list_item *item;
273      for_each_string_list_item(item, config_exclude)
274        string_list_append(&decorate_refs_exclude_config,
275               item->string);
276    }
277
278    if (decoration_style)
279      rev->show_decorations = 1;
280
281    load_ref_decorations(&decoration_filter, decoration_style);
282  }
283
284  if (rev->line_level_traverse)
285    line_log_init(rev, line_cb.prefix, &line_cb.args);
286
287  setup_pager();
288}

303/*
304 * This gives a rough estimate for how many commits we
305 * will print out in the list.
306 */
307static int estimate_commit_count(struct commit_list *list)
308{
309  int n = 0;
310
311  while (list) {
312    struct commit *commit = list->item;
313    unsigned int flags = commit->object.flags;
314    list = list->next;
315    if (!(flags & (TREESAME | UNINTERESTING)))
316      n++;
317  }
318  return n;
319}

333static void log_show_early(struct rev_info *revs, struct commit_list *list)
334{
335  int i = revs->early_output;
336  int show_header = 1;
337  int no_free = revs->diffopt.no_free;
338
339  revs->diffopt.no_free = 0;
340  sort_in_topological_order(&list, revs->sort_order);
341  while (list && i) {
342    struct commit *commit = list->item;
343    switch (simplify_commit(revs, commit)) {
344    case commit_show:
345      if (show_header) {
346        int n = estimate_commit_count(list);
347        show_early_header(revs, "incomplete", n);
348        show_header = 0;
349      }
350      log_tree_commit(revs, commit);
351      i--;
352      break;
353    case commit_ignore:
354      break;
355    case commit_error:
356      revs->diffopt.no_free = no_free;
357      diff_free(&revs->diffopt);
358      return;
359    }
360    list = list->next;
361  }
362
363  /* Did we already get enough commits for the early output? */
364  if (!i) {
365    revs->diffopt.no_free = 0;
366    diff_free(&revs->diffopt);
367    return;
368  }
369
370  /*
371   * ..if no, then repeat it twice a second until we
372   * do.
373   *
374   * NOTE! We don't use "it_interval", because if the
375   * reader isn't listening, we want our output to be
376   * throttled by the writing, and not have the timer
377   * trigger every second even if we're blocked on a
378   * reader!
379   */
380  early_output_timer.it_value.tv_sec = 0;
381  early_output_timer.it_value.tv_usec = 500000;
382  setitimer(ITIMER_REAL, &early_output_timer, NULL);
383}

390static void setup_early_output(void)
391{
392  struct sigaction sa;
393
394  /*
395   * Set up the signal handler, minimally intrusively:
396   * we only set a single volatile integer word (not
397   * using sigatomic_t - trying to avoid unnecessary
398   * system dependencies and headers), and using
399   * SA_RESTART.
400   */
401  memset(&sa, 0, sizeof(sa));
402  sa.sa_handler = early_output;
403  sigemptyset(&sa.sa_mask);
404  sa.sa_flags = SA_RESTART;
405  sigaction(SIGALRM, &sa, NULL);
406
407  /*
408   * If we can get the whole output in less than a
409   * tenth of a second, don't even bother doing the
410   * early-output thing..
411   *
412   * This is a one-time-only trigger.
413   */
414  early_output_timer.it_value.tv_sec = 0;
415  early_output_timer.it_value.tv_usec = 100000;
416  setitimer(ITIMER_REAL, &early_output_timer, NULL);
417}

426static int cmd_log_walk_no_free(struct rev_info *rev)
427{
428  struct commit *commit;
429  int saved_nrl = 0;
430  int saved_dcctc = 0;
431
432  if (rev->remerge_diff) {
433    rev->remerge_objdir = tmp_objdir_create("remerge-diff");
434    if (!rev->remerge_objdir)
435      die(_("unable to create temporary object directory"));
436    tmp_objdir_replace_primary_odb(rev->remerge_objdir, 1);
437  }
438
439  if (rev->early_output)
440    setup_early_output();
441
442  if (prepare_revision_walk(rev))
443    die(_("revision walk setup failed"));
444
445  if (rev->early_output)
446    finish_early_output(rev);
447
448  /*
449   * For --check and --exit-code, the exit code is based on CHECK_FAILED
450   * and HAS_CHANGES being accumulated in rev->diffopt, so be careful to
451   * retain that state information if replacing rev->diffopt in this loop
452   */
453  while ((commit = get_revision(rev)) != NULL) {
454    if (!log_tree_commit(rev, commit) && rev->max_count >= 0)
455      /*
456       * We decremented max_count in get_revision,
457       * but we didn't actually show the commit.
458       */
459      rev->max_count++;
460    if (!rev->reflog_info) {
461      /*
462       * We may show a given commit multiple times when
463       * walking the reflogs.
464       */
465      free_commit_buffer(the_repository->parsed_objects,
466             commit);
467      free_commit_list(commit->parents);
468      commit->parents = NULL;
469    }
470    if (saved_nrl < rev->diffopt.needed_rename_limit)
471      saved_nrl = rev->diffopt.needed_rename_limit;
472    if (rev->diffopt.degraded_cc_to_c)
473      saved_dcctc = 1;
474  }
475  rev->diffopt.degraded_cc_to_c = saved_dcctc;
476  rev->diffopt.needed_rename_limit = saved_nrl;
477
478  if (rev->remerge_diff) {
479    tmp_objdir_destroy(rev->remerge_objdir);
480    rev->remerge_objdir = NULL;
481  }
482
483  if (rev->diffopt.output_format & DIFF_FORMAT_CHECKDIFF &&
484      rev->diffopt.flags.check_failed) {
485    return 02;
486  }
487  return diff_result_code(&rev->diffopt, 0);
488}

501static int git_log_config(const char *var, const char *value, void *cb)
502{
503  const char *slot_name;
504
505  if (!strcmp(var, "format.pretty"))
506    return git_config_string(&fmt_pretty, var, value);
507  if (!strcmp(var, "format.subjectprefix"))
508    return git_config_string(&fmt_patch_subject_prefix, var, value);
509  if (!strcmp(var, "format.filenamemaxlength")) {
510    fmt_patch_name_max = git_config_int(var, value);
511    return 0;
512  }
513  if (!strcmp(var, "format.encodeemailheaders")) {
514    default_encode_email_headers = git_config_bool(var, value);
515    return 0;
516  }
517  if (!strcmp(var, "log.abbrevcommit")) {
518    default_abbrev_commit = git_config_bool(var, value);
519    return 0;
520  }
521  if (!strcmp(var, "log.date"))
522    return git_config_string(&default_date_mode, var, value);
523  if (!strcmp(var, "log.decorate")) {
524    decoration_style = parse_decoration_style(value);
525    if (decoration_style < 0)
526      decoration_style = 0; /* maybe warn? */
527    return 0;
528  }
529  if (!strcmp(var, "log.diffmerges"))
530    return diff_merges_config(value);
531  if (!strcmp(var, "log.showroot")) {
532    default_show_root = git_config_bool(var, value);
533    return 0;
534  }
535  if (!strcmp(var, "log.follow")) {
536    default_follow = git_config_bool(var, value);
537    return 0;
538  }
539  if (skip_prefix(var, "color.decorate.", &slot_name))
540    return parse_decorate_color_config(var, slot_name, value);
541  if (!strcmp(var, "log.mailmap")) {
542    use_mailmap_config = git_config_bool(var, value);
543    return 0;
544  }
545  if (!strcmp(var, "log.showsignature")) {
546    default_show_signature = git_config_bool(var, value);
547    return 0;
548  }
549
550  if (git_gpg_config(var, value, cb) < 0)
551    return -1;
552  return git_diff_ui_config(var, value, cb);

667int cmd_show(int argc, const char **argv, const char *prefix)
668{
669  struct rev_info rev;
670  struct object_array_entry *objects;
671  struct setup_revision_opt opt;
672  struct pathspec match_all;
673  int i, count, ret = 0;
674
675  init_log_defaults();
676  git_config(git_log_config, NULL);
677
678  if (the_repository->gitdir) {
679    prepare_repo_settings(the_repository);
680    the_repository->settings.command_requires_full_index = 0;
681  }
682
683  memset(&match_all, 0, sizeof(match_all));
684  repo_init_revisions(the_repository, &rev, prefix);
685  git_config(grep_config, &rev.grep_filter);
686
687  rev.diff = 1;
688  rev.always_show_header = 1;
689  rev.no_walk = 1;
690  rev.diffopt.stat_width = -1;   /* Scale to real terminal size */
691
692  memset(&opt, 0, sizeof(opt));
693  opt.def = "HEAD";
694  opt.tweak = show_setup_revisions_tweak;
695  cmd_log_init(argc, argv, prefix, &rev, &opt);
696
697  if (!rev.no_walk)
698    return cmd_log_deinit(cmd_log_walk(&rev), &rev);
699
700  count = rev.pending.nr;
701  objects = rev.pending.objects;
702  rev.diffopt.no_free = 1;
703  for (i = 0; i < count && !ret; i++) {
704    struct object *o = objects[i].item;
705    const char *name = objects[i].name;
706    switch (o->type) {
707    case OBJ_BLOB:
708      ret = show_blob_object(&o->oid, &rev, name);
709      break;
710    case OBJ_TAG: {
711      struct tag *t = (struct tag *)o;
712      struct object_id *oid = get_tagged_oid(t);
713
714      if (rev.shown_one)
715        putchar('\n');
716      fprintf(rev.diffopt.file, "%stag %s%s\n",
717          diff_get_color_opt(&rev.diffopt, DIFF_COMMIT),
718          t->tag,
719          diff_get_color_opt(&rev.diffopt, DIFF_RESET));
720      ret = show_tag_object(&o->oid, &rev);
721      rev.shown_one = 1;
722      if (ret)
723        break;
724      o = parse_object(the_repository, oid);
725      if (!o)
726        ret = error(_("could not read object %s"),
727              oid_to_hex(oid));
728      objects[i].item = o;
729      i--;
730      break;
731    }
732    case OBJ_TREE:
733      if (rev.shown_one)
734        putchar('\n');
735      fprintf(rev.diffopt.file, "%stree %s%s\n\n",
736          diff_get_color_opt(&rev.diffopt, DIFF_COMMIT),
737          name,
738          diff_get_color_opt(&rev.diffopt, DIFF_RESET));
739      read_tree(the_repository, (struct tree *)o,
740          &match_all, show_tree_object,
741          rev.diffopt.file);
742      rev.shown_one = 1;
743      break;
744    case OBJ_COMMIT:
745      rev.pending.nr = rev.pending.alloc = 0;
746      rev.pending.objects = NULL;
747      add_object_array(o, name, &rev.pending);
748      ret = cmd_log_walk_no_free(&rev);
749      break;
750    default:
751      ret = error(_("unknown type: %d"), o->type);
752    }
753  }
754
755  rev.diffopt.no_free = 0;
756  diff_free(&rev.diffopt);
757
758  return cmd_log_deinit(ret, &rev);
759}

1048static void get_patch_ids(struct rev_info *rev, struct patch_ids *ids)
1049{
1050  struct rev_info check_rev;
1051  struct commit *commit, *c1, *c2;
1052  struct object *o1, *o2;
1053  unsigned flags1, flags2;
1054
1055  if (rev->pending.nr != 2)
1056    die(_("need exactly one range"));
1057
1058  o1 = rev->pending.objects[0].item;
1059  o2 = rev->pending.objects[1].item;
1060  flags1 = o1->flags;
1061  flags2 = o2->flags;
1062  c1 = lookup_commit_reference(the_repository, &o1->oid);
1063  c2 = lookup_commit_reference(the_repository, &o2->oid);
1064
1065  if ((flags1 & UNINTERESTING) == (flags2 & UNINTERESTING))
1066    die(_("not a range"));
1067
1068  init_patch_ids(the_repository, ids);
1069
1070  /* given a range a..b get all patch ids for b..a */
1071  repo_init_revisions(the_repository, &check_rev, rev->prefix);
1072  check_rev.max_parents = 1;
1073  o1->flags ^= UNINTERESTING;
1074  o2->flags ^= UNINTERESTING;
1075  add_pending_object(&check_rev, o1, "o1");
1076  add_pending_object(&check_rev, o2, "o2");
1077  if (prepare_revision_walk(&check_rev))
1078    die(_("revision walk setup failed"));
1079
1080  while ((commit = get_revision(&check_rev)) != NULL) {
1081    add_commit_patch_id(commit, ids);
1082  }
1083
1084  /* reset for next revision walk */
1085  clear_commit_marks(c1, SEEN | UNINTERESTING | SHOWN | ADDED);
1086  clear_commit_marks(c2, SEEN | UNINTERESTING | SHOWN | ADDED);
1087  o1->flags = flags1;
1088  o2->flags = flags2;
1089}

1525static struct commit *get_base_commit(const char *base_commit,
1526              struct commit **list,
1527              int total)
1528{

1531  int i = 0, rev_nr = 0, auto_select, die_on_failure;
1532
1533  switch (auto_base) {
1534  case AUTO_BASE_NEVER:
1535    if (base_commit) {
1536      auto_select = 0;
1537      die_on_failure = 1;
1538    } else {
1539      /* no base information is requested */
1540      return NULL;
1541    }
1542    break;
1543  case AUTO_BASE_ALWAYS:
1544  case AUTO_BASE_WHEN_ABLE:
1545    if (base_commit) {
1546      BUG("requested automatic base selection but a commit was provided");
1547    } else {
1548      auto_select = 1;
1549      die_on_failure = auto_base == AUTO_BASE_ALWAYS;
1550    }
1551    break;
1552  default:
1553    BUG("unexpected automatic base selection method");
1554  }
1555
1556  if (!auto_select) {
1557    base = lookup_commit_reference_by_name(base_commit);
1558    if (!base)
1559      die(_("unknown commit %s"), base_commit);
1560  } else {
1561    struct branch *curr_branch = branch_get(NULL);
1562    const char *upstream = branch_get_upstream(curr_branch, NULL);
1563    if (upstream) {
1564      struct commit_list *base_list;
1565      struct commit *commit;
1566      struct object_id oid;
1567
1568      if (get_oid(upstream, &oid)) {
1569        if (die_on_failure)
1570          die(_("failed to resolve '%s' as a valid ref"), upstream);
1571        else
1572          return NULL;
1573      }
1574      commit = lookup_commit_or_die(&oid, "upstream base");
1575      base_list = get_merge_bases_many(commit, total, list);
1576      /* There should be one and only one merge base. */
1577      if (!base_list || base_list->next) {
1578        if (die_on_failure) {
1579          die(_("could not find exact merge base"));
1580        } else {
1581          free_commit_list(base_list);
1582          return NULL;
1583        }
1584      }
1585      base = base_list->item;
1586      free_commit_list(base_list);
1587    } else {
1588      if (die_on_failure)
1589        die(_("failed to get upstream, if you want to record base commit automatically,\n"
1590              "please use git branch --set-upstream-to to track a remote branch.\n"
1591              "Or you could specify base commit by --base=<base-commit-id> manually"));
1592      else
1593        return NULL;
1594    }
1595  }
1596
1597  ALLOC_ARRAY(rev, total);
1598  for (i = 0; i < total; i++)
1599    rev[i] = list[i];
1600
1601  rev_nr = total;
1602  /*
1603   * Get merge base through pair-wise computations
1604   * and store it in rev[0].
1605   */
1606  while (rev_nr > 1) {
1607    for (i = 0; i < rev_nr / 2; i++) {
1608      struct commit_list *merge_base;
1609      merge_base = get_merge_bases(rev[2 * i], rev[2 * i + 1]);
1610      if (!merge_base || merge_base->next) {
1611        if (die_on_failure) {
1612          die(_("failed to find exact merge base"));
1613        } else {
1614          free(rev);
1615          return NULL;
1616        }
1617      }
1618
1619      rev[i] = merge_base->item;
1620    }
1621
1622    if (rev_nr % 2)
1623      rev[i] = rev[2 * i];
1624    rev_nr = DIV_ROUND_UP(rev_nr, 2);
1625  }
1626

2114  if (prepare_revision_walk(&rev))

2117  while ((commit = get_revision(&rev)) != NULL) {
2118    if (commit->object.flags & BOUNDARY) {
2119      boundary_count++;

2121      continue;
2122    }
2123
2124    if (ignore_if_in_upstream && has_commit_patch_id(commit, &ids))
2125      continue;
2126
2127    nr++;
2128    REALLOC_ARRAY(list, nr);
2129    list[nr - 1] = commit;
2130  }

2133    goto done;

2135  if (cover_letter == -1) {

2138    else

2140  }

2152               _("Interdiff:"),
2153               _("Interdiff against v%d:"));
2154  }
2155

2157    creation_factor = RANGE_DIFF_CREATION_FACTOR_DEFAULT;
2158  else if (!rdiff_prev)

2160

2163      die(_("--range-diff requires --cover-letter or single patch"));
2164
2165    infer_range_diff_ranges(&rdiff1, &rdiff2, rdiff_prev,
2166          origin, list[0]);

2168    rev.rdiff2 = rdiff2.buf;
2169    rev.creation_factor = creation_factor;
2170    rev.rdiff_title = diff_title(&rdiff_title, reroll_count,
2171               _("Range-diff:"),
2172               _("Range-diff against v%d:"));
2173  }
2174
2175  if (!signature) {
2176    ; /* --no-signature inhibits all signatures */

2180    struct strbuf buf = STRBUF_INIT;
2181
2182    if (strbuf_read_file(&buf, signature_file, 128) < 0)
2183      die_errno(_("unable to read signature file '%s'"), signature_file);

2185  }
2186
2187  memset(&bases, 0, sizeof(bases));
2188  base = get_base_commit(base_commit, list, nr);
2189  if (base) {
2190    reset_revision_walk();

2192    prepare_bases(&bases, base, list, nr);
2193  }
2194
2195  if (in_reply_to || thread || cover_letter) {
2196    rev.ref_message_ids = xmalloc(sizeof(*rev.ref_message_ids));
2197    string_list_init_nodup(rev.ref_message_ids);
2198  }
2199  if (in_reply_to) {

2202  }

2204  rev.patch_suffix = fmt_patch_suffix;

2209          origin, nr, list, branch_name, quiet);
2210    print_bases(&bases, rev.diffopt.file);
2211    print_signature(rev.diffopt.file);
2212    total++;
2213    start_number--;
2214    /* interdiff/range-diff in cover-letter; omit from patches */
2215    rev.idiff_oid1 = NULL;
2216    rev.rdiff1 = NULL;
2217  }
2218  rev.add_signoff = do_signoff;
2219
2220  if (show_progress)

2222  while (0 <= --nr) {

2224    display_progress(progress, total - nr);

2227    /* Make the second and subsequent mails replies to the first */
2228    if (thread) {

2234         * matter what other options are set.
2235         *
2236         * For shallow threading:
2237         *
2238         * Without --cover-letter and
2239         * --in-reply-to, make every mail a

2245         *

2249         * letter is a reply to the
2250         * --in-reply-to, if specified.
2251         */

2253            && rev.ref_message_ids->nr > 0
2254            && (!cover_letter || rev.nr > 1))
2255          free(rev.message_id);
2256        else
2257          string_list_append(rev.ref_message_ids,
2258                 rev.message_id);
2259      }
2260      gen_message_id(&rev, oid_to_hex(&commit->object.oid));
2261    }
2262
2263    if (output_directory &&
2264        open_next_file(rev.numbered_files ? NULL : commit, NULL, &rev, quiet))
2265      die(_("failed to create output files"));

2268           commit);

2280      if (rev.mime_boundary)
2281        fprintf(rev.diffopt.file, "\n--%s%s--\n\n\n",
2282               mime_boundary_leader,
2283               rev.mime_boundary);
2284      else
2285        print_signature(rev.diffopt.file);
2286    }

2288      fclose(rev.diffopt.file);
2289  }

2294  string_list_clear(&extra_cc, 0);
