11static struct treeent {
12  unsigned mode;
13  struct object_id oid;
14  int len;
15  char name[FLEX_ARRAY];
16} **entries;
17static int alloc, used;

19static void append_to_tree(unsigned mode, struct object_id *oid, char *path)
20{
21  struct treeent *ent;
22  size_t len = strlen(path);
23  if (strchr(path, '/'))
24    die("path %s contains slash", path);
25
26  FLEX_ALLOC_MEM(ent, name, path, len);
27  ent->mode = mode;
28  ent->len = len;
29  oidcpy(&ent->oid, oid);
30
31  ALLOC_GROW(entries, used + 1, alloc);
32  entries[used++] = ent;
33}

35static int ent_compare(const void *a_, const void *b_)
36{
37  struct treeent *a = *(struct treeent **)a_;
38  struct treeent *b = *(struct treeent **)b_;
39  return base_name_compare(a->name, a->len, a->mode,
40         b->name, b->len, b->mode);
41}

43static void write_tree(struct object_id *oid)
44{
45  struct strbuf buf;
46  size_t size;
47  int i;
48
49  QSORT(entries, used, ent_compare);
50  for (size = i = 0; i < used; i++)
51    size += 32 + entries[i]->len;
52
53  strbuf_init(&buf, size);
54  for (i = 0; i < used; i++) {
55    struct treeent *ent = entries[i];
56    strbuf_addf(&buf, "%o %s%c", ent->mode, ent->name, '\0');
57    strbuf_add(&buf, ent->oid.hash, the_hash_algo->rawsz);
58  }
59
60  write_object_file(buf.buf, buf.len, OBJ_TREE, oid);
61  strbuf_release(&buf);
62}

69static void mktree_line(char *buf, int nul_term_line, int allow_missing)
70{
71  char *ptr, *ntr;
72  const char *p;
73  unsigned mode;
74  enum object_type mode_type; /* object type derived from mode */
75  enum object_type obj_type; /* object type derived from sha */
76  struct object_info oi = OBJECT_INFO_INIT;
77  char *path, *to_free = NULL;
78  struct object_id oid;
79
80  ptr = buf;
81  /*
82   * Read non-recursive ls-tree output format:
83   *     mode SP type SP sha1 TAB name
84   */
85  mode = strtoul(ptr, &ntr, 8);
86  if (ptr == ntr || !ntr || *ntr != ' ')
87    die("input format error: %s", buf);
88  ptr = ntr + 1; /* type */
89  ntr = strchr(ptr, ' ');
90  if (!ntr || parse_oid_hex(ntr + 1, &oid, &p) ||
91      *p != '\t')
92    die("input format error: %s", buf);
93
94  /* It is perfectly normal if we do not have a commit from a submodule */
95  if (S_ISGITLINK(mode))
96    allow_missing = 1;
97
98
99  *ntr++ = 0; /* now at the beginning of SHA1 */
100
101  path = (char *)p + 1;  /* at the beginning of name */
102  if (!nul_term_line && path[0] == '"') {
103    struct strbuf p_uq = STRBUF_INIT;
104    if (unquote_c_style(&p_uq, path, NULL))
105      die("invalid quoting");
106    path = to_free = strbuf_detach(&p_uq, NULL);
107  }
108
109  /*
110   * Object type is redundantly derivable three ways.
111   * These should all agree.
112   */
113  mode_type = object_type(mode);
114  if (mode_type != type_from_string(ptr)) {
115    die("entry '%s' object type (%s) doesn't match mode type (%s)",
116      path, ptr, type_name(mode_type));
117  }
118
119  /* Check the type of object identified by oid without fetching objects */
120  oi.typep = &obj_type;
121  if (oid_object_info_extended(the_repository, &oid, &oi,
122             OBJECT_INFO_LOOKUP_REPLACE |
123             OBJECT_INFO_QUICK |
124             OBJECT_INFO_SKIP_FETCH_OBJECT) < 0)
125    obj_type = -1;
126
127  if (obj_type < 0) {
128    if (allow_missing) {
129      ; /* no problem - missing objects are presumed to be of the right type */
130    } else {
131      die("entry '%s' object %s is unavailable", path, oid_to_hex(&oid));
132    }
133  } else {
134    if (obj_type != mode_type) {
135      /*
136       * The object exists but is of the wrong type.
137       * This is a problem regardless of allow_missing
138       * because the new tree entry will never be correct.
139       */
140      die("entry '%s' object %s is a %s but specified type was (%s)",
141        path, oid_to_hex(&oid), type_name(obj_type), type_name(mode_type));
142    }
143  }
144
145  append_to_tree(mode, &oid, path);
146  free(to_free);
147}

149int cmd_mktree(int ac, const char **av, const char *prefix)
150{
151  struct strbuf sb = STRBUF_INIT;
152  struct object_id oid;
153  int nul_term_line = 0;
154  int allow_missing = 0;
155  int is_batch_mode = 0;
156  int got_eof = 0;
157  strbuf_getline_fn getline_fn;
158
159  const struct option option[] = {
160    OPT_BOOL('z', NULL, &nul_term_line, N_("input is NUL terminated")),
161    OPT_SET_INT( 0 , "missing", &allow_missing, N_("allow missing objects"), 1),
162    OPT_SET_INT( 0 , "batch", &is_batch_mode, N_("allow creation of more than one tree"), 1),
163    OPT_END()
164  };
165
166  ac = parse_options(ac, av, prefix, option, mktree_usage, 0);
167  getline_fn = nul_term_line ? strbuf_getline_nul : strbuf_getline_lf;
168
169  while (!got_eof) {
170    while (1) {
171      if (getline_fn(&sb, stdin) == EOF) {
172        got_eof = 1;
173        break;
174      }
175      if (sb.buf[0] == '\0') {
176        /* empty lines denote tree boundaries in batch mode */
177        if (is_batch_mode)
178          break;
179        die("input format error: (blank line only valid in batch mode)");
180      }
181      mktree_line(sb.buf, nul_term_line, allow_missing);
182    }
183    if (is_batch_mode && got_eof && used < 1) {
184      /*
185       * Execution gets here if the last tree entry is terminated with a
186       * new-line.  The final new-line has been made optional to be
187       * consistent with the original non-batch behaviour of mktree.
188       */
189      ; /* skip creating an empty tree */
190    } else {
191      write_tree(&oid);
192      puts(oid_to_hex(&oid));
193      fflush(stdout);
194    }
195    used=0; /* reset tree entry buffer for re-use in batch mode */
196  }
197  strbuf_release(&sb);
198  return 0;
199}
