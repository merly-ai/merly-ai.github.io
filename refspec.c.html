18/*
19 * Parses the provided refspec 'refspec' and populates the refspec_item 'item'.
20 * Returns 1 if successful and 0 if the refspec is invalid.
21 */
22static int parse_refspec(struct refspec_item *item, const char *refspec, int fetch)
23{
24  size_t llen;
25  int is_glob;
26  const char *lhs, *rhs;
27  int flags;
28
29  is_glob = 0;
30
31  lhs = refspec;
32  if (*lhs == '+') {
33    item->force = 1;
34    lhs++;
35  } else if (*lhs == '^') {
36    item->negative = 1;
37    lhs++;
38  }
39
40  rhs = strrchr(lhs, ':');
41
42  /* negative refspecs only have one side */
43  if (item->negative && rhs)
44    return 0;
45
46  /*
47   * Before going on, special case ":" (or "+:") as a refspec
48   * for pushing matching refs.
49   */
50  if (!fetch && rhs == lhs && rhs[1] == '\0') {
51    item->matching = 1;
52    return 1;
53  }
54
55  if (rhs) {
56    size_t rlen = strlen(++rhs);
57    is_glob = (1 <= rlen && strchr(rhs, '*'));
58    item->dst = xstrndup(rhs, rlen);
59  } else {
60    item->dst = NULL;
61  }
62
63  llen = (rhs ? (rhs - lhs - 1) : strlen(lhs));
64  if (1 <= llen && memchr(lhs, '*', llen)) {
65    if ((rhs && !is_glob) || (!rhs && !item->negative && fetch))
66      return 0;
67    is_glob = 1;
68  } else if (rhs && is_glob) {
69    return 0;
70  }
71
72  item->pattern = is_glob;
73  if (llen == 1 && *lhs == '@')
74    item->src = xstrdup("HEAD");
75  else
76    item->src = xstrndup(lhs, llen);
77  flags = REFNAME_ALLOW_ONELEVEL | (is_glob ? REFNAME_REFSPEC_PATTERN : 0);
78
79  if (item->negative) {
80    struct object_id unused;
81
82    /*
83     * Negative refspecs only have a LHS, which indicates a ref
84     * (or pattern of refs) to exclude from other matches. This
85     * can either be a simple ref, or a glob pattern. Exact sha1
86     * match is not currently supported.
87     */
88    if (!*item->src)
89      return 0; /* negative refspecs must not be empty */
90    else if (llen == the_hash_algo->hexsz && !get_oid_hex(item->src, &unused))
91      return 0; /* negative refpsecs cannot be exact sha1 */
92    else if (!check_refname_format(item->src, flags))
93      ; /* valid looking ref is ok */
94    else
95      return 0;
96
97    /* the other rules below do not apply to negative refspecs */
98    return 1;
99  }
100
101  if (fetch) {
102    struct object_id unused;
103
104    /* LHS */
105    if (!*item->src)
106      ; /* empty is ok; it means "HEAD" */
107    else if (llen == the_hash_algo->hexsz && !get_oid_hex(item->src, &unused))
108      item->exact_sha1 = 1; /* ok */
109    else if (!check_refname_format(item->src, flags))
110      ; /* valid looking ref is ok */
111    else
112      return 0;
113    /* RHS */
114    if (!item->dst)
115      ; /* missing is ok; it is the same as empty */
116    else if (!*item->dst)
117      ; /* empty is ok; it means "do not store" */
118    else if (!check_refname_format(item->dst, flags))
119      ; /* valid looking ref is ok */
120    else
121      return 0;
122  } else {
123    /*
124     * LHS
125     * - empty is allowed; it means delete.
126     * - when wildcarded, it must be a valid looking ref.
127     * - otherwise, it must be an extended SHA-1, but
128     *   there is no existing way to validate this.
129     */
130    if (!*item->src)
131      ; /* empty is ok */
132    else if (is_glob) {
133      if (check_refname_format(item->src, flags))
134        return 0;
135    }
136    else
137      ; /* anything goes, for now */
138    /*
139     * RHS
140     * - missing is allowed, but LHS then must be a
141     *   valid looking ref.
142     * - empty is not allowed.
143     * - otherwise it must be a valid looking ref.
144     */
145    if (!item->dst) {
146      if (check_refname_format(item->src, flags))
147        return 0;
148    } else if (!*item->dst) {
149      return 0;
150    } else {
151      if (check_refname_format(item->dst, flags))
152        return 0;
153    }
154  }
155
156  return 1;
157}

188static void refspec_append_nodup(struct refspec *rs, char *refspec)
189{
190  struct refspec_item item;
191
192  refspec_item_init_or_die(&item, refspec, rs->fetch);
193
194  ALLOC_GROW(rs->items, rs->nr + 1, rs->alloc);
195  rs->items[rs->nr++] = item;

197  ALLOC_GROW(rs->raw, rs->raw_nr + 1, rs->raw_alloc);

260void refspec_ref_prefixes(const struct refspec *rs,
261        struct strvec *ref_prefixes)
262{
263  int i;
264  for (i = 0; i < rs->nr; i++) {
265    const struct refspec_item *item = &rs->items[i];
266    const char *prefix = NULL;
267
268    if (item->exact_sha1 || item->negative)
269      continue;
270    if (rs->fetch == REFSPEC_FETCH)
271      prefix = item->src;
272    else if (item->dst)
273      prefix = item->dst;
274    else if (item->src && !item->exact_sha1)
275      prefix = item->src;
276
277    if (!prefix)
278      continue;
279
280    if (item->pattern) {
281      const char *glob = strchr(prefix, '*');
282      strvec_pushf(ref_prefixes, "%.*s",
283             (int)(glob - prefix),
284             prefix);
285    } else {
286      expand_ref_prefix(ref_prefixes, prefix);
287    }
288  }
