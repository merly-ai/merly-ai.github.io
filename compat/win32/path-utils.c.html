2int win32_has_dos_drive_prefix(const char *path)
3{
4  int i;
5
6  /*
7   * Does it start with an ASCII letter (i.e. highest bit not set),
8   * followed by a colon?
9   */
10  if (!(0x80 & (unsigned char)*path))
11    return *path && path[1] == ':' ? 2 : 0;
12
13  /*
14   * While drive letters must be letters of the English alphabet, it is
15   * possible to assign virtually _any_ Unicode character via `subst` as
16   * a drive letter to "virtual drives". Even `1`, or `√§`. Or fun stuff
17   * like this:
18   *
19   *      subst ÷ç: %USERPROFILE%\Desktop
20   */
21  for (i = 1; i < 4 && (0x80 & (unsigned char)path[i]); i++)
22    ; /* skip first UTF-8 character */
23  return path[i] == ':' ? i + 1 : 0;
24}
