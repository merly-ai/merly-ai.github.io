75static Function* NewFunction (struct SymEntry* Sym, FuncDesc* D)
76/* Create a new function activation structure and return it */
77{
78    /* Allocate a new structure */
79    Function* F = (Function*) xmalloc (sizeof (Function));
80
81    /* Initialize the fields */
82    F->FuncEntry  = Sym;
83    F->ReturnType = GetFuncReturn (Sym->Type);
84    F->Desc       = D;
85    F->Reserved   = 0;
86    F->RetLab     = 0;
87    F->TopLevelSP = 0;
88    F->RegOffs    = RegisterSpace;
89    F->Flags      = IsTypeVoid (F->ReturnType) ? FF_VOID_RETURN : FF_NONE;
90
91    InitCollection (&F->LocalsBlockStack);
92
93    /* Return the new structure */
94    return F;
95}

108int F_CheckParamList (FuncDesc* D, int RequireAll)
109/* Check and set the parameter sizes.
110** If RequireAll is true, emit errors on parameters of incomplete types.
111** Return true if all parameters have complete types.
112*/
113{
114    unsigned    I = 0;
115    unsigned    Offs;
116    SymEntry*   Param;
117    unsigned    ParamSize = 0;
118    unsigned    IncompleteCount = 0;
119
120    /* Don't bother to check unnecessarily */
121    if ((D->Flags & FD_INCOMPLETE_PARAM) == 0) {
122        return 1;
123    }
124
125    /* Assign offsets. If the function has a variable parameter list,
126    ** there's one additional byte (the arg size).
127    */
128    Offs  = (D->Flags & FD_VARIADIC) ? 1 : 0;
129    Param = D->LastParam;
130    while (Param) {
131        unsigned Size = SizeOf (Param->Type);
132        if (RequireAll && IsIncompleteESUType (Param->Type)) {
133            if (D->Flags & FD_UNNAMED_PARAMS) {
134                Error ("Parameter %u has incomplete type '%s'",
135                       D->ParamCount - I,
136                       GetFullTypeName (Param->Type));
137            } else {
138                Error ("Parameter '%s' has incomplete type '%s'",
139                       Param->Name,
140                       GetFullTypeName (Param->Type));
141            }
142            ++IncompleteCount;
143        }
144        if (SymIsRegVar (Param)) {
145            Param->V.R.SaveOffs = Offs;
146        } else {
147            Param->V.Offs = Offs;
148        }
149        Offs += Size;
150        ParamSize += Size;
151        Param = Param->PrevSym;
152        ++I;
153    }
154
155    /* If all parameters have complete types, set the total size description,
156    ** clear the FD_INCOMPLETE_PARAM flag and return true.
157    */
158    if (IncompleteCount == 0) {
159        D->ParamSize = ParamSize;
160        D->Flags &= ~FD_INCOMPLETE_PARAM;
161        return 1;
162    }
163
164    /* Otherwise return false */
165    return 0;
166}

304/* Allocate any local space previously reserved. The function will do

323int F_AllocRegVar (Function* F, const Type* Type)
324/* Allocate a register variable for the given variable type. If the allocation
325** was successful, return the offset of the register variable in the register
326** bank (zero page storage). If there is no register space left, return -1.
327*/
328{
329    /* Allow register variables only on top level and if enabled */
330    if (IS_Get (&EnableRegVars) && GetLexicalLevel () == LEX_LEVEL_FUNCTION) {
331
332        /* Get the size of the variable */
333        unsigned Size = CheckedSizeOf (Type);
334
335        /* Do we have space left? */
336        if (F->RegOffs >= Size) {
337            /* Space left. We allocate the variables from high to low addresses,
338            ** so the addressing is compatible with the saved values on stack.
339            ** This allows shorter code when saving/restoring the variables.
340            */
341            F->RegOffs -= Size;
342            return F->RegOffs;
343        }
344    }
345
346    /* No space left or no allocation */
347    return -1;
348}

352static void F_RestoreRegVars (Function* F)
353/* Restore the register variables for the local function if there are any. */
354{
355    const SymEntry* Sym;
356
357    /* If we don't have register variables in this function, bail out early */
358    if (F->RegOffs == RegisterSpace) {
359        return;
360    }
361
362    /* Save the accumulator if needed */
363    if (!F_HasVoidReturn (F)) {
364        g_save (CF_CHAR | CF_FORCECHAR);
365    }
366
367    /* Get the first symbol from the function symbol table */
368    Sym = F->Desc->SymTab->SymHead;
369
370    /* Walk through all symbols checking for register variables */
371    while (Sym) {
372        if (SymIsRegVar (Sym)) {
373
374            /* Check for more than one variable */
375            int Offs       = Sym->V.R.SaveOffs;
376            unsigned Bytes = CheckedSizeOf (Sym->Type);
377
378            while (1) {
379
380                /* Find next register variable */
381                const SymEntry* NextSym = Sym->NextSym;
382                while (NextSym && !SymIsRegVar (NextSym)) {
383                    NextSym = NextSym->NextSym;
384                }
385
386                /* If we have a next one, compare the stack offsets */
387                if (NextSym) {
388
389                    /* We have a following register variable. Get the size */
390                    int Size = CheckedSizeOf (NextSym->Type);
391
392                    /* Adjacent variable? */
393                    if (NextSym->V.R.SaveOffs + Size != Offs) {
394                        /* No */
395                        break;
396                    }
397
398                    /* Adjacent variable */
399                    Bytes += Size;
400                    Offs  -= Size;
401                    Sym   = NextSym;
402
403                } else {
404                    break;
405                }
406            }
407
408            /* Restore the memory range */
409            g_restore_regvars (Offs, Sym->V.R.RegOffs, Bytes);
410
411        }
412
413        /* Check next symbol */
414        Sym = Sym->NextSym;
415    }
416
417    /* Restore the accumulator if needed */
418    if (!F_HasVoidReturn (F)) {
419        g_restore (CF_CHAR | CF_FORCECHAR);
420    }
421}

448void NewFunc (SymEntry* Func, FuncDesc* D)
449/* Parse argument declarations and function body. */
450{
451    int         ParamComplete;  /* If all paramemters have complete types */
452    int         C99MainFunc = 0;/* Flag for C99 main function returning int */
453    SymEntry*   Param;
454    const Type* RType;          /* Real type used for struct parameters */
455    const Type* ReturnType;     /* Return type */
456
457    /* Remember this function descriptor used for definition */
458    GetFuncDesc (Func->Type)->FuncDef = D;
459
460    /* Allocate the function activation record for the function */
461    CurrentFunc = NewFunction (Func, D);
462
463    /* Reenter the lexical level */
464    ReenterFunctionLevel (D);
465
466    /* Check return type */
467    ReturnType = F_GetReturnType (CurrentFunc);
468    if (IsIncompleteESUType (ReturnType)) {
469        /* There are already diagnostics on returning arrays or functions */
470        if (!IsTypeArray (ReturnType) && !IsTypeFunc (ReturnType)) {
471            Error ("Function has incomplete return type '%s'",
472                    GetFullTypeName (ReturnType));
473        }
474    }
475
476    /* Check and set the parameter sizes. All parameter must have complete
477    ** types now.
478    */
479    ParamComplete = F_CheckParamList (D, 1);
480
481    /* Check if the function header contains unnamed parameters. These are
482    ** only allowed in cc65 mode.
483    */
484    if ((D->Flags & FD_UNNAMED_PARAMS) != 0 && (IS_Get (&Standard) != STD_CC65)) {
485        Error ("Parameter name omitted");
486    }
487
488    /* Declare two special functions symbols: __fixargs__ and __argsize__.
489    ** The latter is different depending on the type of the function (variadic
490    ** or not).
491    */
492    AddConstSym ("__fixargs__", type_uint, SC_DEF | SC_CONST, D->ParamSize);
493    if (D->Flags & FD_VARIADIC) {
494        /* Variadic function. The variable must be const. */
495        static const Type T[] = { TYPE(T_UCHAR | T_QUAL_CONST), TYPE(T_END) };
496        AddLocalSym ("__argsize__", T, SC_DEF | SC_REF | SC_AUTO, 0);
497    } else {
498        /* Non variadic */
499        AddConstSym ("__argsize__", type_uchar, SC_DEF | SC_CONST, D->ParamSize);
500    }
501
502    /* Function body now defined */
503    Func->Flags |= SC_DEF;
504
505    /* Special handling for main() */
506    if (strcmp (Func->Name, "main") == 0) {
507
508        /* Mark this as the main function */
509        CurrentFunc->Flags |= FF_IS_MAIN;
510
511        /* Main cannot be a fastcall function */
512        if (IsQualFastcall (Func->Type)) {
513            Error ("'main' cannot be declared as __fastcall__");
514        }
515
516        /* If cc65 extensions aren't enabled, don't allow a main function that
517        ** doesn't return an int.
518        */
519        if (IS_Get (&Standard) != STD_CC65 && ReturnType[0].C != T_INT) {
520            Error ("'main' must always return an int");
521        }
522
523        /* Add a forced import of a symbol that is contained in the startup
524        ** code. This will force the startup code to be linked in.
525        */
526        g_importstartup ();
527
528        /* If main() takes parameters, generate a forced import to a function
529        ** that will setup these parameters. This way, programs that do not
530        ** need the additional code will not get it.
531        */
532        if (D->ParamCount > 0 || (D->Flags & FD_VARIADIC) != 0) {
533            g_importmainargs ();
534
535            /* The start-up code doesn't fast-call main(). */
536            Func->Type->C |= T_QUAL_CDECL;
537        }
538
539        /* Determine if this is a main function in a C99 environment that
540        ** returns an int.
541        */
542        if (IsRawTypeInt (F_GetReturnType (CurrentFunc)) &&
543            IS_Get (&Standard) == STD_C99) {
544            C99MainFunc = 1;
545        }
546    }
547
548    /* Allocate code and data segments for this function */
549    Func->V.F.Seg = PushSegments (Func);
550
551    /* Use the info in the segments for generating new local labels */
552    UseLabelPoolFromSegments (Func->V.F.Seg);
553
554    /* Set return label. This has to be done after the segments are pushed */
555    F_SetRetLab (CurrentFunc, GetLocalLabel ());
556
557    /* Allocate a new literal pool */
558    PushLiteralPool (Func);
559
560    /* If this is a fastcall function, push the last parameter onto the stack */
561    if (D->ParamCount > 0 && IsFastcallFunc (Func->Type)) {
562        unsigned Flags;
563
564        /* Generate the push */
565        /* Handle struct/union specially */
566        if (IsClassStruct (D->LastParam->Type)) {
567            Flags = TypeOf (GetStructReplacementType (D->LastParam->Type)) | CF_FORCECHAR;
568        } else {
569            Flags = TypeOf (D->LastParam->Type) | CF_FORCECHAR;
570        }
571        g_push (Flags, 0);
572    }
573
574    /* Generate function entry code if needed */
575    g_enter (FuncTypeOf (Func->Type), F_GetParamSize (CurrentFunc));
576
577    /* If stack checking code is requested, emit a call to the helper routine */
578    if (IS_Get (&CheckStack)) {
579        g_stackcheck ();
580    }
581
582    /* Setup the stack */
583    StackPtr = 0;
584
585    /* Emit code to handle the parameters if all of them have complete types */
586    if (ParamComplete) {
587        /* Walk through the parameter list and allocate register variable space
588        ** for parameters declared as register. Generate code to swap the contents
589        ** of the register bank with the save area on the stack.
590        */
591        Param = D->SymTab->SymHead;
592        while (Param && (Param->Flags & SC_PARAM) != 0) {
593

635    CollAppend (&CurrentFunc->LocalsBlockStack, 0);
636

638    DeclareLocals ();
639
640    /* Remember the current stack pointer. All variables allocated elsewhere
641    ** must be dropped when doing a return from an inner block.
642    */
643    CurrentFunc->TopLevelSP = StackPtr;
644
645    /* Now process statements in this block */
646    while (CurTok.Tok != TOK_RCURLY && CurTok.Tok != TOK_CEOF) {
647        AnyStatement (0);
648    }
649
650    /* If this is not a void function, and not the main function in a C99

668    g_defcodelabel (F_GetRetLab (CurrentFunc));
669
670    /* Restore the register variables */
671    F_RestoreRegVars (CurrentFunc);
672
673    /* Generate the exit code */
674    g_leave ();

701    FreeFunction (CurrentFunc);
