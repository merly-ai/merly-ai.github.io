58static void DoConversion (ExprDesc* Expr, const Type* NewType)
59/* Emit code to convert the given expression to a new type. */
60{
61    const Type* OldType;
62    unsigned    OldBits;
63    unsigned    NewBits;
64
65
66    /* Remember the old type */
67    OldType = Expr->Type;
68
69    /* If we're converting to void, we're done. Note: This does also cover a
70    ** conversion void -> void.
71    */
72    if (IsTypeVoid (NewType)) {
73        ED_MarkExprAsRVal (Expr);     /* Never an lvalue */
74        goto ExitPoint;
75    }
76
77    /* Don't allow casts from void to something else. */
78    if (IsTypeVoid (OldType)) {
79        Error ("Cannot convert from 'void' to something else");
80        goto ExitPoint;
81    }
82
83    /* Get the sizes of the types. Since we've excluded void types, checking
84    ** for known sizes makes sense here.
85    */
86    if (IsTypeBitField (OldType)) {
87        OldBits = OldType->A.B.Width;
88    } else {
89        OldBits = CheckedSizeOf (OldType) * CHAR_BITS;
90    }
91
92    /* If the new type is a bit-field, we use its underlying type instead */
93    if (IsTypeBitField (NewType)) {
94        NewType = GetUnderlyingType (NewType);
95    }
96    NewBits = CheckedSizeOf (NewType) * CHAR_BITS;
97
98    /* lvalue? */
99    if (ED_IsLVal (Expr)) {
100
101        /* We have an lvalue. If the new size is smaller than the old one,
102        ** we don't need to do anything. The compiler will generate code
103        ** to load only the portion of the value that is actually needed.
104        ** This works only on a little endian architecture, but that's
105        ** what we support.
106        ** If both sizes are equal, do also leave the value alone.
107        ** If the new size is larger, we must convert the value.
108        */
109        if (NewBits > OldBits) {
110            /* Load the value into the primary */
111            LoadExpr (CF_NONE, Expr);
112
113            /* Emit typecast code */
114            g_typecast (TypeOf (NewType), TypeOf (OldType));
115
116            /* Value is now in primary and an rvalue */
117            ED_FinalizeRValLoad (Expr);
118        }
119
120    } else if (ED_IsConstAbs (Expr)) {
121
122        /* A cast of a constant numeric value to another type. Be sure
123        ** to handle sign extension correctly.
124        */
125
126        /* Check if the new datatype will have a smaller range. If it
127        ** has a larger range, things are OK, since the value is
128        ** internally already represented by a long.
129        */
130        if (NewBits <= OldBits) {
131
132            /* Cut the value to the new size */
133            Expr->IVal &= (0xFFFFFFFFUL >> (32 - NewBits));
134
135            /* If the new type is signed, sign extend the value */
136            if (IsSignSigned (NewType)) {
137                if (Expr->IVal & (0x01UL << (NewBits-1))) {
138                    /* Beware: Use the safe shift routine here. */
139                    Expr->IVal |= shl_l (~0UL, NewBits);
140                }
141            }
142        }
143
144        /* Do the integer constant <-> absolute address conversion if necessary */
145        if (IsClassPtr (NewType)) {
146            Expr->Flags &= ~E_MASK_LOC;
147            Expr->Flags |= E_LOC_ABS | E_ADDRESS_OF;
148        } else if (IsClassInt (NewType)) {
149            Expr->Flags &= ~(E_MASK_LOC | E_ADDRESS_OF);
150            Expr->Flags |= E_LOC_NONE;
151        }
152
153    } else {
154
155        /* The value is not a constant. If the sizes of the types are
156        ** not equal, add conversion code. Be sure to convert chars
157        ** correctly.
158        */
159        if (OldBits != NewBits) {
160
161            /* Load the value into the primary */
162            LoadExpr (CF_NONE, Expr);
163
164            /* Emit typecast code. */
165            g_typecast (TypeOf (NewType), TypeOf (OldType));
166
167            /* Value is now an rvalue in the primary */
168            ED_FinalizeRValLoad (Expr);
169        }
170    }
171
172ExitPoint:
173    /* The expression has always the new type */
174    ReplaceType (Expr, NewType);
175}

289            ** could just silently fail here to avoid excess errors, but to
290            ** be safe, we must ensure that we do have errors.
291            */
292            if (IsIncompleteESUType (NewType)) {
293                Error ("Conversion to incomplete type '%s'", GetFullTypeName (NewType));

295                Error ("Conversion from incomplete type '%s'", GetFullTypeName (Expr->Type));
296            }
297        }
298    }

303void TypeCast (ExprDesc* Expr)
304/* Handle an explicit cast. */
305{
306    Type    NewType[MAXTYPELEN];
307
308    /* Skip the left paren */
309    NextToken ();
310
311    /* Read the type */
312    ParseType (NewType);
313
314    /* Closing paren */
315    ConsumeRParen ();
316
317    /* Read the expression we have to cast */
318    hie10 (Expr);
319
320    /* Only allow casts to arithmetic, pointer or void types */
321    if (IsCastType (NewType)) {
322        if (!IsIncompleteESUType (NewType)) {
323            /* Convert functions and arrays to "pointer to" object */
324            Expr->Type = PtrConversion (Expr->Type);
325
326            if (TypeCmp (NewType, Expr->Type).C >= TC_PTR_INCOMPATIBLE) {
327                /* If the new type has the same underlying presentation, just
328                ** use it to replace the old one.
329                */
330                ReplaceType (Expr, NewType);
331            } else if (IsCastType (Expr->Type)) {
332                /* Convert the value. The result has always the new type */
333                DoConversion (Expr, NewType);
334            } else {
335                TypeCompatibilityDiagnostic (NewType, Expr->Type, 1,
336                    "Cast to incompatible type '%s' from '%s'");
337            }
338        } else {
339            Error ("Cast to incomplete type '%s'",
340                   GetFullTypeName (NewType));
341        }
342    } else {
343        Error ("Arithmetic or pointer type expected but %s is used",
344               GetBasicTypeName (NewType));
345    }
346
347    /* If the new type is void, the cast expression can have no effects */
348    if (IsTypeVoid (NewType)) {
349        Expr->Flags |= E_EVAL_MAYBE_UNUSED;
350    }
351
352    /* The result is always an rvalue */
353    ED_MarkExprAsRVal (Expr);
354}

415void TypeComposition (Type* lhs, const Type* rhs)
416/* Recursively compose two types into lhs. The two types must have compatible
417** type or this fails with a critical check.
418*/
419{
420    FuncDesc*   F1;
421    FuncDesc*   F2;
422    long LeftCount, RightCount;
423
424    /* Compose two types */
425    while (lhs->C != T_END) {
426
427        /* Check if the end of the type string is reached */
428        if (rhs->C == T_END) {
429            break;
430        }
431
432        /* Check for sanity */
433        CHECK (GetUnderlyingTypeCode (lhs) == GetUnderlyingTypeCode (rhs));
434
435        /* Check for special type elements */
436        if (IsTypeFunc (lhs)) {
437            /* Compose the function descriptors */
438            F1 = GetFuncDesc (lhs);
439            F2 = GetFuncDesc (rhs);

442            ** not a function definition, because the flag is reset in this
443            ** case), its declaration is replaced by the other declaration. If
444            ** neither of the parameter lists is empty, we have to compose them

465
466            /* Set composite type if it is requested */
467            if (LeftCount != UNSPECIFIED) {
468                SetElementCount (lhs, LeftCount);
469            } else if (RightCount != UNSPECIFIED) {
470                SetElementCount (lhs, RightCount);
471            }
472        } else {
473            /* Combine the qualifiers */
474            if (IsClassPtr (lhs)) {
475                ++lhs;
476                ++rhs;
477                lhs->C |= GetQualifier (rhs);
478            }
479        }
480
481        /* Next type string element */
482        ++lhs;
483        ++rhs;
484    }
485
486    return;
487}

491FuncDesc* RefineFuncDesc (Type* OldType, const Type* NewType)
492/* Refine the existing function descriptor with a new one */
493{
494    FuncDesc* Old = GetFuncDesc (OldType);
495    FuncDesc* New = GetFuncDesc (NewType);
496
497    CHECK (Old != 0 && New != 0);
498
499    if ((New->Flags & FD_EMPTY) == 0) {
500        if ((Old->Flags & FD_EMPTY) == 0) {
501            TypeComposition (OldType, NewType);
502        } else {
503            TypeCopy (OldType, NewType);
504            Old->Flags &= ~FD_EMPTY;
505        }
506    }
507
508    return Old;
509}
