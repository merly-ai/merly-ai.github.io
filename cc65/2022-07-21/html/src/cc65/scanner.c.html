160static int CmpKey (const void* Key, const void* Elem)
161/* Compare function for bsearch */
162{
163    return strcmp ((const char*) Key, ((const struct Keyword*) Elem)->Key);
164}

168static token_t FindKey (const char* Key)
169/* Find a keyword and return the token. Return IDENT if the token is not a
170** keyword.
171*/
172{
173    struct Keyword* K;
174    K = bsearch (Key, Keywords, KEY_COUNT, sizeof (Keywords [0]), CmpKey);
175    if (K && (K->Std & (0x01 << IS_Get (&Standard))) != 0) {
176        return K->Tok;
177    } else {
178        return TOK_IDENT;
179    }
180}

184static int SkipWhite (void)
185/* Skip white space in the input stream, reading and preprocessing new lines
186** if necessary. Return 0 if end of file is reached, return 1 otherwise.
187*/
188{
189    while (1) {
190        while (CurC == '\0') {
191            if (NextLine () == 0) {
192                return 0;
193            }
194            Preprocess ();
195        }
196        if (IsSpace (CurC)) {
197            NextChar ();
198        } else {
199            return 1;
200        }
201    }
202}

216void SymName (char* S)
217/* Read a symbol from the input stream. The first character must have been
218** checked before calling this function. The buffer is expected to be at
219** least of size MAX_IDENTLEN+1.
220*/
221{
222    unsigned Len = 0;
223    do {
224        if (Len < MAX_IDENTLEN) {
225            ++Len;
226            *S++ = CurC;
227        }
228        NextChar ();
229    } while (IsIdent (CurC) || IsDigit (CurC));
230    *S = '\0';
231}

266static int ParseChar (void)
267/* Parse a character. Converts escape chars into character codes. */
268{
269    int C;
270    int HadError;
271    int Count;
272
273    /* Check for escape chars */
274    if (CurC == '\\') {
275        NextChar ();
276        switch (CurC) {
277            case '?':
278                C = '\?';
279                break;
280            case 'a':
281                C = '\a';
282                break;
283            case 'b':
284                C = '\b';
285                break;
286            case 'f':
287                C = '\f';
288                break;
289            case 'r':
290                C = '\r';
291                break;
292            case 'n':
293                C = '\n';
294                break;
295            case 't':
296                C = '\t';
297                break;
298            case 'v':
299                C = '\v';
300                break;
301            case '\"':
302                C = '\"';
303                break;
304            case '\'':
305                C = '\'';
306                break;
307            case '\\':
308                C = '\\';
309                break;
310            case 'x':
311            case 'X':
312                /* Hex character constant */
313                if (!IsXDigit (NextC)) {
314                    Error ("\\x used with no following hex digits");
315                    C = ' ';
316                } else {
317                    HadError = 0;
318                    C = 0;
319                    while (IsXDigit (NextC)) {
320                        if ((C << 4) >= 256) {
321                            if (!HadError) {
322                                Error ("Hex character constant out of range");
323                                HadError = 1;
324                            }
325                        } else {
326                            C = (C << 4) | HexVal (NextC);
327                        }
328                        NextChar ();
329                    }
330                }
331                break;
332            case '0':
333            case '1':
334            case '2':
335            case '3':
336            case '4':
337            case '5':
338            case '6':
339            case '7':
340                /* Octal constant */
341                Count = 1;
342                C = HexVal (CurC);
343                while (IsODigit (NextC) && Count++ < 3) {
344                    C = (C << 3) | HexVal (NextC);
345                    NextChar ();
346                }
347                if (C >= 256)
348                    Error ("Octal character constant out of range");
349                break;
350            default:
351                C = CurC;
352                Error ("Illegal escaped character: 0x%02X", CurC);
353                break;
354        }
355    } else {
356        C = CurC;
357    }
358
359    /* Skip the character read */
360    NextChar ();
361
362    /* Do correct sign extension */
363    return SignExtendChar (C);
364}

368static void CharConst (void)
369/* Parse a character constant. */
370{
371    int C;
372
373    /* Skip the quote */
374    NextChar ();
375
376    /* Get character */
377    C = ParseChar ();
378
379    /* Check for closing quote */
380    if (CurC != '\'') {
381        Error ("'\'' expected");
382    } else {
383        /* Skip the quote */
384        NextChar ();
385    }
386
387    /* Setup values and attributes */
388    NextTok.Tok  = TOK_CCONST;
389
390    /* Translate into target charset */
391    NextTok.IVal = SignExtendChar (TgtTranslateChar (C));
392
393    /* Character constants have type int */
394    NextTok.Type = type_int;
395}

399static void StringConst (void)
400/* Parse a quoted string */
401{
402    /* String buffer */
403    StrBuf S = AUTO_STRBUF_INITIALIZER;
404
405    /* Assume next token is a string constant */
406    NextTok.Tok  = TOK_SCONST;
407
408    /* Concatenate strings. If at least one of the concenated strings is a wide
409    ** character literal, the whole string is a wide char literal, otherwise
410    ** it's a normal string literal.
411    */
412    while (1) {
413
414        /* Check if this is a normal or a wide char string */
415        if (CurC == 'L' && NextC == '\"') {
416            /* Wide character literal */
417            NextTok.Tok = TOK_WCSCONST;
418            NextChar ();
419            NextChar ();
420        } else if (CurC == '\"') {
421            /* Skip the quote char */
422            NextChar ();
423        } else {
424            /* No string */
425            break;
426        }
427
428        /* Read until end of string */
429        while (CurC != '\"') {
430            if (CurC == '\0') {
431                Error ("Unexpected newline");
432                break;
433            }
434            SB_AppendChar (&S, ParseChar ());
435        }
436
437        /* Skip closing quote char if there was one */
438        NextChar ();
439
440        /* Skip white space, read new input */
441        SkipWhite ();
442
443    }
444
445    /* Terminate the string */
446    SB_AppendChar (&S, '\0');
447
448    /* Add the whole string to the literal pool */
449    NextTok.SVal = AddLiteralStr (&S);
450
451    /* Free the buffer */
452    SB_Done (&S);
453}

457static void NumericConst (void)
458/* Parse a numeric constant */
459{
460    unsigned Base;              /* Temporary number base */
461    unsigned Prefix;            /* Base according to prefix */
462    StrBuf   S = STATIC_STRBUF_INITIALIZER;
463    int      IsFloat;
464    char     C;
465    unsigned DigitVal;
466    unsigned long IVal;         /* Value */
467
468    /* Check for a leading hex, octal or binary prefix and determine the
469    ** possible integer types.
470    */
471    if (CurC == '0') {
472        /* Gobble 0 and examine next char */
473        NextChar ();
474        if (toupper (CurC) == 'X') {
475            Base = Prefix = 16;
476            NextChar ();        /* gobble "x" */
477        } else if (toupper (CurC) == 'B' && IS_Get (&Standard) >= STD_CC65) {
478            Base = Prefix = 2;
479            NextChar ();        /* gobble 'b' */
480        } else {
481            Base = 10;          /* Assume 10 for now - see below */
482            Prefix = 8;         /* Actual prefix says octal */
483        }
484    } else {
485        Base  = Prefix = 10;
486    }
487
488    /* Because floating point numbers don't have octal prefixes (a number
489    ** with a leading zero is decimal), we first have to read the number
490    ** before converting it, so we can determine if it's a float or an
491    ** integer.
492    */
493    while (IsXDigit (CurC) && HexVal (CurC) < Base) {
494        SB_AppendChar (&S, CurC);
495        NextChar ();
496    }
497    SB_Terminate (&S);
498
499    /* The following character tells us if we have an integer or floating
500    ** point constant. Note: Hexadecimal floating point constants aren't
501    ** supported in C89.
502    */
503    IsFloat = (CurC == '.' ||
504               (Base == 10 && toupper (CurC) == 'E') ||
505               (Base == 16 && toupper (CurC) == 'P' && IS_Get (&Standard) >= STD_C99));
506
507    /* If we don't have a floating point type, an octal prefix results in an
508    ** octal base.
509    */
510    if (!IsFloat && Prefix == 8) {
511        Base = 8;
512    }
513
514    /* Since we do now know the correct base, convert the remembered input
515    ** into a number.
516    */
517    SB_Reset (&S);
518    IVal = 0;
519    while ((C = SB_Get (&S)) != '\0') {
520        DigitVal = HexVal (C);
521        if (DigitVal >= Base) {
522            Error ("Numeric constant contains digits beyond the radix");
523        }
524        IVal = (IVal * Base) + DigitVal;

528    SB_Done (&S);

532

534        unsigned WarnTypes = 0;
535
536        /* Check for a suffix and determine the possible types. It is always
537        ** possible to convert the data to unsigned long even if the IT_ULONG
538        ** flag were not set, but we are not doing that.
539        */
540        if (toupper (CurC) == 'U') {
541            /* Unsigned type */
542            NextChar ();
543            if (toupper (CurC) != 'L') {
544                Types = IT_UINT | IT_ULONG;
545            } else {
546                NextChar ();
547                Types = IT_ULONG;
548            }
549        } else if (toupper (CurC) == 'L') {
550            /* Long type */
551            NextChar ();
552            if (toupper (CurC) != 'U') {
553                Types = IT_LONG | IT_ULONG;
554                WarnTypes = IT_ULONG;
555            } else {
556                NextChar ();
557                Types = IT_ULONG;
558            }
559        } else {
560            if (Prefix == 10) {
561                /* Decimal constants are of any type but uint */
562                Types = IT_INT | IT_LONG | IT_ULONG;
563                WarnTypes = IT_LONG | IT_ULONG;
564            } else {
565                /* Binary, octal and hex constants can be of any type */
566                Types = IT_INT | IT_UINT | IT_LONG | IT_ULONG;
567            }
568        }

594            ** because the constant is of type unsigned long while the user

606        } else if (Types & IT_UINT) {
607            NextTok.Type = type_uint;
608        } else if (Types & IT_LONG) {
609            NextTok.Type = type_long;
610        } else {
611            NextTok.Type = type_ulong;
612        }

614        /* Set the value and the token */

624        if (CurC == '.') {

652            /* Skip the exponent notifier */
653            NextChar ();
654
655            /* Read an optional sign */
656            if (CurC == '-') {
657                NextChar ();
658            } else if (CurC == '+') {
659                NextChar ();
660            }
661
662            /* Read exponent digits. Since we support only 32 bit floats
663            ** with a maximum exponent of +-/127, we read the exponent
664            ** part as integer with up to 3 digits and drop the remainder.
665            ** This avoids an overflow of Exp. The exponent is always
666            ** decimal, even for hex float consts.
667            */
668            Digits = 0;
669            Exp    = 0;
670            while (IsDigit (CurC)) {
671                if (++Digits <= 3) {
672                    Exp = Exp * 10 + HexVal (CurC);
673                }
674                NextChar ();
675            }
676
677            /* Check for errors: We must have exponent digits, and not more
678            ** than three.
679            */
680            if (Digits == 0) {
681                Error ("Floating constant exponent has no digits");
682            } else if (Digits > 3) {
683                Warning ("Floating constant exponent is too large");
684            }
685
686            /* Scale the exponent and adjust the value accordingly */
687            if (Exp) {
688                FVal = FP_D_Mul (FVal, FP_D_Make (pow (10, Exp)));
689            }
690        }
691
692        /* Check for a suffix and determine the type of the constant */
693        if (toupper (CurC) == 'F') {
694            NextChar ();
695            NextTok.Type = type_float;
696        } else {
697            NextTok.Type = type_double;
698        }
699
700        /* Set the value and the token */
701        NextTok.FVal = FVal;
702        NextTok.Tok  = TOK_FCONST;
703
704    }
