122static unsigned OpeningCurlyBraces (unsigned BracesNeeded)
123/* Accept any number of opening curly braces around an initialization, skip
124** them and return the number. If the number of curly braces is less than
125** BracesNeeded, issue a warning.
126*/
127{
128    unsigned BraceCount = 0;
129    while (CurTok.Tok == TOK_LCURLY) {
130        ++BraceCount;
131        NextToken ();
132    }
133    if (BraceCount < BracesNeeded) {
134        Error ("'{' expected");
135    }
136    return BraceCount;
137}

164static void DefineData (ExprDesc* Expr)
165/* Output a data definition for the given expression */
166{
167    switch (ED_GetLoc (Expr)) {
168
169        case E_LOC_NONE:
170            /* Immediate numeric value with no storage */
171            g_defdata (CF_IMM | TypeOf (Expr->Type) | CF_CONST, Expr->IVal, 0);
172            break;
173
174        case E_LOC_ABS:
175            /* Absolute numeric address */
176            g_defdata (CF_ABSOLUTE | TypeOf (Expr->Type) | CF_CONST, Expr->IVal, 0);
177            break;
178
179        case E_LOC_GLOBAL:
180            /* Global variable */
181            g_defdata (CF_EXTERNAL, Expr->Name, Expr->IVal);
182            break;
183
184        case E_LOC_STATIC:
185            /* Static variable */
186            g_defdata (CF_STATIC, Expr->Name, Expr->IVal);
187            break;
188
189        case E_LOC_LITERAL:
190            /* Literal in the literal pool */
191            g_defdata (CF_LITERAL, Expr->Name, Expr->IVal);
192            break;
193
194        case E_LOC_REGISTER:
195            /* Register variable. Taking the address is usually not
196            ** allowed.
197            */
198            if (IS_Get (&AllowRegVarAddr) == 0) {
199                Error ("Cannot take the address of a register variable");
200            }
201            g_defdata (CF_REGVAR, Expr->Name, Expr->IVal);
202            break;
203
204        case E_LOC_CODE:
205            /* Code label location */
206            g_defdata (CF_CODE, Expr->Name, Expr->IVal);
207            break;
208
209        case E_LOC_STACK:
210        case E_LOC_PRIMARY:
211        case E_LOC_EXPR:
212            Error ("Non constant initializer");
213            break;
214
215        default:
216            Internal ("Unknown constant type: 0x%04X", ED_GetLoc (Expr));
217    }
218}

222static void DefineBitFieldData (StructInitData* SI)
223/* Output bit field data */
224{
225    /* Ignore if we have no data */
226    if (SI->ValBits > 0) {
227
228        /* Output the data */
229        g_defdata (CF_CHAR | CF_UNSIGNED | CF_CONST, SI->BitVal, 0);
230
231        /* Update the data from SI and account for the size */
232        if (SI->ValBits >= CHAR_BITS) {
233            SI->BitVal >>= CHAR_BITS;
234            SI->ValBits -= CHAR_BITS;
235        } else {
236            SI->BitVal  = 0;
237            SI->ValBits = 0;
238        }
239        SI->Offs += SIZEOF_CHAR;
240    }
241}

256static ExprDesc ParseScalarInitInternal (const Type* T)
257/* Parse initializaton for scalar data types. This function will not output the
258** data but return it in ED.
259*/
260{
261    /* Optional opening brace */
262    unsigned BraceCount = OpeningCurlyBraces (0);
263
264    /* We warn if an initializer for a scalar contains braces, because this is
265    ** quite unusual and often a sign for some problem in the input.
266    */
267    if (BraceCount > 0) {
268        Warning ("Braces around scalar initializer");
269    }
270
271    /* Get the expression and convert it to the target type */
272    ExprDesc ED = NoCodeConstExpr (hie1);
273    TypeConversion (&ED, T);
274
275    /* Close eventually opening braces */
276    ClosingCurlyBraces (BraceCount);
277
278    return ED;
279}

518            }
519            /* Avoid consuming the comma if any */
520            goto NextMember;
521        }
522
523        /* Skip comma this round */
524        if (SkipComma) {
525            NextToken ();
526            SkipComma = 0;
527        }
528
529        if (SymIsBitField (Sym)) {
530
531            /* Parse initialization of one field. Bit-fields need a special
532            ** handling.
533            */
534            ExprDesc Field;
535            ED_Init (&Field);
536            unsigned long Val;
537            unsigned Shift;
538
539            /* Calculate the bitmask from the bit-field data */
540            unsigned long Mask = shl_l (1UL, Sym->Type->A.B.Width) - 1UL;
541
542            /* Safety ... */
543            CHECK (Sym->V.Offs * CHAR_BITS + Sym->Type->A.B.Offs ==
544                   SI.Offs     * CHAR_BITS + SI.ValBits);
545
546            /* Read the data, check for a constant integer, do a range check */
547            Field = ParseScalarInitInternal (IntPromotion (Sym->Type));
548            if (!ED_IsConstAbsInt (&Field)) {
549                Error ("Constant initializer expected");
550                ED_MakeConstAbsInt (&Field, 1);
551            }
552
553            /* Truncate the initializer value to the width of the bit-field and check if we lost
554            ** any useful bits.
555            */
556            Val = (unsigned long) Field.IVal & Mask;
557            if (IsSignUnsigned (Sym->Type)) {
558                if (Field.IVal < 0 || (unsigned long) Field.IVal != Val) {
559                    Warning (IsSignUnsigned (Field.Type) ?
560                             "Implicit truncation from '%s' to '%s : %u' in bit-field initializer"
561                             " changes value from %lu to %lu" :
562                             "Implicit truncation from '%s' to '%s : %u' in bit-field initializer"
563                             " changes value from %ld to %lu",
564                             GetFullTypeName (Field.Type), GetFullTypeName (Sym->Type),
565                             Sym->Type->A.B.Width, Field.IVal, Val);
566                }
567            } else {
568                /* Sign extend back to full width of host long. */
569                unsigned ShiftBits = sizeof (long) * CHAR_BIT - Sym->Type->A.B.Width;
570                long RestoredVal = asr_l (asl_l (Val, ShiftBits), ShiftBits);
571                if (Field.IVal != RestoredVal) {
572                    Warning (IsSignUnsigned (Field.Type) ?
573                             "Implicit truncation from '%s' to '%s : %u' in bit-field initializer"
574                             " changes value from %lu to %ld" :
575                             "Implicit truncation from '%s' to '%s : %u' in bit-field initializer"
576                             " changes value from %ld to %ld",
577                             GetFullTypeName (Field.Type), GetFullTypeName (Sym->Type),
578                             Sym->Type->A.B.Width, Field.IVal, RestoredVal);
579                }
580            }
581
582            /* Add the value to the currently stored bit-field value */
583            Shift = (Sym->V.Offs - SI.Offs) * CHAR_BITS + Sym->Type->A.B.Offs;
584            SI.BitVal |= (Val << Shift);
585
586            /* Account for the data and output any full bytes we have. */
587            SI.ValBits += Sym->Type->A.B.Width;
588            /* Make sure unsigned is big enough to hold the value, 32 bits.
589            ** This cannot be more than 32 bits because a 16-bit or 32-bit
590            ** bit-field will always be byte-aligned with padding before it
591            ** if there are bits from prior fields that haven't been output
592            ** yet.
593            */
594            CHECK (SI.ValBits <= CHAR_BIT * sizeof(SI.BitVal));
595            /* TODO: Generalize this so any type can be used. */
596            CHECK (SI.ValBits <= LONG_BITS);
597            while (SI.ValBits >= CHAR_BITS) {
598                DefineBitFieldData (&SI);
599            }
600
601        } else {
602
603            /* Standard member. We should never have stuff from a
604            ** bit-field left because an anonymous member was added
605            ** for padding by ParseStructDecl.
606            */
607            CHECK (SI.ValBits == 0);
608
609            /* Flexible array members may only be initialized if they are
610            ** the last field (or part of the last struct field).
611            */
612            SI.Offs += ParseInitInternal (Sym->Type, Braces, AllowFlexibleMembers && Sym->NextSym == 0);
613        }
614
615        /* More initializers? */
616        if (CurTok.Tok != TOK_COMMA) {
617            break;
618        }
619
620        /* Skip the comma next round */
621        SkipComma = 1;
622
623NextMember:
624        /* Next member. For unions, only the first one can be initialized */
625        if (IsTypeUnion (T)) {
626            /* Union */
627            Sym = 0;
628        } else {
629            /* Struct */
630            Sym = Sym->NextSym;
631        }
632    }
633
634    if (HasCurly) {
635        /* Consume the closing curly brace */
636        ConsumeRCurly ();
637    }
638
639    /* If we have data from a bit-field left, output it now */
640    CHECK (SI.ValBits < CHAR_BITS);
641    DefineBitFieldData (&SI);
642
643    /* If there are struct fields left, reserve additional storage */
644    if (SI.Offs < SI.Size) {
645        g_zerobytes (SI.Size - SI.Offs);
646        SI.Offs = SI.Size;
647    }
648
649    /* Return the actual number of bytes initialized. This number may be
650    ** larger than sizeof (Struct) if flexible array members are present and
651    ** were initialized (possible in non ANSI mode).
652    */
653    return SI.Offs;
654}

658static unsigned ParseVoidInit (Type* T)
659/* Parse an initialization of a void variable (special cc65 extension).
660** Return the number of bytes initialized.
661*/
662{
663    unsigned Size;
664
665    /* Opening brace */
666    ConsumeLCurly ();
667
668    /* Allow an arbitrary list of values */
669    Size = 0;
670    do {
671        ExprDesc Expr = NoCodeConstExpr (hie1);
672        switch (GetUnderlyingTypeCode (&Expr.Type[0])) {
673
674            case T_SCHAR:
675            case T_UCHAR:
676                if (ED_IsConstAbsInt (&Expr)) {
677                    /* Make it byte sized */
678                    Expr.IVal &= 0xFF;
679                }
680                DefineData (&Expr);
681                Size += SIZEOF_CHAR;
682                break;
683
684            case T_SHORT:
685            case T_USHORT:
686            case T_INT:
687            case T_UINT:
688            case T_PTR:
689            case T_ARRAY:
690                if (ED_IsConstAbsInt (&Expr)) {
691                    /* Make it word sized */
692                    Expr.IVal &= 0xFFFF;
693                }
694                DefineData (&Expr);
695                Size += SIZEOF_INT;
696                break;
697
698            case T_LONG:
699            case T_ULONG:
700                if (ED_IsConstAbsInt (&Expr)) {
701                    /* Make it dword sized */
702                    Expr.IVal &= 0xFFFFFFFF;
703                }
704                DefineData (&Expr);
705                Size += SIZEOF_LONG;
706                break;
707
708            default:
709                Error ("Illegal type in initialization");
710                break;
711
712        }
713
714        if (CurTok.Tok != TOK_COMMA) {
715            break;
716        }
717        NextToken ();
718
719    } while (CurTok.Tok != TOK_RCURLY);
720
721    /* Closing brace */
722    ConsumeRCurly ();
723
724    /* Number of bytes determined by initializer */
725    if (T->A.U != 0 && T->A.U != Size) {
726        Error ("'void' array initialized with elements of variant sizes");
727    } else {
728        T->A.U = Size;
729    }
730
731    /* Return the number of bytes initialized */
732    return Size;
733}

736
737static unsigned ParseInitInternal (Type* T, int *Braces, int AllowFlexibleMembers)
738/* Parse initialization of variables. Return the number of data bytes. */
739{
740    switch (GetUnderlyingTypeCode (T)) {
741
742        case T_SCHAR:
743        case T_UCHAR:
744        case T_SHORT:
745        case T_USHORT:
746        case T_INT:
747        case T_UINT:
748        case T_LONG:
749        case T_ULONG:
750        case T_FLOAT:
751        case T_DOUBLE:
752            return ParseScalarInit (T);
753
754        case T_PTR:
755            return ParsePointerInit (T);
756
757        case T_ARRAY:
758            return ParseArrayInit (T, Braces, AllowFlexibleMembers);
759
760        case T_STRUCT:
761        case T_UNION:
762            return ParseStructInit (T, Braces, AllowFlexibleMembers);
763
764        case T_ENUM:
765            /* Incomplete enum type must have already raised errors.
766            ** Just proceed to consume the value.
767            */
768            return ParseScalarInit (T);
769
770        case T_VOID:
771            if (IS_Get (&Standard) == STD_CC65) {
772                /* Special cc65 extension in non-ANSI mode */
773                return ParseVoidInit (T);
774            }
775            /* FALLTHROUGH */
776
777        default:
778            Error ("Illegal type");
779            return SIZEOF_CHAR;
780
781    }
782}

785
786unsigned ParseInit (Type* T)
787/* Parse initialization of variables. Return the number of data bytes. */
788{
789    /* Current curly braces layers */
790    int Braces = 0;
791
792    /* Parse the initialization. Flexible array members can only be initialized
793    ** in cc65 mode.
794    */
795    unsigned Size = ParseInitInternal (T, &Braces, IS_Get (&Standard) == STD_CC65);
796
797    /* The initialization may not generate code on global level, because code
798    ** outside function scope will never get executed.
799    */
800    if (HaveGlobalCode ()) {
801        Error ("Non constant initializers");
802        RemoveGlobalCode ();
803    }
804
805    /* Return the size needed for the initialization */
806    return Size;
807}
