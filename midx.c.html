321/* Match "foo.idx" against either "foo.pack" _or_ "foo.idx". */
322static int cmp_idx_or_pack_name(const char *idx_or_pack_name,
323        const char *idx_name)
324{
325  /* Skip past any initial matching prefix. */
326  while (*idx_name && *idx_name == *idx_or_pack_name) {
327    idx_name++;
328    idx_or_pack_name++;
329  }
330
331  /*
332   * If we didn't match completely, we may have matched "pack-1234." and
333   * be left with "idx" and "pack" respectively, which is also OK. We do
334   * not have to check for "idx" and "idx", because that would have been
335   * a complete match (and in that case these strcmps will be false, but
336   * we'll correctly return 0 from the final strcmp() below.
337   *
338   * Technically this matches "fooidx" and "foopack", but we'd never have
339   * such names in the first place.
340   */
341  if (!strcmp(idx_name, "idx") && !strcmp(idx_or_pack_name, "pack"))
342    return 0;
343
344  /*
345   * This not only checks for a complete match, but also orders based on
346   * the first non-identical character, which means our ordering will
347   * match a raw strcmp(). That makes it OK to use this to binary search
348   * a naively-sorted list.
349   */
350  return strcmp(idx_or_pack_name, idx_name);
351}

353int midx_contains_pack(struct multi_pack_index *m, const char *idx_or_pack_name)
354{
355  uint32_t first = 0, last = m->num_packs;
356
357  while (first < last) {
358    uint32_t mid = first + (last - first) / 2;
359    const char *current;
360    int cmp;
361
362    current = m->pack_names[mid];
363    cmp = cmp_idx_or_pack_name(idx_or_pack_name, current);
364    if (!cmp)
365      return 1;
366    if (cmp > 0) {
367      first = mid + 1;
368      continue;
369    }
370    last = mid;
371  }
372
373  return 0;
374}

526  const struct pack_midx_entry *a = (const struct pack_midx_entry *)_a;
527  const struct pack_midx_entry *b = (const struct pack_midx_entry *)_b;
528  int cmp = oidcmp(&a->oid, &b->oid);
529
530  if (cmp)
531    return cmp;
532
533  /* Sort objects in a preferred pack first when multiple copies exist. */
534  if (a->preferred > b->preferred)
535    return -1;
536  if (a->preferred < b->preferred)
537    return 1;
538
539  if (a->pack_mtime > b->pack_mtime)
540    return -1;
541  else if (a->pack_mtime < b->pack_mtime)
542    return 1;
543
544  return a->pack_int_id - b->pack_int_id;
545}

579/*
580 * It is possible to artificially get into a state where there are many
581 * duplicate copies of objects. That can create high memory pressure if
582 * we are to create a list of all objects before de-duplication. To reduce
583 * this memory pressure without a significant performance drop, automatically
584 * group objects by the first byte of their object id. Use the IDX fanout
585 * tables to group the data, copy to a local array, then sort.
586 *
587 * Copy only the de-duplicated entries (selected by most-recent modified time
588 * of a packfile containing the object).
589 */
590static struct pack_midx_entry *get_sorted_entries(struct multi_pack_index *m,
591              struct pack_info *info,
592              uint32_t nr_packs,
593              uint32_t *nr_objects,
594              int preferred_pack)
595{
596  uint32_t cur_fanout, cur_pack, cur_object;
597  uint32_t alloc_fanout, alloc_objects, total_objects = 0;
598  struct pack_midx_entry *entries_by_fanout = NULL;
599  struct pack_midx_entry *deduplicated_entries = NULL;
600  uint32_t start_pack = m ? m->num_packs : 0;
601
602  for (cur_pack = start_pack; cur_pack < nr_packs; cur_pack++)
603    total_objects += info[cur_pack].p->num_objects;
604
605  /*
606   * As we de-duplicate by fanout value, we expect the fanout
607   * slices to be evenly distributed, with some noise. Hence,
608   * allocate slightly more than one 256th.
609   */
610  alloc_objects = alloc_fanout = total_objects > 3200 ? total_objects / 200 : 16;
611
612  ALLOC_ARRAY(entries_by_fanout, alloc_fanout);
613  ALLOC_ARRAY(deduplicated_entries, alloc_objects);
614  *nr_objects = 0;
615
616  for (cur_fanout = 0; cur_fanout < 256; cur_fanout++) {
617    uint32_t nr_fanout = 0;
618
619    if (m) {
620      uint32_t start = 0, end;
621
622      if (cur_fanout)
623        start = ntohl(m->chunk_oid_fanout[cur_fanout - 1]);
624      end = ntohl(m->chunk_oid_fanout[cur_fanout]);
625
626      for (cur_object = start; cur_object < end; cur_object++) {
627        ALLOC_GROW(entries_by_fanout, nr_fanout + 1, alloc_fanout);
628        nth_midxed_pack_midx_entry(m,
629                 &entries_by_fanout[nr_fanout],
630                 cur_object);
631        if (nth_midxed_pack_int_id(m, cur_object) == preferred_pack)
632          entries_by_fanout[nr_fanout].preferred = 1;
633        else
634          entries_by_fanout[nr_fanout].preferred = 0;
635        nr_fanout++;
636      }
637    }
638
639    for (cur_pack = start_pack; cur_pack < nr_packs; cur_pack++) {
640      uint32_t start = 0, end;
641      int preferred = cur_pack == preferred_pack;
642
643      if (cur_fanout)
644        start = get_pack_fanout(info[cur_pack].p, cur_fanout - 1);
645      end = get_pack_fanout(info[cur_pack].p, cur_fanout);
646
647      for (cur_object = start; cur_object < end; cur_object++) {
648        ALLOC_GROW(entries_by_fanout, nr_fanout + 1, alloc_fanout);
649        fill_pack_entry(cur_pack,
650            info[cur_pack].p,
651            cur_object,
652            &entries_by_fanout[nr_fanout],
653            preferred);
654        nr_fanout++;
655      }
656    }
657
658    QSORT(entries_by_fanout, nr_fanout, midx_oid_compare);
659
660    /*
661     * The batch is now sorted by OID and then mtime (descending).
662     * Take only the first duplicate.
663     */
664    for (cur_object = 0; cur_object < nr_fanout; cur_object++) {
665      if (cur_object && oideq(&entries_by_fanout[cur_object - 1].oid,
666            &entries_by_fanout[cur_object].oid))
667        continue;
668
669      ALLOC_GROW(deduplicated_entries, *nr_objects + 1, alloc_objects);
670      memcpy(&deduplicated_entries[*nr_objects],
671             &entries_by_fanout[cur_object],
672             sizeof(struct pack_midx_entry));
673      (*nr_objects)++;
674    }
675  }
676
677  free(entries_by_fanout);
678  return deduplicated_entries;
679}

681static int write_midx_pack_names(struct hashfile *f, void *data)
682{
683  struct write_midx_context *ctx = data;

706  if (i < MIDX_CHUNK_ALIGNMENT) {

714static int write_midx_oid_fanout(struct hashfile *f,
715         void *data)
716{
717  struct write_midx_context *ctx = data;
718  struct pack_midx_entry *list = ctx->entries;
719  struct pack_midx_entry *last = ctx->entries + ctx->entries_nr;
720  uint32_t count = 0;
721  uint32_t i;
722
723  /*
724  * Write the first-level table (the list is sorted,
725  * but we use a 256-entry lookup to be able to avoid
726  * having to do eight extra binary search iterations).
727  */
728  for (i = 0; i < 256; i++) {
729    struct pack_midx_entry *next = list;
730
731    while (next < last && next->oid.hash[0] == i) {
732      count++;
733      next++;
734    }
735
736    hashwrite_be32(f, count);
737    list = next;
738  }
739
740  return 0;
741}

1012static struct commit **find_commits_for_midx_bitmap(uint32_t *indexed_commits_nr_p,
1013                const char *refs_snapshot,
1014                struct write_midx_context *ctx)
1015{
1016  struct rev_info revs;
1017  struct bitmap_commit_cb cb = {0};
1018

1020
1021  repo_init_revisions(the_repository, &revs, NULL);
1022  if (refs_snapshot) {
1023    read_refs_snapshot(refs_snapshot, &revs);
1024  } else {
1025    setup_revisions(0, NULL, &revs, NULL);
1026    for_each_ref(add_ref_to_pending, &revs);
1027  }

1030   * Skipping promisor objects here is intentional, since it only excludes
1031   * them from the list of reachable commits that we want to select from
1032   * when computing the selection of MIDX'd commits to receive bitmaps.

1035   * reachability, but fetching any objects missing from promisors at this
1036   * point is too late. But, if one of those objects can be reached from
1037   * an another object that is included in the bitmap, then we will
1038   * complain later that we don't have reachability closure (and fail
1039   * appropriately).
1040   */
1041  fetch_if_missing = 0;
1042  revs.exclude_promisor_objects = 1;
1043
1044  if (prepare_revision_walk(&revs))

1597int verify_midx_file(struct repository *r, const char *object_dir, unsigned flags)
1598{
1599  struct pair_pos_vs_id *pairs = NULL;
1600  uint32_t i;
1601  struct progress *progress = NULL;
1602  struct multi_pack_index *m = load_multi_pack_index(object_dir, 1);
1603  verify_midx_error = 0;
1604
1605  if (!m) {
1606    int result = 0;
1607    struct stat sb;
1608    struct strbuf filename = STRBUF_INIT;
1609
1610    get_midx_filename(&filename, object_dir);
1611
1612    if (!stat(filename.buf, &sb)) {
1613      error(_("multi-pack-index file exists, but failed to parse"));
1614      result = 1;
1615    }
1616    strbuf_release(&filename);
1617    return result;
1618  }
1619
1620  if (!midx_checksum_valid(m))
1621    midx_report(_("incorrect checksum"));
1622
1623  if (flags & MIDX_PROGRESS)
1624    progress = start_delayed_progress(_("Looking for referenced packfiles"),
1625            m->num_packs);
1626  for (i = 0; i < m->num_packs; i++) {
1627    if (prepare_midx_pack(r, m, i))
1628      midx_report("failed to load pack in position %d", i);
1629
1630    display_progress(progress, i + 1);
1631  }
1632  stop_progress(&progress);
1633
1634  for (i = 0; i < 255; i++) {
1635    uint32_t oid_fanout1 = ntohl(m->chunk_oid_fanout[i]);
1636    uint32_t oid_fanout2 = ntohl(m->chunk_oid_fanout[i + 1]);
1637
1638    if (oid_fanout1 > oid_fanout2)
1639      midx_report(_("oid fanout out of order: fanout[%d] = %"PRIx32" > %"PRIx32" = fanout[%d]"),
1640            i, oid_fanout1, oid_fanout2, i + 1);
1641  }
1642
1643  if (m->num_objects == 0) {
1644    midx_report(_("the midx contains no oid"));
1645    /*
1646     * Remaining tests assume that we have objects, so we can
1647     * return here.
1648     */
1649    goto cleanup;
1650  }
1651
1652  if (flags & MIDX_PROGRESS)
1653    progress = start_sparse_progress(_("Verifying OID order in multi-pack-index"),
1654             m->num_objects - 1);
1655  for (i = 0; i < m->num_objects - 1; i++) {
1656    struct object_id oid1, oid2;
1657
1658    nth_midxed_object_oid(&oid1, m, i);
1659    nth_midxed_object_oid(&oid2, m, i + 1);
1660
1661    if (oidcmp(&oid1, &oid2) >= 0)
1662      midx_report(_("oid lookup out of order: oid[%d] = %s >= %s = oid[%d]"),
1663            i, oid_to_hex(&oid1), oid_to_hex(&oid2), i + 1);
1664
1665    midx_display_sparse_progress(progress, i + 1);
1666  }
1667  stop_progress(&progress);
1668
1669  /*
1670   * Create an array mapping each object to its packfile id.  Sort it
1671   * to group the objects by packfile.  Use this permutation to visit
1672   * each of the objects and only require 1 packfile to be open at a
1673   * time.
1674   */
1675  ALLOC_ARRAY(pairs, m->num_objects);
1676  for (i = 0; i < m->num_objects; i++) {
1677    pairs[i].pos = i;
1678    pairs[i].pack_int_id = nth_midxed_pack_int_id(m, i);
1679  }
1680
1681  if (flags & MIDX_PROGRESS)
1682    progress = start_sparse_progress(_("Sorting objects by packfile"),
1683             m->num_objects);
1684  display_progress(progress, 0); /* TODO: Measure QSORT() progress */
1685  QSORT(pairs, m->num_objects, compare_pair_pos_vs_id);
1686  stop_progress(&progress);
1687
1688  if (flags & MIDX_PROGRESS)
1689    progress = start_sparse_progress(_("Verifying object offsets"), m->num_objects);
1690  for (i = 0; i < m->num_objects; i++) {
1691    struct object_id oid;
1692    struct pack_entry e;
1693    off_t m_offset, p_offset;
1694
1695    if (i > 0 && pairs[i-1].pack_int_id != pairs[i].pack_int_id &&
1696        m->packs[pairs[i-1].pack_int_id])
1697    {
1698      close_pack_fd(m->packs[pairs[i-1].pack_int_id]);
1699      close_pack_index(m->packs[pairs[i-1].pack_int_id]);
1700    }
1701
1702    nth_midxed_object_oid(&oid, m, pairs[i].pos);
1703
1704    if (!fill_midx_entry(r, &oid, &e, m)) {
1705      midx_report(_("failed to load pack entry for oid[%d] = %s"),
1706            pairs[i].pos, oid_to_hex(&oid));
1707      continue;
1708    }
1709
1710    if (open_pack_index(e.p)) {
1711      midx_report(_("failed to load pack-index for packfile %s"),
1712            e.p->pack_name);
1713      break;
1714    }
1715
1716    m_offset = e.offset;
1717    p_offset = find_pack_entry_one(oid.hash, e.p);
1718
1719    if (m_offset != p_offset)
1720      midx_report(_("incorrect object offset for oid[%d] = %s: %"PRIx64" != %"PRIx64),
1721            pairs[i].pos, oid_to_hex(&oid), m_offset, p_offset);
1722
1723    midx_display_sparse_progress(progress, i + 1);
1724  }
1725  stop_progress(&progress);
1726
1727cleanup:
1728  free(pairs);
1729  close_midx(m);
1730
1731  return verify_midx_error;
1732}

1833static int fill_included_packs_batch(struct repository *r,
1834             struct multi_pack_index *m,
1835             unsigned char *include_pack,
1836             size_t batch_size)
1837{
1838  uint32_t i, packs_to_repack;
1839  size_t total_size;
1840  struct repack_info *pack_info = xcalloc(m->num_packs, sizeof(struct repack_info));
1841  int pack_kept_objects = 0;
1842
1843  repo_config_get_bool(r, "repack.packkeptobjects", &pack_kept_objects);
1844
1845  for (i = 0; i < m->num_packs; i++) {
1846    pack_info[i].pack_int_id = i;
1847
1848    if (prepare_midx_pack(r, m, i))
1849      continue;
1850
1851    pack_info[i].mtime = m->packs[i]->mtime;
1852  }
1853
1854  for (i = 0; batch_size && i < m->num_objects; i++) {
1855    uint32_t pack_int_id = nth_midxed_pack_int_id(m, i);
1856    pack_info[pack_int_id].referenced_objects++;
1857  }
1858
1859  QSORT(pack_info, m->num_packs, compare_by_mtime);
1860
1861  total_size = 0;
1862  packs_to_repack = 0;
1863  for (i = 0; total_size < batch_size && i < m->num_packs; i++) {
1864    int pack_int_id = pack_info[i].pack_int_id;
1865    struct packed_git *p = m->packs[pack_int_id];
1866    size_t expected_size;
1867
1868    if (!p)
1869      continue;
1870    if (!pack_kept_objects && p->pack_keep)
1871      continue;
1872    if (open_pack_index(p) || !p->num_objects)
1873      continue;
1874
1875    expected_size = (size_t)(p->pack_size
1876           * pack_info[i].referenced_objects);
1877    expected_size /= p->num_objects;
1878
1879    if (expected_size >= batch_size)
1880      continue;
1881
1882    packs_to_repack++;
1883    total_size += expected_size;
1884    include_pack[pack_int_id] = 1;
1885  }
1886
1887  free(pack_info);
1888
1889  if (packs_to_repack < 2)
1890    return 1;

1892  return 0;

1895int midx_repack(struct repository *r, const char *object_dir, size_t batch_size, unsigned flags)
1896{
1897  int result = 0;
1898  uint32_t i;
1899  unsigned char *include_pack;
1900  struct child_process cmd = CHILD_PROCESS_INIT;
1901  FILE *cmd_in;
1902  struct strbuf base_name = STRBUF_INIT;
1903  struct multi_pack_index *m = lookup_multi_pack_index(r, object_dir);
1904
1905  /*
1906   * When updating the default for these configuration
1907   * variables in builtin/repack.c, these must be adjusted
1908   * to match.
1909   */
1910  int delta_base_offset = 1;
1911  int use_delta_islands = 0;
1912
1913  if (!m)
1914    return 0;
1915
1916  CALLOC_ARRAY(include_pack, m->num_packs);
1917
1918  if (batch_size) {
1919    if (fill_included_packs_batch(r, m, include_pack, batch_size))
1920      goto cleanup;
1921  } else if (fill_included_packs_all(r, m, include_pack))
1922    goto cleanup;
1923
1924  repo_config_get_bool(r, "repack.usedeltabaseoffset", &delta_base_offset);
1925  repo_config_get_bool(r, "repack.usedeltaislands", &use_delta_islands);
1926
1927  strvec_push(&cmd.args, "pack-objects");
1928
1929  strbuf_addstr(&base_name, object_dir);
1930  strbuf_addstr(&base_name, "/pack/pack");
1931  strvec_push(&cmd.args, base_name.buf);
1932
1933  if (delta_base_offset)
1934    strvec_push(&cmd.args, "--delta-base-offset");

1936    strvec_push(&cmd.args, "--delta-islands");
1937
1938  if (flags & MIDX_PROGRESS)
1939    strvec_push(&cmd.args, "--progress");
1940  else
1941    strvec_push(&cmd.args, "-q");
1942
1943  strbuf_release(&base_name);
1944
1945  cmd.git_cmd = 1;
1946  cmd.in = cmd.out = -1;
1947
1948  if (start_command(&cmd)) {
1949    error(_("could not start pack-objects"));
1950    result = 1;
1951    goto cleanup;
1952  }
1953
1954  cmd_in = xfdopen(cmd.in, "w");
1955
1956  for (i = 0; i < m->num_objects; i++) {
1957    struct object_id oid;
1958    uint32_t pack_int_id = nth_midxed_pack_int_id(m, i);
1959
1960    if (!include_pack[pack_int_id])
1961      continue;

1964    fprintf(cmd_in, "%s\n", oid_to_hex(&oid));
1965  }
1966  fclose(cmd_in);
1967
1968  if (finish_command(&cmd)) {
1969    error(_("could not finish pack-objects"));
1970    result = 1;
1971    goto cleanup;
1972  }
1973
1974  result = write_midx_internal(object_dir, NULL, NULL, NULL, NULL, flags);
1975
1976cleanup:
1977  free(include_pack);
1978  return result;
