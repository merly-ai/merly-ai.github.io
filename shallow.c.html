54int is_repository_shallow(struct repository *r)
55{
56  FILE *fp;
57  char buf[1024];
58  const char *path = r->parsed_objects->alternate_shallow_file;
59
60  if (r->parsed_objects->is_shallow >= 0)
61    return r->parsed_objects->is_shallow;
62
63  if (!path)
64    path = git_path_shallow(r);
65  /*
66   * fetch-pack sets '--shallow-file ""' as an indicator that no
67   * shallow file should be used. We could just open it and it
68   * will likely fail. But let's do an explicit check instead.
69   */
70  if (!*path || (fp = fopen(path, "r")) == NULL) {
71    stat_validity_clear(r->parsed_objects->shallow_stat);
72    r->parsed_objects->is_shallow = 0;
73    return r->parsed_objects->is_shallow;
74  }
75  stat_validity_update(r->parsed_objects->shallow_stat, fileno(fp));
76  r->parsed_objects->is_shallow = 1;
77
78  while (fgets(buf, sizeof(buf), fp)) {
79    struct object_id oid;
80    if (get_oid_hex(buf, &oid))
81      die("bad shallow line: %s", buf);
82    register_shallow(r, &oid);
83  }
84  fclose(fp);
85  return r->parsed_objects->is_shallow;
86}

124struct commit_list *get_shallow_commits(struct object_array *heads, int depth,
125    int shallow_flag, int not_shallow_flag)
126{
127  int i = 0, cur_depth = 0;
128  struct commit_list *result = NULL;
129  struct object_array stack = OBJECT_ARRAY_INIT;
130  struct commit *commit = NULL;
131  struct commit_graft *graft;
132  struct commit_depth depths;
133
134  init_commit_depth(&depths);
135  while (commit || i < heads->nr || stack.nr) {
136    struct commit_list *p;
137    if (!commit) {
138      if (i < heads->nr) {
139        int **depth_slot;
140        commit = (struct commit *)
141          deref_tag(the_repository,
142              heads->objects[i++].item,
143              NULL, 0);
144        if (!commit || commit->object.type != OBJ_COMMIT) {
145          commit = NULL;
146          continue;
147        }
148        depth_slot = commit_depth_at(&depths, commit);
149        if (!*depth_slot)
150          *depth_slot = xmalloc(sizeof(int));
151        **depth_slot = 0;
152        cur_depth = 0;
153      } else {
154        commit = (struct commit *)
155          object_array_pop(&stack);
156        cur_depth = **commit_depth_at(&depths, commit);
157      }
158    }
159    parse_commit_or_die(commit);
160    cur_depth++;
161    if ((depth != INFINITE_DEPTH && cur_depth >= depth) ||
162        (is_repository_shallow(the_repository) && !commit->parents &&
163         (graft = lookup_commit_graft(the_repository, &commit->object.oid)) != NULL &&
164         graft->nr_parent < 0)) {
165      commit_list_insert(commit, &result);
166      commit->object.flags |= shallow_flag;
167      commit = NULL;
168      continue;
169    }
170    commit->object.flags |= not_shallow_flag;
171    for (p = commit->parents, commit = NULL; p; p = p->next) {
172      int **depth_slot = commit_depth_at(&depths, p->item);
173      if (!*depth_slot) {
174        *depth_slot = xmalloc(sizeof(int));
175        **depth_slot = cur_depth;
176      } else {
177        if (cur_depth >= **depth_slot)
178          continue;
179        **depth_slot = cur_depth;
180      }
181      if (p->next)
182        add_object_array(&p->item->object,
183            NULL, &stack);
184      else {
185        commit = p->item;
186        cur_depth = **commit_depth_at(&depths, commit);
187      }
188    }
189  }
190  deep_clear_commit_depth(&depths, free_depth_in_slab);
191
192  return result;
193}

200/*
201 * Given rev-list arguments, run rev-list. All reachable commits
202 * except border ones are marked with not_shallow_flag. Border commits
203 * are marked with shallow_flag. The list of border/shallow commits
204 * are also returned.
205 */
206struct commit_list *get_shallow_commits_by_rev_list(int ac, const char **av,
207                int shallow_flag,
208                int not_shallow_flag)
209{
210  struct commit_list *result = NULL, *p;
211  struct commit_list *not_shallow_list = NULL;
212  struct rev_info revs;
213  int both_flags = shallow_flag | not_shallow_flag;
214
215  /*
216   * SHALLOW (excluded) and NOT_SHALLOW (included) should not be
217   * set at this point. But better be safe than sorry.
218   */
219  clear_object_flags(both_flags);
220
221  is_repository_shallow(the_repository); /* make sure shallows are read */
222
223  repo_init_revisions(the_repository, &revs, NULL);
224  save_commit_buffer = 0;
225  setup_revisions(ac, av, &revs, NULL);
226
227  if (prepare_revision_walk(&revs))
228    die("revision walk setup failed");
229  traverse_commit_list(&revs, show_commit, NULL, &not_shallow_list);
230
231  if (!not_shallow_list)
232    die("no commits selected for shallow requests");
233
234  /* Mark all reachable commits as NOT_SHALLOW */
235  for (p = not_shallow_list; p; p = p->next)
236    p->item->object.flags |= not_shallow_flag;
237
238  /*
239   * mark border commits SHALLOW + NOT_SHALLOW.
240   * We cannot clear NOT_SHALLOW right now. Imagine border
241   * commit A is processed first, then commit B, whose parent is
242   * A, later. If NOT_SHALLOW on A is cleared at step 1, B
243   * itself is considered border at step 2, which is incorrect.
244   */
245  for (p = not_shallow_list; p; p = p->next) {
246    struct commit *c = p->item;
247    struct commit_list *parent;
248
249    if (parse_commit(c))
250      die("unable to parse commit %s",
251          oid_to_hex(&c->object.oid));
252
253    for (parent = c->parents; parent; parent = parent->next)
254      if (!(parent->item->object.flags & not_shallow_flag)) {
255        c->object.flags |= shallow_flag;
256        commit_list_insert(c, &result);
257        break;
258      }
259  }
260  free_commit_list(not_shallow_list);
261
262  /*
263   * Now we can clean up NOT_SHALLOW on border commits. Having
264   * both flags set can confuse the caller.
265   */
266  for (p = result; p; p = p->next) {
267    struct object *o = &p->item->object;
268    if ((o->flags & both_flags) == both_flags)
269      o->flags &= ~not_shallow_flag;
270  }
271  release_revisions(&revs);
272  return result;
273}

296static int write_one_shallow(const struct commit_graft *graft, void *cb_data)
297{
298  struct write_shallow_data *data = cb_data;
299  const char *hex = oid_to_hex(&graft->oid);
300  if (graft->nr_parent != -1)
301    return 0;
302  if (data->flags & QUICK) {
303    if (!has_object_file(&graft->oid))
304      return 0;
305  } else if (data->flags & SEEN_ONLY) {
306    struct commit *c = lookup_commit(the_repository, &graft->oid);
307    if (!c || !(c->object.flags & SEEN)) {
308      if (data->flags & VERBOSE)
309        printf("Removing %s from .git/shallow\n",
310               oid_to_hex(&c->object.oid));
311      return 0;
312    }
313  }
314  data->count++;
315  if (data->use_pack_protocol)
316    packet_buf_write(data->out, "shallow %s", hex);
317  else {
318    strbuf_addstr(data->out, hex);
319    strbuf_addch(data->out, '\n');
320  }
321  return 0;
322}

324static int write_shallow_commits_1(struct strbuf *out, int use_pack_protocol,

413/*
414 * mark_reachable_objects() should have been run prior to this and all
415 * reachable commits marked as "SEEN", except when quick_prune is non-zero,
416 * in which case lines are excised from the shallow file if they refer to
417 * commits that do not exist (any longer).
418 */
419void prune_shallow(unsigned options)
420{
421  struct shallow_lock shallow_lock = SHALLOW_LOCK_INIT;
422  struct strbuf sb = STRBUF_INIT;
423  unsigned flags = SEEN_ONLY;
424  int fd;
425
426  if (options & PRUNE_QUICK)
427    flags |= QUICK;
428
429  if (options & PRUNE_SHOW_ONLY) {
430    flags |= VERBOSE;
431    write_shallow_commits_1(&sb, 0, NULL, flags);
432    strbuf_release(&sb);
433    return;
434  }
435  fd = hold_lock_file_for_update(&shallow_lock.lock,
436               git_path_shallow(the_repository),
437               LOCK_DIE_ON_ERROR);
438  check_shallow_file_for_update(the_repository);
439  if (write_shallow_commits_1(&sb, 0, NULL, flags)) {
440    if (write_in_full(fd, sb.buf, sb.len) < 0)
441      die_errno("failed to write to %s",
442          get_lock_file_path(&shallow_lock.lock));
443    commit_shallow_file(the_repository, &shallow_lock);
444  } else {
445    unlink(git_path_shallow(the_repository));
446    rollback_shallow_file(the_repository, &shallow_lock);
447  }
448  strbuf_release(&sb);
449}

453/*
454 * Step 1, split sender shallow commits into "ours" and "theirs"
455 * Step 2, clean "ours" based on .git/shallow
456 */
457void prepare_shallow_info(struct shallow_info *info, struct oid_array *sa)
458{
459  int i;
460  trace_printf_key(&trace_shallow, "shallow: prepare_shallow_info\n");
461  memset(info, 0, sizeof(*info));
462  info->shallow = sa;
463  if (!sa)
464    return;
465  ALLOC_ARRAY(info->ours, sa->nr);
466  ALLOC_ARRAY(info->theirs, sa->nr);
467  for (i = 0; i < sa->nr; i++) {
468    if (has_object_file(sa->oid + i)) {
469      struct commit_graft *graft;
470      graft = lookup_commit_graft(the_repository,
471                &sa->oid[i]);
472      if (graft && graft->nr_parent < 0)
473        continue;
474      info->ours[info->nr_ours++] = i;
475    } else
476      info->theirs[info->nr_theirs++] = i;
477  }
478}

514static uint32_t *paint_alloc(struct paint_info *info)
515{
516  unsigned nr = DIV_ROUND_UP(info->nr_bits, 32);
517  unsigned size = nr * sizeof(uint32_t);
518  void *p;
519  if (!info->pool_count || size > info->end - info->free) {
520    if (size > POOL_SIZE)
521      BUG("pool size too small for %d in paint_alloc()",
522          size);
523    info->pool_count++;
524    REALLOC_ARRAY(info->pools, info->pool_count);
525    info->free = xmalloc(POOL_SIZE);
526    info->pools[info->pool_count - 1] = info->free;
527    info->end = info->free + POOL_SIZE;
528  }
529  p = info->free;
530  info->free += size;
531  return p;
532}

534/*
535 * Given a commit SHA-1, walk down to parents until either SEEN,
536 * UNINTERESTING or BOTTOM is hit. Set the id-th bit in ref_bitmap for
537 * all walked commits.
538 */
539static void paint_down(struct paint_info *info, const struct object_id *oid,
540           unsigned int id)
541{
542  unsigned int i, nr;
543  struct commit_list *head = NULL;
544  int bitmap_nr = DIV_ROUND_UP(info->nr_bits, 32);
545  size_t bitmap_size = st_mult(sizeof(uint32_t), bitmap_nr);
546  struct commit *c = lookup_commit_reference_gently(the_repository, oid,
547                1);
548  uint32_t *tmp; /* to be freed before return */
549  uint32_t *bitmap;
550
551  if (!c)
552    return;
553
554  tmp = xmalloc(bitmap_size);
555  bitmap = paint_alloc(info);
556  memset(bitmap, 0, bitmap_size);
557  bitmap[id / 32] |= (1U << (id % 32));
558  commit_list_insert(c, &head);
559  while (head) {
560    struct commit_list *p;
561    struct commit *c = pop_commit(&head);
562    uint32_t **refs = ref_bitmap_at(&info->ref_bitmap, c);
563
564    /* XXX check "UNINTERESTING" from pack bitmaps if available */
565    if (c->object.flags & (SEEN | UNINTERESTING))
566      continue;
567    else
568      c->object.flags |= SEEN;
569
570    if (!*refs)
571      *refs = bitmap;
572    else {
573      memcpy(tmp, *refs, bitmap_size);
574      for (i = 0; i < bitmap_nr; i++)
575        tmp[i] |= bitmap[i];
576      if (memcmp(tmp, *refs, bitmap_size)) {
577        *refs = paint_alloc(info);
578        memcpy(*refs, tmp, bitmap_size);
579      }
580    }
581
582    if (c->object.flags & BOTTOM)
583      continue;
584
585    if (parse_commit(c))
586      die("unable to parse commit %s",
587          oid_to_hex(&c->object.oid));
588
589    for (p = c->parents; p; p = p->next) {
590      if (p->item->object.flags & SEEN)
591        continue;
592      commit_list_insert(p->item, &head);
593    }
594  }
595
596  nr = get_max_object_index();
597  for (i = 0; i < nr; i++) {
598    struct object *o = get_indexed_object(i);
599    if (o && o->type == OBJ_COMMIT)
600      o->flags &= ~SEEN;
601  }
602
603  free(tmp);
604}

606static int mark_uninteresting(const char *refname, const struct object_id *oid,
607            int flags, void *cb_data)
608{
609  struct commit *commit = lookup_commit_reference_gently(the_repository,
610                     oid, 1);
611  if (!commit)
612    return 0;
613  commit->object.flags |= UNINTERESTING;
614  mark_parents_uninteresting(NULL, commit);
615  return 0;
616}
617
618static void post_assign_shallow(struct shallow_info *info,
619        struct ref_bitmap *ref_bitmap,
620        int *ref_status);
621/*
622 * Step 6(+7), associate shallow commits with new refs
623 *
624 * info->ref must be initialized before calling this function.
625 *
626 * If used is not NULL, it's an array of info->shallow->nr
627 * bitmaps. The n-th bit set in the m-th bitmap if ref[n] needs the
628 * m-th shallow commit from info->shallow.
629 *
630 * If used is NULL, "ours" and "theirs" are updated. And if ref_status
631 * is not NULL it's an array of ref->nr ints. ref_status[i] is true if
632 * the ref needs some shallow commits from either info->ours or
633 * info->theirs.
634 */
635void assign_shallow_commits_to_refs(struct shallow_info *info,
636            uint32_t **used, int *ref_status)
637{
638  struct object_id *oid = info->shallow->oid;
639  struct oid_array *ref = info->ref;
640  unsigned int i, nr;
641  int *shallow, nr_shallow = 0;
642  struct paint_info pi;
643
644  trace_printf_key(&trace_shallow, "shallow: assign_shallow_commits_to_refs\n");
645  ALLOC_ARRAY(shallow, info->nr_ours + info->nr_theirs);
646  for (i = 0; i < info->nr_ours; i++)
647    shallow[nr_shallow++] = info->ours[i];
648  for (i = 0; i < info->nr_theirs; i++)
649    shallow[nr_shallow++] = info->theirs[i];
650
651  /*
652   * Prepare the commit graph to track what refs can reach what
653   * (new) shallow commits.
654   */
655  nr = get_max_object_index();
656  for (i = 0; i < nr; i++) {
657    struct object *o = get_indexed_object(i);
658    if (!o || o->type != OBJ_COMMIT)
659      continue;
660
661    o->flags &= ~(UNINTERESTING | BOTTOM | SEEN);
662  }
663
664  memset(&pi, 0, sizeof(pi));
665  init_ref_bitmap(&pi.ref_bitmap);
666  pi.nr_bits = ref->nr;
667
668  /*
669   * "--not --all" to cut short the traversal if new refs
670   * connect to old refs. If not (e.g. force ref updates) it'll
671   * have to go down to the current shallow commits.
672   */
673  head_ref(mark_uninteresting, NULL);
674  for_each_ref(mark_uninteresting, NULL);
675
676  /* Mark potential bottoms so we won't go out of bound */
677  for (i = 0; i < nr_shallow; i++) {
678    struct commit *c = lookup_commit(the_repository,
679             &oid[shallow[i]]);
680    c->object.flags |= BOTTOM;
681  }
682
683  for (i = 0; i < ref->nr; i++)
684    paint_down(&pi, ref->oid + i, i);
685
686  if (used) {
687    int bitmap_size = DIV_ROUND_UP(pi.nr_bits, 32) * sizeof(uint32_t);
688    memset(used, 0, sizeof(*used) * info->shallow->nr);
689    for (i = 0; i < nr_shallow; i++) {
690      const struct commit *c = lookup_commit(the_repository,
691                     &oid[shallow[i]]);
692      uint32_t **map = ref_bitmap_at(&pi.ref_bitmap, c);
693      if (*map)
694        used[shallow[i]] = xmemdupz(*map, bitmap_size);
695    }
696    /*
697     * unreachable shallow commits are not removed from
698     * "ours" and "theirs". The user is supposed to run
699     * step 7 on every ref separately and not trust "ours"
700     * and "theirs" any more.
701     */
702  } else
703    post_assign_shallow(info, &pi.ref_bitmap, ref_status);
704
705  clear_ref_bitmap(&pi.ref_bitmap);
706  for (i = 0; i < pi.pool_count; i++)
707    free(pi.pools[i]);
708  free(pi.pools);
709  free(shallow);
710}

712struct commit_array {
713  struct commit **commits;
714  int nr, alloc;
715};
716
717static int add_ref(const char *refname, const struct object_id *oid,
718       int flags, void *cb_data)
719{
720  struct commit_array *ca = cb_data;
721  ALLOC_GROW(ca->commits, ca->nr + 1, ca->alloc);
722  ca->commits[ca->nr] = lookup_commit_reference_gently(the_repository,
723                   oid, 1);
724  if (ca->commits[ca->nr])
725    ca->nr++;
726  return 0;
727}
728
729static void update_refstatus(int *ref_status, int nr, uint32_t *bitmap)
730{
731  unsigned int i;
732  if (!ref_status)
733    return;
734  for (i = 0; i < nr; i++)
735    if (bitmap[i / 32] & (1U << (i % 32)))
736      ref_status[i]++;
737}
738
739/*
740 * Step 7, reachability test on "ours" at commit level
741 */
742static void post_assign_shallow(struct shallow_info *info,
743        struct ref_bitmap *ref_bitmap,
744        int *ref_status)
745{
746  struct object_id *oid = info->shallow->oid;
747  struct commit *c;
748  uint32_t **bitmap;
749  int dst, i, j;
750  int bitmap_nr = DIV_ROUND_UP(info->ref->nr, 32);
751  struct commit_array ca;
752
753  trace_printf_key(&trace_shallow, "shallow: post_assign_shallow\n");
754  if (ref_status)
755    memset(ref_status, 0, sizeof(*ref_status) * info->ref->nr);
756
757  /* Remove unreachable shallow commits from "theirs" */
758  for (i = dst = 0; i < info->nr_theirs; i++) {
759    if (i != dst)
760      info->theirs[dst] = info->theirs[i];
761    c = lookup_commit(the_repository, &oid[info->theirs[i]]);
762    bitmap = ref_bitmap_at(ref_bitmap, c);
763    if (!*bitmap)
764      continue;
765    for (j = 0; j < bitmap_nr; j++)
766      if (bitmap[0][j]) {
767        update_refstatus(ref_status, info->ref->nr, *bitmap);
768        dst++;
769        break;
770      }
771  }
772  info->nr_theirs = dst;
773
774  memset(&ca, 0, sizeof(ca));
775  head_ref(add_ref, &ca);
776  for_each_ref(add_ref, &ca);
777
778  /* Remove unreachable shallow commits from "ours" */
779  for (i = dst = 0; i < info->nr_ours; i++) {
780    if (i != dst)
781      info->ours[dst] = info->ours[i];
782    c = lookup_commit(the_repository, &oid[info->ours[i]]);
783    bitmap = ref_bitmap_at(ref_bitmap, c);
784    if (!*bitmap)
785      continue;
786    for (j = 0; j < bitmap_nr; j++)
787      if (bitmap[0][j] &&
788          /* Step 7, reachability test at commit level */
789          !in_merge_bases_many(c, ca.nr, ca.commits)) {
790        update_refstatus(ref_status, info->ref->nr, *bitmap);
791        dst++;
792        break;
793      }
794  }
795  info->nr_ours = dst;
796
797  free(ca.commits);
798}
799
800/* (Delayed) step 7, reachability test at commit level */
801int delayed_reachability_test(struct shallow_info *si, int c)
802{
803  if (si->need_reachability_test[c]) {
804    struct commit *commit = lookup_commit(the_repository,
805                  &si->shallow->oid[c]);
806
807    if (!si->commits) {
808      struct commit_array ca;
809
810      memset(&ca, 0, sizeof(ca));
811      head_ref(add_ref, &ca);
812      for_each_ref(add_ref, &ca);
813      si->commits = ca.commits;
814      si->nr_commits = ca.nr;
815    }
816
817    si->reachable[c] = in_merge_bases_many(commit,
818                   si->nr_commits,
819                   si->commits);
820    si->need_reachability_test[c] = 0;
821  }
822  return si->reachable[c];
823}
