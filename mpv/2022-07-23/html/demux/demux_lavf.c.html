278// At least mp4 has name="mov,mp4,m4a,3gp,3g2,mj2", so we split the name
279// on "," in general.
280static bool matches_avinputformat_name(struct lavf_priv *priv,
281                                       const char *name)
282{
283    const char *avifname = priv->avif->name;
284    while (1) {
285        const char *next = strchr(avifname, ',');
286        if (!next)
287            return !strcmp(avifname, name);
288        int len = next - avifname;
289        if (len == strlen(name) && !memcmp(avifname, name, len))
290            return true;
291        avifname = next + 1;
292    }
293}

310static int64_t mp_seek(void *opaque, int64_t pos, int whence)
311{
312    struct demuxer *demuxer = opaque;
313    lavf_priv_t *priv = demuxer->priv;
314    struct stream *stream = priv->stream;
315    if (!stream)
316        return -1;
317
318    MP_TRACE(demuxer, "mp_seek(%p, %"PRId64", %s)\n", stream, pos,
319             whence == SEEK_END ? "end" :
320             whence == SEEK_CUR ? "cur" :
321             whence == SEEK_SET ? "set" : "size");
322    if (whence == SEEK_END || whence == AVSEEK_SIZE) {
323        int64_t end = stream_get_size(stream);
324        if (end < 0)
325            return -1;
326        if (whence == AVSEEK_SIZE)
327            return end;
328        pos += end;
329    } else if (whence == SEEK_CUR) {
330        pos += stream_tell(stream);
331    } else if (whence != SEEK_SET) {
332        return -1;
333    }
334
335    if (pos < 0)
336        return -1;
337
338    int64_t current_pos = stream_tell(stream);
339    if (stream_seek(stream, pos) == 0) {
340        stream_seek(stream, current_pos);
341        return -1;
342    }
343
344    return pos;
345}

347static int64_t mp_read_seek(void *opaque, int stream_idx, int64_t ts, int flags)
348{

354        .stream_index = stream_idx,
355        .timestamp = ts,
356        .flags = flags,
357    };

363    return AVERROR(ENOSYS);

375static void convert_charset(struct demuxer *demuxer)
376{
377    lavf_priv_t *priv = demuxer->priv;
378    char *cp = priv->opts->sub_cp;
379    if (!cp || !cp[0] || mp_charset_is_utf8(cp))
380        return;
381    bstr data = stream_read_complete(priv->stream, NULL, 128 * 1024 * 1024);
382    if (!data.start) {
383        MP_WARN(demuxer, "File too big (or error reading) - skip charset probing.\n");
384        return;
385    }
386    void *alloc = data.start;
387    cp = (char *)mp_charset_guess(priv, demuxer->log, data, cp, 0);
388    if (cp && !mp_charset_is_utf8(cp))
389        MP_INFO(demuxer, "Using subtitle charset: %s\n", cp);
390    // libavformat transparently converts UTF-16 to UTF-8
391    if (!mp_charset_is_utf16(cp) && !mp_charset_is_utf8(cp)) {
392        bstr conv = mp_iconv_to_utf8(demuxer->log, data, cp, MP_ICONV_VERBOSE);
393        if (conv.start && conv.start != data.start)
394            talloc_steal(alloc, conv.start);
395        if (conv.start)
396            data = conv;
397    }
398    if (data.start) {
399        priv->stream = stream_memory_open(demuxer->global, data.start, data.len);
400        priv->own_stream = true;
401    }
402    talloc_free(alloc);
403}

418static int lavf_check_file(demuxer_t *demuxer, enum demux_check check)
419{
420    lavf_priv_t *priv = demuxer->priv;
421    struct demux_lavf_opts *lavfdopts = priv->opts;
422    struct stream *s = priv->stream;
423
424    priv->filename = remove_prefix(s->url, prefixes);
425
426    char *avdevice_format = NULL;
427    if (s->info && strcmp(s->info->name, "avdevice") == 0) {
428        // always require filename in the form "format:filename"
429        char *sep = strchr(priv->filename, ':');
430        if (!sep) {
431            MP_FATAL(demuxer, "Must specify filename in 'format:filename' form\n");
432            return -1;
433        }
434        avdevice_format = talloc_strndup(priv, priv->filename,
435                                         sep - priv->filename);
436        priv->filename = sep + 1;
437    }
438
439    char *mime_type = s->mime_type;
440    if (!lavfdopts->allow_mimetype || !mime_type)
441        mime_type = "";
442
443    const AVInputFormat *forced_format = NULL;
444    const char *format = lavfdopts->format;
445    if (!format)
446        format = s->lavf_type;
447    if (!format)
448        format = avdevice_format;
449    if (format) {
450        if (strcmp(format, "help") == 0) {
451            list_formats(demuxer);
452            return -1;
453        }
454        forced_format = av_find_input_format(format);
455        if (!forced_format) {
456            MP_FATAL(demuxer, "Unknown lavf format %s\n", format);
457            return -1;
458        }
459    }
460
461    AVProbeData avpd = {
462        // Disable file-extension matching with normal checks
463        .filename = check <= DEMUX_CHECK_REQUEST ? priv->filename : "",
464        .buf_size = 0,
465        .buf = av_mallocz(PROBE_BUF_SIZE + AV_INPUT_BUFFER_PADDING_SIZE),
466    };
467    if (!avpd.buf)
468        return -1;
469
470    bool final_probe = false;
471    do {
472        int score = 0;
473
474        if (forced_format) {
475            priv->avif = forced_format;
476            score = AVPROBE_SCORE_MAX;
477        } else {
478            int nsize = av_clip(avpd.buf_size * 2, INITIAL_PROBE_SIZE,
479                                PROBE_BUF_SIZE);
480            nsize = stream_read_peek(s, avpd.buf, nsize);
481            if (nsize <= avpd.buf_size)
482                final_probe = true;
483            avpd.buf_size = nsize;
484
485            priv->avif = av_probe_input_format2(&avpd, avpd.buf_size > 0, &score);
486        }
487
488        if (priv->avif) {
489            MP_VERBOSE(demuxer, "Found '%s' at score=%d size=%d%s.\n",
490                       priv->avif->name, score, avpd.buf_size,
491                       forced_format ? " (forced)" : "");
492
493            for (int n = 0; lavfdopts->hacks && format_hacks[n].ff_name; n++) {
494                const struct format_hack *entry = &format_hacks[n];
495                if (!matches_avinputformat_name(priv, entry->ff_name))
496                    continue;
497                if (entry->mime_type && strcasecmp(entry->mime_type, mime_type) != 0)
498                    continue;
499                priv->format_hack = *entry;
500                break;
501            }
502
503            if (score >= lavfdopts->probescore)
504                break;
505
506            if (priv->format_hack.probescore &&
507                score >= priv->format_hack.probescore &&
508                (!priv->format_hack.max_probe || final_probe))
509                break;
510        }
511
512        priv->avif = NULL;
513        priv->format_hack = (struct format_hack){0};
514    } while (!final_probe);
515
516    av_free(avpd.buf);
517
518    if (priv->avif && !forced_format && priv->format_hack.ignore) {
519        MP_VERBOSE(demuxer, "Format blacklisted.\n");
520        priv->avif = NULL;
521    }
522
523    if (!priv->avif) {
524        MP_VERBOSE(demuxer, "No format found, try lowering probescore or forcing the format.\n");
525        return -1;
526    }
527
528    if (lavfdopts->hacks)
529        priv->avif_flags = priv->avif->flags | priv->format_hack.if_flags;
530
531    priv->linearize_ts = lavfdopts->linearize_ts;
532    if (
 Recommendations (Experimental)  R1: (priv->linearize_ts < 0 && !format_hack)
R2: (priv < 0 && !linearize_ts)
R3: ((priv < 0) && !linearize_ts)
R4: ((priv < {const}) && (!linearize_ts))
R5: (priv < 0 && !linearize_ts->format_hack)
 Score  Issue:  1 unfamiliar pattern(s) detected 
Cost: 14.50, Total Score: 1014.50
Class: complex_2
priv->linearize_ts < 0 && !priv->format_hack.linearize_audio_ts)
533        priv->linearize_ts = 0;
534
535    demuxer->filetype = priv->avif->name;
536
537    if (priv->format_hack.detect_charset)
538        convert_charset(demuxer);
539
540    return 0;
541}

592static void export_replaygain(demuxer_t *demuxer, struct sh_stream *sh,
593                              AVStream *st)
594{
595    for (int i = 0; i < st->nb_side_data; i++) {
596        AVReplayGain *av_rgain;
597        struct replaygain_data *rgain;
598        AVPacketSideData *src_sd = &st->side_data[i];
599
600        if (src_sd->type != AV_PKT_DATA_REPLAYGAIN)
601            continue;
602
603        av_rgain = (AVReplayGain*)src_sd->data;
604        rgain    = talloc_ptrtype(demuxer, rgain);
605        rgain->track_gain = rgain->album_gain = 0;
606        rgain->track_peak = rgain->album_peak = 1;
607
608        // Set values in *rgain, using track gain as a fallback for album gain
609        // if the latter is not present. This behavior matches that in
610        // demux/demux.c's decode_rgain; if you change this, please make
611        // equivalent changes there too.
612        if (av_rgain->track_gain != INT32_MIN && av_rgain->track_peak != 0.0) {
613            // Track gain is defined.
614            rgain->track_gain = av_rgain->track_gain / 100000.0f;
615            rgain->track_peak = av_rgain->track_peak / 100000.0f;
616
617            if (av_rgain->album_gain != INT32_MIN &&
618                av_rgain->album_peak != 0.0)
619            {
620                // Album gain is also defined.
621                rgain->album_gain = av_rgain->album_gain / 100000.0f;
622                rgain->album_peak = av_rgain->album_peak / 100000.0f;
623            } else {
624                // Album gain is undefined; fall back to track gain.
625                rgain->album_gain = rgain->track_gain;
626                rgain->album_peak = rgain->track_peak;
627            }
628        }
629
630        // This must be run only before the stream was added, otherwise there
631        // will be race conditions with accesses from the user thread.
632        assert(!sh->ds);
633        sh->codec->replaygain_data = rgain;
634    }
635}

650static void handle_new_stream(demuxer_t *demuxer, int i)
651{
652    lavf_priv_t *priv = demuxer->priv;
653    AVFormatContext *avfc = priv->avfc;
654    AVStream *st = avfc->streams[i];
655    struct sh_stream *sh = NULL;
656    AVCodecParameters *codec = st->codecpar;
657    int lavc_delay = codec->initial_padding;
658
659    switch (codec->codec_type) {
660    case AVMEDIA_TYPE_AUDIO: {
661        sh = demux_alloc_sh_stream(STREAM_AUDIO);
662
663#if !HAVE_AV_CHANNEL_LAYOUT
664        // probably unneeded
665        mp_chmap_set_unknown(&sh->codec->channels, codec->channels);
666        if (codec->channel_layout)
667            mp_chmap_from_lavc(&sh->codec->channels, codec->channel_layout);
668#else
669        if (!mp_chmap_from_av_layout(&sh->codec->channels, &codec->ch_layout)) {
670            char layout[128] = {0};
671            MP_WARN(demuxer,
672                    "Failed to convert channel layout %s to mpv one!\n",
673                    av_channel_layout_describe(&codec->ch_layout,
674                                               layout, 128) < 0 ?
675                    "undefined" : layout);
676        }
677#endif
678
679        sh->codec->samplerate = codec->sample_rate;
680        sh->codec->bitrate = codec->bit_rate;
681
682        double delay = 0;
683        if (codec->sample_rate > 0)
684            delay = lavc_delay / (double)codec->sample_rate;
685        priv->seek_delay = MPMAX(priv->seek_delay, delay);
686
687        export_replaygain(demuxer, sh, st);
688
689        sh->seek_preroll = delay;
690
691        break;
692    }
693    case AVMEDIA_TYPE_VIDEO: {
694        sh = demux_alloc_sh_stream(STREAM_VIDEO);
695
696        if ((st->disposition & AV_DISPOSITION_ATTACHED_PIC) &&
697            !(st->disposition & AV_DISPOSITION_TIMED_THUMBNAILS))
698        {
699            sh->attached_picture =
700                new_demux_packet_from_avpacket(&st->attached_pic);
701            if (sh->attached_picture) {
702                sh->attached_picture->pts = 0;
703                talloc_steal(sh, sh->attached_picture);
704                sh->attached_picture->keyframe = true;
705            }
706        }
707
708        if (!sh->attached_picture) {
709            // A real video stream probably means it's a packet based format.
710            priv->pcm_seek_hack_disabled = true;
711            priv->pcm_seek_hack = NULL;
712            // Also, we don't want to do this shit for ogv videos.
713            if (priv->linearize_ts < 0)
714                priv->linearize_ts = 0;
715        }
716
717        sh->codec->disp_w = codec->width;
718        sh->codec->disp_h = codec->height;
719        if (st->avg_frame_rate.num)
720            sh->codec->fps = av_q2d(st->avg_frame_rate);
721        if (st->nb_frames <= 1 && (
722                sh->attached_picture ||
723                bstr_endswith0(bstr0(priv->avif->name), "_pipe") ||
724                strcmp(priv->avif->name, "alias_pix") == 0 ||
725                strcmp(priv->avif->name, "gif") == 0 ||
726                strcmp(priv->avif->name, "image2pipe") == 0
727            )) {
728            MP_VERBOSE(demuxer, "Assuming this is an image format.\n");
729            sh->image = true;
730            sh->codec->fps = priv->mf_fps;
731        }
732        sh->codec->par_w = st->sample_aspect_ratio.num;
733        sh->codec->par_h = st->sample_aspect_ratio.den;
734
735        uint8_t *sd = av_stream_get_side_data(st, AV_PKT_DATA_DISPLAYMATRIX, NULL);
736        if (sd) {
737            double r = av_display_rotation_get((uint32_t *)sd);
738            if (!isnan(r))
739                sh->codec->rotate = (((int)(-r) % 360) + 360) % 360;
740        }
741
742#if LIBAVUTIL_VERSION_INT >= AV_VERSION_INT(56, 43, 100)
743        if ((sd = av_stream_get_side_data(st, AV_PKT_DATA_DOVI_CONF, NULL))) {
744            const AVDOVIDecoderConfigurationRecord *cfg = (void *) sd;
745            MP_VERBOSE(demuxer, "Found Dolby Vision config record: profile "
746                       "%d level %d\n", cfg->dv_profile, cfg->dv_level);
747            av_format_inject_global_side_data(avfc);
748        }
749#endif
750
751        // This also applies to vfw-muxed mkv, but we can't detect these easily.
752        sh->codec->avi_dts = matches_avinputformat_name(priv, "avi");
753
754        break;
755    }
756    case AVMEDIA_TYPE_SUBTITLE: {
757        sh = demux_alloc_sh_stream(STREAM_SUB);
758
759        if (codec->extradata_size) {
760            sh->codec->extradata = talloc_size(sh, codec->extradata_size);
761            memcpy(sh->codec->extradata, codec->extradata, codec->extradata_size);
762            sh->codec->extradata_size = codec->extradata_size;
763        }
764
765        if (matches_avinputformat_name(priv, "microdvd")) {
766            AVRational r;
767            if (av_opt_get_q(avfc, "subfps", AV_OPT_SEARCH_CHILDREN, &r) >= 0) {
768                // File headers don't have a FPS set.
769                if (r.num < 1 || r.den < 1)
770                    sh->codec->frame_based = 23.976; // default timebase
771            }
772        }
773        break;
774    }
775    case AVMEDIA_TYPE_ATTACHMENT: {
776        AVDictionaryEntry *ftag = av_dict_get(st->metadata, "filename", NULL, 0);
777        char *filename = ftag ? ftag->value : NULL;
778        AVDictionaryEntry *mt = av_dict_get(st->metadata, "mimetype", NULL, 0);
779        char *mimetype = mt ? mt->value : NULL;
780        if (mimetype) {
781            demuxer_add_attachment(demuxer, filename, mimetype,
782                                   codec->extradata, codec->extradata_size);
783        }
784        break;
785    }
786    default: ;
787    }
788
789    struct stream_info *info = talloc_zero(priv, struct stream_info);
790    *info = (struct stream_info){
791        .sh = sh,
792        .last_key_pts = MP_NOPTS_VALUE,
793        .highest_pts = MP_NOPTS_VALUE,
794    };
795    assert(priv->num_streams == i); // directly mapped
796    MP_TARRAY_APPEND(priv, priv->streams, priv->num_streams, info);
797
798    if (sh) {
799        sh->ff_index = st->index;
800        sh->codec->codec = mp_codec_from_av_codec_id(codec->codec_id);
801        sh->codec->codec_tag = codec->codec_tag;
802        sh->codec->lav_codecpar = avcodec_parameters_alloc();
803        if (sh->codec->lav_codecpar)
804            avcodec_parameters_copy(sh->codec->lav_codecpar, codec);
805        sh->codec->native_tb_num = st->time_base.num;
806        sh->codec->native_tb_den = st->time_base.den;
807
808        if (st->disposition & AV_DISPOSITION_DEFAULT)
809            sh->default_track = true;
810        if (st->disposition & AV_DISPOSITION_FORCED)
811            sh->forced_track = true;
812        if (st->disposition & AV_DISPOSITION_DEPENDENT)
813            sh->dependent_track = true;
814        if (st->disposition & AV_DISPOSITION_VISUAL_IMPAIRED)
815            sh->visual_impaired_track = true;
816        if (st->disposition & AV_DISPOSITION_HEARING_IMPAIRED)
817            sh->hearing_impaired_track = true;
818        if (st->disposition & AV_DISPOSITION_STILL_IMAGE)
819            sh->still_image = true;
820        if (priv->format_hack.use_stream_ids)
821            sh->demuxer_id = st->id;
822        AVDictionaryEntry *title = av_dict_get(st->metadata, "title", NULL, 0);
823        if (title && title->value)
824            sh->title = talloc_strdup(sh, title->value);
825        if (!sh->title && st->disposition & AV_DISPOSITION_VISUAL_IMPAIRED)
826            sh->title = talloc_asprintf(sh, "visual impaired");
827        if (!sh->title && st->disposition & AV_DISPOSITION_HEARING_IMPAIRED)
828            sh->title = talloc_asprintf(sh, "hearing impaired");
829        AVDictionaryEntry *lang = av_dict_get(st->metadata, "language", NULL, 0);
830        if (lang && lang->value && strcmp(lang->value, "und") != 0)
831            sh->lang = talloc_strdup(sh, lang->value);
832        sh->hls_bitrate = dict_get_decimal(st->metadata, "variant_bitrate", 0);
833        sh->missing_timestamps = !!(priv->avif_flags & AVFMT_NOTIMESTAMPS);
834        mp_tags_copy_from_av_dictionary(sh->tags, st->metadata);
835        demux_add_sh_stream(demuxer, sh);
836
837        // Unfortunately, there is no better way to detect PCM codecs, other
838        // than listing them all manually. (Or other "frameless" codecs. Or
839        // rather, codecs with frames so small libavformat will put multiple of
840        // them into a single packet, but not preserve these artificial packet
841        // boundaries on seeking.)
842        if (sh->codec->codec && strncmp(sh->codec->codec, "pcm_", 4) == 0 &&
843            codec->block_align && !priv->pcm_seek_hack_disabled &&
844            priv->opts->hacks && !priv->format_hack.no_pcm_seek &&
845            st->time_base.num == 1 && st->time_base.den == codec->sample_rate)
846        {
847            if (priv->pcm_seek_hack) {
848                // More than 1 audio stream => usually doesn't apply.
849                priv->pcm_seek_hack_disabled = true;
850                priv->pcm_seek_hack = NULL;
851            } else {
852                priv->pcm_seek_hack = st;
853            }
854        }
855    }
856
857    select_tracks(demuxer, i);
858}

1323            int64_t align = seek_pts_av % samples;
1324            seek_pts_av -= align;
1325            seek_stream = priv->pcm_seek_hack->index;
1326        }
1327    }
1328
1329    int r = av_seek_frame(priv->avfc, seek_stream, seek_pts_av, avsflags);
1330    if (r < 0 && (avsflags & AVSEEK_FLAG_BACKWARD)) {
1331        // When seeking before the beginning of the file, and seeking fails,
1332        // try again without the backwards flag to make it seek to the
1333        // beginning.
1334        avsflags &= ~AVSEEK_FLAG_BACKWARD;
1335        r = av_seek_frame(priv->avfc, seek_stream, seek_pts_av, avsflags);
1336    }
1337
1338    if (r < 0) {
1339        char buf[180];
1340        av_strerror(r, buf, sizeof(buf));
1341        MP_VERBOSE(demuxer, "Seek failed (%s)\n", buf);
1342    }
1343
1344    update_read_stats(demuxer);
1345}

1348{
1349    select_tracks(demuxer, 0);
1350}

1352static void demux_close_lavf(demuxer_t *demuxer)
1353{
1354    lavf_priv_t *priv = demuxer->priv;
1355    if (priv) {
1356        // This will be a dangling pointer; but see below.
1357        AVIOContext *leaking = priv->avfc ? priv->avfc->pb : NULL;
1358        avformat_close_input(&priv->avfc);
1359        // The ffmpeg garbage breaks its own API yet again: hls.c will call
1360        // io_open on the main playlist, but never calls io_close. This happens
1361        // to work out for us (since we don't really use custom I/O), but it's
1362        // still weird. Compensate.
1363        if (priv->num_nested == 1 && priv->nested[0].id == leaking)
1364            priv->num_nested = 0;
1365        if (priv->num_nested) {
1366            MP_WARN(demuxer, "Leaking %d nested connections (FFmpeg bug).\n",
1367                    priv->num_nested);
1368        }
1369        if (priv->pb)
1370            av_freep(&priv->pb->buffer);
1371        av_freep(&priv->pb);
1372        for (int n = 0; n < priv->num_streams; n++) {
1373            struct stream_info *info = priv->streams[n];
1374            if (info->sh)
1375                avcodec_parameters_free(&info->sh->codec->lav_codecpar);
1376        }
1377        if (priv->own_stream)
1378            free_stream(priv->stream);
1379        if (priv->av_opts)
1380            av_dict_free(&priv->av_opts);
1381        talloc_free(priv);
1382        demuxer->priv = NULL;
1383    }
1384}
