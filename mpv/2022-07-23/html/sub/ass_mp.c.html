43// res_y should be track->PlayResY
44// It determines scaling of font sizes and more.
45void mp_ass_set_style(ASS_Style *style, double res_y,
46                      const struct osd_style_opts *opts)
47{
48    if (!style)
49        return;
50
51    if (opts->font) {
52        if (!style->FontName || strcmp(style->FontName, opts->font) != 0) {
53            free(style->FontName);
54            style->FontName = strdup(opts->font);
55        }
56    }
57
58    // libass_font_size = FontSize * (window_height / res_y)
59    // scale translates parameters from PlayResY=720 to res_y
60    double scale = res_y / 720.0;
61
62    style->FontSize = opts->font_size * scale;
63    style->PrimaryColour = MP_ASS_COLOR(opts->color);
64    style->SecondaryColour = style->PrimaryColour;
65    style->OutlineColour = MP_ASS_COLOR(opts->border_color);
66    if (opts->back_color.a) {
67        style->BackColour = MP_ASS_COLOR(opts->back_color);
68        style->BorderStyle = 4; // opaque box
69    } else {
70        style->BackColour = MP_ASS_COLOR(opts->shadow_color);
71        style->BorderStyle = 1; // outline
72    }
73    style->Outline = opts->border_size * scale;
74    style->Shadow = opts->shadow_offset * scale;
75    style->Spacing = opts->spacing * scale;
76    style->MarginL = opts->margin_x * scale;
77    style->MarginR = style->MarginL;
78    style->MarginV = opts->margin_y * scale;
79    style->ScaleX = 1.;
80    style->ScaleY = 1.;
81    style->Alignment = 1 + (opts->align_x + 1) + (opts->align_y + 2) % 3 * 4;
82#ifdef ASS_JUSTIFY_LEFT
83    style->Justify = opts->justify;
84#endif
85    style->Blur = opts->blur;
86    style->Bold = opts->bold;
87    style->Italic = opts->italic;
88}

90void mp_ass_configure_fonts(ASS_Renderer *priv, struct osd_style_opts *opts,
91                            struct mpv_global *global, struct mp_log *log)
92{
93    void *tmp = talloc_new(NULL);
94    char *default_font = mp_find_config_file(tmp, global, "subfont.ttf");
95    char *config       = mp_find_config_file(tmp, global, "fonts.conf");
96
97    if (default_font && !mp_path_exists(default_font))
98        default_font = NULL;
99
100    int font_provider = ASS_FONTPROVIDER_AUTODETECT;
101    if (opts->font_provider == 1)
102        font_provider = ASS_FONTPROVIDER_NONE;
103    if (opts->font_provider == 2)
104        font_provider = ASS_FONTPROVIDER_FONTCONFIG;
105
106    mp_verbose(log, "Setting up fonts...\n");
107    ass_set_fonts(priv, default_font, opts->font, font_provider, config, 1);
108    mp_verbose(log, "Done.\n");
109
110    talloc_free(tmp);
111}

151{
152    int n = 0;
153    for (; n < track->n_events; n++) {
154        if ((track->events[n].Start + track->events[n].Duration) >= ts)
155            break;
156        ass_free_event(track, n);
157        track->n_events--;
158    }
159    for (int i = 0; n > 0 && i < track->n_events; i++) {
160        track->events[i] = track->events[i+n];
161    }
162}

164static void draw_ass_rgba(unsigned char *src, int src_w, int src_h,
165                          int src_stride, unsigned char *dst, size_t dst_stride,
166                          int dst_x, int dst_y, uint32_t color)
167{
168    const unsigned int r = (color >> 24) & 0xff;
169    const unsigned int g = (color >> 16) & 0xff;
170    const unsigned int b = (color >>  8) & 0xff;
171    const unsigned int a = 0xff - (color & 0xff);
172
173    dst += dst_y * dst_stride + dst_x * 4;
174
175    for (int y = 0; y < src_h; y++, dst += dst_stride, src += src_stride) {
176        uint32_t *dstrow = (uint32_t *) dst;
177        for (int x = 0; x < src_w; x++) {
178            const unsigned int v = src[x];
179            int rr = (r * a * v);
180            int gg = (g * a * v);
181            int bb = (b * a * v);
182            int aa =      a * v;
183            uint32_t dstpix = dstrow[x];
184            unsigned int dstb =  dstpix        & 0xFF;
185            unsigned int dstg = (dstpix >>  8) & 0xFF;
186            unsigned int dstr = (dstpix >> 16) & 0xFF;
187            unsigned int dsta = (dstpix >> 24) & 0xFF;
188            dstb = (bb       + dstb * (255 * 255 - aa)) / (255 * 255);
189            dstg = (gg       + dstg * (255 * 255 - aa)) / (255 * 255);
190            dstr = (rr       + dstr * (255 * 255 - aa)) / (255 * 255);
191            dsta = (aa * 255 + dsta * (255 * 255 - aa)) / (255 * 255);
192            dstrow[x] = dstb | (dstg << 8) | (dstr << 16) | (dsta << 24);
193        }
194    }
195}

261static bool pack_libass(struct mp_ass_packer *p, struct sub_bitmaps *res)
262{
263    if (!pack(p, res, IMGFMT_Y8))
264        return false;
265
266    for (int n = 0; n < res->num_parts; n++) {
267        struct sub_bitmap *b = &res->parts[n];
268
269        int stride = res->packed->stride[0];
270        void *pdata =
271            (uint8_t *)res->packed->planes[0] + b->src_y * stride + b->src_x;
272        memcpy_pic(pdata, b->bitmap, b->w, b->h, stride, b->stride);
273
274        b->bitmap = pdata;
275        b->stride = stride;
276    }
277
278    return true;
279}

281static bool pack_rgba(struct mp_ass_packer *p, struct sub_bitmaps *res)
282{
283    struct mp_rect bb_list[MP_SUB_BB_LIST_MAX];
284    int num_bb = mp_get_sub_bb_list(res, bb_list, MP_SUB_BB_LIST_MAX);
285
286    struct sub_bitmaps imgs = {
287        .change_id = res->change_id,
288        .format = SUBBITMAP_BGRA,
289        .parts = p->rgba_imgs,
290        .num_parts = num_bb,
291    };
292
293    for (int n = 0; n < imgs.num_parts; n++) {
294        imgs.parts[n].w = bb_list[n].x1 - bb_list[n].x0;
295        imgs.parts[n].h = bb_list[n].y1 - bb_list[n].y0;
296    }
297
298    if (!pack(p, &imgs, IMGFMT_BGRA))
299        return false;
300
301    for (int n = 0; n < num_bb; n++) {
302        struct mp_rect bb = bb_list[n];
303        struct sub_bitmap *b = &imgs.parts[n];
304
305        b->x = bb.x0;
306        b->y = bb.y0;
307        b->w = b->dw = bb.x1 - bb.x0;
308        b->h = b->dh = bb.y1 - bb.y0;
309        b->stride = imgs.packed->stride[0];
310        b->bitmap = (uint8_t *)imgs.packed->planes[0] +
311                    b->stride * b->src_y + b->src_x * 4;
312
313        memset_pic(b->bitmap, 0, b->w * 4, b->h, b->stride);
314
315        for (int i = 0; i < res->num_parts; i++) {
316            struct sub_bitmap *s = &res->parts[i];
317
318            // Assume mp_get_sub_bb_list() never splits sub bitmaps
319            // So we don't clip/adjust the size of the sub bitmap
320            if (s->x > bb.x1 || s->x + s->w < bb.x0 ||
321                s->y > bb.y1 || s->y + s->h < bb.y0)
322                continue;
323
324            draw_ass_rgba(s->bitmap, s->w, s->h, s->stride,
325                          b->bitmap, b->stride,
326                          s->x - bb.x0, s->y - bb.y0,
327                          s->libass.color);
328        }
329    }
330
331    *res = imgs;
332    return true;
333}

335// Pack the contents of image_lists[0] to image_lists[num_image_lists-1] into
336// a single image, and make *out point to it. *out is completely overwritten.
337// If libass reported any change, image_lists_changed must be set (it then
338// repacks all images). preferred_osd_format can be set to a desired
339// sub_bitmap_format. Currently, only SUBBITMAP_LIBASS is supported.
340void mp_ass_packer_pack(struct mp_ass_packer *p, ASS_Image **image_lists,
341                        int num_image_lists, bool image_lists_changed,
342                        int preferred_osd_format, struct sub_bitmaps *out)
343{
344    int format = preferred_osd_format == SUBBITMAP_BGRA ? SUBBITMAP_BGRA
345                                                        : SUBBITMAP_LIBASS;
346
347    if (p->cached_subs_valid && !image_lists_changed &&
348        p->cached_subs.format == format)
349    {
350        *out = p->cached_subs;
351        return;
352    }
353
354    *out = (struct sub_bitmaps){.change_id = 1};
355    p->cached_subs_valid = false;
356
357    struct sub_bitmaps res = {
358        .change_id = image_lists_changed,
359        .format = SUBBITMAP_LIBASS,
360        .parts = p->cached_parts,
361    };
362
363    for (int n = 0; n < num_image_lists; n++) {
364        for (struct ass_image *img = image_lists[n]; img; img = img->next) {
365            if (img->w == 0 || img->h == 0)
366                continue;
367            MP_TARRAY_GROW(p, p->cached_parts, res.num_parts);
368            res.parts = p->cached_parts;
369            struct sub_bitmap *b = &res.parts[res.num_parts];
370            b->bitmap = img->bitmap;
371            b->stride = img->stride;
372            b->libass.color = img->color;
373            b->dw = b->w = img->w;
374            b->dh = b->h = img->h;
375            b->x = img->dst_x;
376            b->y = img->dst_y;
377            res.num_parts++;
378        }
379    }
380
381    bool r = false;
382    if (format == SUBBITMAP_BGRA) {
383        r = pack_rgba(p, &res);
384    } else {
385        r = pack_libass(p, &res);
386    }
387
388    if (!r)
389        return;
390
391    *out = res;
392    p->cached_subs = res;
393    p->cached_subs.change_id = 0;
394    p->cached_subs_valid = true;
395}

397// Set *out_rc to [x0, y0, x1, y1] of the graphical bounding box in script
398// coordinates.
399// Set it to [inf, inf, -inf, -inf] if empty.
400void mp_ass_get_bb(ASS_Image *image_list, ASS_Track *track,
401                   struct mp_osd_res *res, double *out_rc)
402{
403    double rc[4] = {INFINITY, INFINITY, -INFINITY, -INFINITY};
404
405    for (ASS_Image *img = image_list; img; img = img->next) {
406        if (img->w == 0 || img->h == 0)
407            continue;
408        rc[0] = MPMIN(rc[0], img->dst_x);
409        rc[1] = MPMIN(rc[1], img->dst_y);
410        rc[2] = MPMAX(rc[2], img->dst_x + img->w);
411        rc[3] = MPMAX(rc[3], img->dst_y + img->h);
412    }
413
414    double scale = track->PlayResY / (double)MPMAX(res->h, 1);
415    if (scale > 0) {
416        for (int i = 0; i < 4; i++)
417            out_rc[i] = rc[i] * scale;
418    }
