75static void update_subtitle_speed(struct dec_sub *sub)
76{
77    struct mp_subtitle_opts *opts = sub->opts;
78    sub->sub_speed = 1.0;
79
80    if (sub->video_fps > 0 && sub->codec->frame_based > 0) {
81        MP_VERBOSE(sub, "Frame based format, dummy FPS: %f, video FPS: %f\n",
82                   sub->codec->frame_based, sub->video_fps);
83        sub->sub_speed *= sub->codec->frame_based / sub->video_fps;
84    }
85
86    if (opts->sub_fps && sub->video_fps)
87        sub->sub_speed *= opts->sub_fps / sub->video_fps;
88
89    sub->sub_speed *= opts->sub_speed;
90}

92// Return the subtitle PTS used for a given video PTS.
93static double pts_to_subtitle(struct dec_sub *sub, double pts)
94{
95    struct mp_subtitle_opts *opts = sub->opts;
96
97    if (pts != MP_NOPTS_VALUE)
98        pts = (pts * sub->play_dir - opts->sub_delay) / sub->sub_speed;
99
100    return pts;
101}

103static double pts_from_subtitle(struct dec_sub *sub, double pts)
104{
105    struct mp_subtitle_opts *opts = sub->opts;
106
107    if (pts != MP_NOPTS_VALUE)
108        pts = (pts * sub->sub_speed + opts->sub_delay) * sub->play_dir;
109
110    return pts;
111}

133static struct sd *init_decoder(struct dec_sub *sub)
134{
135    for (int n = 0; sd_list[n]; n++) {
136        const struct sd_functions *driver = sd_list[n];
137        struct sd *sd = talloc(NULL, struct sd);
138        *sd = (struct sd){
139            .global = sub->global,
140            .log = mp_log_new(sd, sub->log, driver->name),
141            .opts = sub->opts,
142            .driver = driver,
143            .attachments = sub->attachments,
144            .codec = sub->codec,
145            .preload_ok = true,
146        };
147
148        if (sd->driver->init(sd) >= 0)
149            return sd;
150
151        talloc_free(sd);
152    }
153
154    MP_ERR(sub, "Could not find subtitle decoder for format '%s'.\n",
155           sub->codec->codec);
156    return NULL;
157}

159// Thread-safety of the returned object: all functions are thread-safe,
160// except sub_get_bitmaps() and sub_get_text(). Decoder backends (sd_*)
161// do not need to acquire locks.
162// Ownership of attachments goes to the callee, and is released with
163// talloc_free() (even on failure).
164struct dec_sub *sub_create(struct mpv_global *global, struct sh_stream *sh,
165                           struct attachment_list *attachments, int order)
166{
167    assert(sh && sh->type == STREAM_SUB);
168
169    struct dec_sub *sub = talloc(NULL, struct dec_sub);
170    *sub = (struct dec_sub){
171        .log = mp_log_new(sub, global->log, "sub"),
172        .global = global,
173        .opts_cache = m_config_cache_alloc(sub, global, &mp_subtitle_sub_opts),
174        .sh = sh,
175        .codec = sh->codec,
176        .attachments = talloc_steal(sub, attachments),
177        .play_dir = 1,
178        .order = order,
179        .last_pkt_pts = MP_NOPTS_VALUE,
180        .last_vo_pts = MP_NOPTS_VALUE,
181        .start = MP_NOPTS_VALUE,
182        .end = MP_NOPTS_VALUE,
183    };
184    sub->opts = sub->opts_cache->opts;
185    mpthread_mutex_init_recursive(&sub->lock);
186
187    sub->sd = init_decoder(sub);
188    if (sub->sd) {
189        update_subtitle_speed(sub);
190        return sub;
191    }
192
193    sub_destroy(sub);
194    return NULL;
195}

236void sub_preload(struct dec_sub *sub)
237{
238    pthread_mutex_lock(&sub->lock);
239
240    struct mp_dispatch_queue *demux_waiter = mp_dispatch_create(NULL);
241    demux_set_stream_wakeup_cb(sub->sh, wakeup_demux, demux_waiter);
242
243    sub->preload_attempted = true;
244
245    for (;;) {
246        struct demux_packet *pkt = NULL;
247        int r = demux_read_packet_async(sub->sh, &pkt);
248        if (r == 0) {
249            mp_dispatch_queue_process(demux_waiter, INFINITY);
250            continue;
251        }
252        if (!pkt)
253            break;
254        sub->sd->driver->decode(sub->sd, pkt);
255        talloc_free(pkt);
256    }
257
258    demux_set_stream_wakeup_cb(sub->sh, NULL, NULL);
259    talloc_free(demux_waiter);
260
261    pthread_mutex_unlock(&sub->lock);
262}

264static bool is_new_segment(struct dec_sub *sub, struct demux_packet *p)
265{
266    return p->segmented &&

270// Read packets from the demuxer stream passed to sub_create(). Return true if
271// enough packets were read, false if the player should wait until the demuxer
272// signals new packets available (and then should retry).
273bool sub_read_packets(struct dec_sub *sub, double video_pts)
274{
275    bool r = true;
276    pthread_mutex_lock(&sub->lock);
277    video_pts = pts_to_subtitle(sub, video_pts);
278    while (1) {
279        bool read_more = true;
280        if (sub->sd->driver->accepts_packet)
281            read_more = sub->sd->driver->accepts_packet(sub->sd, video_pts);
282
283        if (!read_more)
284            break;
285
286        if (sub->new_segment && sub->new_segment->start < video_pts) {
287            sub->last_vo_pts = video_pts;
288            update_segment(sub);
289        }
290
291        if (sub->new_segment)
292            break;
293
294        // (Use this mechanism only if sub_delay matters to avoid corner cases.)
295        double min_pts = sub->opts->sub_delay < 0 ? video_pts : MP_NOPTS_VALUE;
296
297        struct demux_packet *pkt;
298        int st = demux_read_packet_async_until(sub->sh, min_pts, &pkt);
299        // Note: "wait" (st==0) happens with non-interleaved streams only, and
300        // then we should stop the playloop until a new enough packet has been
301        // seen (or the subtitle decoder's queue is full). This usually does not
302        // happen for interleaved subtitle streams, which never return "wait"
303        // when reading, unless min_pts is set.
304        if (st <= 0) {
305            r = st < 0 || (sub->last_pkt_pts != MP_NOPTS_VALUE &&
306                           sub->last_pkt_pts > video_pts);
307            break;
308        }
309
310        if (sub->recorder_sink)
311            mp_recorder_feed_packet(sub->recorder_sink, pkt);
312
313        sub->last_pkt_pts = pkt->pts;
314
315        if (is_new_segment(sub, pkt)) {
316            sub->new_segment = pkt;
317            // Note that this can be delayed to a much later point in time.
318            update_segment(sub);
319            break;
320        }
321
322        if (!(sub->preload_attempted && sub->sd->preload_ok))
323            sub->sd->driver->decode(sub->sd, pkt);
324
325        talloc_free(pkt);
326    }
327    pthread_mutex_unlock(&sub->lock);
328    return r;
329}

406int sub_control(struct dec_sub *sub, enum sd_ctrl cmd, void *arg)
407{
408    int r = CONTROL_UNKNOWN;
409    pthread_mutex_lock(&sub->lock);
410    bool propagate = false;
411    switch (cmd) {
412    case SD_CTRL_SET_VIDEO_DEF_FPS:
413        sub->video_fps = *(double *)arg;
414        update_subtitle_speed(sub);
415        break;
416    case SD_CTRL_SUB_STEP: {
417        double *a = arg;
418        double arg2[2] = {a[0], a[1]};
419        arg2[0] = pts_to_subtitle(sub, arg2[0]);
420        if (sub->sd->driver->control)
421            r = sub->sd->driver->control(sub->sd, cmd, arg2);
422        if (r == CONTROL_OK)
423            a[0] = pts_from_subtitle(sub, arg2[0]);
424        break;
425    case SD_CTRL_UPDATE_OPTS:
426        if (m_config_cache_update(sub->opts_cache))
427            update_subtitle_speed(sub);
428        propagate = true;
429        break;
430    }
431    default:
432        propagate = true;
433    }
434    if (propagate && sub->sd->driver->control)
435        r = sub->sd->driver->control(sub->sd, cmd, arg);
436    pthread_mutex_unlock(&sub->lock);
437    return r;
438}
