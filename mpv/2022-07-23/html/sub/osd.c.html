118struct osd_state *osd_create(struct mpv_global *global)
119{
120    assert(MAX_OSD_PARTS >= OSDTYPE_COUNT);
121
122    struct osd_state *osd = talloc_zero(NULL, struct osd_state);
123    *osd = (struct osd_state) {
124        .opts_cache = m_config_cache_alloc(osd, global, &mp_osd_render_sub_opts),
125        .global = global,
126        .log = mp_log_new(osd, global->log, "osd"),
127        .force_video_pts = ATOMIC_VAR_INIT(MP_NOPTS_VALUE),
128        .stats = stats_ctx_create(osd, global, "osd"),
129    };
130    pthread_mutex_init(&osd->lock, NULL);
131    osd->opts = osd->opts_cache->opts;
132
133    for (int n = 0; n < MAX_OSD_PARTS; n++) {
134        struct osd_object *obj = talloc(osd, struct osd_object);
135        *obj = (struct osd_object) {
136            .type = n,
137            .text = talloc_strdup(obj, ""),
138            .progbar_state = {.type = -1},
139            .vo_change_id = 1,
140        };
141        osd->objs[n] = obj;
142    }
143
144    osd->objs[OSDTYPE_SUB]->is_sub = true;
145    osd->objs[OSDTYPE_SUB2]->is_sub = true;
146
147    osd_init_backend(osd);
148    return osd;
149}

259// Optional. Can be called for faster reaction of OSD-generating scripts like
260// osc.lua. This can achieve that the resize happens first, so that the OSD is
261// generated at the correct resolution the first time the resized frame is
262// rendered. Since the OSD doesn't (and can't) wait for the script, this
263// increases the time in which the script can react, and also gets rid of the
264// unavoidable redraw delay (though it will still be racy).
265// Unnecessary for anything else.
266void osd_resize(struct osd_state *osd, struct mp_osd_res res)
267{
268    pthread_mutex_lock(&osd->lock);
269    int types[] = {OSDTYPE_OSD, OSDTYPE_EXTERNAL, OSDTYPE_EXTERNAL2, -1};
270    for (int n = 0; types[n] >= 0; n++)
271        check_obj_resize(osd, res, osd->objs[types[n]]);
272    pthread_mutex_unlock(&osd->lock);
273}

275static struct sub_bitmaps *render_object(struct osd_state *osd,
276                                         struct osd_object *obj,
277                                         struct mp_osd_res osdres, double video_pts,
278                                         const bool sub_formats[SUBBITMAP_COUNT])
279{
280    int format = SUBBITMAP_LIBASS;
281    if (!sub_formats[format] || osd->opts->force_rgba_osd)
282        format = SUBBITMAP_BGRA;
283
284    struct sub_bitmaps *res = NULL;
285
286    check_obj_resize(osd, osdres, obj);
287
288    if (obj->type == OSDTYPE_SUB) {
289        if (obj->sub && sub_is_primary_visible(obj->sub))
290            res = sub_get_bitmaps(obj->sub, obj->vo_res, format, video_pts);
291    } else if (obj->type == OSDTYPE_SUB2) {
292        if (obj->sub && sub_is_secondary_visible(obj->sub))
293            res = sub_get_bitmaps(obj->sub, obj->vo_res, format, video_pts);
294    } else if (obj->type == OSDTYPE_EXTERNAL2) {
295        if (obj->external2 && obj->external2->format) {
296            res = sub_bitmaps_copy(NULL, obj->external2); // need to be owner
297            obj->external2->change_id = 0;
298        }
299    } else {
300        res = osd_object_get_bitmaps(osd, obj, format);
301    }
302
303    if (obj->vo_had_output != !!res) {
304        obj->vo_had_output = !!res;
305        obj->vo_change_id += 1;
306    }
307
308    if (res) {
309        obj->vo_change_id += res->change_id;
310
311        res->render_index = obj->type;
312        res->change_id = obj->vo_change_id;
313    }
314
315    return res;
316}

452// The interesting part about this is that OSD has to compensate the aspect
453// ratio if the image does not have a 1:1 pixel aspect ratio.
454struct mp_osd_res osd_res_from_image_params(const struct mp_image_params *p)
455{
456    return (struct mp_osd_res) {
457        .w = p->w,
458        .h = p->h,
459        .display_par = p->p_h / (double)p->p_w,
460    };
461}

463// Typically called to react to OSD style changes.
464void osd_changed(struct osd_state *osd)
465{
466    pthread_mutex_lock(&osd->lock);
467    osd->objs[OSDTYPE_OSD]->osd_changed = true;
468    osd->want_redraw_notification = true;
469    // Done here for a lack of a better place.
470    m_config_cache_update(osd->opts_cache);
471    pthread_mutex_unlock(&osd->lock);
472}

474bool osd_query_and_reset_want_redraw(struct osd_state *osd)
475{
476    pthread_mutex_lock(&osd->lock);
477    bool r = osd->want_redraw_notification;
478    osd->want_redraw_notification = false;
479    pthread_mutex_unlock(&osd->lock);
480    return r;

485    pthread_mutex_lock(&osd->lock);

492// Position the subbitmaps in imgs on the screen. Basically, this fits the
493// subtitle canvas (of size frame_w x frame_h) onto the screen, such that it
494// fills the whole video area (especially if the video is magnified, e.g. on
495// fullscreen). If compensate_par is >0, adjust the way the subtitles are
496// "stretched" on the screen, and letter-box the result. If compensate_par
497// is <0, strictly letter-box the subtitles. If it is 0, stretch them.
498void osd_rescale_bitmaps(struct sub_bitmaps *imgs, int frame_w, int frame_h,
499                         struct mp_osd_res res, double compensate_par)
500{
501    int vidw = res.w - res.ml - res.mr;
502    int vidh = res.h - res.mt - res.mb;
503    double xscale = (double)vidw / frame_w;
504    double yscale = (double)vidh / frame_h;
505    if (compensate_par < 0) {
506        assert(res.display_par);
507        compensate_par = xscale / yscale / res.display_par;
508    }
509    if (compensate_par > 0)
510        xscale /= compensate_par;
511    int cx = vidw / 2 - (int)(frame_w * xscale) / 2;
512    int cy = vidh / 2 - (int)(frame_h * yscale) / 2;
513    for (int i = 0; i < imgs->num_parts; i++) {
514        struct sub_bitmap *bi = &imgs->parts[i];
515        bi->x = (int)(bi->x * xscale) + cx + res.ml;
516        bi->y = (int)(bi->y * yscale) + cy + res.mt;
517        bi->dw = (int)(bi->w * xscale + 0.5);
518        bi->dh = (int)(bi->h * yscale + 0.5);
519    }
520}

522// Copy *in and return a new allocation of it. Free with talloc_free(). This
523// will contain a refcounted copy of the image data.
524//
525// in->packed must be set and must be a refcounted image, unless there is no
526// data (num_parts==0).
527//
528//  p_cache: if not NULL, then this points to a struct sub_bitmap_copy_cache*
529//           variable. The function may set this to an allocation and may later
530//           read it. You have to free it with talloc_free() when done.
531//  in: valid struct, or NULL (in this case it also returns NULL)
532//  returns: new copy, or NULL if there was no data in the input
533struct sub_bitmaps *sub_bitmaps_copy(struct sub_bitmap_copy_cache **p_cache,
534                                     struct sub_bitmaps *in)
535{
536    if (!in || !in->num_parts)
537        return NULL;
538
539    struct sub_bitmaps *res = talloc(NULL, struct sub_bitmaps);
540    *res = *in;
541
542    // Note: the p_cache thing is a lie and unused.
543
544    // The bitmaps being refcounted is essential for performance, and for
545    // not invalidating in->parts[*].bitmap pointers.
546    assert(in->packed && in->packed->bufs[0]);
547
548    res->packed = mp_image_new_ref(res->packed);
549    MP_HANDLE_OOM(res->packed);
550    talloc_steal(res, res->packed);
551
552    res->parts = NULL;
553    MP_RESIZE_ARRAY(res, res->parts, res->num_parts);
554    memcpy(res->parts, in->parts, sizeof(res->parts[0]) * res->num_parts);
555
556    return res;
557}
