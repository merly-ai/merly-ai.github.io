75unsigned int dvb_get_tuner_delsys_mask(int fe_fd, struct mp_log *log)
76{
77    unsigned int ret_mask = 0, delsys;
78    struct dtv_property prop[1];
79    struct dtv_properties cmdseq = {.num = 1, .props = prop};
80    struct dvb_frontend_info fe_info;
81
82#ifdef DVB_USE_S2API
83    /* S2API is the DVB API new since 2.6.28.
84       It allows to query frontends with multiple delivery systems. */
85    mp_verbose(log, "Querying tuner frontend type via DVBv5 API for frontend FD %d\n",
86               fe_fd);
87    prop[0].cmd = DTV_ENUM_DELSYS;
88    if (ioctl(fe_fd, FE_GET_PROPERTY, &cmdseq) < 0) {
89        mp_err(log, "DVBv5: FE_GET_PROPERTY(DTV_ENUM_DELSYS) error: %d, FD: %d\n\n", errno, fe_fd);
90        goto old_api;
91    }
92    unsigned int i, delsys_count = prop[0].u.buffer.len;
93    mp_verbose(log, "DVBv5: Number of supported delivery systems: %d\n", delsys_count);
94    if (delsys_count == 0) {
95        mp_err(log, "DVBv5: Frontend FD %d returned no delivery systems!\n", fe_fd);
96        goto old_api;
97    }
98    for (i = 0; i < delsys_count; i++) {
99        delsys = (unsigned int)prop[0].u.buffer.data[i];
100        DELSYS_SET(ret_mask, delsys);
101        mp_verbose(log, "DVBv5: Tuner frontend type seems to be %s\n", get_dvb_delsys(delsys));
102    }
103
104    return ret_mask;
105
106old_api:
107#endif
108    mp_verbose(log, "Querying tuner frontend type via pre-DVBv5 API for frontend FD %d\n",
109               fe_fd);
110
111    memset(&fe_info, 0x00, sizeof(struct dvb_frontend_info));
112    if (ioctl(fe_fd, FE_GET_INFO, &fe_info) < 0) {
113        mp_err(log, "DVBv3: FE_GET_INFO error: %d, FD: %d\n\n", errno, fe_fd);
114        return ret_mask;
115    }
116    /* Try to get kernel DVB API version. */
117    prop[0].cmd = DTV_API_VERSION;
118    if (ioctl(fe_fd, FE_GET_PROPERTY, &cmdseq) < 0) {
119        prop[0].u.data = 0x0300; /* Fail, assume 3.0 */
120    }
121
122    mp_verbose(log, "DVBv3: Queried tuner frontend type of device named '%s', FD: %d\n",
123               fe_info.name, fe_fd);
124    switch (fe_info.type) {
125    case FE_OFDM:
126        DELSYS_SET(ret_mask, SYS_DVBT);
127        if (prop[0].u.data < 0x0500)
128            break;
129        if (FE_CAN_2G_MODULATION & fe_info.caps) {
130            DELSYS_SET(ret_mask, SYS_DVBT2);
131        }
132        break;
133    case FE_QPSK:
134        DELSYS_SET(ret_mask, SYS_DVBS);
135        if (prop[0].u.data < 0x0500)
136            break;
137        if (FE_CAN_2G_MODULATION & fe_info.caps) {
138            DELSYS_SET(ret_mask, SYS_DVBS2);
139        }
140#if 0 /* Not used now. */
141        if (FE_CAN_TURBO_FEC & fe_info.caps) {
142            DELSYS_SET(ret_mask, SYS_TURBO);
143        }
144#endif
145        break;
146    case FE_QAM:
147        DELSYS_SET(ret_mask, SYS_DVBC_ANNEX_A);
148        DELSYS_SET(ret_mask, SYS_DVBC_ANNEX_C);
149        break;
150#ifdef DVB_ATSC
151    case FE_ATSC:
152        if ((FE_CAN_8VSB | FE_CAN_16VSB) & fe_info.caps) {
153            DELSYS_SET(ret_mask, SYS_ATSC);
154        }
155        if ((FE_CAN_QAM_64 | FE_CAN_QAM_256 | FE_CAN_QAM_AUTO) & fe_info.caps) {
156            DELSYS_SET(ret_mask, SYS_DVBC_ANNEX_B);
157        }
158        break;
159#endif
160    default:
161        mp_err(log, "DVBv3: Unknown tuner frontend type: %d\n", fe_info.type);
162        return ret_mask;
163    }
164
165    for (delsys = 0; delsys < SYS_DVB__COUNT__; delsys ++) {
166        if (!DELSYS_IS_SET(ret_mask, delsys))
167            continue; /* Skip unsupported. */
168        mp_verbose(log, "DVBv3: Tuner frontend type seems to be %s\n", get_dvb_delsys(delsys));
169    }
170
171    return ret_mask;
172}

174int dvb_open_devices(dvb_priv_t *priv, unsigned int adapter,
175    unsigned int frontend, unsigned int demux_cnt)
176{
177    unsigned int i;
178    char frontend_dev[PATH_MAX], dvr_dev[PATH_MAX], demux_dev[PATH_MAX];
179    dvb_state_t* state = priv->state;
180
181    snprintf(frontend_dev, sizeof(frontend_dev), "/dev/dvb/adapter%u/frontend%u", adapter, frontend);
182    snprintf(dvr_dev, sizeof(dvr_dev), "/dev/dvb/adapter%u/dvr0", adapter);
183    snprintf(demux_dev, sizeof(demux_dev), "/dev/dvb/adapter%u/demux0", adapter);
184    MP_VERBOSE(priv, "DVB_OPEN_DEVICES: frontend: %s\n", frontend_dev);
185    state->fe_fd = open(frontend_dev, O_RDWR | O_NONBLOCK | O_CLOEXEC);
186    if (state->fe_fd < 0) {
187        MP_ERR(priv, "ERROR OPENING FRONTEND DEVICE %s: ERRNO %d\n",
188               frontend_dev, errno);
189        return 0;
190    }
191    state->demux_fds_cnt = 0;
192    MP_VERBOSE(priv, "DVB_OPEN_DEVICES(%d)\n", demux_cnt);
193    for (i = 0; i < demux_cnt; i++) {
194        state->demux_fds[i] = open(demux_dev, O_RDWR | O_NONBLOCK | O_CLOEXEC);
195        if (state->demux_fds[i] < 0) {
196            MP_ERR(priv, "ERROR OPENING DEMUX 0: %d\n", errno);
197            return 0;
198        } else {
199            MP_VERBOSE(priv, "OPEN(%d), file %s: FD=%d, CNT=%d\n", i, demux_dev,
200                       state->demux_fds[i], state->demux_fds_cnt);
201            state->demux_fds_cnt++;
202        }
203    }
204
205    state->dvr_fd = open(dvr_dev, O_RDONLY | O_NONBLOCK | O_CLOEXEC);
206    if (state->dvr_fd < 0) {
207        MP_ERR(priv, "ERROR OPENING DVR DEVICE %s: %d\n", dvr_dev, errno);
208        return 0;
209    }
210
211    return 1;
212}

215int dvb_fix_demuxes(dvb_priv_t *priv, unsigned int cnt)
216{
217    int i;
218    char demux_dev[PATH_MAX];
219
220    dvb_state_t* state = priv->state;
221
222    snprintf(demux_dev, sizeof(demux_dev), "/dev/dvb/adapter%d/demux0",
223            state->adapters[state->cur_adapter].devno);
224    MP_VERBOSE(priv, "FIX %d -> %d\n", state->demux_fds_cnt, cnt);
225    if (state->demux_fds_cnt >= cnt) {
226        for (i = state->demux_fds_cnt - 1; i >= (int)cnt; i--) {
227            MP_VERBOSE(priv, "FIX, CLOSE fd(%d): %d\n", i, state->demux_fds[i]);
228            close(state->demux_fds[i]);
229        }
230        state->demux_fds_cnt = cnt;
231    } else {
232        for (i = state->demux_fds_cnt; i < cnt; i++) {
233            state->demux_fds[i] = open(demux_dev,
234                                      O_RDWR | O_NONBLOCK | O_CLOEXEC);
235            MP_VERBOSE(priv, "FIX, OPEN fd(%d): %d\n", i, state->demux_fds[i]);
236            if (state->demux_fds[i] < 0) {
237                MP_ERR(priv, "ERROR OPENING DEMUX 0: %d\n", errno);
238                return 0;
239            } else
240                state->demux_fds_cnt++;
241        }
242    }
243
244    return 1;
245}

248                    dmx_pes_type_t pestype)
249{
250    int i;
251    struct dmx_pes_filter_params pesFilterParams;
252
253    pesFilterParams.pid     = pid;
254    pesFilterParams.input   = DMX_IN_FRONTEND;
255    pesFilterParams.output  = DMX_OUT_TS_TAP;
256    pesFilterParams.pes_type = pestype;
257    pesFilterParams.flags   = DMX_IMMEDIATE_START;
258
259    {
260        int buffersize = 256 * 1024;
261        if (ioctl(fd, DMX_SET_BUFFER_SIZE, buffersize) < 0)
262            MP_ERR(priv, "ERROR IN DMX_SET_BUFFER_SIZE %i for fd %d: ERRNO: %d\n",
263                   pid, fd, errno);
264    }
265
266    errno = 0;
267    if ((i = ioctl(fd, DMX_SET_PES_FILTER, &pesFilterParams)) < 0) {
268        MP_ERR(priv, "ERROR IN SETTING DMX_FILTER %i for fd %d: ERRNO: %d\n",
269               pid, fd, errno);

271    }
272
273    MP_VERBOSE(priv, "SET PES FILTER ON PID %d to fd %d, RESULT: %d, ERRNO: %d\n",
274               pid, fd, i, errno);
275    return 1;
276}

278int dvb_get_pmt_pid(dvb_priv_t *priv, int devno, int service_id)
279{
280    /* We need special filters on the demux,
281       so open one locally, and close also here. */
282    char demux_dev[PATH_MAX];
283    snprintf(demux_dev, sizeof(demux_dev), "/dev/dvb/adapter%d/demux0", devno);
284
285    struct dmx_sct_filter_params fparams;
286
287    memset(&fparams, 0x00, sizeof(fparams));
288    fparams.pid = 0;
289    fparams.filter.filter[0] = 0x00;
290    fparams.filter.mask[0] = 0xff;
291    fparams.timeout = 0;
292    fparams.flags = DMX_IMMEDIATE_START | DMX_CHECK_CRC;
293
294    int pat_fd;
295    if ((pat_fd = open(demux_dev, O_RDWR)) < 0) {
296        MP_ERR(priv, "Opening PAT DEMUX failed, error: %d", errno);
297        return -1;
298    }
299
300    if (ioctl(pat_fd, DMX_SET_FILTER, &fparams) < 0) {
301        MP_ERR(priv, "ioctl DMX_SET_FILTER failed, error: %d", errno);
302        close(pat_fd);
303        return -1;
304    }
305
306    int bytes_read;
307    int section_length;
308    unsigned char buft[4096];
309    unsigned char *bufptr = buft;
310
311    int pmt_pid = -1;
312
313    bool pat_read = false;
314    while (!pat_read) {
315        if (((bytes_read =
316                  read(pat_fd, bufptr,
317                       sizeof(buft))) < 0) && errno == EOVERFLOW)
318            bytes_read = read(pat_fd, bufptr, sizeof(buft));
319        if (bytes_read < 0) {
320            MP_ERR(priv, "PAT: read_sections: read error: %d", errno);
321            close(pat_fd);
322            return -1;
323        }
324
325        section_length = ((bufptr[1] & 0x0f) << 8) | bufptr[2];
326        if (bytes_read != section_length + 3)
327            continue;
328
329        bufptr += 8;
330        section_length -= 8;
331
332        /* assumes one section contains the whole pat */
333        pat_read = true;
334        while (section_length > 0) {
335            int this_service_id = (bufptr[0] << 8) | bufptr[1];
336            if (this_service_id == service_id) {
337                pmt_pid = ((bufptr[2] & 0x1f) << 8) | bufptr[3];
338                section_length = 0;
339            }
340            bufptr += 4;
341            section_length -= 4;
342        }
343    }
344    close(pat_fd);
345
346    return pmt_pid;
347}

367static int check_status(dvb_priv_t *priv, int fd_frontend, int tmout)
368{
369    int32_t strength;
370    fe_status_t festatus;
371    struct pollfd pfd[1];
372    int ok = 0, locks = 0;
373    time_t tm1, tm2;
374
375    pfd[0].fd = fd_frontend;
376    pfd[0].events = POLLPRI;
377
378    MP_VERBOSE(priv, "Getting frontend status\n");
379    tm1 = tm2 = time((time_t *) NULL);
380    while (!ok) {
381        festatus = 0;
382        if (poll(pfd, 1, tmout * 1000) > 0) {
383            if (pfd[0].revents & POLLPRI) {
384                if (ioctl(fd_frontend, FE_READ_STATUS, &festatus) >= 0) {
385                    if (festatus & FE_HAS_LOCK)
386                        locks++;
387                }
388            }
389        }
390        usleep(10000);
391        tm2 = time((time_t *) NULL);
392        if (
 Recommendations (Experimental)  R1: (!(festatus >= 2 && festatus <= 123))
R2: (!(festatus == 123 || festatus == 123))
R3: (!(festatus == FE_TIMEDOUT && locks == "string"))
R4: (!(festatus == 123 || festatus == 123 || festatus == 123))
R5: (!(festatus == 123 || festatus == 123 || festatus == 123 || festatus == 123))
 Score  Issue:  1 unfamiliar pattern(s) detected 
Cost: 13.35, Total Score: 1013.35
Class: complex_2
(festatus & FE_TIMEDOUT) || (locks >= 2) || (tm2 - tm1 >= tmout))
393            ok = 1;
394    }
395
396    if (festatus & FE_HAS_LOCK) {
397        strength = 0;
398        if (ioctl(fd_frontend, FE_READ_BER, &strength) >= 0)
399            MP_VERBOSE(priv, "Bit error rate: %d\n", strength);
400
401        strength = 0;
402        if (ioctl(fd_frontend, FE_READ_SIGNAL_STRENGTH, &strength) >= 0)
403            MP_VERBOSE(priv, "Signal strength: %d\n", strength);
404
405        strength = 0;
406        if (ioctl(fd_frontend, FE_READ_SNR, &strength) >= 0)
407            MP_VERBOSE(priv, "SNR: %d\n", strength);
408
409        strength = 0;
410        if (ioctl(fd_frontend, FE_READ_UNCORRECTED_BLOCKS, &strength) >= 0)
411            MP_VERBOSE(priv, "UNC: %d\n", strength);
412
413        print_status(priv, festatus);
414    } else {
415        MP_ERR(priv, "Not able to lock to the signal on the given frequency, "
416               "timeout: %d\n", tmout);
417        return -1;
418    }
419    return 0;
420}

427static int diseqc_send_msg(int fd, fe_sec_voltage_t v, struct diseqc_cmd *cmd,
428                           fe_sec_tone_mode_t t, fe_sec_mini_cmd_t b)
429{
430    if (ioctl(fd, FE_SET_TONE, SEC_TONE_OFF) < 0)
431        return -1;
432    if (ioctl(fd, FE_SET_VOLTAGE, v) < 0)
433        return -1;
434    usleep(15 * 1000);
435    if (ioctl(fd, FE_DISEQC_SEND_MASTER_CMD, &cmd->cmd) < 0)
436        return -1;
437    usleep(cmd->wait * 1000);
438    usleep(15 * 1000);
439    if (ioctl(fd, FE_DISEQC_SEND_BURST, b) < 0)
440        return -1;
441    usleep(15 * 1000);
442    if (ioctl(fd, FE_SET_TONE, t) < 0)
443        return -1;
444    usleep(100000);
445
446    return 0;
447}

449/* digital satellite equipment control,
450 * specification is available from http://www.eutelsat.com/
451 */
452static int do_diseqc(int secfd, int sat_no, int polv, int hi_lo)
453{
454    struct diseqc_cmd cmd =  { {{0xe0, 0x10, 0x38, 0xf0, 0x00, 0x00}, 4}, 0 };
455
456    /* param: high nibble: reset bits, low nibble set bits,
457     * bits are: option, position, polarizaion, band
458     */
459    cmd.cmd.msg[3] = 0xf0 | (((sat_no * 4) & 0x0f) | (hi_lo ? 1 : 0) | (polv ? 0 : 2));
460
461    return diseqc_send_msg(secfd, polv ? SEC_VOLTAGE_13 : SEC_VOLTAGE_18,
462                           &cmd, hi_lo ? SEC_TONE_ON : SEC_TONE_OFF,
463                           ((sat_no / 4) % 2) ? SEC_MINI_B : SEC_MINI_A);
464}

486static int tune_it(dvb_priv_t *priv, int fd_frontend, unsigned int delsys,
487                   unsigned int freq, unsigned int srate, char pol,
488                   int stream_id,
489                   fe_spectral_inversion_t specInv, unsigned int diseqc,
490                   fe_modulation_t modulation,
491                   fe_code_rate_t HP_CodeRate,
492                   fe_transmit_mode_t TransmissionMode,
493                   fe_guard_interval_t guardInterval,
494                   fe_bandwidth_t bandwidth,
495                   fe_code_rate_t LP_CodeRate, fe_hierarchy_t hier,
496                   int timeout)
497{
498    int hi_lo = 0, bandwidth_hz = 0;
499    dvb_state_t* state = priv->state;
500    struct dvb_frontend_parameters feparams;
501
502
503    MP_VERBOSE(priv, "TUNE_IT, fd_frontend %d, %s freq %lu, srate %lu, "
504               "pol %c, diseqc %u\n", fd_frontend,
505               get_dvb_delsys(delsys),
506               (long unsigned int)freq, (long unsigned int)srate,
507               (pol > ' ' ? pol : '-'), diseqc);
508
509    MP_VERBOSE(priv, "Using %s adapter %d\n",
510        get_dvb_delsys(delsys),
511        state->adapters[state->cur_adapter].devno);
512
513    {
514        /* discard stale QPSK events */
515        struct dvb_frontend_event ev;
516        while (true) {
517            if (ioctl(fd_frontend, FE_GET_EVENT, &ev) < 0)
518                break;
519        }
520    }
521
522   /* Prepare params, be verbose. */
523    switch (delsys) {
524    case SYS_DVBT2:
525#ifndef DVB_USE_S2API
526        MP_ERR(priv, "ERROR: Can not tune to T2 channel, S2-API not "
527                     "available, will tune to DVB-T!\n");
528#endif
529        /* PASSTROUTH. */
530    case SYS_DVBT:
531        if (freq < 1000000)
532            freq *= 1000UL;
533        switch (bandwidth) {
534        case BANDWIDTH_5_MHZ:
535            bandwidth_hz = 5000000;
536            break;
537        case BANDWIDTH_6_MHZ:
538            bandwidth_hz = 6000000;
539            break;
540        case BANDWIDTH_7_MHZ:
541            bandwidth_hz = 7000000;
542            break;
543        case BANDWIDTH_8_MHZ:
544            bandwidth_hz = 8000000;
545            break;
546        case BANDWIDTH_10_MHZ:
547            bandwidth_hz = 10000000;
548            break;
549        case BANDWIDTH_AUTO:
550            if (freq < 474000000) {
551                bandwidth_hz = 7000000;
552            } else {
553                bandwidth_hz = 8000000;
554            }
555            break;
556        default:
557            bandwidth_hz = 0;
558            break;
559        }
560
561        MP_VERBOSE(priv, "tuning %s to %d Hz, bandwidth: %d\n",
562                   get_dvb_delsys(delsys), freq, bandwidth_hz);
563        break;
564    case SYS_DVBS2:
565#ifndef DVB_USE_S2API
566        MP_ERR(priv, "ERROR: Can not tune to S2 channel, S2-API not "
567                     "available, will tune to DVB-S!\n");
568#endif
569        /* PASSTROUTH. */
570    case SYS_DVBS:
571        if (freq > 2200000) {
572            // this must be an absolute frequency
573            if (freq < SLOF) {
574                freq -= LOF1;
575                hi_lo = 0;
576            } else {
577                freq -= LOF2;
578                hi_lo = 1;
579            }
580        }
581        MP_VERBOSE(priv, "tuning %s to Freq: %u, Pol: %c Srate: %d, "
582                   "22kHz: %s, LNB:  %d\n", get_dvb_delsys(delsys), freq,
583                   pol, srate, hi_lo ? "on" : "off", diseqc);
584
585        if (do_diseqc(fd_frontend, diseqc, (pol == 'V' ? 1 : 0), hi_lo) == 0) {
586            MP_VERBOSE(priv, "DISEQC setting succeeded\n");
587        } else {
588            MP_ERR(priv, "DISEQC setting failed\n");
589            return -1;
590        }
591
592        break;
593    case SYS_DVBC_ANNEX_A:
594    case SYS_DVBC_ANNEX_C:
595        MP_VERBOSE(priv, "tuning %s to %d, srate=%d\n",
596                   get_dvb_delsys(delsys), freq, srate);
597        break;
598#ifdef DVB_ATSC
599    case SYS_ATSC:
600    case SYS_DVBC_ANNEX_B:
601        MP_VERBOSE(priv, "tuning %s to %d, modulation=%d\n",
602                   get_dvb_delsys(delsys), freq, modulation);
603        break;
604#endif
605    default:
606        MP_VERBOSE(priv, "Unknown FE type. Aborting\n");
607        return 0;
608    }
609
610#ifdef DVB_USE_S2API
611    /* S2API is the DVB API new since 2.6.28.
612     * It is needed to tune to new delivery systems, e.g. DVB-S2.
613     * It takes a struct with a list of pairs of command + parameter.
614     */
615
616    /* Reset before tune. */
617    struct dtv_property p_clear[] = {
618        { .cmd = DTV_CLEAR },
619    };
620    struct dtv_properties cmdseq_clear = {
621        .num = 1,
622        .props = p_clear
623    };
624    if (ioctl(fd_frontend, FE_SET_PROPERTY, &cmdseq_clear) < 0) {
625        MP_ERR(priv, "FE_SET_PROPERTY DTV_CLEAR failed\n");
626    }
627
628    /* Tune. */
629    switch (delsys) {
630    case SYS_DVBS:
631    case SYS_DVBS2:
632        {
633            struct dtv_property p[] = {
634                { .cmd = DTV_DELIVERY_SYSTEM, .u.data = delsys },
635                { .cmd = DTV_FREQUENCY, .u.data = freq },
636                { .cmd = DTV_MODULATION, .u.data = modulation },
637                { .cmd = DTV_SYMBOL_RATE, .u.data = srate },
638                { .cmd = DTV_INNER_FEC, .u.data = HP_CodeRate },
639                { .cmd = DTV_INVERSION, .u.data = specInv },
640                { .cmd = DTV_ROLLOFF, .u.data = ROLLOFF_AUTO },
641                { .cmd = DTV_PILOT, .u.data = PILOT_AUTO },
642                { .cmd = DTV_TUNE },
643            };
644            struct dtv_properties cmdseq = {
645                .num = sizeof(p) / sizeof(p[0]),
646                .props = p
647            };
648            if (dvbv5_tune(priv, fd_frontend, delsys, &cmdseq) != 0) {
649                goto old_api;
650            }
651        }
652        break;
653    case SYS_DVBT:
654    case SYS_DVBT2:
655        {
656            struct dtv_property p[] = {
657                { .cmd = DTV_DELIVERY_SYSTEM, .u.data = delsys },
658                { .cmd = DTV_FREQUENCY, .u.data = freq },
659                { .cmd = DTV_MODULATION, .u.data = modulation },
660                { .cmd = DTV_SYMBOL_RATE, .u.data = srate },
661                { .cmd = DTV_CODE_RATE_HP, .u.data = HP_CodeRate },
662                { .cmd = DTV_CODE_RATE_LP, .u.data = LP_CodeRate },
663                { .cmd = DTV_INVERSION, .u.data = specInv },
664                { .cmd = DTV_BANDWIDTH_HZ, .u.data = bandwidth_hz },
665                { .cmd = DTV_TRANSMISSION_MODE, .u.data = TransmissionMode },
666                { .cmd = DTV_GUARD_INTERVAL, .u.data = guardInterval },
667                { .cmd = DTV_HIERARCHY, .u.data = hier },
668                { .cmd = DTV_STREAM_ID, .u.data = stream_id },
669                { .cmd = DTV_TUNE },
670            };
671            struct dtv_properties cmdseq = {
672                .num = sizeof(p) / sizeof(p[0]),
673                .props = p
674            };
675            if (dvbv5_tune(priv, fd_frontend, delsys, &cmdseq) != 0) {
676                goto old_api;
677            }
678        }
679        break;
680    case SYS_DVBC_ANNEX_A:
681    case SYS_DVBC_ANNEX_C:
682        {
683            struct dtv_property p[] = {
684                { .cmd = DTV_DELIVERY_SYSTEM, .u.data = delsys },
685                { .cmd = DTV_FREQUENCY, .u.data = freq },
686                { .cmd = DTV_MODULATION, .u.data = modulation },
687                { .cmd = DTV_SYMBOL_RATE, .u.data = srate },
688                { .cmd = DTV_INNER_FEC, .u.data = HP_CodeRate },
689                { .cmd = DTV_INVERSION, .u.data = specInv },
690                { .cmd = DTV_TUNE },
691            };
692            struct dtv_properties cmdseq = {
693                .num = sizeof(p) / sizeof(p[0]),
694                .props = p
695            };
696            if (dvbv5_tune(priv, fd_frontend, delsys, &cmdseq) != 0) {
697                goto old_api;
698            }
699        }
700        break;
701#ifdef DVB_ATSC
702    case SYS_ATSC:
703    case SYS_DVBC_ANNEX_B:
704        {
705            struct dtv_property p[] = {
706                { .cmd = DTV_DELIVERY_SYSTEM, .u.data = delsys },
707                { .cmd = DTV_FREQUENCY, .u.data = freq },
708                { .cmd = DTV_INVERSION, .u.data = specInv },
709                { .cmd = DTV_MODULATION, .u.data = modulation },
710                { .cmd = DTV_TUNE },
711            };
712            struct dtv_properties cmdseq = {
713                .num = sizeof(p) / sizeof(p[0]),
714                .props = p
715            };
716            if (dvbv5_tune(priv, fd_frontend, delsys, &cmdseq) != 0) {
717                goto old_api;
718            }
719        }
720        break;
721#endif
722    }
723
724    int tune_status = check_status(priv, fd_frontend, timeout);
725    if (tune_status != 0) {
726        MP_ERR(priv, "ERROR locking to channel when tuning with S2API, clearing and falling back to DVBv3-tuning.\n");
727        if (ioctl(fd_frontend, FE_SET_PROPERTY, &cmdseq_clear) < 0) {
728            MP_ERR(priv, "FE_SET_PROPERTY DTV_CLEAR failed\n");
729        }
730        goto old_api;
731    } else {
732        return tune_status;
733    }
734
735old_api:
736#endif
737
738    MP_VERBOSE(priv, "Tuning via DVB-API version 3.\n");
739
740    if (stream_id != NO_STREAM_ID_FILTER && stream_id != 0) {
741        MP_ERR(priv, "DVB-API version 3 does not support stream_id (PLP).\n");
742        return -1;
743    }
744    memset(&feparams, 0x00, sizeof(feparams));
745    feparams.frequency = freq;
746    feparams.inversion = specInv;
747
748    switch (delsys) {
749    case SYS_DVBT:
750    case SYS_DVBT2:
751        feparams.u.ofdm.bandwidth = bandwidth;
752        feparams.u.ofdm.code_rate_HP = HP_CodeRate;
753        feparams.u.ofdm.code_rate_LP = LP_CodeRate;
754        feparams.u.ofdm.constellation = modulation;
755        feparams.u.ofdm.transmission_mode = TransmissionMode;
756        feparams.u.ofdm.guard_interval = guardInterval;
757        feparams.u.ofdm.hierarchy_information = hier;
758        break;
759    case SYS_DVBS:
760    case SYS_DVBS2:
761        feparams.u.qpsk.symbol_rate = srate;
762        feparams.u.qpsk.fec_inner = HP_CodeRate;
763        break;
764    case SYS_DVBC_ANNEX_A:
765    case SYS_DVBC_ANNEX_C:
766        feparams.u.qam.symbol_rate = srate;
767        feparams.u.qam.fec_inner = HP_CodeRate;
768        feparams.u.qam.modulation = modulation;
769        break;
770#ifdef DVB_ATSC
771    case SYS_ATSC:
772    case SYS_DVBC_ANNEX_B:
773        feparams.u.vsb.modulation = modulation;
774        break;
775#endif
776    }
777
778    if (ioctl(fd_frontend, FE_SET_FRONTEND, &feparams) < 0) {
779        MP_ERR(priv, "ERROR tuning channel\n");
780        return -1;
781    }
782
783    return check_status(priv, fd_frontend, timeout);
784}

786int dvb_tune(dvb_priv_t *priv, unsigned int delsys,
787             int freq, char pol, int srate, int diseqc,
788             int stream_id, fe_spectral_inversion_t specInv,
789             fe_modulation_t modulation, fe_guard_interval_t guardInterval,
790             fe_transmit_mode_t TransmissionMode, fe_bandwidth_t bandWidth,
791             fe_code_rate_t HP_CodeRate,
792             fe_code_rate_t LP_CodeRate, fe_hierarchy_t hier,
793             int timeout)
794{
795    MP_INFO(priv, "dvb_tune %s Freq: %lu\n",
796            get_dvb_delsys(delsys), (long unsigned int) freq);
797
798    dvb_state_t* state = priv->state;
799
800    int ris = tune_it(priv, state->fe_fd, delsys, freq, srate, pol,
801                      stream_id, specInv, diseqc, modulation,
802                      HP_CodeRate, TransmissionMode, guardInterval,
803                      bandWidth, LP_CodeRate, hier, timeout);
804
805    if (ris != 0)
806        MP_INFO(priv, "dvb_tune, TUNING FAILED\n");
807
808    return ris == 0;
809}
