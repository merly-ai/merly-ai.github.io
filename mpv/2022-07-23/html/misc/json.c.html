18/* JSON parser:
19 *
20 * Unlike standard JSON, \u escapes don't allow you to specify UTF-16 surrogate
21 * pairs. There may be some differences how numbers are parsed (this parser
22 * doesn't verify what's passed to strtod(), and also prefers parsing numbers
23 * as integers with stroll() if possible).
24 *
25 * It has some non-standard extensions which shouldn't conflict with JSON:
26 *  - a list or object item can have a trailing ","
27 *  - object syntax accepts "=" in addition of ":"
28 *  - object keys can be unquoted, if they start with a character in [A-Za-z_]
29 *    and contain only characters in [A-Za-z0-9_]
30 *  - byte escapes with "\xAB" are allowed (with AB being a 2 digit hex number)
31 *
32 * Also see: http://tools.ietf.org/html/rfc8259
33 *
34 * JSON writer:
35 *
36 * Doesn't insert whitespace. It's literally a waste of space.
37 *
38 * Can output invalid UTF-8, if input is invalid UTF-8. Consumers are supposed
39 * to deal with somehow: either by using byte-strings for JSON, or by running
40 * a "fixup" pass on the input data. The latter could for example change
41 * invalid UTF-8 sequences to replacement characters.
42 */
43

57static bool eat_c(char **s, char c)
58{
59    if (**s == c) {
60        *s += 1;
61        return true;
62    }
63    return false;
64}

66static void eat_ws(char **src)
67{
68    while (1) {
69        char c = **src;
70        if (c != ' ' && c != '\t' && c != '\n' && c != '\r')
71            return;
72        *src += 1;
73    }
74}

76void json_skip_whitespace(char **src)
77{
78    eat_ws(src);
79}

81static int read_id(void *ta_parent, struct mpv_node *dst, char **src)
82{
83    char *start = *src;
84    if (!mp_isalpha(**src) && **src != '_')
85        return -1;
86    while (mp_isalnum(**src) || **src == '_')
87        *src += 1;
88    if (**src == ' ') {
89        **src = '\0'; // we're allowed to mutate it => can avoid the strndup
90        *src += 1;
91    } else {
92        start = talloc_strndup(ta_parent, start, *src - start);
93    }
94    dst->format = MPV_FORMAT_STRING;
95    dst->u.string = start;
96    return 0;
97}

99static int read_str(void *ta_parent, struct mpv_node *dst, char **src)
100{
101    if (!eat_c(src, '"'))
102        return -1; // not a string
103    char *str = *src;
104    char *cur = str;
105    bool has_escapes = false;
106    while (cur[0] && cur[0] != '"') {
107        if (cur[0] == '\\') {
108            has_escapes = true;
109            // skip >\"< and >\\< (latter to handle >\\"< correctly)
110            if (cur[1] == '"' || cur[1] == '\\')
111                cur++;
112        }
113        cur++;
114    }
115    if (cur[0] != '"')
116        return -1; // invalid termination
117    // Mutate input string so we have a null-terminated string to the literal.
118    // This is a stupid micro-optimization, so we can avoid allocation.
119    cur[0] = '\0';
120    *src = cur + 1;
121    if (has_escapes) {
122        bstr unescaped = {0};
123        bstr r = bstr0(str);
124        if (!mp_append_escaped_string(ta_parent, &unescaped, &r))
125            return -1; // broken escapes
126        str = unescaped.start; // the function guarantees null-termination
127    }
128    dst->format = MPV_FORMAT_STRING;
129    dst->u.string = str;
130    return 0;
131}

133static int read_sub(void *ta_parent, struct mpv_node *dst, char **src,
134                    int max_depth)
135{
136    bool is_arr = eat_c(src, '[');
137    bool is_obj = !is_arr && eat_c(src, '{');
138    if (!is_arr && !is_obj)
139        return -1; // not an array or object
140    char term = is_obj ? '}' : ']';
141    struct mpv_node_list *list = talloc_zero(ta_parent, struct mpv_node_list);
142    while (1) {
143        eat_ws(src);
144        if (eat_c(src, term))
145            break;
146        if (list->num > 0 && !eat_c(src, ','))
147            return -1; // missing ','
148        eat_ws(src);
149        // non-standard extension: allow a trailing ","
150        if (eat_c(src, term))
151            break;
152        if (is_obj) {
153            struct mpv_node keynode;
154            // non-standard extension: allow unquoted strings as keys
155            if (read_id(list, &keynode, src) < 0 &&
156                read_str(list, &keynode, src) < 0)
157                return -1; // key is not a string
158            eat_ws(src);
159            // non-standard extension: allow "=" instead of ":"
160            if (!eat_c(src, ':') && !eat_c(src, '='))
161                return -1; // ':' missing
162            eat_ws(src);
163            MP_TARRAY_GROW(list, list->keys, list->num);
164            list->keys[list->num] = keynode.u.string;
165        }
166        MP_TARRAY_GROW(list, list->values, list->num);
167        if (json_parse(ta_parent, &list->values[list->num], src, max_depth) < 0)
168            return -1;
169        list->num++;
170    }
171    dst->format = is_obj ? MPV_FORMAT_NODE_MAP : MPV_FORMAT_NODE_ARRAY;
172    dst->u.list = list;
173    return 0;
174}

176/* Parse the string in *src as JSON, and write the result into *dst.
177 * max_depth limits the recursion and JSON tree depth.
178 * Warning: this overwrites the input string (what *src points to)!
179 * Returns:
180 *   0: success, *dst is valid, *src points to the end (the caller must check
181 *      whether *src really terminates)
182 *  -1: failure, *dst is invalid, there may be dead allocs under ta_parent
183 *      (ta_free_children(ta_parent) is the only way to free them)
184 * The input string can be mutated in both cases. *dst might contain string
185 * elements, which point into the (mutated) input string.
186 */
187int json_parse(void *ta_parent, struct mpv_node *dst, char **src, int max_depth)
188{
189    max_depth -= 1;
190    if (max_depth < 0)
191        return -1;
192
193    eat_ws(src);
194
195    char c = **src;
196    if (!c)
197        return -1; // early EOF
198    if (c == 'n' && strncmp(*src, "null", 4) == 0) {
199        *src += 4;
200        dst->format = MPV_FORMAT_NONE;
201        return 0;
202    } else if (c == 't' && strncmp(*src, "true", 4) == 0) {
203        *src += 4;
204        dst->format = MPV_FORMAT_FLAG;
205        dst->u.flag = 1;
206        return 0;
207    } else if (c == 'f' && strncmp(*src, "false", 5) == 0) {
208        *src += 5;
209        dst->format = MPV_FORMAT_FLAG;
210        dst->u.flag = 0;
211        return 0;
212    } else if (c == '"') {
213        return read_str(ta_parent, dst, src);
214    } else if (c == '[' || c == '{') {
215        return read_sub(ta_parent, dst, src, max_depth);
216    } else if (c == '-' || (c >= '0' && c <= '9')) {
217        // The number could be either a float or an int. JSON doesn't make a
218        // difference, but the client API does.
219        char *nsrci = *src, *nsrcf = *src;
220        errno = 0;

230            dst->format = MPV_FORMAT_INT64; // long long is usually 64 bits
231            dst->u.int64 = numi;
232            return 0;
233        }

281static void add_indent(bstr *b, int indent)
282{
283    if (indent < 0)
284        return;
285    bstr_xappend(NULL, b, bstr0("\n"));
286    for (int n = 0; n < indent; n++)
287        bstr_xappend(NULL, b, bstr0(" "));
288}

290static int json_append(bstr *b, const struct mpv_node *src, int indent)
291{
292    switch (src->format) {
293    case MPV_FORMAT_NONE:
294        APPEND(b, "null");
295        return 0;
296    case MPV_FORMAT_FLAG:
297        APPEND(b, src->u.flag ? "true" : "false");
298        return 0;
299    case MPV_FORMAT_INT64:
300        bstr_xappend_asprintf(NULL, b, "%"PRId64, src->u.int64);
301        return 0;
302    case MPV_FORMAT_DOUBLE: {
303        const char *px = isfinite(src->u.double_) ? "" : "\"";
304        bstr_xappend_asprintf(NULL, b, "%s%f%s", px, src->u.double_, px);
305        return 0;
306    }
307    case MPV_FORMAT_STRING:
308        write_json_str(b, src->u.string);
309        return 0;
310    case MPV_FORMAT_NODE_ARRAY:
311    case MPV_FORMAT_NODE_MAP: {
312        struct mpv_node_list *list = src->u.list;
313        bool is_obj = src->format == MPV_FORMAT_NODE_MAP;
314        APPEND(b, is_obj ? "{" : "[");
315        int next_indent = indent >= 0 ? indent + 1 : -1;
316        for (int n = 0; n < list->num; n++) {
317            if (n)
318                APPEND(b, ",");
319            add_indent(b, next_indent);
320            if (is_obj) {
321                write_json_str(b, list->keys[n]);
322                APPEND(b, ":");
323            }
324            json_append(b, &list->values[n], next_indent);
325        }
326        add_indent(b, indent);
327        APPEND(b, is_obj ? "}" : "]");
328        return 0;
329    }
330    }
331    return -1; // unknown format
332}

334static int json_append_str(char **dst, struct mpv_node *src, int indent)
335{
336    bstr buffer = bstr0(*dst);
337    int r = json_append(&buffer, src, indent);
338    *dst = buffer.start;
339    return r;
340}

342/* Write the contents of *src as JSON, and append the JSON string to *dst.
343 * This will use strlen() to determine the start offset, and ta_get_size()
344 * and ta_realloc() to extend the memory allocation of *dst.
345 * Returns: 0 on success, <0 on failure.
346 */
347int json_write(char **dst, struct mpv_node *src)
348{
349    return json_append_str(dst, src, -1);
350}
