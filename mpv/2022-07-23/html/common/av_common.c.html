67AVCodecParameters *mp_codec_params_to_av(struct mp_codec_params *c)
68{
69    AVCodecParameters *avp = avcodec_parameters_alloc();
70    if (!avp)
71        return NULL;
72
73    // If we have lavf demuxer params, they overwrite by definition any others.
74    if (c->lav_codecpar) {
75        if (avcodec_parameters_copy(avp, c->lav_codecpar) < 0)
76            goto error;
77        return avp;
78    }
79
80    avp->codec_type = mp_to_av_stream_type(c->type);
81    avp->codec_id = mp_codec_to_av_codec_id(c->codec);
82    avp->codec_tag = c->codec_tag;
83    if (c->extradata_size) {
84        uint8_t *extradata = c->extradata;
85        int size = c->extradata_size;
86
87        if (avp->codec_id == AV_CODEC_ID_FLAC) {
88            // ffmpeg expects FLAC extradata to be just the STREAMINFO,
89            // so grab only that (and assume it'll be the first block)
90            if (size >= 8 && !memcmp(c->extradata, "fLaC", 4)) {
91                extradata += 8;
92                size = MPMIN(34, size - 8); // FLAC_STREAMINFO_SIZE
93            }
94        }
95
96        avp->extradata = av_mallocz(size + AV_INPUT_BUFFER_PADDING_SIZE);
97        if (!avp->extradata)
98            goto error;
99        avp->extradata_size = size;
100        memcpy(avp->extradata, extradata, size);
101    }
102    avp->bits_per_coded_sample = c->bits_per_coded_sample;
103
104    // Video only
105    avp->width = c->disp_w;
106    avp->height = c->disp_h;
107
108    // Audio only
109    avp->sample_rate = c->samplerate;
110    avp->bit_rate = c->bitrate;
111    avp->block_align = c->block_align;
112
113#if !HAVE_AV_CHANNEL_LAYOUT
114    avp->channels = c->channels.num;
115    if (!mp_chmap_is_unknown(&c->channels))
116        avp->channel_layout = mp_chmap_to_lavc(&c->channels);
117#else
118    mp_chmap_to_av_layout(&avp->ch_layout, &c->channels);
119#endif
120
121    return avp;
122error:
123    avcodec_parameters_free(&avp);
124    return NULL;
125}

127// Set avctx codec headers for decoding. Returns <0 on failure.
128int mp_set_avctx_codec_headers(AVCodecContext *avctx, struct mp_codec_params *c)
129{
130    enum AVMediaType codec_type = avctx->codec_type;
131    enum AVCodecID codec_id = avctx->codec_id;
132    AVCodecParameters *avp = mp_codec_params_to_av(c);
133    if (!avp)
134        return -1;
135
136    int r = avcodec_parameters_to_context(avctx, avp) < 0 ? -1 : 0;
137    avcodec_parameters_free(&avp);
138
139    if (avctx->codec_type != AVMEDIA_TYPE_UNKNOWN)
140        avctx->codec_type = codec_type;
141    if (avctx->codec_id != AV_CODEC_ID_NONE)
142        avctx->codec_id = codec_id;
143    return r;
144}

146// Pick a "good" timebase, which will be used to convert double timestamps
147// back to fractions for passing them through libavcodec.
148AVRational mp_get_codec_timebase(struct mp_codec_params *c)
149{
150    AVRational tb = {c->native_tb_num, c->native_tb_den};
151    if (tb.num < 1 || tb.den < 1) {
152        if (c->reliable_fps)
153            tb = av_inv_q(av_d2q(c->fps, 1000000));
154        if (tb.num < 1 || tb.den < 1)
155            tb = AV_TIME_BASE_Q;
156    }
157
158    // If the timebase is too coarse, raise its precision, or small adjustments
159    // to timestamps done between decoder and demuxer could be lost.
160    if (av_q2d(tb) > 0.001) {
161        AVRational r = av_div_q(tb, (AVRational){1, 1000});
162        tb.den *= (r.num + r.den - 1) / r.den;
163    }
164
165    av_reduce(&tb.num, &tb.den, tb.num, tb.den, INT_MAX);
166
167    if (tb.num < 1 || tb.den < 1)
168        tb = AV_TIME_BASE_Q;
169
170    return tb;
171}

173static AVRational get_def_tb(AVRational *tb)
174{
175    return tb && tb->num > 0 && tb->den > 0 ? *tb : AV_TIME_BASE_Q;
176}

178// Convert the mpv style timestamp (seconds as double) to a libavcodec style
179// timestamp (integer units in a given timebase).
180int64_t mp_pts_to_av(double mp_pts, AVRational *tb)
181{
182    AVRational b = get_def_tb(tb);
183    return mp_pts == MP_NOPTS_VALUE ? AV_NOPTS_VALUE : llrint(mp_pts / av_q2d(b));
184}
185
186// Inverse of mp_pts_to_av(). (The timebases must be exactly the same.)
187double mp_pts_from_av(int64_t av_pts, AVRational *tb)
188{
189    AVRational b = get_def_tb(tb);
190    return av_pts == AV_NOPTS_VALUE ? MP_NOPTS_VALUE : av_pts * av_q2d(b);
191}

193// Set dst from mpkt. Note that dst is not refcountable.
194// mpkt can be NULL to generate empty packets (used to flush delayed data).
195// Sets pts/dts using mp_pts_to_av(ts, tb). (Be aware of the implications.)
196// Set duration field only if tb is set.
197void mp_set_av_packet(AVPacket *dst, struct demux_packet *mpkt, AVRational *tb)
198{
199    av_init_packet(dst);
200    dst->data = mpkt ? mpkt->buffer : NULL;
201    dst->size = mpkt ? mpkt->len : 0;
202    /* Some codecs (ZeroCodec, some cases of PNG) may want keyframe info
203     * from demuxer. */
204    if (mpkt && mpkt->keyframe)
205        dst->flags |= AV_PKT_FLAG_KEY;
206    if (mpkt && mpkt->avpacket) {
207        dst->side_data = mpkt->avpacket->side_data;
208        dst->side_data_elems = mpkt->avpacket->side_data_elems;
209        if (dst->data == mpkt->avpacket->data)
210            dst->buf = mpkt->avpacket->buf;
211        dst->flags |= mpkt->avpacket->flags;
212    }
213    if (mpkt && tb && tb->num > 0 && tb->den > 0)
214        dst->duration = mpkt->duration / av_q2d(*tb);
215    dst->pts = mp_pts_to_av(mpkt ? mpkt->pts : MP_NOPTS_VALUE, tb);
216    dst->dts = mp_pts_to_av(mpkt ? mpkt->dts : MP_NOPTS_VALUE, tb);
217}

219void mp_set_avcodec_threads(struct mp_log *l, AVCodecContext *avctx, int threads)
220{
221    if (threads == 0) {
222        threads = av_cpu_count();
223        if (threads < 1) {
224            mp_warn(l, "Could not determine thread count to use, defaulting to 1.\n");
225            threads = 1;
226        } else {
227            mp_verbose(l, "Detected %d logical cores.\n", threads);
228            if (threads > 1)
229                threads += 1; // extra thread for better load balancing
230        }
231        // Apparently some libavcodec versions have or had trouble with more
232        // than 16 threads, and/or print a warning when using > 16.
233        threads = MPMIN(threads, 16);
234    }
235    mp_verbose(l, "Requesting %d threads for decoding.\n", threads);
236    avctx->thread_count = threads;
237}

239static void add_codecs(struct mp_decoder_list *list, enum AVMediaType type,
240                       bool decoders)
241{
242    void *iter = NULL;
243    for (;;) {
244        const AVCodec *cur = av_codec_iterate(&iter);
245        if (!cur)
246            break;
247        if (av_codec_is_decoder(cur) == decoders &&
248            (type == AVMEDIA_TYPE_UNKNOWN || cur->type == type))
249        {
250            mp_add_decoder(list, mp_codec_from_av_codec_id(cur->id),
251                           cur->name, cur->long_name);
252        }
253    }
254}

282int mp_codec_to_av_codec_id(const char *codec)
283{
284    int id = AV_CODEC_ID_NONE;
285    if (codec) {
286        const AVCodecDescriptor *desc = avcodec_descriptor_get_by_name(codec);
287        if (desc)
288            id = desc->id;
289        if (id == AV_CODEC_ID_NONE) {
290            const AVCodec *avcodec = avcodec_find_decoder_by_name(codec);
291            if (avcodec)
292                id = avcodec->id;
293        }
294    }
295    return id;
296}

306        if (avcodec)
307            name = avcodec->name;
308    }
309    return name;

336// If the name starts with "@", try to interpret it as a number, and set *name
337// to the name of the n-th parameter.
338static void resolve_positional_arg(void *avobj, char **name)
339{
340    if (!*name || (*name)[0] != '@' || !avobj)
341        return;
342
343    char *end = NULL;
344    int pos = strtol(*name + 1, &end, 10);
345    if (!end || *end)
346        return;
347
348    const AVOption *opt = NULL;
349    int offset = -1;
350    while (1) {
351        opt = av_opt_next(avobj, opt);
352        if (!opt)
353            return;
354        // This is what libavfilter's parser does to skip aliases.
355        if (opt->offset != offset && opt->type != AV_OPT_TYPE_CONST)
356            pos--;
357        if (pos < 0) {
358            *name = (char *)opt->name;
359            return;
360        }
361        offset = opt->offset;
362    }
363}
