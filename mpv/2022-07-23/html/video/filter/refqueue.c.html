29struct mp_refqueue {
30    struct mp_filter *filter;
31    struct mp_autoconvert *conv;
32    struct mp_pin *in, *out;
33
34    struct mp_image *in_format;
35
36    // Buffered frame in case of format changes.
37    struct mp_image *next;
38
39    int needed_past_frames;
40    int needed_future_frames;
41    int flags;
42
43    bool second_field; // current frame has to output a second field yet
44    bool eof;
45
46    // Queue of input frames, used to determine past/current/future frames.
47    // queue[0] is the newest frame, queue[num_queue - 1] the oldest.
48    struct mp_image **queue;
49    int num_queue;
50    // queue[pos] is the current frame, unless pos is an invalid index.
51    int pos;
52};

56static void refqueue_dtor(void *p)
57{
58    struct mp_refqueue *q = p;
59    mp_refqueue_flush(q);
60    mp_image_unrefp(&q->in_format);
61    talloc_free(q->conv->f);
62}

64struct mp_refqueue *mp_refqueue_alloc(struct mp_filter *f)
65{
66    struct mp_refqueue *q = talloc_zero(f, struct mp_refqueue);
67    talloc_set_destructor(q, refqueue_dtor);
68    q->filter = f;
69
70    q->conv = mp_autoconvert_create(f);
71    if (!q->conv)
72        abort();
73
74    q->in = q->conv->f->pins[1];
75    mp_pin_connect(q->conv->f->pins[0], f->ppins[0]);
76    q->out = f->ppins[1];
77
78    mp_refqueue_flush(q);
79    return q;
80}

87// The minimum number of frames required before and after the current frame.
88void mp_refqueue_set_refs(struct mp_refqueue *q, int past, int future)
89{
90    assert(past >= 0 && future >= 0);
91    q->needed_past_frames = past;
92    q->needed_future_frames = MPMAX(future, 1); // at least 1 for determining PTS
93}

101// Whether the current frame should be deinterlaced.
102bool mp_refqueue_should_deint(struct mp_refqueue *q)
103{
104    if (!mp_refqueue_has_output(q) || !(q->flags & MP_MODE_DEINT))
105        return false;
106
107    return (q->queue[q->pos]->fields & MP_IMGFIELD_INTERLACED) ||
108           !(q->flags & MP_MODE_INTERLACED_ONLY);
109}

111// Whether the current output frame (field) is the top field, bottom field
112// otherwise. (Assumes the caller forces deinterlacing.)
113bool mp_refqueue_is_top_field(struct mp_refqueue *q)
114{
115    if (!mp_refqueue_has_output(q))
116        return false;
117
118    return !!(q->queue[q->pos]->fields & MP_IMGFIELD_TOP_FIRST) ^ q->second_field;
119}

121// Whether top-field-first mode is enabled.
122bool mp_refqueue_top_field_first(struct mp_refqueue *q)
123{
124    if (!mp_refqueue_has_output(q))
125        return false;
126
127    return q->queue[q->pos]->fields & MP_IMGFIELD_TOP_FIRST;
128}

130// Discard all state.
131void mp_refqueue_flush(struct mp_refqueue *q)
132{
133    for (int n = 0; n < q->num_queue; n++)
134        talloc_free(q->queue[n]);
135    q->num_queue = 0;
136    q->pos = -1;
137    q->second_field = false;
138    q->eof = false;
139    mp_image_unrefp(&q->next);
140}

142static void mp_refqueue_add_input(struct mp_refqueue *q, struct mp_image *img)
143{
144    assert(img);
145
146    MP_TARRAY_INSERT_AT(q, q->queue, q->num_queue, 0, img);
147    q->pos++;
148
149    assert(q->pos >= 0 && q->pos < q->num_queue);
150}

152static bool mp_refqueue_need_input(struct mp_refqueue *q)
153{
154    return q->pos < q->needed_future_frames && !q->eof;
155}

157static bool mp_refqueue_has_output(struct mp_refqueue *q)
158{
159    return q->pos >= 0 && !mp_refqueue_need_input(q);
160}

162static bool output_next_field(struct mp_refqueue *q)
163{
164    if (q->second_field)
165        return false;
166    if (!(q->flags & MP_MODE_OUTPUT_FIELDS))
167        return false;
168    if (!mp_refqueue_should_deint(q))
169        return false;
170
171    assert(q->pos >= 0);
172
173    // If there's no (reasonable) timestamp, also skip the field.
174    if (q->pos == 0)
175        return false;
176
177    double pts = q->queue[q->pos]->pts;
178    double next_pts = q->queue[q->pos - 1]->pts;
179    if (pts == MP_NOPTS_VALUE || next_pts == MP_NOPTS_VALUE)
180        return false;
181
182    double frametime = next_pts - pts;
183    if (frametime <= 0.0 || frametime >= 1.0)
184        return false;
185
186    q->queue[q->pos]->pts = pts + frametime / 2;
187    q->second_field = true;
188    return true;
189}

191// Advance to next input frame (skips fields even in field output mode).
192static void mp_refqueue_next(struct mp_refqueue *q)
193{
194    if (!mp_refqueue_has_output(q))
195        return;
196
197    q->pos--;
198    q->second_field = false;
199
200    assert(q->pos >= -1 && q->pos < q->num_queue);
201
202    // Discard unneeded past frames.
203    while (q->num_queue - (q->pos + 1) > q->needed_past_frames) {
204        assert(q->num_queue > 0);
205        talloc_free(q->queue[q->num_queue - 1]);
206        q->num_queue--;
207    }
208
209    assert(q->pos >= -1 && q->pos < q->num_queue);
210}

212// Advance current field, depending on interlace flags.
213static void mp_refqueue_next_field(struct mp_refqueue *q)
214{
215    if (!mp_refqueue_has_output(q))
216        return;
217
218    if (!output_next_field(q))
219        mp_refqueue_next(q);
220}

222// Return a frame by relative position:
223//  -1: first past frame
224//   0: current frame
225//   1: first future frame
226// Caller doesn't get ownership. Return NULL if unavailable.
227struct mp_image *mp_refqueue_get(struct mp_refqueue *q, int pos)
228{
229    int i = q->pos - pos;
230    return i >= 0 && i < q->num_queue ? q->queue[i] : NULL;
231}
232
233// Same as mp_refqueue_get(), but return the frame which contains a field
234// relative to the current field's position.
235struct mp_image *mp_refqueue_get_field(struct mp_refqueue *q, int pos)
236{
237    // If the current field is the second field (conceptually), then pos=1
238    // needs to get the next frame. Similarly, pos=-1 needs to get the current
239    // frame, so round towards negative infinity.
240    int round = mp_refqueue_top_field_first(q) != mp_refqueue_is_top_field(q);
241    int frame = (pos < 0 ? pos - (1 - round) : pos + round) / 2;
242    return mp_refqueue_get(q, frame);
243}

250// Return non-NULL if a format change happened. A format change is defined by
251// a change in image parameters, using broad enough checks that happen to be
252// sufficient for all users of refqueue.
253// On format change, the refqueue transparently drains remaining frames, and
254// once that is done, this function returns a mp_image reference of the new
255// frame. Reinit the low level video processor based on it, and then leave the
256// reference alone and continue normally.
257// All frames returned in the future will have a compatible format.
258struct mp_image *mp_refqueue_execute_reinit(struct mp_refqueue *q)
259{
260    if (mp_refqueue_has_output(q) || !q->next)
261        return NULL;
262
263    struct mp_image *cur = q->next;
264    q->next = NULL;
265
266    mp_image_unrefp(&q->in_format);
267    mp_refqueue_flush(q);
268
269    q->in_format = mp_image_new_ref(cur);
270    if (!q->in_format)
271        abort();
272    mp_image_unref_data(q->in_format);
273
274    mp_refqueue_add_input(q, cur);
275    return cur;
276}

278// Main processing function. Call this in the filter process function.
279// Returns if enough input frames are available for filtering, and output pin
280// needs data; in other words, if this returns true, you render a frame and
281// output it.
282// If this returns true, you must call mp_refqueue_write_out_pin() to make
283// progress.
284bool mp_refqueue_can_output(struct mp_refqueue *q)
285{
286    if (!mp_pin_in_needs_data(q->out))
287        return false;
288
289    // Strictly return any output first to reduce latency.
290    if (mp_refqueue_has_output(q))
291        return true;
292
293    if (q->next) {
294        // Make it call again for mp_refqueue_execute_reinit().
295        mp_filter_internal_mark_progress(q->filter);
296        return false;
297    }
298
299    struct mp_frame frame = mp_pin_out_read(q->in);
300    if (frame.type == MP_FRAME_NONE)
301        return false;
302
303    if (frame.type == MP_FRAME_EOF) {
304        q->eof = true;
305        if (mp_refqueue_has_output(q)) {
306            mp_pin_out_unread(q->in, frame);
307            return true;
308        }
309        mp_pin_in_write(q->out, frame);
310        mp_refqueue_flush(q);
311        return false;
312    }
313
314    if (frame.type != MP_FRAME_VIDEO) {
315        MP_ERR(q->filter, "unsupported frame type\n");
316        mp_frame_unref(&frame);
317        mp_filter_internal_mark_failed(q->filter);
318        return false;
319    }
320
321    struct mp_image *img = frame.data;
322
323    if (!q->in_format || !!q->in_format->hwctx != !!img->hwctx ||
324        (img->hwctx && img->hwctx->data != q->in_format->hwctx->data) ||
325        !mp_image_params_equal(&q->in_format->params, &img->params))
326    {
327        q->next = img;
328        q->eof = true;
329        mp_filter_internal_mark_progress(q->filter);
330        return false;
331    }
332
333    mp_refqueue_add_input(q, img);
334
335    if (mp_refqueue_has_output(q))
336        return true;
337
338    mp_pin_out_request_data(q->in);
339    return false;
340}

342// (Accepts NULL for generic errors.)
343void mp_refqueue_write_out_pin(struct mp_refqueue *q, struct mp_image *mpi)
344{
345    if (mpi) {
346        mp_pin_in_write(q->out, MAKE_FRAME(MP_FRAME_VIDEO, mpi));
347    } else {
348        MP_WARN(q->filter, "failed to output frame\n");
349        mp_filter_internal_mark_failed(q->filter);
350    }
351    mp_refqueue_next_field(q);
352}
