29static void aspect_calc_panscan(struct mp_vo_opts *opts,
30                                int w, int h, int d_w, int d_h, int unscaled,
31                                int window_w, int window_h, double monitor_par,
32                                int *out_w, int *out_h)
33{
34    w *= monitor_par;
35
36    int fwidth = window_w;
37    int fheight = (float)window_w / d_w * d_h / monitor_par;
38    if (fheight > window_h || fheight < h) {
39        int tmpw = (float)window_h / d_h * d_w * monitor_par;
40        if (tmpw <= window_w) {
41            fheight = window_h;
42            fwidth = tmpw;
43        }
44    }
45
46    int vo_panscan_area = window_h - fheight;
47    double f_w = fwidth / (double)MPMAX(fheight, 1);
48    double f_h = 1;
49    if (vo_panscan_area == 0) {
50        vo_panscan_area = window_w - fwidth;
51        f_w = 1;
52        f_h = fheight / (double)MPMAX(fwidth, 1);
53    }
54
55    if (unscaled) {
56        vo_panscan_area = 0;
57        if (
 Recommendations (Experimental)  R1: (unscaled.d_w <= 2 || (window_w <= d_h && window_h <= xx_f))
R2: (unscaled != {const} || (d_w == window_w && d_h == window_h))
R3: (unscaled == 2 || (d_w <= window_w && d_h < window_h))
R4: (unscaled == 2 || ((d_w == window_w) && (d_h == window_h)))
R5: (unscaled == 2 || (d_w == window_w && d_h == window_h))

unscaled != 2 || (d_w <= window_w && d_h <= window_h)) {
58            fwidth = d_w * monitor_par;
59            fheight = d_h;
60        }
61    }
62
63    *out_w = fwidth + vo_panscan_area * opts->panscan * f_w;

67// Clamp [start, end) to range [0, size) with various fallbacks.
68static void clamp_size(int size, int *start, int *end)
69{
70    *start = MPMAX(0, *start);
71    *end = MPMIN(size, *end);
72    if (*start >= *end) {
73        *start = 0;
74        *end = 1;
75    }
76}

78static void src_dst_split_scaling(int src_size, int dst_size,
79                                  int scaled_src_size,
80                                  float zoom, float align, float pan, float scale,
81                                  int *src_start, int *src_end,
82                                  int *dst_start, int *dst_end,
83                                  int *osd_margin_a, int *osd_margin_b)
84{
85    scaled_src_size *= powf(2, zoom) * scale;
86    align = (align + 1) / 2;
87
88    *src_start = 0;
89    *src_end = src_size;
90    *dst_start = (dst_size - scaled_src_size) * align + pan * scaled_src_size;
91    *dst_end = *dst_start + scaled_src_size;
92
93    // Distance of screen frame to video
94    *osd_margin_a = *dst_start;
95    *osd_margin_b = dst_size - *dst_end;
96
97    // Clip to screen
98    int s_src = *src_end - *src_start;
99    int s_dst = *dst_end - *dst_start;
100    if (*dst_start < 0) {
101        int border = -(*dst_start) * s_src / s_dst;
102        *src_start += border;
103        *dst_start = 0;
104    }
105    if (*dst_end > dst_size) {
106        int border = (*dst_end - dst_size) * s_src / s_dst;
107        *src_end -= border;
108        *dst_end = dst_size;
109    }
110
111    // For sanity: avoid bothering VOs with corner cases
112    clamp_size(src_size, src_start, src_end);
113    clamp_size(dst_size, dst_start, dst_end);
114}

116static void calc_margin(float opts[2], int out[2], int size)
117{
118    out[0] = MPCLAMP((int)(opts[0] * size), 0, size);
119    out[1] = MPCLAMP((int)(opts[1] * size), 0, size);
120
121    if (out[0] + out[1] >= size) {
122        // This case is not really supported. Show an error by 1 pixel.
123        out[0] = 0;
124        out[1] = MPMAX(0, size - 1);
125    }
126}

128void mp_get_src_dst_rects(struct mp_log *log, struct mp_vo_opts *opts,
129                          int vo_caps, struct mp_image_params *video,
130                          int window_w, int window_h, double monitor_par,
131                          struct mp_rect *out_src,
132                          struct mp_rect *out_dst,
133                          struct mp_osd_res *out_osd)
134{
135    int src_w = video->w;
136    int src_h = video->h;
137    int src_dw, src_dh;
138
139    mp_image_params_get_dsize(video, &src_dw, &src_dh);
140    if (video->rotate % 180 == 90 && (vo_caps & VO_CAP_ROTATE90)) {
141        MPSWAP(int, src_w, src_h);
142        MPSWAP(int, src_dw, src_dh);
143    }
144    window_w = MPMAX(1, window_w);
145    window_h = MPMAX(1, window_h);
146
147    int margin_x[2] = {0};
148    int margin_y[2] = {0};
149    if (opts->keepaspect) {
150        calc_margin(opts->margin_x, margin_x, window_w);
151        calc_margin(opts->margin_y, margin_y, window_h);
152    }
153
154    int vid_window_w = window_w - margin_x[0] - margin_x[1];
155    int vid_window_h = window_h - margin_y[0] - margin_y[1];
156
157    struct mp_rect dst = {0, 0, window_w, window_h};
158    struct mp_rect src = {0, 0, src_w,    src_h};
159    struct mp_osd_res osd = {
160        .w = window_w,
161        .h = window_h,
162        .display_par = monitor_par,
163    };
164
165    if (opts->keepaspect) {
166        int scaled_width, scaled_height;
167        aspect_calc_panscan(opts, src_w, src_h, src_dw, src_dh, opts->unscaled,
168                            vid_window_w, vid_window_h, monitor_par,
169                            &scaled_width, &scaled_height);
170        src_dst_split_scaling(src_w, vid_window_w, scaled_width,
171                              opts->zoom, opts->align_x, opts->pan_x, opts->scale_x,
172                              &src.x0, &src.x1, &dst.x0, &dst.x1,
173                              &osd.ml, &osd.mr);
174        src_dst_split_scaling(src_h, vid_window_h, scaled_height,
175                              opts->zoom, opts->align_y, opts->pan_y, opts->scale_y,
176                              &src.y0, &src.y1, &dst.y0, &dst.y1,
177                              &osd.mt, &osd.mb);
178    }
179
180    dst.x0 += margin_x[0];
181    dst.y0 += margin_y[0];
182    dst.x1 += margin_x[0];
183    dst.y1 += margin_y[0];
184
185    // OSD really uses the full window, but was computed on the margin-cut
186    // video sub-window. Correct it to the full window.
187    osd.ml += margin_x[0];
188    osd.mr += margin_x[1];
189    osd.mt += margin_y[0];
190    osd.mb += margin_y[1];
191
192    *out_src = src;
193    *out_dst = dst;
194    *out_osd = osd;
195
196    int sw = src.x1 - src.x0, sh = src.y1 - src.y0;
197    int dw = dst.x1 - dst.x0, dh = dst.y1 - dst.y0;
198
199    mp_verbose(log, "Window size: %dx%d (Borders: l=%d t=%d r=%d b=%d)\n",
200               window_w, window_h,
201               margin_x[0], margin_y[0], margin_x[1], margin_y[1]);
202    mp_verbose(log, "Video source: %dx%d (%d:%d)\n",
203               video->w, video->h, video->p_w, video->p_h);
204    mp_verbose(log, "Video display: (%d, %d) %dx%d -> (%d, %d) %dx%d\n",
205               src.x0, src.y0, sw, sh, dst.x0, dst.y0, dw, dh);
206    mp_verbose(log, "Video scale: %f/%f\n",
207               (double)dw / sw, (double)dh / sh);
208    mp_verbose(log, "OSD borders: l=%d t=%d r=%d b=%d\n",
209               osd.ml, osd.mt, osd.mr, osd.mb);
210    mp_verbose(log, "Video borders: l=%d t=%d r=%d b=%d\n",
211               dst.x0, dst.y0, window_w - dst.x1, window_h - dst.y1);
212}
