184static LRESULT borderless_nchittest(struct vo_w32_state *w32, int x, int y)
185{
186    if (IsMaximized(w32->window))
187        return HTCLIENT;
188
189    POINT mouse = { x, y };
190    ScreenToClient(w32->window, &mouse);
191
192    // The horizontal frame should be the same size as the vertical frame,
193    // since the NONCLIENTMETRICS structure does not distinguish between them
194    int frame_size = GetSystemMetrics(SM_CXFRAME) +
195                     GetSystemMetrics(SM_CXPADDEDBORDER);
196    // The diagonal size handles are slightly wider than the side borders
197    int diagonal_width = frame_size * 2 + GetSystemMetrics(SM_CXBORDER);
198
199    // Hit-test top border
200    if (mouse.y < frame_size) {
201        if (mouse.x < diagonal_width)
202            return HTTOPLEFT;
203        if (mouse.x >= rect_w(w32->windowrc) - diagonal_width)
204            return HTTOPRIGHT;
205        return HTTOP;
206    }
207
208    // Hit-test bottom border
209    if (mouse.y >= rect_h(w32->windowrc) - frame_size) {
210        if (mouse.x < diagonal_width)
211            return HTBOTTOMLEFT;
212        if (mouse.x >= rect_w(w32->windowrc) - diagonal_width)
213            return HTBOTTOMRIGHT;
214        return HTBOTTOM;
215    }
216
217    // Hit-test side borders
218    if (mouse.x < frame_size)
219        return HTLEFT;
220    if (mouse.x >= rect_w(w32->windowrc) - frame_size)
221        return HTRIGHT;
222    return HTCLIENT;
223}

248{

253        key_state(VK_RMENU) && key_state(VK_LCONTROL);

260        res |= MP_KEY_MODIFIER_ALT;
261    return res;

269static int decode_utf16(struct vo_w32_state *w32, wchar_t c)
270{
271    // Decode UTF-16, keeping state in w32->high_surrogate
272    if (IS_HIGH_SURROGATE(c)) {
273        w32->high_surrogate = c;
274        return 0;
275    }
276    if (IS_LOW_SURROGATE(c)) {
277        if (!w32->high_surrogate) {
278            MP_ERR(w32, "Invalid UTF-16 input\n");
279            return 0;
280        }
281        int codepoint = decode_surrogate_pair(w32->high_surrogate, c);
282        w32->high_surrogate = 0;
283        return codepoint;
284    }
285    if (w32->high_surrogate != 0) {
286        w32->high_surrogate = 0;
287        MP_ERR(w32, "Invalid UTF-16 input\n");
288        return 0;
289    }
290
291    return c;
292}

294static void clear_keyboard_buffer(void)
295{
296    static const UINT vkey = VK_DECIMAL;
297    static const BYTE keys[256] = { 0 };
298    UINT scancode = MapVirtualKey(vkey, MAPVK_VK_TO_VSC);
299    wchar_t buf[10];
300    int ret = 0;
301
302    // Use the method suggested by Michael Kaplan to clear any pending dead
303    // keys from the current keyboard layout. See:
304    // https://web.archive.org/web/20101004154432/http://blogs.msdn.com/b/michkap/archive/2006/04/06/569632.aspx
305    // https://web.archive.org/web/20100820152419/http://blogs.msdn.com/b/michkap/archive/2007/10/27/5717859.aspx
306    do {
307        ret = ToUnicode(vkey, scancode, keys, buf, MP_ARRAY_SIZE(buf), 0);
308    } while (ret < 0);
309}

311static int to_unicode(UINT vkey, UINT scancode, const BYTE keys[256])
312{
313    // This wraps ToUnicode to be stateless and to return only one character
314
315    // Make the buffer 10 code units long to be safe, same as here:
316    // https://web.archive.org/web/20101013215215/http://blogs.msdn.com/b/michkap/archive/2006/03/24/559169.aspx
317    wchar_t buf[10] = { 0 };
318
319    // Dead keys aren't useful for key shortcuts, so clear the keyboard state
320    clear_keyboard_buffer();
321
322    int len = ToUnicode(vkey, scancode, keys, buf, MP_ARRAY_SIZE(buf), 0);
323
324    // Return the last complete UTF-16 code point. A negative return value
325    // indicates a dead key, however there should still be a non-combining
326    // version of the key in the buffer.
327    if (len < 0)
328        len = -len;
329    if (len >= 2 && IS_SURROGATE_PAIR(buf[len - 2], buf[len - 1]))
330        return decode_surrogate_pair(buf[len - 2], buf[len - 1]);
331    if (len >= 1)
332        return buf[len - 1];
333
334    return 0;
335}

337static int decode_key(struct vo_w32_state *w32, UINT vkey, UINT scancode)
338{
339    BYTE keys[256];
340    GetKeyboardState(keys);
341
342    // If mp_input_use_alt_gr is false, detect and remove AltGr so normal
343    // characters are generated. Note that AltGr is represented as
344    // LCONTROL+RMENU on Windows.
345    if ((keys[VK_RMENU] & 0x80) && (keys[VK_LCONTROL] & 0x80) &&
346        !mp_input_use_alt_gr(w32->input_ctx))
347    {
348        keys[VK_RMENU] = keys[VK_LCONTROL] = 0;
349        keys[VK_MENU] = keys[VK_LMENU];
350        keys[VK_CONTROL] = keys[VK_RCONTROL];
351    }
352
353    int c = to_unicode(vkey, scancode, keys);
354
355    // Some shift states prevent ToUnicode from working or cause it to produce
356    // control characters. If this is detected, remove modifiers until it
357    // starts producing normal characters.
358    if (c < 0x20 && (keys[VK_MENU] & 0x80)) {
359        keys[VK_LMENU] = keys[VK_RMENU] = keys[VK_MENU] = 0;
360        c = to_unicode(vkey, scancode, keys);
361    }
362    if (c < 0x20 && (keys[VK_CONTROL] & 0x80)) {
363        keys[VK_LCONTROL] = keys[VK_RCONTROL] = keys[VK_CONTROL] = 0;
364        c = to_unicode(vkey, scancode, keys);
365    }
366    if (c < 0x20)
367        return 0;
368
369    // Decode lone UTF-16 surrogates (VK_PACKET can generate these)
370    if (c < 0x10000)
371        return decode_utf16(w32, c);
372    return c;
373}

494    // https://msdn.microsoft.com/en-us/library/windows/desktop/dd183565%28v=vs.85%29.aspx
495    // mpv validates this value with a threshold of 1, so don't return exactly 1
496    if (dm.dmDisplayFrequency == 1)
497        return 0.0;

500    // highly likely that 23 represents 24/1.001, 59 represents 60/1.001, etc.
501    // A caller can always reproduce the original value by using floor.
502    double rv = dm.dmDisplayFrequency;
503    switch (dm.dmDisplayFrequency) {
504        case  23:
505        case  29:
506        case  47:
507        case  59:
508        case  71:
509        case  89:
510        case  95:
511        case 119:
512        case 143:
513            rv = (rv + 1) / 1.001;
514    }
515
516    return rv;
517}

705static bool snap_to_screen_edges(struct vo_w32_state *w32, RECT *rc)
706{
707    if (w32->parent || w32->current_fs || IsMaximized(w32->window))
708        return false;
709
710    if (!w32->opts->snap_window) {
711        w32->snapped = false;
712        return false;
713    }
714
715    RECT rect;
716    POINT cursor;
717    if (!GetWindowRect(w32->window, &rect) || !GetCursorPos(&cursor))
718        return false;
719    // Check if window is going to be aero-snapped
720    if (rect_w(*rc) != rect_w(rect) || rect_h(*rc) != rect_h(rect))
721        return false;
722
723    // Check if window has already been aero-snapped
724    WINDOWPLACEMENT wp = {0};
725    wp.length = sizeof(wp);
726    if (!GetWindowPlacement(w32->window, &wp))
727        return false;
728    RECT wr = wp.rcNormalPosition;
729    if (rect_w(*rc) != rect_w(wr) || rect_h(*rc) != rect_h(wr))
730        return false;
731
732    // Get the work area to let the window snap to taskbar
733    wr = get_working_area(w32);
734
735    // Check for invisible borders and adjust the work area size
736    RECT frame = {0};
737    if (DwmGetWindowAttribute(w32->window, DWMWA_EXTENDED_FRAME_BOUNDS,
738                              &frame, sizeof(RECT)) == S_OK) {
739        wr.left -= frame.left - rect.left;
740        wr.top -= frame.top - rect.top;
741        wr.right += rect.right - frame.right;
742        wr.bottom += rect.bottom - frame.bottom;
743    }
744
745    // Let the window to unsnap by changing its position,
746    // otherwise it will stick to the screen edges forever
747    rect = *rc;
748    if (w32->snapped) {
749        OffsetRect(&rect, cursor.x - rect.left - w32->snap_dx,
750                          cursor.y - rect.top - w32->snap_dy);
751    }
752
753    int threshold = (w32->dpi * 16) / 96;
754    bool snapped = false;
755    // Adjust X position
756    if (abs(rect.left - wr.left) < threshold) {
757        snapped = true;
758        OffsetRect(&rect, wr.left - rect.left, 0);
759    } else if (abs(rect.right - wr.right) < threshold) {
760        snapped = true;
761        OffsetRect(&rect, wr.right - rect.right, 0);
762    }
763    // Adjust Y position
764    if (abs(rect.top - wr.top) < threshold) {
765        snapped = true;
766        OffsetRect(&rect, 0, wr.top - rect.top);
767    } else if (abs(rect.bottom - wr.bottom) < threshold) {
768        snapped = true;
769        OffsetRect(&rect, 0, wr.bottom - rect.bottom);
770    }
771
772    if (!w32->snapped && snapped) {
773        w32->snap_dx = cursor.x - rc->left;
774        w32->snap_dy = cursor.y - rc->top;
775    }
776
777    w32->snapped = snapped;
778    *rc = rect;
779    return true;
780}

809// If rc is wider/taller than n_w/n_h, shrink rc size while keeping the center.
810// returns true if the rectangle was modified.
811static bool fit_rect_size(RECT *rc, long n_w, long n_h)
812{
813    // nothing to do if we already fit.
814    int o_w = rect_w(*rc), o_h = rect_h(*rc);
815    if (o_w <= n_w && o_h <= n_h)
816        return false;
817
818    // Apply letterboxing
819    const float o_asp = o_w / (float)MPMAX(o_h, 1);
820    const float n_asp = n_w / (float)MPMAX(n_h, 1);
821    if (o_asp > n_asp) {
822        n_h = n_w / o_asp;
823    } else {
824        n_w = n_h * o_asp;
825    }
826
827    // Calculate new position and save the rect
828    const int x = rc->left + o_w / 2 - n_w / 2;
829    const int y = rc->top + o_h / 2 - n_h / 2;
830    SetRect(rc, x, y, x + n_w, y + n_h);
831    return true;
832}

1009        // receives. It allows struct vo_w32_state to be passed from
1010        // CreateWindow's lpParam to the window procedure. However, as a
1011        // longstanding Windows bug, overlapped top-level windows will get a
1012        // WM_GETMINMAXINFO before WM_NCCREATE. This can be ignored.
1013        if (message != WM_NCCREATE)
1014            return DefWindowProcW(hWnd, message, wParam, lParam);

1017        w32 = cs->lpCreateParams;

1023    // playback glitches, since it is likely blocking the VO thread
1024    if (!w32->in_dispatch) {
1025        w32->in_dispatch = true;
1026        mp_dispatch_queue_process(w32->dispatch, 0);
1027        w32->in_dispatch = false;

1039                                   y - w32->windowrc.top);

1045        MP_DBG(w32, "move window: %d:%d\n", x, y);
1046        break;
1047    }
1048    case WM_MOVING: {

1050        RECT *rc = (RECT*)lParam;

1052            return TRUE;

1066        }
1067        break;

1070        break;
1071    case WM_SIZE: {
1072        if (w32->moving)

1101                                         &w32->opts->window_maximized);
1102            }
1103        }
1104

1328static void resize_child_win(HWND parent)
1329{
1330    // Check if an mpv window is a child of this window. This will not
1331    // necessarily be the case because the hook functions will run for all
1332    // windows on the parent window's thread.
1333    ATOM cls = get_window_class();
1334    HWND child = FindWindowExW(parent, NULL, (LPWSTR)MAKEINTATOM(cls), NULL);
1335    if (!child)
1336        return;
1337    // Make sure the window was created by this instance
1338    if (GetWindowLongPtrW(child, GWLP_HINSTANCE) != (LONG_PTR)HINST_THISCOMPONENT)
1339        return;
1340
1341    // Resize the mpv window to match its parent window's size
1342    RECT rm, rp;
1343    if (!GetClientRect(child, &rm))
1344        return;
1345    if (!GetClientRect(parent, &rp))
1346        return;
1347    if (EqualRect(&rm, &rp))
1348        return;
1349    SetWindowPos(child, NULL, 0, 0, rp.right, rp.bottom, SWP_ASYNCWINDOWPOS |
1350        SWP_NOACTIVATE | SWP_NOZORDER | SWP_NOOWNERZORDER | SWP_NOSENDCHANGING);
1351}

1416static void gui_thread_reconfig(void *ptr)
1417{
1418    struct vo_w32_state *w32 = ptr;
1419    struct vo *vo = w32->vo;
1420
1421    RECT r = get_working_area(w32);
1422    // for normal window which is auto-positioned (centered), center the window
1423    // rather than the content (by subtracting the borders from the work area)
1424    if (!w32->current_fs && !IsMaximized(w32->window) && w32->opts->border &&
1425        !w32->opts->geometry.xy_valid /* specific position not requested */)
1426    {
1427        subtract_window_borders(w32, w32->window, &r);
1428    }
1429    struct mp_rect screen = { r.left, r.top, r.right, r.bottom };
1430    struct vo_win_geometry geo;
1431
1432    RECT monrc = get_monitor_info(w32).rcMonitor;
1433    struct mp_rect mon = { monrc.left, monrc.top, monrc.right, monrc.bottom };
1434
1435    if (w32->dpi_scale == 0)
1436        force_update_display_info(w32);
1437
1438    vo_calc_window_geometry3(vo, &screen, &mon, w32->dpi_scale, &geo);
1439    vo_apply_window_geometry(vo, &geo);
1440
1441    bool reset_size = w32->o_dwidth != vo->dwidth ||
1442                      w32->o_dheight != vo->dheight;
1443
1444    w32->o_dwidth = vo->dwidth;
1445    w32->o_dheight = vo->dheight;
1446
1447    if (!w32->parent && !w32->window_bounds_initialized) {
1448        SetRect(&w32->windowrc, geo.win.x0, geo.win.y0,
1449                geo.win.x0 + vo->dwidth, geo.win.y0 + vo->dheight);
1450        w32->prev_windowrc = w32->windowrc;
1451        w32->window_bounds_initialized = true;
1452        w32->fit_on_screen = true;
1453        goto finish;
1454    }
1455
1456    // The rect which size is going to be modified.
1457    RECT *rc = &w32->windowrc;
1458
1459    // The desired size always matches the window size in wid mode.
1460    if (!reset_size || w32->parent) {
1461        GetClientRect(w32->window, &r);
1462        // Restore vo_dwidth and vo_dheight, which were reset in vo_config()

1464        vo->dheight = r.bottom;
1465    } else {
1466        if (w32->current_fs)
1467            rc = &w32->prev_windowrc;
1468        w32->fit_on_screen = true;
1469    }

1472    const int x = rc->left + rect_w(*rc) / 2 - vo->dwidth / 2;
1473    const int y = rc->top + rect_h(*rc) / 2 - vo->dheight / 2;
1474    SetRect(rc, x, y, x + vo->dwidth, y + vo->dheight);
1475
1476finish:
1477    reinit_window_state(w32);
1478}

1509    bool ole_ok = false;
1510    int res = 0;
1511
1512    mpthread_set_name("win32 window");
1513
1514    w32_api_load(w32);
1515
1516    // Disables the IME for windows on this thread
1517    if (w32->api.pImmDisableIME)
1518        w32->api.pImmDisableIME(0);
1519
1520    if (w32->opts->WinID >= 0)
1521        w32->parent = (HWND)(intptr_t)(w32->opts->WinID);
1522
1523    ATOM cls = get_window_class();
1524    if (w32->parent) {
1525        RECT r;
1526        GetClientRect(w32->parent, &r);
1527        CreateWindowExW(WS_EX_NOPARENTNOTIFY, (LPWSTR)MAKEINTATOM(cls), L"mpv",
1528                        WS_CHILD | WS_VISIBLE, 0, 0, r.right, r.bottom,
1529                        w32->parent, 0, HINST_THISCOMPONENT, w32);
1530
1531        // Install a hook to get notifications when the parent changes size
1532        if (w32->window)
1533            install_parent_hook(w32);
1534    } else {
1535        CreateWindowExW(0, (LPWSTR)MAKEINTATOM(cls), L"mpv",
1536                        update_style(w32, 0), CW_USEDEFAULT, SW_HIDE, 100, 100,
1537                        0, 0, HINST_THISCOMPONENT, w32);
1538    }
1539
1540    if (!w32->window) {
1541        MP_ERR(w32, "unable to create window!\n");
1542        goto done;
1543    }
1544
1545    if (SUCCEEDED(OleInitialize(NULL))) {
1546        ole_ok = true;
1547
1548        IDropTarget *dt = mp_w32_droptarget_create(w32->log, w32->input_ctx);
1549        RegisterDragDrop(w32->window, dt);
1550
1551        // ITaskbarList2 has the MarkFullscreenWindow method, which is used to
1552        // make sure the taskbar is hidden when mpv goes fullscreen
1553        if (SUCCEEDED(CoCreateInstance(&CLSID_TaskbarList, NULL,
1554                                       CLSCTX_INPROC_SERVER, &IID_ITaskbarList2,
1555                                       (void**)&w32->taskbar_list)))

1558                ITaskbarList2_Release(w32->taskbar_list);
1559                w32->taskbar_list = NULL;
1560            }
1561        }
1562
1563        // ITaskbarList3 has methods for status indication on taskbar buttons,
1564        // however that interface is only available on Win7/2008 R2 or newer
1565        if (SUCCEEDED(CoCreateInstance(&CLSID_TaskbarList, NULL,
1566                                       CLSCTX_INPROC_SERVER, &IID_ITaskbarList3,
1567                                       (void**)&w32->taskbar_list3)))
1568        {
1569            if (FAILED(ITaskbarList3_HrInit(w32->taskbar_list3))) {
1570                ITaskbarList3_Release(w32->taskbar_list3);
1571                w32->taskbar_list3 = NULL;
1572            } else {
1573                w32->tbtnCreatedMsg = RegisterWindowMessage(L"TaskbarButtonCreated");
1574            }
1575        }
1576    } else {
1577        MP_ERR(w32, "Failed to initialize OLE/COM\n");
1578    }
1579
1580    w32->tracking   = FALSE;
1581    w32->trackEvent = (TRACKMOUSEEVENT){
1582        .cbSize    = sizeof(TRACKMOUSEEVENT),
1583        .dwFlags   = TME_LEAVE,
1584        .hwndTrack = w32->window,
1585    };
1586
1587    if (w32->parent)
1588        EnableWindow(w32->window, 0);
1589
1590    w32->cursor_visible = true;
1591    w32->moving = false;
1592    w32->snapped = false;
1593    w32->snap_dx = w32->snap_dy = 0;
1594
1595    mp_dispatch_set_wakeup_fn(w32->dispatch, wakeup_gui_thread, w32);
1596
1597    res = 1;
1598done:
1599
1600    mp_rendezvous(w32, res); // init barrier
1601
1602    // This blocks until the GUI thread is to be exited.
1603    if (res)
1604        run_message_loop(w32);
1605
1606    MP_VERBOSE(w32, "uninit\n");
1607
1608    remove_parent_hook(w32);
1609    if (w32->window && !w32->destroyed)
1610        DestroyWindow(w32->window);
1611    if (w32->taskbar_list)
1612        ITaskbarList2_Release(w32->taskbar_list);
1613    if (w32->taskbar_list3)
1614        ITaskbarList3_Release(w32->taskbar_list3);
1615    if (ole_ok)
1616        OleUninitialize();
1617    SetThreadExecutionState(ES_CONTINUOUS);
1618    return NULL;
1619}

1706static int gui_thread_control(struct vo_w32_state *w32, int request, void *arg)
1707{
1708    switch (request) {
1709    case VOCTRL_VO_OPTS_CHANGED: {
1710        void *changed_option;
1711
1712        while (m_config_cache_get_next_changed(w32->opts_cache,
1713                                               &changed_option))
1714        {
1715            struct mp_vo_opts *vo_opts = w32->opts_cache->opts;
1716
1717            if (changed_option == &vo_opts->fullscreen) {
1718                reinit_window_state(w32);
1719            } else if (changed_option == &vo_opts->ontop) {
1720                update_window_state(w32);
1721            } else if (changed_option == &vo_opts->border) {
1722                update_window_style(w32);
1723                update_window_state(w32);
1724            } else if (changed_option == &vo_opts->window_minimized) {
1725                update_minimized_state(w32);
1726            } else if (changed_option == &vo_opts->window_maximized) {
1727                update_maximized_state(w32);
1728            }
1729        }
1730
1731        return VO_TRUE;
1732    }
1733    case VOCTRL_GET_HIDPI_SCALE: {
1734        *(double *)arg = w32->dpi_scale;
1735        return VO_TRUE;
1736    }
1737    case VOCTRL_GET_UNFS_WINDOW_SIZE: {
1738        int *s = arg;
1739
1740        if (!w32->window_bounds_initialized)
1741            return VO_FALSE;
1742
1743        RECT *rc = w32->current_fs ? &w32->prev_windowrc : &w32->windowrc;
1744        s[0] = rect_w(*rc) / w32->dpi_scale;
1745        s[1] = rect_h(*rc) / w32->dpi_scale;
1746        return VO_TRUE;
1747    }
1748    case VOCTRL_SET_UNFS_WINDOW_SIZE: {
1749        int *s = arg;
1750
1751        if (!w32->window_bounds_initialized)
1752            return VO_FALSE;
1753
1754        s[0] *= w32->dpi_scale;
1755        s[1] *= w32->dpi_scale;
1756
1757        RECT *rc = w32->current_fs ? &w32->prev_windowrc : &w32->windowrc;
1758        const int x = rc->left + rect_w(*rc) / 2 - s[0] / 2;
1759        const int y = rc->top + rect_h(*rc) / 2 - s[1] / 2;
1760        SetRect(rc, x, y, x + s[0], y + s[1]);
1761
1762        w32->fit_on_screen = true;
1763        reinit_window_state(w32);
1764        return VO_TRUE;
1765    }
1766    case VOCTRL_SET_CURSOR_VISIBILITY:
1767        w32->cursor_visible = *(bool *)arg;
1768
1769        if (w32->can_set_cursor && w32->tracking) {
1770            if (w32->cursor_visible)
1771                SetCursor(LoadCursor(NULL, IDC_ARROW));
1772            else
1773                SetCursor(NULL);
1774        }
1775        return VO_TRUE;
1776    case VOCTRL_KILL_SCREENSAVER:
1777        w32->disable_screensaver = true;
1778        SetThreadExecutionState(ES_CONTINUOUS | ES_SYSTEM_REQUIRED |
1779                                ES_DISPLAY_REQUIRED);
1780        return VO_TRUE;
1781    case VOCTRL_RESTORE_SCREENSAVER:
1782        w32->disable_screensaver = false;
1783        SetThreadExecutionState(ES_CONTINUOUS);
1784        return VO_TRUE;
1785    case VOCTRL_UPDATE_WINDOW_TITLE: {
1786        wchar_t *title = mp_from_utf8(NULL, (char *)arg);
1787        SetWindowTextW(w32->window, title);
1788        talloc_free(title);
1789        return VO_TRUE;
1790    }
1791    case VOCTRL_UPDATE_PLAYBACK_STATE: {
1792        w32->current_pstate = *(struct voctrl_playback_state *)arg;
1793
1794        update_playback_state(w32);
1795        return VO_TRUE;
1796    }
1797    case VOCTRL_GET_DISPLAY_FPS:
1798        update_display_info(w32);
1799        *(double*) arg = w32->display_fps;
1800        return VO_TRUE;
1801    case VOCTRL_GET_DISPLAY_RES: ;
1802        RECT r = get_screen_area(w32);
1803        ((int *)arg)[0] = r.right;
1804        ((int *)arg)[1] = r.bottom;
1805        return VO_TRUE;
1806    case VOCTRL_GET_DISPLAY_NAMES:
1807        *(char ***)arg = get_disp_names(w32);
1808        return VO_TRUE;
1809    case VOCTRL_GET_ICC_PROFILE:
1810        update_display_info(w32);
1811        if (w32->color_profile) {
1812            bstr *p = arg;
1813            *p = stream_read_file(w32->color_profile, NULL,
1814                w32->vo->global, 100000000); // 100 MB
1815            return p->len ? VO_TRUE : VO_FALSE;
1816        }
1817        return VO_FALSE;
1818    case VOCTRL_GET_FOCUSED:
1819        *(bool *)arg = w32->focused;
1820        return VO_TRUE;
1821    }
1822    return VO_NOTIMPL;
1823}

1825static void do_control(void *ptr)
1826{
1827    void **p = ptr;
1828    struct vo_w32_state *w32 = p[0];
1829    int *events = p[1];
1830    int request = *(int *)p[2];
1831    void *arg = p[3];
1832    int *ret = p[4];
1833    *ret = gui_thread_control(w32, request, arg);
1834    *events |= atomic_fetch_and(&w32->event_flags, 0);
1835    // Safe access, since caller (owner of vo) is blocked.
1836    if (*events & VO_EVENT_RESIZE) {
1837        w32->vo->dwidth = rect_w(w32->windowrc);
1838        w32->vo->dheight = rect_h(w32->windowrc);
1839    }
