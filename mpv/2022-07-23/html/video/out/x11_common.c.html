402        int event_base, error_base;
403        if (!XRRQueryExtension(x11->display, &event_base, &error_base)) {
404            MP_VERBOSE(x11, "Couldn't init Xrandr.\n");

437            int nvidia = bstr_find0(provider_name, "nvidia");
438            int radeon = bstr_find0(provider_name, "radeon");

454        if (!out || !out->crtc)

457        if (!crtc)

1311                                               present_event->msc);

1313            }
1314            break;
