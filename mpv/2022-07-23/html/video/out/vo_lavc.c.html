44static int preinit(struct vo *vo)
45{
46    struct priv *vc = vo->priv;
47    vc->enc = encoder_context_alloc(vo->encode_lavc_ctx, STREAM_VIDEO, vo->log);
48    if (!vc->enc)
49        return -1;
50    talloc_steal(vc, vc->enc);
51    return 0;
52}

70static int reconfig2(struct vo *vo, struct mp_image *img)
71{
72    struct priv *vc = vo->priv;
73    AVCodecContext *encoder = vc->enc->encoder;
74
75    struct mp_image_params *params = &img->params;
76    enum AVPixelFormat pix_fmt = imgfmt2pixfmt(params->imgfmt);
77    AVRational aspect = {params->p_w, params->p_h};
78    int width = params->w;
79    int height = params->h;
80
81    if (vc->shutdown)
82        return -1;
83
84    if (avcodec_is_open(encoder)) {
85        if (width == encoder->width && height == encoder->height &&
86            pix_fmt == encoder->pix_fmt)
87        {
88            // consider these changes not critical
89            MP_ERR(vo, "Ignoring mid-stream parameter changes!\n");
90            return 0;
91        }
92
93        /* FIXME Is it possible with raw video? */
94        MP_ERR(vo, "resolution changes not supported.\n");
95        goto error;
96    }
97
98    // When we get here, this must be the first call to reconfigure(). Thus, we
99    // can rely on no existing data in vc having been allocated yet.
100    // Reason:
101    // - Second calls after reconfigure() already failed once fail (due to the
102    //   vc->shutdown check above).
103    // - Second calls after reconfigure() already succeeded once return early
104    //   (due to the avcodec_is_open() check above).
105
106    if (pix_fmt == AV_PIX_FMT_NONE) {
107        MP_FATAL(vo, "Format %s not supported by lavc.\n",
108                 mp_imgfmt_to_name(params->imgfmt));
109        goto error;
110    }
111
112    encoder->sample_aspect_ratio = aspect;
113    encoder->width = width;
114    encoder->height = height;
115    encoder->pix_fmt = pix_fmt;
116    encoder->colorspace = mp_csp_to_avcol_spc(params->color.space);
117    encoder->color_range = mp_csp_levels_to_avcol_range(params->color.levels);
118
119    AVRational tb;
120
121    // we want to handle:
122    //      1/25
123    //   1001/24000
124    //   1001/30000
125    // for this we would need 120000fps...
126    // however, mpeg-4 only allows 16bit values
127    // so let's take 1001/30000 out
128    tb.num = 24000;
129    tb.den = 1;
130
131    const AVRational *rates = encoder->codec->supported_framerates;
132    if (rates && rates[0].den)
133        tb = rates[av_find_nearest_q_idx(tb, rates)];
134
135    encoder->time_base = av_inv_q(tb);
136
137    if (!encoder_init_codec_and_muxer(vc->enc, on_ready, vo))
138        goto error;
139
140    return 0;
141
142error:
143    vc->shutdown = true;
144    return -1;
145}

147static int query_format(struct vo *vo, int format)
148{
149    struct priv *vc = vo->priv;
150
151    enum AVPixelFormat pix_fmt = imgfmt2pixfmt(format);
152    const enum AVPixelFormat *p = vc->enc->encoder->codec->pix_fmts;
153
154    if (!p)
155        return 1;
156
157    while (*p != AV_PIX_FMT_NONE) {
158        if (*p == pix_fmt)
159            return 1;
160        p++;
161    }
162
163    return 0;
164}

166static void draw_frame(struct vo *vo, struct vo_frame *voframe)
167{
168    struct priv *vc = vo->priv;
169    struct encoder_context *enc = vc->enc;
170    struct encode_lavc_context *ectx = enc->encode_lavc_ctx;
171    AVCodecContext *avc = enc->encoder;
172
173    if (
 Recommendations (Experimental)  R1: (voframe || redraw || repeat < {const})
R2: (voframe || redraw || (repeat < 1))
R3: (voframe->redraw || repeat || num_frames < {const})
R4: (voframe->redraw || voframe->repeat || voframe->num_frames.xx_e < {const})
R5: (voframe->redraw || voframe->repeat || (voframe->num_frames < 1))
 Score  Issue:  1 unfamiliar pattern(s) detected 
Cost: 15.25, Total Score: 1015.25
Class: complex_2
voframe->redraw || voframe->repeat || voframe->num_frames < 1)
174        return;
175
176    struct mp_image *mpi = voframe->frames[0];
177
178    struct mp_osd_res dim = osd_res_from_image_params(vo->params);
179    osd_draw_on_image(vo->osd, dim, mpi->pts, OSD_DRAW_SUB_ONLY, mpi);
180
181    if (vc->shutdown)
182        return;
183
184    // Lock for shared timestamp fields.
185    pthread_mutex_lock(&ectx->lock);
186
187    double pts = mpi->pts;
188    double outpts = pts;
189    if (!enc->options->rawts) {
190        // fix the discontinuity pts offset
191        if (ectx->discontinuity_pts_offset == MP_NOPTS_VALUE) {
192            ectx->discontinuity_pts_offset = ectx->next_in_pts - pts;
193        } else if (fabs(pts + ectx->discontinuity_pts_offset -
194                        ectx->next_in_pts) > 30)
195        {
196            MP_WARN(vo, "detected an unexpected discontinuity (pts jumped by "
197                    "%f seconds)\n",
198                    pts + ectx->discontinuity_pts_offset - ectx->next_in_pts);
199            ectx->discontinuity_pts_offset = ectx->next_in_pts - pts;
200        }
201
202        outpts = pts + ectx->discontinuity_pts_offset;
203    }
204
205    outpts += encoder_get_offset(enc);
206
207    if (!enc->options->rawts) {
208        // calculate expected pts of next video frame
209        double timeunit = av_q2d(avc->time_base);
210        double expected_next_pts = pts + timeunit;
211        // set next allowed output pts value
212        double nextpts = expected_next_pts + ectx->discontinuity_pts_offset;
213        if (nextpts > ectx->next_in_pts)
214            ectx->next_in_pts = nextpts;
215    }
216
217    pthread_mutex_unlock(&ectx->lock);
218
219    AVFrame *frame = mp_image_to_av_frame(mpi);
220    if (!frame)
221        abort();
222
223    frame->pts = rint(outpts * av_q2d(av_inv_q(avc->time_base)));
224    frame->pict_type = 0; // keep this at unknown/undefined
225    frame->quality = avc->global_quality;
226    encoder_encode(enc, frame);
227    av_frame_free(&frame);
228}
