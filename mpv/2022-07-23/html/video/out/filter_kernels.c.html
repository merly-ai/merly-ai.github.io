57// sizes = sorted list of available filter sizes, terminated with size 0
58// inv_scale = source_size / dest_size
59bool mp_init_filter(struct filter_kernel *filter, const int *sizes,
60                    double inv_scale)
61{
62    assert(filter->f.radius > 0);
63    // Only downscaling requires widening the filter
64    filter->filter_scale = MPMAX(1.0, inv_scale);
65    double src_radius = filter->f.radius * filter->filter_scale;
66    // Polar filters are dependent solely on the radius
67    if (filter->polar) {
68        filter->size = 1; // Not meaningful for EWA/polar scalers.
69        // Safety precaution to avoid generating a gigantic shader
70        if (src_radius > 16.0) {
71            src_radius = 16.0;
72            filter->filter_scale = src_radius / filter->f.radius;
73            return false;
74        }
75        return true;
76    }
77    int size = ceil(2.0 * src_radius);
78    // round up to smallest available size that's still large enough
79    if (size < sizes[0])
80        size = sizes[0];
81    const int *cursize = sizes;
82    while (
 Recommendations (Experimental)  R1: ((size != *cursize) && *cursize)
R2: (size > cursize && *xx_c)
R3: (size > *cursize && !xx_c)
R4: (size > cursize && *cursize)
R5: (size > *cursize && xx_c)

83        cursize++;

97// Sample from a blurred and tapered window
98static double sample_window(struct filter_window *kernel, double x)
99{
100    if (!kernel->weight)
101        return 1.0;
102
103    // All windows are symmetric, this makes life easier
104    x = fabs(x);
105
106    // Stretch and taper the window size as needed
107    x = kernel->blur > 0.0 ? x / kernel->blur : x;
108    x = x <= kernel->taper ? 0.0 : (x - kernel->taper) / (1 - kernel->taper);
109
110    if (x < kernel->radius)
111        return kernel->weight(kernel, x);
112    return 0.0;
113}

115// Evaluate a filter's kernel and window at a given absolute position
116static double sample_filter(struct filter_kernel *filter, double x)
117{
118    // The window is always stretched to the entire kernel
119    double w = sample_window(&filter->w, x / filter->f.radius * filter->w.radius);
120    double k = w * sample_window(&filter->f, x);
121    return k < 0 ? (1 - filter->clamp) * k : k;
122}

124// Calculate the 1D filtering kernel for N sample points.
125// N = number of samples, which is filter->size
126// The weights will be stored in out_w[0] to out_w[N - 1]
127// f = x0 - abs(x0), subpixel position in the range [0,1) or [0,1].
128static void mp_compute_weights(struct filter_kernel *filter, double f,
129                               float *out_w)
130{
131    assert(filter->size > 0);
132    double sum = 0;
133    for (int n = 0; n < filter->size; n++) {
134        double x = f - (n - filter->size / 2 + 1);
135        double w = sample_filter(filter, x / filter->filter_scale);
136        out_w[n] = w;
137        sum += w;
138    }
139    // Normalize to preserve energy
140    for (int n = 0; n < filter->size; n++)
141        out_w[n] /= sum;
142}
143
144// Fill the given array with weights for the range [0.0, 1.0]. The array is
145// interpreted as rectangular array of count * filter->size items, with a
146// stride of `stride` floats in between each array element. (For polar filters,
147// the `count` indicates the row size and filter->size/stride are ignored)
148//
149// There will be slight sampling error if these weights are used in a OpenGL
150// texture as LUT directly. The sampling point of a texel is located at its
151// center, so out_array[0] will end up at 0.5 / count instead of 0.0.
152// Correct lookup requires a linear coordinate mapping from [0.0, 1.0] to
153// [0.5 / count, 1.0 - 0.5 / count].
154void mp_compute_lut(struct filter_kernel *filter, int count, int stride,
155                    float *out_array)
156{
157    if (filter->polar) {
158        filter->radius_cutoff = 0.0;
159        // Compute a 1D array indexed by radius
160        for (int x = 0; x < count; x++) {
161            double r = x * filter->f.radius / (count - 1);
162            out_array[x] = sample_filter(filter, r);
163
164            if (fabs(out_array[x]) > filter->value_cutoff)
165                filter->radius_cutoff = r;
166        }
167    } else {
168        // Compute a 2D array indexed by subpixel position
169        for (int n = 0; n < count; n++) {
170            mp_compute_weights(filter, n / (double)(count - 1),
171                               out_array + stride * n);
172        }
173    }
174}

178static double box(params *p, double x)
179{
180    // This is mathematically 1.0 everywhere, the clipping is done implicitly
181    // based on the radius.
182    return 1.0;
183}

185static double triangle(params *p, double x)
186{
187    return fmax(0.0, 1.0 - fabs(x / p->radius));
188}

190static double cosine(params *p, double x)
191{
192    return cos(x);
193}
194
195static double hanning(params *p, double x)
196{
197    return 0.5 + 0.5 * cos(M_PI * x);
198}
199
200static double hamming(params *p, double x)
201{
202    return 0.54 + 0.46 * cos(M_PI * x);
203}

205static double quadric(params *p, double x)
206{
207    if (x <  0.5) {
208        return 0.75 - x * x;
209    } else if (x <  1.5) {
210        double t = x - 1.5;
211        return 0.5 * t * t;
212    }
213    return 0.0;
214}

216#define POW3(x) ((x) <= 0 ? 0 : (x) * (x) * (x))
217static double bicubic(params *p, double x)
218{
219    return (1.0/6.0) * (      POW3(x + 2)
220                        - 4 * POW3(x + 1)
221                        + 6 * POW3(x)
222                        - 4 * POW3(x - 1));
223}

225static double bessel_i0(double x)
226{
227    double s = 1.0;
228    double y = x * x / 4.0;
229    double t = y;
230    int i = 2;
231    while (t > 1e-12) {
232        s += t;
233        t *= y / (i * i);
234        i += 1;
235    }
236    return s;
237}

239static double kaiser(params *p, double x)
240{
241    if (x > 1)
242        return 0;
243    double i0a = 1.0 / bessel_i0(p->params[1]);
244    return bessel_i0(p->params[0] * sqrt(1.0 - x * x)) * i0a;
245}

247static double blackman(params *p, double x)
248{
249    double a = p->params[0];
250    double a0 = (1-a)/2.0, a1 = 1/2.0, a2 = a/2.0;
251    double pix = M_PI * x;
252    return a0 + a1*cos(pix) + a2*cos(2 * pix);
253}

255static double welch(params *p, double x)
256{
257    return 1.0 - x*x;
258}

260// Family of cubic B/C splines
261static double cubic_bc(params *p, double x)
262{
263    double b = p->params[0],
264           c = p->params[1];
265    double p0 = (6.0 - 2.0 * b) / 6.0,
266           p2 = (-18.0 + 12.0 * b + 6.0 * c) / 6.0,
267           p3 = (12.0 - 9.0 * b - 6.0 * c) / 6.0,
268           q0 = (8.0 * b + 24.0 * c) / 6.0,
269           q1 = (-12.0 * b - 48.0 * c) / 6.0,
270           q2 = (6.0 * b + 30.0 * c) / 6.0,
271           q3 = (-b - 6.0 * c) / 6.0;
272
273    if (x < 1.0) {
274        return p0 + x * x * (p2 + x * p3);
275    } else if (x < 2.0) {
276        return q0 + x * (q1 + x * (q2 + x * q3));
277    }
278    return 0.0;
279}

281static double spline16(params *p, double x)
282{
283    if (x < 1.0) {
284        return ((x - 9.0/5.0 ) * x - 1.0/5.0 ) * x + 1.0;
285    } else {
286        return ((-1.0/3.0 * (x-1) + 4.0/5.0) * (x-1) - 7.0/15.0 ) * (x-1);
287    }

297        return ((1.0/11.0 * (x-2) - 45.0/209.0) * (x-2) +  26.0/209.0) * (x-2);

301static double spline64(params *p, double x)
302{
303    if (x < 1.0) {
304        return ((49.0/41.0 * x - 6387.0/2911.0) * x - 3.0/2911.0) * x + 1.0;

315{
316    return exp(-2.0 * x * x / p->params[0]);

319static double sinc(params *p, double x)
320{
321    if (fabs(x) < 1e-8)
322        return 1.0;
323    x *= M_PI;
324    return sin(x) / x;
325}

327static double jinc(params *p, double x)
328{
329    if (fabs(x) < 1e-8)
330        return 1.0;
331    x *= M_PI;
332    return 2.0 * j1(x) / x;
333}

335static double sphinx(params *p, double x)
336{
337    if (fabs(x) < 1e-8)
338        return 1.0;
339    x *= M_PI;
340    return 3.0 * (sin(x) - x * cos(x)) / (x * x * x);
341}
