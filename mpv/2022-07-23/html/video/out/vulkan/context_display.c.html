47/**
48 * If a selector is passed, verify that it is valid and return the matching
49 * mode properties. If null is passed, walk all modes and print them out.
50 */
51static bool walk_display_properties(struct mp_log *log,
52                                    int msgl_err,
53                                    VkPhysicalDevice device,
54                                    struct mode_selector *selector) {
55    bool ret = false;
56    VkResult res;
57
58    int msgl_info = selector ? MSGL_TRACE : MSGL_INFO;
59
60    // Use a dummy as parent for all other allocations.
61    void *tmp = talloc_new(NULL);
62
63    VkPhysicalDeviceProperties prop;
64    vkGetPhysicalDeviceProperties(device, &prop);
65    mp_msg(log, msgl_info, "  '%s' (GPU ID %x:%x)\n", prop.deviceName,
66           (unsigned)prop.vendorID, (unsigned)prop.deviceID);
67
68    // Count displays. This must be done before enumerating planes with the
69    // Intel driver, or it will not enumerate any planes. WTF.
70    int num_displays = 0;
71    vkGetPhysicalDeviceDisplayPropertiesKHR(device, &num_displays, NULL);
72    if (!num_displays) {
73        mp_msg(log, msgl_info, "    No available displays for device.\n");
74        goto done;
75    }
76    if (selector && selector->display_idx + 1 > num_displays) {
77        mp_msg(log, msgl_err, "Selected display (%d) not present.\n",
78               selector->display_idx);
79        goto done;
80    }
81
82    // Enumerate Planes
83    int num_planes = 0;
84    vkGetPhysicalDeviceDisplayPlanePropertiesKHR(device, &num_planes, NULL);
85    if (!num_planes) {
86        mp_msg(log, msgl_info, "    No available planes for device.\n");
87        goto done;
88    }
89    if (selector && selector->plane_idx + 1 > num_planes) {
90        mp_msg(log, msgl_err, "Selected plane (%d) not present.\n",
91               selector->plane_idx);
92        goto done;
93    }
94
95    VkDisplayPlanePropertiesKHR *planes =
96        talloc_array(tmp, VkDisplayPlanePropertiesKHR, num_planes);
97    res = vkGetPhysicalDeviceDisplayPlanePropertiesKHR(device, &num_planes,
98                                                       planes);
99    if (res != VK_SUCCESS) {
100        mp_msg(log, msgl_err, "    Failed enumerating planes\n");
101        goto done;
102    }
103
104    // Allocate zeroed arrays so that planes with no displays have a null entry.
105    VkDisplayKHR **planes_to_displays =
106        talloc_zero_array(tmp, VkDisplayKHR *, num_planes);
107    for (int j = 0; j < num_planes; j++) {
108        int num_displays_for_plane = 0;
109        vkGetDisplayPlaneSupportedDisplaysKHR(device, j,
110                                              &num_displays_for_plane, NULL);
111        if (!num_displays_for_plane)
112            continue;
113
114        // Null terminated array
115        VkDisplayKHR *displays =
116            talloc_zero_array(planes_to_displays, VkDisplayKHR,
117                              num_displays_for_plane + 1);
118        res = vkGetDisplayPlaneSupportedDisplaysKHR(device, j,
119                                                    &num_displays_for_plane,
120                                                    displays);
121        if (res != VK_SUCCESS) {
122            mp_msg(log, msgl_err, "      Failed enumerating plane displays\n");
123            continue;
124        }
125        planes_to_displays[j] = displays;
126    }
127
128    // Enumerate Displays and Modes
129    VkDisplayPropertiesKHR *props =
130        talloc_array(tmp, VkDisplayPropertiesKHR, num_displays);
131    res = vkGetPhysicalDeviceDisplayPropertiesKHR(device, &num_displays, props);
132    if (res != VK_SUCCESS) {
133        mp_msg(log, msgl_err, "    Failed enumerating display properties\n");
134        goto done;
135    }
136
137    for (int j = 0; j < num_displays; j++) {
138        if (selector && selector->display_idx != j)
139            continue;
140
141        mp_msg(log, msgl_info, "    Display %d: '%s' (%dx%d)\n",
142               j,
143               props[j].displayName,
144               props[j].physicalResolution.width,
145               props[j].physicalResolution.height);
146
147        VkDisplayKHR display = props[j].display;
148
149        mp_msg(log, msgl_info, "    Modes:\n");
150
151        int num_modes = 0;
152        vkGetDisplayModePropertiesKHR(device, display, &num_modes, NULL);
153        if (!num_modes) {
154            mp_msg(log, msgl_info, "      No available modes for display.\n");
155            continue;
156        }
157        if (selector && selector->mode_idx + 1 > num_modes) {
158            mp_msg(log, msgl_err, "Selected mode (%d) not present.\n",
159                   selector->mode_idx);
160            goto done;
161        }
162
163        VkDisplayModePropertiesKHR *modes =
164            talloc_array(tmp, VkDisplayModePropertiesKHR, num_modes);
165        res = vkGetDisplayModePropertiesKHR(device, display, &num_modes, modes);
166        if (res != VK_SUCCESS) {
167            mp_msg(log, msgl_err, "      Failed enumerating display modes\n");
168            continue;
169        }
170
171        for (int k = 0; k < num_modes; k++) {
172            if (selector && selector->mode_idx != k)
173                continue;
174
175            mp_msg(log, msgl_info, "      Mode %02d: %dx%d (%02d.%03d Hz)\n", k,
176                   modes[k].parameters.visibleRegion.width,
177                   modes[k].parameters.visibleRegion.height,
178                   modes[k].parameters.refreshRate / 1000,
179                   modes[k].parameters.refreshRate % 1000);
180
181            if (selector)
182                selector->out_mode_props = talloc_dup(NULL, &modes[k]);
183        }
184
185        int found_plane = -1;
186        mp_msg(log, msgl_info, "    Planes:\n");
187        for (int k = 0; k < num_planes; k++) {
188            VkDisplayKHR *displays = planes_to_displays[k];
189            if (!displays) {
190                // This plane is not connected to any displays.
191                continue;
192            }
193            for (int d = 0; displays[d]; d++) {
194                if (displays[d] == display) {
195                    if (selector && selector->plane_idx != k)
196                        continue;
197
198                    mp_msg(log, msgl_info, "      Plane: %d\n", k);
199                    found_plane = k;
200                }
201            }
202        }
203        if (selector && selector->plane_idx != found_plane) {
204            mp_msg(log, msgl_err,
205                   "Selected plane (%d) not available on selected display.\n",
206                   selector->plane_idx);
207            goto done;
208        }
209    }
210    ret = true;
211done:
212    talloc_free(tmp);
213    return ret;
214}

216static int print_display_info(struct mp_log *log, const struct m_option *opt,
217                              struct bstr name) {
218    VkResult res;
219    VkPhysicalDevice *devices = NULL;
220
221    // Create a dummy instance to list the resources
222    VkInstanceCreateInfo info = {
223        .sType = VK_STRUCTURE_TYPE_INSTANCE_CREATE_INFO,
224        .enabledExtensionCount = 1,
225        .ppEnabledExtensionNames = (const char*[]) {
226            VK_KHR_DISPLAY_EXTENSION_NAME
227        },
228    };
229
230    VkInstance inst = NULL;
231    res = vkCreateInstance(&info, NULL, &inst);
232    if (res != VK_SUCCESS) {
233        mp_warn(log, "Unable to create Vulkan instance.\n");
234        goto done;
235    }
236
237    uint32_t num_devices = 0;
238    vkEnumeratePhysicalDevices(inst, &num_devices, NULL);
239    if (!num_devices) {
240        mp_info(log, "No Vulkan devices detected.\n");
241        goto done;
242    }
243
244    devices = talloc_array(NULL, VkPhysicalDevice, num_devices);
245    vkEnumeratePhysicalDevices(inst, &num_devices, devices);
246    if (res != VK_SUCCESS) {
247        mp_warn(log, "Failed enumerating physical devices.\n");
248        goto done;
249    }
250
251    mp_info(log, "Vulkan Devices:\n");
252    for (int i = 0; i < num_devices; i++) {
253        walk_display_properties(log, MSGL_WARN, devices[i], NULL);
254    }
255
256done:
257    talloc_free(devices);
258    vkDestroyInstance(inst, NULL);
259    return M_OPT_EXIT;
260}

368static bool display_init(struct ra_ctx *ctx)
369{
370    struct priv *p = ctx->priv = talloc_zero(ctx, struct priv);
