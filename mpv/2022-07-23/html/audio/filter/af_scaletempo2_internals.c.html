9// Algorithm overview (from chromium):
10// Waveform Similarity Overlap-and-add (WSOLA).
11//
12// One WSOLA iteration
13//
14// 1) Extract |target_block| as input frames at indices
15//    [|target_block_index|, |target_block_index| + |ola_window_size|).
16//    Note that |target_block| is the "natural" continuation of the output.
17//
18// 2) Extract |search_block| as input frames at indices
19//    [|search_block_index|,
20//     |search_block_index| + |num_candidate_blocks| + |ola_window_size|).
21//
22// 3) Find a block within the |search_block| that is most similar
23//    to |target_block|. Let |optimal_index| be the index of such block and
24//    write it to |optimal_block|.
25//
26// 4) Update:
27//    |optimal_block| = |transition_window| * |target_block| +
28//    (1 - |transition_window|) * |optimal_block|.
29//
30// 5) Overlap-and-add |optimal_block| to the |wsola_output|.
31//
32// 6) Update:write

66// Energies of sliding windows of channels are interleaved.
67// The number windows is |input_frames| - (|frames_per_window| - 1), hence,
68// the method assumes |energy| must be, at least, of size
69// (|input_frames| - (|frames_per_window| - 1)) * |channels|.
70static void multi_channel_moving_block_energies(
71    float **input, int input_frames, int channels,
72    int frames_per_block, float *energy)
73{
74    int num_blocks = input_frames - (frames_per_block - 1);
75
76    for (int k = 0; k < channels; ++k) {
77        const float* input_channel = input[k];
78
79        energy[k] = 0;
80
81        // First block of channel |k|.
82        for (int m = 0; m < frames_per_block; ++m) {
83            energy[k] += input_channel[m] * input_channel[m];
84        }
85
86        const float* slide_out = input_channel;
87        const float* slide_in = input_channel + frames_per_block;
88        for (int n = 1; n < num_blocks; ++n, ++slide_in, ++slide_out) {
89            energy[k + n * channels] = energy[k + (n - 1) * channels]
90                - *slide_out * *slide_out + *slide_in * *slide_in;
91        }
92    }
93}

95static float multi_channel_similarity_measure(
96    const float* dot_prod_a_b,
97    const float* energy_a, const float* energy_b,
98    int channels)
99{
100    const float epsilon = 1e-12f;
101    float similarity_measure = 0.0f;
102    for (int n = 0; n < channels; ++n) {
103        similarity_measure += dot_prod_a_b[n]
104            / sqrtf(energy_a[n] * energy_b[n] + epsilon);
105    }
106    return similarity_measure;
107}

113// Dot-product of channels of two AudioBus. For each AudioBus an offset is
114// given. |dot_product[k]| is the dot-product of channel |k|. The caller should
115// allocate sufficient space for |dot_product|.
116static void multi_channel_dot_product(
117    float **a, int frame_offset_a,
118    float **b, int frame_offset_b,
119    int channels,
120    int num_frames, float *dot_product)
121{
122    assert(frame_offset_a >= 0);
123    assert(frame_offset_b >= 0);
124
125    for (int k = 0; k < channels; ++k) {
126        const float* ch_a = a[k] + frame_offset_a;
127        const float* ch_b = b[k] + frame_offset_b;
128        float sum = 0.0;
129        if (num_frames < 32)
130            goto rest;
131
132        const v8sf *va = (const v8sf *) ch_a;
133        const v8sf *vb = (const v8sf *) ch_b;
134        v8sf vsum[4] = {
135            // Initialize to product of first 32 floats
136            va[0] * vb[0],
137            va[1] * vb[1],
138            va[2] * vb[2],
139            va[3] * vb[3],
140        };
141        va += 4;
142        vb += 4;
143
144        // Process `va` and `vb` across four vertical stripes
145        for (int n = 1; n < num_frames / 32; n++) {
146            vsum[0] += va[0] * vb[0];
147            vsum[1] += va[1] * vb[1];
148            vsum[2] += va[2] * vb[2];
149            vsum[3] += va[3] * vb[3];
150            va += 4;
151            vb += 4;
152        }
153
154        // Vertical sum across `vsum` entries
155        vsum[0] += vsum[1];
156        vsum[2] += vsum[3];
157        vsum[0] += vsum[2];
158
159        // Horizontal sum across `vsum[0]`, could probably be done better but
160        // this section is not super performance critical
161        float *vf = (float *) &vsum[0];
162        sum = vf[0] + vf[1] + vf[2] + vf[3] + vf[4] + vf[5] + vf[6] + vf[7];
163        ch_a = (const float *) va;
164        ch_b = (const float *) vb;
165
166rest:
167        // Process the remainder
168        for (int n = 0; n < num_frames % 32; n++)
169            sum += *ch_a++ * *ch_b++;
170
171        dot_product[k] = sum;
172    }
173}

198// Fit the curve f(x) = a * x^2 + b * x + c such that
199//   f(-1) = y[0]
200//   f(0) = y[1]
201//   f(1) = y[2]
202// and return the maximum, assuming that y[0] <= y[1] >= y[2].
203static void quadratic_interpolation(
204    const float* y_values, float* extremum, float* extremum_value)
205{
206    float a = 0.5f * (y_values[2] + y_values[0]) - y_values[1];
207    float b = 0.5f * (y_values[2] - y_values[0]);
208    float c = y_values[1];
209
210    if (a == 0.f) {
211        // The coordinates are colinear (within floating-point error).
212        *extremum = 0;
213        *extremum_value = y_values[1];
214    } else {
215        *extremum = -b / (2.f * a);
216        *extremum_value = a * (*extremum) * (*extremum) + b * (*extremum) + c;
217    }
218}

220// Search a subset of all candid blocks. The search is performed every
221// |decimation| frames. This reduces complexity by a factor of about
222// 1 / |decimation|. A cubic interpolation is used to have a better estimate of
223// the best match.
224static int decimated_search(
225    int decimation, struct interval exclude_interval,
226    float **target_block, int target_block_frames,
227    float **search_segment, int search_segment_frames,
228    int channels,
229    const float *energy_target_block, const float *energy_candidate_blocks)
230{
231    int num_candidate_blocks = search_segment_frames - (target_block_frames - 1);
232    float dot_prod [MP_NUM_CHANNELS];
233    float similarity[3];  // Three elements for cubic interpolation.
234
235    int n = 0;
236    multi_channel_dot_product(
237        target_block, 0,
238        search_segment, n,
239        channels,
240        target_block_frames, dot_prod);
241    similarity[0] = multi_channel_similarity_measure(
242        dot_prod, energy_target_block,
243        &energy_candidate_blocks[n * channels], channels);
244
245    // Set the starting point as optimal point.
246    float best_similarity = similarity[0];
247    int optimal_index = 0;
248
249    n += decimation;
250    if (n >= num_candidate_blocks) {
251        return 0;
252    }
253
254    multi_channel_dot_product(
255        target_block, 0,
256        search_segment, n,
257        channels,
258        target_block_frames, dot_prod);
259    similarity[1] = multi_channel_similarity_measure(
260        dot_prod, energy_target_block,
261        &energy_candidate_blocks[n * channels], channels);
262
263    n += decimation;
264    if (n >= num_candidate_blocks) {
265        // We cannot do any more sampling. Compare these two values and return the
266        // optimal index.
267        return similarity[1] > similarity[0] ? decimation : 0;
268    }
269
270    for (; n < num_candidate_blocks; n += decimation) {
271        multi_channel_dot_product(
272            target_block, 0,
273            search_segment, n,
274            channels,
275            target_block_frames, dot_prod);
276
277        similarity[2] = multi_channel_similarity_measure(
278            dot_prod, energy_target_block,
279            &energy_candidate_blocks[n * channels], channels);
280
281        if ((similarity[1] > similarity[0] && similarity[1] >= similarity[2]) ||
282            (similarity[1] >= similarity[0] && similarity[1] > similarity[2]))
283        {
284            // A local maximum is found. Do a cubic interpolation for a better
285            // estimate of candidate maximum.
286            float normalized_candidate_index;
287            float candidate_similarity;
288            quadratic_interpolation(similarity, &normalized_candidate_index,
289                                    &candidate_similarity);
290
291            int candidate_index = n - decimation
292                 + (int)(normalized_candidate_index * decimation +  0.5f);
293            if (candidate_similarity > best_similarity
294                && !in_interval(candidate_index, exclude_interval)) {
295                optimal_index = candidate_index;
296                best_similarity = candidate_similarity;
297            }
298        } else if (n + decimation >= num_candidate_blocks &&
299                   similarity[2] > best_similarity &&
300                   !in_interval(n, exclude_interval))
301        {
302            // If this is the end-point and has a better similarity-measure than
303            // optimal, then we accept it as optimal point.
304            optimal_index = n;
305            best_similarity = similarity[2];
306        }
307        memmove(similarity, &similarity[1], 2 * sizeof(*similarity));
308    }
309    return optimal_index;
310}

312// Search [|low_limit|, |high_limit|] of |search_segment| to find a block that
313// is most similar to |target_block|. |energy_target_block| is the energy of the
314// |target_block|. |energy_candidate_blocks| is the energy of all blocks within
315// |search_block|.
316static int full_search(
317    int low_limit, int high_limit,
318    struct interval exclude_interval,
319    float **target_block, int target_block_frames,
320    float **search_block, int search_block_frames,
321    int channels,
322    const float* energy_target_block,
323    const float* energy_candidate_blocks)
324{
325    // int block_size = target_block->frames;
326    float dot_prod [sizeof(float) * MP_NUM_CHANNELS];
327
328    float best_similarity = -FLT_MAX;//FLT_MIN;
329    int optimal_index = 0;
330
331    for (int n = low_limit; n <= high_limit; ++n) {
332        if (in_interval(n, exclude_interval)) {
333            continue;
334        }
335        multi_channel_dot_product(target_block, 0, search_block, n, channels,
336            target_block_frames, dot_prod);
337
338        float similarity = multi_channel_similarity_measure(
339            dot_prod, energy_target_block,
340            &energy_candidate_blocks[n * channels], channels);
341
342        if (similarity > best_similarity) {
343            best_similarity = similarity;
344            optimal_index = n;
345        }
346    }
347
348    return optimal_index;
349}

351// Find the index of the block, within |search_block|, that is most similar
352// to |target_block|. Obviously, the returned index is w.r.t. |search_block|.
353// |exclude_interval| is an interval that is excluded from the search.
354static int compute_optimal_index(
355    float **search_block, int search_block_frames,
356    float **target_block, int target_block_frames,
357    float *energy_candidate_blocks,
358    int channels,
359    struct interval exclude_interval)
360{
361    int num_candidate_blocks = search_block_frames - (target_block_frames - 1);
362
363    // This is a compromise between complexity reduction and search accuracy. I
364    // don't have a proof that down sample of order 5 is optimal.
365    // One can compute a decimation factor that minimizes complexity given
366    // the size of |search_block| and |target_block|. However, my experiments
367    // show the rate of missing the optimal index is significant.
368    // This value is chosen heuristically based on experiments.
369    const int search_decimation = 5;
370
371    float energy_target_block [MP_NUM_CHANNELS];
372    // energy_candidate_blocks must have at least size
373    // sizeof(float) * channels * num_candidate_blocks
374
375    // Energy of all candid frames.
376    multi_channel_moving_block_energies(
377        search_block,
378        search_block_frames,
379        channels,
380        target_block_frames,
381        energy_candidate_blocks);
382
383    // Energy of target frame.
384    multi_channel_dot_product(
385        target_block, 0,
386        target_block, 0,
387        channels,
388        target_block_frames, energy_target_block);
389
390    int optimal_index = decimated_search(
391        search_decimation, exclude_interval,
392        target_block, target_block_frames,
393        search_block, search_block_frames,
394        channels,
395        energy_target_block,
396        energy_candidate_blocks);
397
398    int lim_low = MPMAX(0, optimal_index - search_decimation);
399    int lim_high = MPMIN(num_candidate_blocks - 1,
400                            optimal_index + search_decimation);
401    return full_search(
402        lim_low, lim_high, exclude_interval,
403        target_block, target_block_frames,
404        search_block, search_block_frames,
405        channels,
406        energy_target_block, energy_candidate_blocks);
407}

482    uint8_t **planes, int frame_size, bool final)
483{
484    int needed = frames_needed(p);
485    int read = MPMIN(needed, frame_size);
486    int total_fill = final ? needed : read;
487    if (total_fill == 0) return 0;
488
489    int required_size = total_fill + p->input_buffer_frames;
490    resize_input_buffer(p, required_size);
491
492    for (int i = 0; i < p->channels; ++i) {
493        memcpy(p->input_buffer[i] + p->input_buffer_frames,
494            planes[i], read * sizeof(float));
495        for (int j = read; j < total_fill; ++j) {
496            p->input_buffer[p->input_buffer_frames + j] = 0;
497        }
498    }
499
500    p->input_buffer_frames += total_fill;
501    return read;
502}

504static bool target_is_within_search_region(struct mp_scaletempo2 *p)
505{
506    return p->target_block_index >= p->search_block_index
507        && p->target_block_index + p->ola_window_size
508            <= p->search_block_index + p->search_block_size;
509}

512static void peek_audio_with_zero_prepend(struct mp_scaletempo2 *p,
513    int read_offset_frames, float **dest, int dest_frames)
514{
515    assert(read_offset_frames + dest_frames <= p->input_buffer_frames);
516
517    int write_offset = 0;
518    int num_frames_to_read = dest_frames;
519    if (read_offset_frames < 0) {
520        int num_zero_frames_appended = MPMIN(
521            -read_offset_frames, num_frames_to_read);
522        read_offset_frames = 0;
523        num_frames_to_read -= num_zero_frames_appended;
524        write_offset = num_zero_frames_appended;
525        zero_2d_partial(dest, p->channels, num_zero_frames_appended);
526    }
527    peek_buffer(p, num_frames_to_read, read_offset_frames, write_offset, dest);
528}

530static void get_optimal_block(struct mp_scaletempo2 *p)
531{
532    int optimal_index = 0;
533
534    // An interval around last optimal block which is excluded from the search.
535    // This is to reduce the buzzy sound. The number 160 is rather arbitrary and
536    // derived heuristically.
537    const int exclude_interval_length_frames = 160;
538    if (target_is_within_search_region(p)) {
539        optimal_index = p->target_block_index;
540        peek_audio_with_zero_prepend(p,
541            optimal_index, p->optimal_block, p->ola_window_size);
542    } else {
543        peek_audio_with_zero_prepend(p,
544            p->target_block_index, p->target_block, p->ola_window_size);
545        peek_audio_with_zero_prepend(p,
546            p->search_block_index, p->search_block, p->search_block_size);
547        int last_optimal = p->target_block_index
548            - p->ola_hop_size - p->search_block_index;
549        struct interval exclude_iterval = {
550            .lo = last_optimal - exclude_interval_length_frames / 2,
551            .hi = last_optimal + exclude_interval_length_frames / 2
552        };
553
554        // |optimal_index| is in frames and it is relative to the beginning of the
555        // |search_block|.
556        optimal_index = compute_optimal_index(
557            p->search_block, p->search_block_size,
558            p->target_block, p->ola_window_size,
559            p->energy_candidate_blocks,
560            p->channels,
561            exclude_iterval);
562
563        // Translate |index| w.r.t. the beginning of |audio_buffer| and extract the
564        // optimal block.
565        optimal_index += p->search_block_index;
566        peek_audio_with_zero_prepend(p,
567            optimal_index, p->optimal_block, p->ola_window_size);
568
569        // Make a transition from target block to the optimal block if different.
570        // Target block has the best continuation to the current output.
571        // Optimal block is the most similar block to the target, however, it might
572        // introduce some discontinuity when over-lap-added. Therefore, we combine
573        // them for a smoother transition. The length of transition window is twice
574        // as that of the optimal-block which makes it like a weighting function
575        // where target-block has higher weight close to zero (weight of 1 at index
576        // 0) and lower weight close the end.
577        for (int k = 0; k < p->channels; ++k) {
578            float* ch_opt = p->optimal_block[k];
579            float* ch_target = p->target_block[k];
580            for (int n = 0; n < p->ola_window_size; ++n) {
581                ch_opt[n] = ch_opt[n] * p->transition_window[n]
582                    + ch_target[n] * p->transition_window[p->ola_window_size + n];
583            }
584        }
585    }
586
587    // Next target is one hop ahead of the current optimal.
588    p->target_block_index = optimal_index + p->ola_hop_size;
589}

591static void update_output_time(struct mp_scaletempo2 *p,
592    float playback_rate, double time_change)
593{
594    p->output_time += time_change;
595    // Center of the search region, in frames.
596    int search_block_center_index = (int)(p->output_time * playback_rate + 0.5);
597    p->search_block_index = search_block_center_index
598        - p->search_block_center_offset;
599}

601static void remove_old_input_frames(struct mp_scaletempo2 *p, float playback_rate)
602{
603    const int earliest_used_index = MPMIN(
604        p->target_block_index, p->search_block_index);
605    if (earliest_used_index <= 0)
606        return;  // Nothing to remove.
607
608    // Remove frames from input and adjust indices accordingly.
609    seek_buffer(p, earliest_used_index);
610    p->target_block_index -= earliest_used_index;
611
612    // Adjust output index.
613    double output_time_change = ((double) earliest_used_index) / playback_rate;
614    assert(p->output_time >= output_time_change);
615    update_output_time(p, playback_rate, -output_time_change);
616}
617
618static bool run_one_wsola_iteration(struct mp_scaletempo2 *p, float playback_rate)
619{
620    if (!can_perform_wsola(p)){
621        return false;
622    }
623
624    get_optimal_block(p);
625
626    // Overlap-and-add.
627    for (int k = 0; k < p->channels; ++k) {
628        float* ch_opt_frame = p->optimal_block[k];
629        float* ch_output = p->wsola_output[k] + p->num_complete_frames;
630        for (int n = 0; n < p->ola_hop_size; ++n) {
631            ch_output[n] = ch_output[n] * p->ola_window[p->ola_hop_size + n] +
632                ch_opt_frame[n] * p->ola_window[n];
633        }
634
635        // Copy the second half to the output.
636        memcpy(&ch_output[p->ola_hop_size], &ch_opt_frame[p->ola_hop_size],
637               sizeof(*ch_opt_frame) * p->ola_hop_size);
638    }
639
640    p->num_complete_frames += p->ola_hop_size;
641    update_output_time(p, playback_rate, p->ola_hop_size);
642    remove_old_input_frames(p, playback_rate);
643    return true;
644}

647    float **dest, int dest_size, float playback_rate)
648{
649    if (playback_rate == 0) return 0;
650
651    // Optimize the muted case to issue a single clear instead of performing
652    // the full crossfade and clearing each crossfaded frame.
653    if (playback_rate < p->opts->min_playback_rate

playback_rate > p->opts->max_playback_rate
655            && p->opts->max_playback_rate > 0))
656    {
657        int frames_to_render = MPMIN(dest_size,
658            (int) (p->input_buffer_frames / playback_rate));
659
660        // Compute accurate number of frames to actually skip in the source data.
661        // Includes the leftover partial frame from last request. However, we can
662        // only skip over complete frames, so a partial frame may remain for next

664        p->muted_partial_frame += frames_to_render * playback_rate;
665        int seek_frames = (int) (p->muted_partial_frame);
666        zero_2d_partial(dest, p->channels, frames_to_render);
667        seek_buffer(p, seek_frames);

670        // the user switches back to playing, it may be off time by this partial
671        // frame, which would be undetectable. If they subsequently switch to
672        // another playback rate that mutes, the code will attempt to line up the
673        // frames again.
674        p->muted_partial_frame -= seek_frames;
675        return frames_to_render;
676    }
677
678    int slower_step = (int) ceilf(p->ola_window_size * playback_rate);
679    int faster_step = (int) ceilf(p->ola_window_size / playback_rate);
680
681    // Optimize the most common |playback_rate| ~= 1 case to use a single copy
682    // instead of copying frame by frame.
683    if (p->ola_window_size <= faster_step && slower_step >= p->ola_window_size) {
684        int frames_to_copy = MPMIN(dest_size, p->input_buffer_frames);
685        read_buffer(p, frames_to_copy, dest);
686        return frames_to_copy;
687    }
688
689    int rendered_frames = 0;
690    do {
691        rendered_frames += write_completed_frames_to(p,
692            dest_size - rendered_frames, rendered_frames, dest);
693    } while (rendered_frames < dest_size
694             && run_one_wsola_iteration(p, playback_rate));
695    return rendered_frames;
696}

715void mp_scaletempo2_reset(struct mp_scaletempo2 *p)
716{
717    p->input_buffer_frames = 0;
718    p->output_time = 0.0;
719    p->search_block_index = 0;
720    p->target_block_index = 0;
721    // Clear the queue of decoded packets.
722    zero_2d(p->wsola_output, p->channels, p->wsola_output_size);
723    p->num_complete_frames = 0;
724}
725
726// Return a "periodic" Hann window. This is the first L samples of an L+1
727// Hann window. It is perfect reconstruction for overlap-and-add.
728static void get_symmetric_hanning_window(int window_length, float* window)
729{
730    const float scale = 2.0f * M_PI / window_length;
731    for (int n = 0; n < window_length; ++n)
732        window[n] = 0.5f * (1.0f - cosf(n * scale));
733}

736void mp_scaletempo2_init(struct mp_scaletempo2 *p, int channels, int rate)
737{
738    p->muted_partial_frame = 0;
739    p->output_time = 0;
740    p->search_block_center_offset = 0;
741    p->search_block_index = 0;
742    p->num_complete_frames = 0;
743    p->channels = channels;
744
745    p->samples_per_second = rate;
746    p->num_candidate_blocks = (int)(p->opts->wsola_search_interval_ms
747        * p->samples_per_second / 1000);
748    p->ola_window_size = (int)(p->opts->ola_window_size_ms
749        * p->samples_per_second / 1000);
750    // Make sure window size in an even number.
751    p->ola_window_size += p->ola_window_size & 1;
752    p->ola_hop_size = p->ola_window_size / 2;
753    // |num_candidate_blocks| / 2 is the offset of the center of the search
754    // block to the center of the first (left most) candidate block. The offset
755    // of the center of a candidate block to its left most point is
756    // |ola_window_size| / 2 - 1. Note that |ola_window_size| is even and in
757    // our convention the center belongs to the left half, so we need to subtract
758    // one frame to get the correct offset.
759    //
760    //                             Search Block
761    //              <------------------------------------------->
762    //
763    //   |ola_window_size| / 2 - 1
764    //              <----
765    //
766    //             |num_candidate_blocks| / 2
767    //                   <----------------
768    //                                 center
769    //              X----X----------------X---------------X-----X
770    //              <---------->                     <---------->
771    //                Candidate      ...               Candidate
772    //                   1,          ...         |num_candidate_blocks|
773    p->search_block_center_offset = p->num_candidate_blocks / 2
774        + (p->ola_window_size / 2 - 1);
775    p->ola_window = realloc(p->ola_window, sizeof(float) * p->ola_window_size);
776    get_symmetric_hanning_window(p->ola_window_size, p->ola_window);
777    p->transition_window = realloc(p->transition_window,
778        sizeof(float) * p->ola_window_size * 2);
779    get_symmetric_hanning_window(2 * p->ola_window_size, p->transition_window);
780
781    p->wsola_output_size = p->ola_window_size + p->ola_hop_size;
782    p->wsola_output = realloc_2d(p->wsola_output, p->channels, p->wsola_output_size);
783    // Initialize for overlap-and-add of the first block.
784    zero_2d(p->wsola_output, p->channels, p->wsola_output_size);
785
786    // Auxiliary containers.
787    p->optimal_block = realloc_2d(p->optimal_block, p->channels, p->ola_window_size);
788    p->search_block_size = p->num_candidate_blocks + (p->ola_window_size - 1);
789    p->search_block = realloc_2d(p->search_block, p->channels, p->search_block_size);
790    p->target_block = realloc_2d(p->target_block, p->channels, p->ola_window_size);
791
792    resize_input_buffer(p, 4 * MPMAX(p->ola_window_size, p->search_block_size));
793    p->input_buffer_frames = 0;
794
795    p->energy_candidate_blocks = realloc(p->energy_candidate_blocks,
796        sizeof(float) * p->channels * p->num_candidate_blocks);
797}
