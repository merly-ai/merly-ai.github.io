120static int control(struct ao *ao, enum aocontrol cmd, void *arg)
121{
122    struct priv *p = ao->priv;
123    snd_mixer_t *handle = NULL;
124    switch (cmd) {
125    case AOCONTROL_GET_MUTE:
126    case AOCONTROL_SET_MUTE:
127    case AOCONTROL_GET_VOLUME:
128    case AOCONTROL_SET_VOLUME:
129    {
130        int err;
131        snd_mixer_elem_t *elem;
132        snd_mixer_selem_id_t *sid;
133
134        long pmin, pmax;
135        long get_vol, set_vol;
136        float f_multi;
137
138        if (!af_fmt_is_pcm(ao->format))
139            return CONTROL_FALSE;
140
141        snd_mixer_selem_id_alloca(&sid);
142
143        snd_mixer_selem_id_set_index(sid, p->opts->mixer_index);
144        snd_mixer_selem_id_set_name(sid, p->opts->mixer_name);
145
146        err = snd_mixer_open(&handle, 0);
147        CHECK_ALSA_ERROR("Mixer open error");
148
149        err = snd_mixer_attach(handle, p->opts->mixer_device);
150        CHECK_ALSA_ERROR("Mixer attach error");
151
152        err = snd_mixer_selem_register(handle, NULL, NULL);
153        CHECK_ALSA_ERROR("Mixer register error");
154
155        err = snd_mixer_load(handle);
156        CHECK_ALSA_ERROR("Mixer load error");
157
158        elem = snd_mixer_find_selem(handle, sid);
159        if (!elem) {
160            MP_VERBOSE(ao, "Unable to find simple control '%s',%i.\n",
161                       snd_mixer_selem_id_get_name(sid),
162                       snd_mixer_selem_id_get_index(sid));
163            goto alsa_error;
164        }
165
166        snd_mixer_selem_get_playback_volume_range(elem, &pmin, &pmax);
167        f_multi = (100 / (float)(pmax - pmin));
168
169        switch (cmd) {
170        case AOCONTROL_SET_VOLUME: {
171            ao_control_vol_t *vol = arg;
172            set_vol = vol->left / f_multi + pmin + 0.5;
173
174            err = snd_mixer_selem_set_playback_volume(elem, 0, set_vol);
175            CHECK_ALSA_ERROR("Error setting left channel");
176            MP_DBG(ao, "left=%li, ", set_vol);
177
178            set_vol = vol->right / f_multi + pmin + 0.5;
179
180            err = snd_mixer_selem_set_playback_volume(elem, 1, set_vol);
181            CHECK_ALSA_ERROR("Error setting right channel");
182            MP_DBG(ao, "right=%li, pmin=%li, pmax=%li, mult=%f\n",
183                   set_vol, pmin, pmax, f_multi);
184            break;
185        }
186        case AOCONTROL_GET_VOLUME: {
187            ao_control_vol_t *vol = arg;
188            snd_mixer_selem_get_playback_volume(elem, 0, &get_vol);
189            vol->left = (get_vol - pmin) * f_multi;
190            snd_mixer_selem_get_playback_volume(elem, 1, &get_vol);
191            vol->right = (get_vol - pmin) * f_multi;
192            MP_DBG(ao, "left=%f, right=%f\n", vol->left, vol->right);
193            break;
194        }
195        case AOCONTROL_SET_MUTE: {
196            bool *mute = arg;
197            if (!snd_mixer_selem_has_playback_switch(elem))
198                goto alsa_error;
199            if (!snd_mixer_selem_has_playback_switch_joined(elem)) {
200                snd_mixer_selem_set_playback_switch(elem, 1, !*mute);
201            }
202            snd_mixer_selem_set_playback_switch(elem, 0, !*mute);
203            break;
204        }
205        case AOCONTROL_GET_MUTE: {
206            bool *mute = arg;
207            if (!snd_mixer_selem_has_playback_switch(elem))
208                goto alsa_error;
209            int tmp = 1;
210            snd_mixer_selem_get_playback_switch(elem, 0, &tmp);
211            *mute = !tmp;
212            if (!snd_mixer_selem_has_playback_switch_joined(elem)) {
213                snd_mixer_selem_get_playback_switch(elem, 1, &tmp);
214                *mute &= !tmp;
215            }
216            break;
217        }
218        }
219        snd_mixer_close(handle);
220        return CONTROL_OK;
221    }
222
223    } //end switch
224    return CONTROL_UNKNOWN;
225
226alsa_error:
227    if (handle)
228        snd_mixer_close(handle);
229    return CONTROL_ERROR;
230}

307// Replace each channel in a with b (a->num == b->num)
308static void replace_submap(struct mp_chmap *dst, struct mp_chmap *a,
309                           struct mp_chmap *b)
310{
311    struct mp_chmap t = *dst;
312    if (!mp_chmap_is_valid(&t) || mp_chmap_diffn(a, &t) != 0)
313        return;
314    assert(a->num == b->num);
315    for (int n = 0; n < t.num; n++) {
316        for (int i = 0; i < a->num; i++) {
317            if (t.speaker[n] == a->speaker[i]) {
318                t.speaker[n] = b->speaker[i];
319                break;
320            }
321        }
322    }
323    if (mp_chmap_is_valid(&t))
324        *dst = t;
325}

327static bool mp_chmap_from_alsa(struct mp_chmap *dst, snd_pcm_chmap_t *src)
328{
329    *dst = (struct mp_chmap) {0};
330
331    if (src->channels > MP_NUM_CHANNELS)
332        return false;
333
334    dst->num = src->channels;
335    for (int c = 0; c < dst->num; c++)
336        dst->speaker[c] = find_mp_channel(src->pos[c]);
337
338    // Assume anything with 1 channel is mono.
339    if (dst->num == 1)
340        dst->speaker[0] = MP_SP(FC);
341
342    // Remap weird Intel HDA HDMI 7.1 layouts correctly.
343    replace_submap(dst, CHMAP(6, FL, FR, BL, BR, SDL, SDR),
344                        CHMAP(6, FL, FR, SL, SR, BL,  BR));
345
346    return mp_chmap_is_valid(dst);
347}

349static bool query_chmaps(struct ao *ao, struct mp_chmap *chmap)
350{
351    struct priv *p = ao->priv;
352    struct mp_chmap_sel chmap_sel = {.tmp = p};
353
354    snd_pcm_chmap_query_t **maps = snd_pcm_query_chmaps(p->alsa);
355    if (!maps) {
356        MP_VERBOSE(ao, "snd_pcm_query_chmaps() returned NULL\n");
357        return false;
358    }
359
360    for (int i = 0; maps[i] != NULL; i++) {
361        char aname[128];
362        if (snd_pcm_chmap_print(&maps[i]->map, sizeof(aname), aname) <= 0)
363            aname[0] = '\0';
364
365        struct mp_chmap entry;
366        if (mp_chmap_from_alsa(&entry, &maps[i]->map)) {
367            struct mp_chmap reorder = entry;
368            mp_chmap_reorder_norm(&reorder);
369
370            MP_DBG(ao, "got ALSA chmap: %s (%s) -> %s", aname,
371                   snd_pcm_chmap_type_name(maps[i]->type),
372                   mp_chmap_to_str(&entry));
373            if (!mp_chmap_equals(&entry, &reorder))
374                MP_DBG(ao, " -> %s", mp_chmap_to_str(&reorder));
375            MP_DBG(ao, "\n");
376
377            struct mp_chmap final =
378                maps[i]->type == SND_CHMAP_TYPE_VAR ? reorder : entry;
379            mp_chmap_sel_add_map(&chmap_sel, &final);
380        } else {
381            MP_VERBOSE(ao, "skipping unknown ALSA channel map: %s\n", aname);
382        }
383    }
384
385    snd_pcm_free_chmaps(maps);
386
387    return ao_chmap_sel_adjust2(ao, &chmap_sel, chmap, false);
388}

390// Map back our selected channel layout to an ALSA one. This is done this way so
391// that our ALSA->mp_chmap mapping function only has to go one way.
392// The return value is to be freed with free().
393static snd_pcm_chmap_t *map_back_chmap(struct ao *ao, struct mp_chmap *chmap)
394{
395    struct priv *p = ao->priv;
396    if (!mp_chmap_is_valid(chmap))
397        return NULL;
398
399    snd_pcm_chmap_query_t **maps = snd_pcm_query_chmaps(p->alsa);
400    if (!maps)
401        return NULL;
402
403    snd_pcm_chmap_t *alsa_chmap = NULL;
404
405    for (int i = 0; maps[i] != NULL; i++) {
406        struct mp_chmap entry;
407        if (!mp_chmap_from_alsa(&entry, &maps[i]->map))
408            continue;
409
410        if (mp_chmap_equals(chmap, &entry) ||
411            (mp_chmap_equals_reordered(chmap, &entry) &&
412                maps[i]->type == SND_CHMAP_TYPE_VAR))
413        {
414            alsa_chmap = calloc(1, sizeof(*alsa_chmap) +
415                                   sizeof(alsa_chmap->pos[0]) * entry.num);
416            if (!alsa_chmap)
417                break;
418            alsa_chmap->channels = entry.num;
419
420            // Undo if mp_chmap_reorder() was called on the result.
421            int reorder[MP_NUM_CHANNELS];
422            mp_chmap_get_reorder(reorder, chmap, &entry);
423            for (int n = 0; n < entry.num; n++)
424                alsa_chmap->pos[n] = maps[i]->map.pos[reorder[n]];
425            break;
426        }
427    }
428
429    snd_pcm_free_chmaps(maps);
430    return alsa_chmap;
431}

434static int set_chmap(struct ao *ao, struct mp_chmap *dev_chmap, int num_channels)
435{
436    struct priv *p = ao->priv;
437    int err;
438
439    snd_pcm_chmap_t *alsa_chmap = map_back_chmap(ao, dev_chmap);
440    if (alsa_chmap) {
441        char tmp[128];
442        if (snd_pcm_chmap_print(alsa_chmap, sizeof(tmp), tmp) > 0)
443            MP_VERBOSE(ao, "trying to set ALSA channel map: %s\n", tmp);
444
445        err = snd_pcm_set_chmap(p->alsa, alsa_chmap);
446        if (err == -ENXIO) {
447            // A device my not be able to set any channel map, even channel maps
448            // that were reported as supported. This is either because the ALSA
449            // device is broken (dmix), or because the driver has only 1
450            // channel map per channel count, and setting the map is not needed.
451            MP_VERBOSE(ao, "device returned ENXIO when setting channel map %s\n",
452                       mp_chmap_to_str(dev_chmap));
453        } else {
454            CHECK_ALSA_WARN("Channel map setup failed");
455        }
456
457        free(alsa_chmap);
458    }
459
460    alsa_chmap = snd_pcm_get_chmap(p->alsa);
461    if (alsa_chmap) {
462        char tmp[128];
463        if (snd_pcm_chmap_print(alsa_chmap, sizeof(tmp), tmp) > 0)
464            MP_VERBOSE(ao, "channel map reported by ALSA: %s\n", tmp);
465
466        struct mp_chmap chmap;
467        mp_chmap_from_alsa(&chmap, alsa_chmap);
468
469        MP_VERBOSE(ao, "which we understand as: %s\n", mp_chmap_to_str(&chmap));
470
471        if (p->opts->ignore_chmap) {
472            MP_VERBOSE(ao, "user set ignore-chmap; ignoring the channel map.\n");
473        } else if (af_fmt_is_spdif(ao->format)) {
474            MP_VERBOSE(ao, "using spdif passthrough; ignoring the channel map.\n");
475        } else if (!mp_chmap_is_valid(&chmap)) {
476            MP_WARN(ao, "Got unknown channel map from ALSA.\n");
477        } else if (chmap.num != num_channels) {
478            MP_WARN(ao, "ALSA channel map conflicts with channel count!\n");
479        } else {
480            if (mp_chmap_equals(&chmap, &ao->channels)) {
481                MP_VERBOSE(ao, "which is what we requested.\n");
482            } else if (!mp_chmap_is_valid(dev_chmap)) {
483                MP_VERBOSE(ao, "ignoring the ALSA channel map.\n");
484            } else {
485                MP_VERBOSE(ao, "using the ALSA channel map.\n");
486                ao->channels = chmap;
487            }
488        }
489
490        free(alsa_chmap);
491    }
492
493    return 0;
494}

543// ALSA device strings can have parameters. They are usually appended to the
544// device name. There can be various forms, and we (sometimes) want to append
545// them to unknown device strings, which possibly already include params.
546static char *append_params(void *ta_parent, const char *device, const char *p)
547{
548    if (!p || !p[0])
549        return talloc_strdup(ta_parent, device);
550
551    int len = strlen(device);
552    char *end = strchr(device, ':');
553    if (!end) {
554        /* no existing parameters: add it behind device name */
555        return talloc_asprintf(ta_parent, "%s:%s", device, p);
556    } else if (end[1] == '\0') {
557        /* ":" but no parameters */
558        return talloc_asprintf(ta_parent, "%s%s", device, p);
559    } else if (end[1] == '{' && device[len - 1] == '}') {
560        /* parameters in config syntax: add it inside the { } block */
561        return talloc_asprintf(ta_parent, "%.*s %s}", len - 1, device, p);
562    } else {
563        /* a simple list of parameters: add it at the end of the list */
564        return talloc_asprintf(ta_parent, "%s,%s", device, p);
565    }
566    abort();
567}

569static int try_open_device(struct ao *ao, const char *device, int mode)
570{
571    struct priv *p = ao->priv;
572    int err;
573
574    if (af_fmt_is_spdif(ao->format)) {
575        void *tmp = talloc_new(NULL);
576        char *params = talloc_asprintf(tmp,
577                        "AES0=%d,AES1=%d,AES2=0,AES3=%d",
578                        IEC958_AES0_NONAUDIO | IEC958_AES0_PRO_EMPHASIS_NONE,
579                        IEC958_AES1_CON_ORIGINAL | IEC958_AES1_CON_PCM_CODER,
580                        map_iec958_srate(ao->samplerate));
581        const char *ac3_device = append_params(tmp, device, params);
582        MP_VERBOSE(ao, "opening device '%s' => '%s'\n", device, ac3_device);
583        err = snd_pcm_open(&p->alsa, ac3_device, SND_PCM_STREAM_PLAYBACK, mode);
584        if (err < 0) {
585            // Some spdif-capable devices do not accept the AES0 parameter,
586            // and instead require the iec958 pseudo-device (they will play
587            // noise otherwise). Unfortunately, ALSA gives us no way to map
588            // these devices, so try it for the default device only.
589            bstr dev;
590            bstr_split_tok(bstr0(device), ":", &dev, &(bstr){0});
591            if (bstr_equals0(dev, "default")) {
592                const char *const fallbacks[] = {"hdmi", "iec958", NULL};
593                for (int n = 0; fallbacks[n]; n++) {
594                    char *ndev = append_params(tmp, fallbacks[n], params);
595                    MP_VERBOSE(ao, "got error '%s'; opening iec fallback "
596                               "device '%s'\n", snd_strerror(err), ndev);
597                    err = snd_pcm_open
598                                (&p->alsa, ndev, SND_PCM_STREAM_PLAYBACK, mode);
599                    if (err >= 0)
600                        break;
601                }
602            }
603        }
604        talloc_free(tmp);
605    } else {
606        MP_VERBOSE(ao, "opening device '%s'\n", device);
607        err = snd_pcm_open(&p->alsa, device, SND_PCM_STREAM_PLAYBACK, mode);
608    }
609
610    return err;
611}

632#define INIT_DEVICE_ERR_GENERIC -1
633#define INIT_DEVICE_ERR_HWPARAMS -2
634static int init_device(struct ao *ao, int mode)
635{
636    struct priv *p = ao->priv;
637    struct ao_alsa_opts *opts = p->opts;
638    int ret = INIT_DEVICE_ERR_GENERIC;
639    char *tmp;
640    size_t tmp_s;
641    int err;
642
643    p->alsa_fmt = SND_PCM_FORMAT_UNKNOWN;
644
645    err = snd_output_buffer_open(&p->output);
646    CHECK_ALSA_ERROR("Unable to create output buffer");
647
648    const char *device = "default";
649    if (ao->device)
650        device = ao->device;
651
652    err = try_open_device(ao, device, mode);
653    CHECK_ALSA_ERROR("Playback open error");
654
655    err = snd_pcm_dump(p->alsa, p->output);
656    CHECK_ALSA_WARN("Dump PCM error");
657    tmp_s = snd_output_buffer_string(p->output, &tmp);
658    if (tmp)
659        MP_DBG(ao, "PCM setup:\n---\n%.*s---\n", (int)tmp_s, tmp);
660    snd_output_flush(p->output);
661
662    err = snd_pcm_nonblock(p->alsa, 0);
663    CHECK_ALSA_WARN("Unable to set blocking mode");
664
665    snd_pcm_hw_params_t *alsa_hwparams;
666    snd_pcm_hw_params_alloca(&alsa_hwparams);
667
668    err = snd_pcm_hw_params_any(p->alsa, alsa_hwparams);
669    CHECK_ALSA_ERROR("Unable to get initial parameters");
670
671    dump_hw_params(ao, "Start HW params:\n", alsa_hwparams);
672
673    // Some ALSA drivers have broken delay reporting, so disable the ALSA
674    // resampling plugin by default.
675    if (!p->opts->resample) {
676        err = snd_pcm_hw_params_set_rate_resample(p->alsa, alsa_hwparams, 0);
677        CHECK_ALSA_ERROR("Unable to disable resampling");
678    }
679    dump_hw_params(ao, "HW params after rate:\n", alsa_hwparams);
680
681    snd_pcm_access_t access = af_fmt_is_planar(ao->format)
682                                    ? SND_PCM_ACCESS_RW_NONINTERLEAVED
683                                    : SND_PCM_ACCESS_RW_INTERLEAVED;
684    err = snd_pcm_hw_params_set_access(p->alsa, alsa_hwparams, access);
685    if (err < 0 && af_fmt_is_planar(ao->format)) {
686        ao->format = af_fmt_from_planar(ao->format);
687        access = SND_PCM_ACCESS_RW_INTERLEAVED;
688        err = snd_pcm_hw_params_set_access(p->alsa, alsa_hwparams, access);
689    }
690    CHECK_ALSA_ERROR("Unable to set access type");
691    dump_hw_params(ao, "HW params after access:\n", alsa_hwparams);
692
693    bool found_format = false;
694    int try_formats[AF_FORMAT_COUNT + 1];
695    af_get_best_sample_formats(ao->format, try_formats);
696    for (int n = 0; try_formats[n] && !found_format; n++) {
697        int mp_format = try_formats[n];
698        if (af_fmt_is_planar(ao->format) != af_fmt_is_planar(mp_format))
699            continue; // implied SND_PCM_ACCESS mismatches
700        int mp_pformat = af_fmt_from_planar(mp_format);
701        if (af_fmt_is_spdif(mp_pformat))
702            mp_pformat = AF_FORMAT_S16;
703        const struct alsa_fmt *fmt = find_alsa_format(mp_pformat);
704        if (!fmt)
705            continue;
706        for (; fmt->mp_format == mp_pformat; fmt++) {
707            p->alsa_fmt = fmt->alsa_format;
708            p->convert = (struct ao_convert_fmt){
709                .src_fmt = mp_format,
710                .dst_bits = fmt->bits ? fmt->bits : af_fmt_to_bytes(mp_format) * 8,
711                .pad_msb = fmt->pad_msb,
712            };
713            if (!ao_can_convert_inplace(&p->convert))
714                continue;
715            MP_VERBOSE(ao, "trying format %s/%d\n", af_fmt_to_str(mp_pformat),
716                       p->alsa_fmt);
717            if (snd_pcm_hw_params_test_format(p->alsa, alsa_hwparams,
718                                              p->alsa_fmt) >= 0)
719            {
720                ao->format = mp_format;
721                found_format = true;
722                break;
723            }
724        }
725    }
726
727    if (!found_format) {
728        MP_ERR(ao, "Can't find appropriate sample format.\n");
729        goto alsa_error;
730    }
731
732    err = snd_pcm_hw_params_set_format(p->alsa, alsa_hwparams, p->alsa_fmt);
733    CHECK_ALSA_ERROR("Unable to set format");
734    dump_hw_params(ao, "HW params after format:\n", alsa_hwparams);
735
736    // Stereo, or mono if input is 1 channel.
737    struct mp_chmap reduced;
738    mp_chmap_from_channels(&reduced, MPMIN(2, ao->channels.num));
739
740    struct mp_chmap dev_chmap = {0};
741    if (!af_fmt_is_spdif(ao->format) && !p->opts->ignore_chmap &&
742        !mp_chmap_equals(&ao->channels, &reduced))
743    {
744        struct mp_chmap res = ao->channels;
745        if (query_chmaps(ao, &res))
746            dev_chmap = res;
747
748        // Whatever it is, we dumb it down to mono or stereo. Some drivers may
749        // return things like bl-br, but the user (probably) still wants stereo.
750        // This also handles the failure case (dev_chmap.num==0).
751        if (dev_chmap.num <= 2) {
752            dev_chmap.num = 0;
753            ao->channels = reduced;
754        } else if (dev_chmap.num) {
755            ao->channels = dev_chmap;
756        }
757    }
758
759    int num_channels = ao->channels.num;
760    err = snd_pcm_hw_params_set_channels_near
761            (p->alsa, alsa_hwparams, &num_channels);
762    CHECK_ALSA_ERROR("Unable to set channels");
763    dump_hw_params(ao, "HW params after channels:\n", alsa_hwparams);
764
765    if (num_channels > MP_NUM_CHANNELS) {
766        MP_FATAL(ao, "Too many audio channels (%d).\n", num_channels);
767        goto alsa_error;
768    }
769
770    err = snd_pcm_hw_params_set_rate_near
771            (p->alsa, alsa_hwparams, &ao->samplerate, NULL);
772    CHECK_ALSA_ERROR("Unable to set samplerate-2");
773    dump_hw_params(ao, "HW params after rate-2:\n", alsa_hwparams);
774
775    snd_pcm_hw_params_t *hwparams_backup;
776    snd_pcm_hw_params_alloca(&hwparams_backup);
777    snd_pcm_hw_params_copy(hwparams_backup, alsa_hwparams);
778
779    // Cargo-culted buffer settings; might still be useful for PulseAudio.
780    err = 0;
781    if (opts->buffer_time) {
782        err = snd_pcm_hw_params_set_buffer_time_near
783                (p->alsa, alsa_hwparams, &(unsigned int){opts->buffer_time}, NULL);
784        CHECK_ALSA_WARN("Unable to set buffer time near");
785    }
786    if (err >= 0 && opts->frags) {
787        err = snd_pcm_hw_params_set_periods_near
788                    (p->alsa, alsa_hwparams, &(unsigned int){opts->frags}, NULL);
789        CHECK_ALSA_WARN("Unable to set periods");
790    }
791    if (err < 0)
792        snd_pcm_hw_params_copy(alsa_hwparams, hwparams_backup);
793
794    dump_hw_params(ao, "Going to set final HW params:\n", alsa_hwparams);
795
796    /* finally install hardware parameters */
797    err = snd_pcm_hw_params(p->alsa, alsa_hwparams);
798    ret = INIT_DEVICE_ERR_HWPARAMS;
799    CHECK_ALSA_ERROR("Unable to set hw-parameters");
800    ret = INIT_DEVICE_ERR_GENERIC;
801    dump_hw_params(ao, "Final HW params:\n", alsa_hwparams);
802
803    if (set_chmap(ao, &dev_chmap, num_channels) < 0)
804        goto alsa_error;
805
806    if (num_channels != ao->channels.num) {
807        int req = ao->channels.num;
808        mp_chmap_from_channels(&ao->channels, MPMIN(2, num_channels));
809        mp_chmap_fill_na(&ao->channels, num_channels);
810        MP_ERR(ao, "Asked for %d channels, got %d - fallback to %s.\n", req,
811               num_channels, mp_chmap_to_str(&ao->channels));
812        if (num_channels != ao->channels.num) {
813            MP_FATAL(ao, "mismatching channel counts.\n");
814            goto alsa_error;
815        }
816    }
817
818    err = snd_pcm_hw_params_get_buffer_size(alsa_hwparams, &p->buffersize);
819    CHECK_ALSA_ERROR("Unable to get buffersize");
820
821    err = snd_pcm_hw_params_get_period_size(alsa_hwparams, &p->outburst, NULL);
822    CHECK_ALSA_ERROR("Unable to get period size");
823
824    p->can_pause = snd_pcm_hw_params_can_pause(alsa_hwparams);
825
826    snd_pcm_sw_params_t *alsa_swparams;
827    snd_pcm_sw_params_alloca(&alsa_swparams);
828
829    err = snd_pcm_sw_params_current(p->alsa, alsa_swparams);
830    CHECK_ALSA_ERROR("Unable to get sw-parameters");
831
832    snd_pcm_uframes_t boundary;
833    err = snd_pcm_sw_params_get_boundary(alsa_swparams, &boundary);
834    CHECK_ALSA_ERROR("Unable to get boundary");
835
836    // Manual trigger; INT_MAX as suggested by ALSA doxygen (they call it MAXINT).
837    err = snd_pcm_sw_params_set_start_threshold(p->alsa, alsa_swparams, INT_MAX);
838    CHECK_ALSA_ERROR("Unable to set start threshold");
839
840    /* play silence when there is an underrun */
841    err = snd_pcm_sw_params_set_silence_size
842            (p->alsa, alsa_swparams, boundary);
843    CHECK_ALSA_ERROR("Unable to set silence size");
844
845    err = snd_pcm_sw_params(p->alsa, alsa_swparams);
846    CHECK_ALSA_ERROR("Unable to set sw-parameters");
847
848    MP_VERBOSE(ao, "hw pausing supported: %s\n", p->can_pause ? "yes" : "no");
849    MP_VERBOSE(ao, "buffersize: %d samples\n", (int)p->buffersize);
850    MP_VERBOSE(ao, "period size: %d samples\n", (int)p->outburst);
851
852    ao->device_buffer = p->buffersize;
853
854    p->convert.channels = ao->channels.num;
855
856    err = snd_pcm_prepare(p->alsa);
857    CHECK_ALSA_ERROR("pcm prepare error");
858
859    return 0;
860
861alsa_error:
862    uninit(ao);
863    return ret;
864}

866static int init(struct ao *ao)
867{
868    struct priv *p = ao->priv;
869    p->opts = mp_get_config_group(ao, ao->global, &ao_alsa_conf);
870
871    if (!p->opts->ni)
872        ao->format = af_fmt_from_planar(ao->format);
873
874    MP_VERBOSE(ao, "using ALSA version: %s\n", snd_asoundlib_version());
875
876    int mode = 0;
877    int r = init_device(ao, mode);
878    if (r == INIT_DEVICE_ERR_HWPARAMS) {
879        // With some drivers, ALSA appears to be unable to set valid hwparams,
880        // but they work if at least SND_PCM_NO_AUTO_FORMAT is set. Also, it
881        // appears you can set this flag only on opening a device, thus there
882        // is the need to retry opening the device.
883        MP_WARN(ao, "Attempting to work around even more ALSA bugs...\n");
884        mode |= SND_PCM_NO_AUTO_CHANNELS | SND_PCM_NO_AUTO_FORMAT |
885                SND_PCM_NO_AUTO_RESAMPLE;
886        r = init_device(ao, mode);
887    }
888
889    // Sometimes, ALSA will advertise certain chmaps, but it's not possible to
890    // set them. This can happen with dmix: as of alsa 1.0.29, dmix can do
891    // stereo only, but advertises the surround chmaps of the underlying device.
892    // In this case, e.g. setting 6 channels will succeed, but requesting  5.1
893    // afterwards will fail. Then it will return something like "FL FR NA NA NA NA"
894    // as channel map. This means we would have to pad stereo output to 6
895    // channels with silence, which would require lots of extra processing. You
896    // can't change the number of channels to 2 either, because the hw params
897    // are already set! So just fuck it and reopen the device with the chmap
898    // "cleaned out" of NA entries.

916// Function for dealing with playback state. This attempts to recover the ALSA
917// state (bring it into SND_PCM_STATE_{PREPARED,RUNNING,PAUSED,UNDERRUN}). If
918// state!=NULL, fill it after recovery is attempted.
919// Returns true if PCM is in one the expected states.
920static bool recover_and_get_state(struct ao *ao, struct mp_pcm_state *state)
921{
922    struct priv *p = ao->priv;
923    int err;
924
925    snd_pcm_status_t *st;
926    snd_pcm_status_alloca(&st);
927
928    bool state_ok = false;
929    snd_pcm_state_t pcmst = SND_PCM_STATE_DISCONNECTED;
930
931    // Give it a number of chances to recover. This tries to deal with the fact
932    // that the API is asynchronous, and to account for some past cargo-cult
933    // (where things were retried in a loop).
934    for (int n = 0; n < 10; n++) {
935        err = snd_pcm_status(p->alsa, st);
936        if (err == -EPIPE) {
937            // ALSA APIs can return -EPIPE when an XRUN happens,
938            // we skip right to handling it by setting pcmst
939            // manually.
940            pcmst = SND_PCM_STATE_XRUN;
941        } else {
942            // Otherwise do error checking and query the PCM state properly.
943            CHECK_ALSA_ERROR("snd_pcm_status");
944
945            pcmst = snd_pcm_status_get_state(st);
946        }
947
948        if (pcmst == SND_PCM_STATE_PREPARED ||
949            pcmst == SND_PCM_STATE_RUNNING ||
950            pcmst == SND_PCM_STATE_PAUSED)
951        {
952            state_ok = true;
953            break;
954        }
955
956        MP_VERBOSE(ao, "attempt %d to recover from state '%s'...\n",
957                   n + 1, snd_pcm_state_name(pcmst));
958
959        switch (pcmst) {
960        // Underrun; recover. (We never use draining.)
961        case SND_PCM_STATE_XRUN:
962        case SND_PCM_STATE_DRAINING:
963            err = snd_pcm_prepare(p->alsa);
964            CHECK_ALSA_ERROR("pcm prepare error");
965            continue;
966        // Hardware suspend.
967        case SND_PCM_STATE_SUSPENDED:
968            MP_INFO(ao, "PCM in suspend mode, trying to resume.\n");
969            err = snd_pcm_resume(p->alsa);
970            if (err == -EAGAIN) {
971                // Cargo-cult from decades ago, with a cargo cult timeout.
972                MP_INFO(ao, "PCM resume EAGAIN - retrying.\n");
973                sleep(1);
974                continue;
975            }
976            if (err == -ENOSYS) {
977                // As suggested by ALSA doxygen.
978                MP_VERBOSE(ao, "ENOSYS, retrying with snd_pcm_prepare().\n");
979                err = snd_pcm_prepare(p->alsa);
980            }
981            if (err < 0)
982                MP_ERR(ao, "resuming from SUSPENDED: %s\n", snd_strerror(err));
983            continue;
984        // Device lost. OPEN/SETUP are states we never enter after init, so
985        // treat them like DISCONNECTED.
986        case SND_PCM_STATE_DISCONNECTED:
987        case SND_PCM_STATE_OPEN:
988        case SND_PCM_STATE_SETUP:
989        default:
990            if (!p->device_lost) {
991                MP_WARN(ao, "Device lost, trying to recover...\n");
992                ao_request_reload(ao);
993                p->device_lost = true;
994            }
995            goto alsa_error;
996        }
997    }
998
999    if (!state_ok) {
1000        MP_ERR(ao, "could not recover\n");
1001    }
1002
1003alsa_error:
1004
1005    if (state) {
1006        snd_pcm_sframes_t del = state_ok ? snd_pcm_status_get_delay(st) : 0;
1007        state->delay = MPMAX(del, 0) / (double)ao->samplerate;
1008        state->free_samples = state_ok ? snd_pcm_status_get_avail(st) : 0;
1009        state->free_samples = MPCLAMP(state->free_samples, 0, ao->device_buffer);
1010        // Align to period size.
1011        state->free_samples = state->free_samples / p->outburst * p->outburst;
1012        state->queued_samples = ao->device_buffer - state->free_samples;
1013        state->playing = pcmst == SND_PCM_STATE_RUNNING ||
1014                         pcmst == SND_PCM_STATE_PAUSED;
1015    }
1016
1017    return state_ok;
1018}
