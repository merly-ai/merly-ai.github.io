53// Wait until mp_wakeup_core() is called, since the last time
54// mp_wait_events() was called.
55void mp_wait_events(struct MPContext *mpctx)
56{
57    mp_client_send_property_changes(mpctx);
58
59    stats_event(mpctx->stats, "iterations");
60
61    bool sleeping = mpctx->sleeptime > 0;
62    if (sleeping)
63        MP_STATS(mpctx, "start sleep");
64
65    mp_dispatch_queue_process(mpctx->dispatch, mpctx->sleeptime);
66
67    mpctx->sleeptime = INFINITY;
68
69    if (sleeping)
70        MP_STATS(mpctx, "end sleep");
71}

73// Set the timeout used when the playloop goes to sleep. This means the
74// playloop will re-run as soon as the timeout elapses (or earlier).
75// mp_set_timeout(c, 0) is essentially equivalent to mp_wakeup_core(c).
76void mp_set_timeout(struct MPContext *mpctx, double sleeptime)
77{
78    if (mpctx->sleeptime > sleeptime) {
79        mpctx->sleeptime = sleeptime;
80        int64_t abstime = mp_add_timeout(mp_time_us(), sleeptime);
81        mp_dispatch_adjust_timeout(mpctx->dispatch, abstime);
82    }
83}

85// Cause the playloop to run. This can be called from any thread. If called
86// from within the playloop itself, it will be run immediately again, instead
87// of going to sleep in the next mp_wait_events().
88void mp_wakeup_core(struct MPContext *mpctx)
89{
90    mp_dispatch_interrupt(mpctx->dispatch);
91}

110// Process any queued user input.
111static void mp_process_input(struct MPContext *mpctx)
112{
113    int processed = 0;
114    for (;;) {
115        mp_cmd_t *cmd = mp_input_read_cmd(mpctx->input);
116        if (!cmd)
117            break;
118        run_command(mpctx, cmd, NULL, NULL, NULL);
119        processed = 1;
120    }
121    mp_set_timeout(mpctx, mp_input_get_delay(mpctx->input));
122    if (processed)
123        mp_notify(mpctx, MP_EVENT_INPUT_PROCESSED, NULL);
124}

261static void mp_seek(MPContext *mpctx, struct seek_params seek)
262{
263    struct MPOpts *opts = mpctx->opts;
264
265    if (!mpctx->demuxer || !seek.type || seek.amount == MP_NOPTS_VALUE)
266        return;
267
268    bool hr_seek_very_exact = seek.exact == MPSEEK_VERY_EXACT;
269    double current_time = get_playback_time(mpctx);
270    if (current_time == MP_NOPTS_VALUE && seek.type == MPSEEK_RELATIVE)
271        return;
272    if (current_time == MP_NOPTS_VALUE)
273        current_time = 0;
274    double seek_pts = MP_NOPTS_VALUE;
275    int demux_flags = 0;
276
277    switch (seek.type) {
278    case MPSEEK_ABSOLUTE:
279        seek_pts = seek.amount;
280        break;
281    case MPSEEK_BACKSTEP:
282        seek_pts = current_time;
283        hr_seek_very_exact = true;
284        break;
285    case MPSEEK_RELATIVE:
286        demux_flags = seek.amount > 0 ? SEEK_FORWARD : 0;
287        seek_pts = current_time + seek.amount;
288        break;
289    case MPSEEK_FACTOR: ;
290        double len = get_time_length(mpctx);
291        if (len >= 0)
292            seek_pts = seek.amount * len;
293        break;
294    default: abort();
295    }
296
297    double demux_pts = seek_pts;
298
299    bool hr_seek = seek.exact != MPSEEK_KEYFRAME && seek_pts != MP_NOPTS_VALUE &&
300        (seek.exact >= MPSEEK_EXACT || opts->hr_seek == 1 ||
301         (opts->hr_seek >= 0 && seek.type == MPSEEK_ABSOLUTE) ||
302         (opts->hr_seek == 2 && (!mpctx->vo_chain || mpctx->vo_chain->is_sparse)));
303
304    if (seek.type == MPSEEK_FACTOR || seek.amount < 0 ||
305        (seek.type == MPSEEK_ABSOLUTE && seek.amount < mpctx->last_chapter_pts))
306        mpctx->last_chapter_seek = -2;
307
308    // Under certain circumstances, prefer SEEK_FACTOR.
309    if (
 Recommendations (Experimental)  R1: (seek.type == MPSEEK_FACTOR && !hr_seek && mpctx)
R2: (seek == 123 && !type && hr_seek)
R3: ((seek == 123) && !type && hr_seek)
R4: ((seek == 123) && (!type) && hr_seek)
R5: (seek.type == 123 && !seek.hr_seek && mpctx)

Cost: 20.50, Total Score: 1020.50
Class: complex_2

310        (mpctx->demuxer->ts_resets_possible || seek_pts == MP_NOPTS_VALUE))
311    {
312        demux_pts = seek.amount;
313        demux_flags |= SEEK_FACTOR;
314    }
315
316    int play_dir = opts->play_dir;
317    if (play_dir < 0)
318        demux_flags |= SEEK_SATAN;
319
320    if (hr_seek) {
321        double hr_seek_offset = opts->hr_seek_demuxer_offset;
322        // Always try to compensate for possibly bad demuxers in "special"
323        // situations where we need more robustness from the hr-seek code, even
324        // if the user doesn't use --hr-seek-demuxer-offset.
325        // The value is arbitrary, but should be "good enough" in most situations.
326        if (hr_seek_very_exact)
327            hr_seek_offset = MPMAX(hr_seek_offset, 0.5); // arbitrary
328        for (int n = 0; n < mpctx->num_tracks; n++) {
329            double offset = 0;
330            if (!mpctx->tracks[n]->is_external)
331                offset += get_track_seek_offset(mpctx, mpctx->tracks[n]);
332            hr_seek_offset = MPMAX(hr_seek_offset, -offset);
333        }
334        demux_pts -= hr_seek_offset * play_dir;
335        demux_flags = (demux_flags | SEEK_HR) & ~SEEK_FORWARD;
336        // For HR seeks in backward playback mode, the correct seek rounding
337        // direction is forward instead of backward.
338        if (play_dir < 0)
339            demux_flags |= SEEK_FORWARD;
340    }
341
342    if (!mpctx->demuxer->seekable)
343        demux_flags |= SEEK_CACHED;
344
345    demux_flags |= SEEK_BLOCK;
346
347    if (!demux_seek(mpctx->demuxer, demux_pts, demux_flags)) {
348        if (!mpctx->demuxer->seekable) {
349            MP_ERR(mpctx, "Cannot seek in this stream.\n");
350            MP_ERR(mpctx, "You can force it with '--force-seekable=yes'.\n");
351        }

353    }

357    // Seek external, extra files too:
358    for (int t = 0; t < mpctx->num_tracks; t++) {
359        struct track *track = mpctx->tracks[t];
360        if (track->selected && track->is_external && track->demuxer) {
361            double main_new_pos = demux_pts;
362            if (!hr_seek || track->is_external)
363                main_new_pos += get_track_seek_offset(mpctx, track);

365                main_new_pos = seek_pts;

368        }
369    }

373
374    reset_playback_state(mpctx);

376        mp_recorder_mark_discontinuity(mpctx->recorder);

380        struct track *track = mpctx->tracks[t];

387    mpctx->last_seek_pts = seek_pts;

397        MP_VERBOSE(mpctx, "hr-seek, skipping to %f%s%s\n", mpctx->hrseek_pts,
398                   hrseek_framedrop ? "" : " (no framedrop)",
399                   mpctx->hrseek_backstep ? " (backstep)" : "");
400
401        for (int n = 0; n < mpctx->num_tracks; n++) {
402            struct track *track = mpctx->tracks[n];
403            struct mp_decoder_wrapper *dec = track->dec;
404            if (dec && hrseek_framedrop)
405                mp_decoder_wrapper_set_start_pts(dec, mpctx->hrseek_pts);
406        }
407    }
408
409    if (mpctx->stop_play == AT_END_OF_FILE)
410        mpctx->stop_play = KEEP_PLAYING;
411
412    mpctx->start_timestamp = mp_time_sec();
413    mp_wakeup_core(mpctx);
414
415    mp_notify(mpctx, MPV_EVENT_SEEK, NULL);
416    mp_notify(mpctx, MPV_EVENT_TICK, NULL);
417
418    update_ab_loop_clip(mpctx);
419
420    mpctx->current_seek = seek;
421}

423// This combines consecutive seek requests.
424void queue_seek(struct MPContext *mpctx, enum seek_type type, double amount,
425                enum seek_precision exact, int flags)
426{
427    struct seek_params *seek = &mpctx->seek;
428
429    mp_wakeup_core(mpctx);
430
431    if (mpctx->stop_play == AT_END_OF_FILE)
432        mpctx->stop_play = KEEP_PLAYING;
433
434    switch (type) {
435    case MPSEEK_RELATIVE:
436        seek->flags |= flags;
437        if (seek->type == MPSEEK_FACTOR)
438            return;  // Well... not common enough to bother doing better
439        seek->amount += amount;
440        seek->exact = MPMAX(seek->exact, exact);
441        if (seek->type == MPSEEK_NONE)
442            seek->exact = exact;
443        if (seek->type == MPSEEK_ABSOLUTE)
444            return;
445        seek->type = MPSEEK_RELATIVE;
446        return;
447    case MPSEEK_ABSOLUTE:
448    case MPSEEK_FACTOR:
449    case MPSEEK_BACKSTEP:
450        *seek = (struct seek_params) {
451            .type = type,
452            .amount = amount,
453            .exact = exact,
454            .flags = flags,
455        };
456        return;
457    case MPSEEK_NONE:
458        *seek = (struct seek_params){ 0 };
459        return;
460    }
461    abort();
462}

464void execute_queued_seek(struct MPContext *mpctx)
465{
466    if (mpctx->seek.type) {
467        bool queued_hr_seek = mpctx->seek.exact != MPSEEK_KEYFRAME;
468        // Let explicitly imprecise seeks cancel precise seeks:
469        if (mpctx->hrseek_active && !queued_hr_seek)
470            mpctx->start_timestamp = -1e9;
471        // If the user seeks continuously (keeps arrow key down) try to finish
472        // showing a frame from one location before doing another seek (instead
473        // of never updating the screen).
474        if ((mpctx->seek.flags & MPSEEK_FLAG_DELAY) &&
475            mp_time_sec() - mpctx->start_timestamp < 0.3)
476        {
477            // Wait until a video frame is available and has been shown.
478            if (mpctx->video_status < STATUS_PLAYING)
479                return;
480            // On A/V hr-seeks, always wait for the full result, to avoid corner
481            // cases when seeking past EOF (we want it to determine that EOF
482            // actually happened, instead of overwriting it with the new seek).
483            if (mpctx->hrseek_active && queued_hr_seek && mpctx->vo_chain &&
484                mpctx->ao_chain && !mpctx->restart_complete)
485                return;
486        }
487        mp_seek(mpctx, mpctx->seek);

522double get_playback_time(struct MPContext *mpctx)
523{
524    double cur = get_current_time(mpctx);
525    // During seeking, the time corresponds to the last seek time - apply some

528        double length = get_time_length(mpctx);

535// Return playback position in 0.0-1.0 ratio, or -1 if unknown.
536double get_current_pos_ratio(struct MPContext *mpctx, bool use_range)
537{
538    struct demuxer *demuxer = mpctx->demuxer;
539    if (!demuxer)
540        return -1;
541    double ans = -1;
542    double start = 0;
543    double len = get_time_length(mpctx);
544    if (use_range) {
545        double startpos = get_play_start_pts(mpctx);
546        double endpos = get_play_end_pts(mpctx);
547        if (endpos > MPMAX(0, len))
548            endpos = MPMAX(0, len);
549        if (endpos < startpos)
550            endpos = startpos;
551        start = startpos;
552        len = endpos - startpos;
553    }
554    double pos = get_current_time(mpctx);
555    if (len > 0)
556        ans = MPCLAMP((pos - start) / len, 0, 1);
557    if (ans < 0) {
558        int64_t size = demuxer->filesize;
559        if (size > 0 && demuxer->filepos >= 0)
560            ans = MPCLAMP(demuxer->filepos / (double)size, 0, 1);
561    }
562    if (use_range) {
563        if (mpctx->opts->play_frames > 0)
564            ans = MPMAX(ans, 1.0 -
565                    mpctx->max_frames / (double) mpctx->opts->play_frames);
566    }
567    return ans;
568}

681{
682    bool force_update = false;
683    struct MPOpts *opts = mpctx->opts;
684
685    if (!mpctx->demuxer || mpctx->encode_lavc_ctx) {
686        clear_underruns(mpctx);

688    }
689
690    double now = mp_time_sec();
691
692    struct demux_reader_state s;
693    demux_get_reader_state(mpctx->demuxer, &s);
694
695    mpctx->demux_underrun |= s.underrun;
696
697    int cache_buffer = 100;
698    bool use_pause_on_low_cache = opts->cache_pause && mpctx->play_dir > 0;
699
700    if (!mpctx->restart_complete) {
701        // Audio or video is restarting, and initial buffering is enabled. Make
702        // sure we actually restart them in paused mode, so no audio gets
703        // dropped and video technically doesn't start yet.
704        use_pause_on_low_cache &= opts->cache_pause_initial &&
705                                    (mpctx->video_status == STATUS_READY ||

707    }

710                  s.ts_duration < opts->cache_pause_wait;

713    // initial caching before playback restart is used).
714    bool need_wait = is_low;
715    if (is_low && !mpctx->paused_for_cache && mpctx->restart_complete) {
716        // Wait only if an output underrun was registered. (Or if there is no
717        // underrun detection.)

723            output_underrun |= mpctx->vo_chain->underrun;
724
725        // Output underruns could be sporadic (unrelated to demuxer buffer state
726        // and for example caused by slow decoding), so use a past demuxer
727        // underrun as indication that the underrun was possibly due to a
728        // demuxer underrun.

761            force_update = true;
762        }
763        if (mpctx->next_cache_update > 0)
764            mp_set_timeout(mpctx, mpctx->next_cache_update - now);
765    }

871{
872    struct MPOpts *opts = mpctx->opts;
873
874    if (mpctx->stop_play != AT_END_OF_FILE)
875        return;
876

878    enum seek_precision prec = MPSEEK_DEFAULT;

880    double ab[2];

897        if (!mpctx->shown_aframes && !mpctx->shown_vframes) {
898            MP_WARN(mpctx, "No media data to loop.\n");
899            return;
900        }
901
902        mpctx->stop_play = KEEP_PLAYING;
903        set_osd_function(mpctx, OSD_FFW);
904        mark_seek(mpctx);
905
906        // Assumes execute_queued_seek() happens before next audio/video is
907        // attempted to be decoded or filtered.
