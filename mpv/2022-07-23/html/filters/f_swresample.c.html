120// Modify out_layout and return the new value. The intention is reducing the
121// loss libswresample's rematrixing will cause by exchanging similar, but
122// strictly speaking incompatible channel pairs. For example, 7.1 should be
123// changed to 7.1(wide) without dropping the SL/SR channels. (We still leave
124// it to libswresample to create the remix matrix.)
125static uint64_t fudge_layout_conversion(struct priv *p,
126                                        uint64_t in, uint64_t out)
127{
128    for (int n = 0; n < MP_ARRAY_SIZE(fudge_pairs); n++) {
129        uint64_t a = mp_chmap_to_lavc(&fudge_pairs[n][0]);
130        uint64_t b = mp_chmap_to_lavc(&fudge_pairs[n][1]);
131        if (
 Recommendations (Experimental)  R1: ((in * in * in) == a && (b % in) == 0 && (out % in) == 0)
R2: (((in == a) || in == b) && out == 0)
R3: (((in == a) || in == b) && out > 0)
R4: (((in == a) || in == b) && out != {const})
R5: (((in == a) || (in == b)) && (out == 0))
 Score  Issue:  1 unfamiliar pattern(s) detected 
Cost: 30.75, Total Score: 1030.75
Class: complex_2
(in & a) == a && (in & b) == 0 &&
132            (out & a) == 0 && (out & b) == b)
133        {
134            out = (out & ~b) | a;
135
136            MP_VERBOSE(p, "Fudge: %s -> %s\n",
137                       mp_chmap_to_str(&fudge_pairs[n][0]),
138                       mp_chmap_to_str(&fudge_pairs[n][1]));
139        }
140    }
141    return out;
142}

144// mp_chmap_get_reorder() performs:
145//  to->speaker[n] = from->speaker[src[n]]
146// but libavresample does:
147//  to->speaker[dst[n]] = from->speaker[n]
148static void transpose_order(int *map, int num)
149{
150    int nmap[MP_NUM_CHANNELS] = {0};
151    for (int n = 0; n < num; n++) {
152        for (int i = 0; i < num; i++) {
153            if (map[n] == i)
154                nmap[i] = n;
155        }
156    }
157    memcpy(map, nmap, sizeof(nmap));
158}

160static bool configure_lavrr(struct priv *p, bool verbose)
161{
162    close_lavrr(p);
163
164    p->in_rate = rate_from_speed(p->in_rate_user, p->speed);
165
166    MP_VERBOSE(p, "%dHz %s %s -> %dHz %s %s\n",
167               p->in_rate, mp_chmap_to_str(&p->in_channels),
168               af_fmt_to_str(p->in_format),
169               p->out_rate, mp_chmap_to_str(&p->out_channels),
170               af_fmt_to_str(p->out_format));
171
172    p->avrctx = swr_alloc();
173    p->avrctx_out = swr_alloc();
174    if (!p->avrctx || !p->avrctx_out)
175        goto error;
176
177    enum AVSampleFormat in_samplefmt = af_to_avformat(p->in_format);
178    enum AVSampleFormat out_samplefmt = af_to_avformat(p->out_format);
179    enum AVSampleFormat out_samplefmtp = av_get_planar_sample_fmt(out_samplefmt);
180
181    if (in_samplefmt == AV_SAMPLE_FMT_NONE ||
182        out_samplefmt == AV_SAMPLE_FMT_NONE ||
183        out_samplefmtp == AV_SAMPLE_FMT_NONE)
184    {
185        MP_ERR(p, "unsupported conversion: %s -> %s\n",
186               af_fmt_to_str(p->in_format), af_fmt_to_str(p->out_format));
187        goto error;
188    }
189
190    av_opt_set_int(p->avrctx, "filter_size",        p->opts->filter_size, 0);
191    av_opt_set_int(p->avrctx, "phase_shift",        p->opts->phase_shift, 0);
192    av_opt_set_int(p->avrctx, "linear_interp",      p->opts->linear, 0);
193
194    double cutoff = p->opts->cutoff;
195    if (cutoff <= 0.0)
196        cutoff = MPMAX(1.0 - 6.5 / (p->opts->filter_size + 8), 0.80);
197    av_opt_set_double(p->avrctx, "cutoff",          cutoff, 0);
198
199    int normalize = p->opts->normalize;
200    av_opt_set_double(p->avrctx, "rematrix_maxval", normalize ? 1 : 1000, 0);
201
202    if (mp_set_avopts(p->log, p->avrctx, p->opts->avopts) < 0)
203        goto error;
204
205    struct mp_chmap map_in = p->in_channels;
206    struct mp_chmap map_out = p->out_channels;
207
208    // Try not to do any remixing if at least one is "unknown". Some corner
209    // cases also benefit from disabling all channel handling logic if the
210    // src/dst layouts are the same (like fl-fr-na -> fl-fr-na).
211    if (mp_chmap_is_unknown(&map_in) || mp_chmap_is_unknown(&map_out) ||
212        mp_chmap_equals(&map_in, &map_out))
213    {
214        mp_chmap_set_unknown(&map_in, map_in.num);
215        mp_chmap_set_unknown(&map_out, map_out.num);
216    }
217
218    // unchecked: don't take any channel reordering into account
219    uint64_t in_ch_layout = mp_chmap_to_lavc_unchecked(&map_in);
220    uint64_t out_ch_layout = mp_chmap_to_lavc_unchecked(&map_out);
221
222    struct mp_chmap in_lavc, out_lavc;
223    mp_chmap_from_lavc(&in_lavc, in_ch_layout);
224    mp_chmap_from_lavc(&out_lavc, out_ch_layout);
225
226    if (verbose && !mp_chmap_equals(&in_lavc, &out_lavc)) {
227        MP_VERBOSE(p, "Remix: %s -> %s\n", mp_chmap_to_str(&in_lavc),
228                                            mp_chmap_to_str(&out_lavc));
229    }
230
231    if (in_lavc.num != map_in.num) {
232        // For handling NA channels, we would have to add a planarization step.
233        MP_FATAL(p, "Unsupported input channel layout %s.\n",
234                 mp_chmap_to_str(&map_in));
235        goto error;
236    }
237
238    mp_chmap_get_reorder(p->reorder_in, &map_in, &in_lavc);
239    transpose_order(p->reorder_in, map_in.num);
240
241    if (mp_chmap_equals(&out_lavc, &map_out)) {
242        // No intermediate step required - output new format directly.
243        out_samplefmtp = out_samplefmt;
244    } else {
245        // Verify that we really just reorder and/or insert NA channels.
246        struct mp_chmap withna = out_lavc;
247        mp_chmap_fill_na(&withna, map_out.num);
248        if (withna.num != map_out.num)
249            goto error;
250    }
251    mp_chmap_get_reorder(p->reorder_out, &out_lavc, &map_out);
252
253    p->pre_out_fmt = mp_aframe_create();
254    mp_aframe_set_rate(p->pre_out_fmt, p->out_rate);
255    mp_aframe_set_chmap(p->pre_out_fmt, &p->out_channels);
256    mp_aframe_set_format(p->pre_out_fmt, p->out_format);
257
258    p->avrctx_fmt = mp_aframe_create();
259    mp_aframe_config_copy(p->avrctx_fmt, p->pre_out_fmt);
260    mp_aframe_set_chmap(p->avrctx_fmt, &out_lavc);
261    mp_aframe_set_format(p->avrctx_fmt, af_from_avformat(out_samplefmtp));
262
263    // If there are NA channels, the final output will have more channels than
264    // the avrctx output. Also, avrctx will output planar (out_samplefmtp was
265    // not overwritten). Allocate the output frame with more channels, so the
266    // NA channels can be trivially added.
267    p->pool_fmt = mp_aframe_create();
268    mp_aframe_config_copy(p->pool_fmt, p->avrctx_fmt);
269    if (map_out.num > out_lavc.num)
270        mp_aframe_set_chmap(p->pool_fmt, &map_out);
271
272    out_ch_layout = fudge_layout_conversion(p, in_ch_layout, out_ch_layout);
273
274    // Real conversion; output is input to avrctx_out.
275    av_opt_set_int(p->avrctx, "in_channel_layout",  in_ch_layout, 0);
276    av_opt_set_int(p->avrctx, "out_channel_layout", out_ch_layout, 0);
277    av_opt_set_int(p->avrctx, "in_sample_rate",     p->in_rate, 0);
278    av_opt_set_int(p->avrctx, "out_sample_rate",    p->out_rate, 0);
279    av_opt_set_int(p->avrctx, "in_sample_fmt",      in_samplefmt, 0);
280    av_opt_set_int(p->avrctx, "out_sample_fmt",     out_samplefmtp, 0);
281
282    // Just needs the correct number of channels for deplanarization.
283    struct mp_chmap fake_chmap;
284    mp_chmap_set_unknown(&fake_chmap, map_out.num);
285    uint64_t fake_out_ch_layout = mp_chmap_to_lavc_unchecked(&fake_chmap);
286    if (!fake_out_ch_layout)
287        goto error;
288    av_opt_set_int(p->avrctx_out, "in_channel_layout",  fake_out_ch_layout, 0);
289    av_opt_set_int(p->avrctx_out, "out_channel_layout", fake_out_ch_layout, 0);
290
291    av_opt_set_int(p->avrctx_out, "in_sample_fmt",      out_samplefmtp, 0);
292    av_opt_set_int(p->avrctx_out, "out_sample_fmt",     out_samplefmt, 0);
293    av_opt_set_int(p->avrctx_out, "in_sample_rate",     p->out_rate, 0);
294    av_opt_set_int(p->avrctx_out, "out_sample_rate",    p->out_rate, 0);
295
296    // API has weird requirements, quoting avresample.h:
297    //  * This function can only be called when the allocated context is not open.
298    //  * Also, the input channel layout must have already been set.
299    swr_set_channel_mapping(p->avrctx, p->reorder_in);
300
301    p->is_resampling = false;
302
303    if (swr_init(p->avrctx) < 0 || swr_init(p->avrctx_out) < 0) {
304        MP_ERR(p, "Cannot open Libavresample context.\n");
305        goto error;
306    }
307    return true;
308
309error:
310    close_lavrr(p);
311    mp_filter_internal_mark_failed(p->public.f);
312    MP_FATAL(p, "libswresample failed to initialize.\n");
313    return false;
314}

330// This relies on the tricky way mpa was allocated.
331static bool reorder_planes(struct mp_aframe *mpa, int *reorder,
332                           struct mp_chmap *newmap)
333{
334    if (!mp_aframe_set_chmap(mpa, newmap))
335        return false;
336
337    int num_planes = mp_aframe_get_planes(mpa);
338    uint8_t **planes = mp_aframe_get_data_rw(mpa);
339    uint8_t *old_planes[MP_NUM_CHANNELS];
340    assert(num_planes <= MP_NUM_CHANNELS);
341    for (int n = 0; n < num_planes; n++)
342        old_planes[n] = planes[n];
343
344    int next_na = 0;
345    for (int n = 0; n < num_planes; n++)
346        next_na += newmap->speaker[n] != MP_SPEAKER_ID_NA;
347
348    for (int n = 0; n < num_planes; n++) {
349        int src = reorder[n];
350        assert(src >= -1 && src < num_planes);
351        if (src >= 0) {
352            planes[n] = old_planes[src];
353        } else {
354            assert(next_na < num_planes);
355            planes[n] = old_planes[next_na++];
356            // The NA planes were never written by avrctx, so clear them.
357            af_fill_silence(planes[n],
358                            mp_aframe_get_sstride(mpa) * mp_aframe_get_size(mpa),
359                            mp_aframe_get_format(mpa));
360        }
361    }
362
363    return true;
364}

366static int resample_frame(struct SwrContext *r,
367                          struct mp_aframe *out, struct mp_aframe *in,
368                          int consume_in)
369{
370    // Be aware that the channel layout and count can be different for in and
371    // out frames. In some situations the caller will fix up the frames before
372    // or after conversion. The sample rates can also be different.
373    AVFrame *av_i = in ? mp_aframe_get_raw_avframe(in) : NULL;
374    AVFrame *av_o = out ? mp_aframe_get_raw_avframe(out) : NULL;
375    return swr_convert(r,
376        av_o ? av_o->extended_data : NULL,
377        av_o ? av_o->nb_samples : 0,
378        (const uint8_t **)(av_i ? av_i->extended_data : NULL),
379        av_i ? MPMIN(av_i->nb_samples, consume_in) : 0);
380}

382static struct mp_frame filter_resample_output(struct priv *p,
383                                              struct mp_aframe *in)
384{
385    struct mp_aframe *out = NULL;
386
387    if (!p->avrctx)
388        goto error;
389
390    // Limit the filtered data size for better latency when changing speed.
391    // Avoid buffering data within the resampler => restrict input size.
392    // p->in_rate already includes the speed factor.
393    double s = p->opts->max_output_frame_size / 1000 * p->in_rate;
394    int max_in = lrint(MPCLAMP(s, 128, INT_MAX));
395    int consume_in = in ? mp_aframe_get_size(in) : 0;
396    consume_in = MPMIN(consume_in, max_in);
397
398    int samples = get_out_samples(p, consume_in);
399    out = mp_aframe_create();
400    mp_aframe_config_copy(out, p->pool_fmt);
401    if (mp_aframe_pool_allocate(p->out_pool, out, samples) < 0)
402        goto error;
403
404    int out_samples = 0;
405    if (samples) {
406        out_samples = resample_frame(p->avrctx, out, in, consume_in);
407        if (out_samples < 0 || out_samples > samples)
408            goto error;
409        mp_aframe_set_size(out, out_samples);
410    }
411
412    struct mp_chmap out_chmap;
413    if (!mp_aframe_get_chmap(p->pool_fmt, &out_chmap))
414        goto error;
415    if (!reorder_planes(out, p->reorder_out, &out_chmap))
416        goto error;
417
418    if (!mp_aframe_config_equals(out, p->pre_out_fmt)) {
419        struct mp_aframe *new = mp_aframe_create();
420        mp_aframe_config_copy(new, p->pre_out_fmt);
421        if (mp_aframe_pool_allocate(p->reorder_buffer, new, out_samples) < 0) {
422            talloc_free(new);
423            goto error;
424        }
425        int got = 0;
426        if (out_samples)
427            got = resample_frame(p->avrctx_out, new, out, out_samples);
428        talloc_free(out);
429        out = new;
430        if (got != out_samples)
431            goto error;
432    }
433
434    if (in) {
435        mp_aframe_copy_attributes(out, in);
436        p->current_pts = mp_aframe_end_pts(in);
437        mp_aframe_skip_samples(in, consume_in);
438    }
439
440    if (out_samples) {
441        if (p->current_pts != MP_NOPTS_VALUE) {
442            double delay = get_delay(p) * mp_aframe_get_speed(out) +
443                           mp_aframe_duration(out) +
444                           (p->input ? mp_aframe_duration(p->input) : 0);
445            mp_aframe_set_pts(out, p->current_pts - delay);
446            mp_aframe_mul_speed(out, p->speed);
447        }
448    } else {
449        TA_FREEP(&out);
450    }
451
452    return out ? MAKE_FRAME(MP_FRAME_AUDIO, out) : MP_NO_FRAME;
453error:
454    talloc_free(out);
455    MP_ERR(p, "Error on resampling.\n");
456    mp_filter_internal_mark_failed(p->public.f);
457    return MP_NO_FRAME;
458}
459
460static void process(struct mp_filter *f)
461{
462    struct priv *p = f->priv;
463
464    if (!mp_pin_in_needs_data(f->ppins[1]))
465        return;
466
467    p->speed = p->cmd_speed * p->public.speed;
468
469    struct mp_aframe *input = NULL;
470    if (!p->input) {
471        struct mp_frame frame = mp_pin_out_read(f->ppins[0]);
472
473        if (frame.type == MP_FRAME_AUDIO) {
474            input = frame.data;
475        } else if (!frame.type) {
476            return; // no new data
477        } else if (frame.type != MP_FRAME_EOF) {
478            MP_ERR(p, "Unsupported frame type.\n");
479            mp_frame_unref(&frame);
480            mp_filter_internal_mark_failed(f);
481            return;
482        }
483
484        if (!input && !p->avrctx) {
485            // Obviously no draining needed.
486            mp_pin_in_write(f->ppins[1], MP_EOF_FRAME);
487            return;
488        }
489    }
490
491    if (input) {
492        assert(!p->input);
493
494        struct mp_swresample *s = &p->public;
495
496        int in_rate = mp_aframe_get_rate(input);
497        int in_format = mp_aframe_get_format(input);
498        struct mp_chmap in_channels = {0};
499        mp_aframe_get_chmap(input, &in_channels);
500
501        if (!in_rate || !in_format || !in_channels.num) {
502            MP_ERR(p, "Frame with invalid format unsupported\n");
503            talloc_free(input);
504            mp_filter_internal_mark_failed(f);
505            return;
506        }
507
508        int out_rate = s->out_rate ? s->out_rate : in_rate;
509        int out_format = s->out_format ? s->out_format : in_format;
510        struct mp_chmap out_channels =
511            s->out_channels.num ? s->out_channels : in_channels;
512
513        if (p->in_rate_user != in_rate ||
514            p->in_format != in_format ||
515            !mp_chmap_equals(&p->in_channels, &in_channels) ||
516            p->out_rate != out_rate ||
517            p->out_format != out_format ||
518            !mp_chmap_equals(&p->out_channels, &out_channels) ||
519            !p->avrctx)
520        {
521            if (p->avrctx) {
522                // drain remaining audio
523                struct mp_frame out = filter_resample_output(p, NULL);
524                if (out.type) {
525                    mp_pin_in_write(f->ppins[1], out);
526                    // continue filtering next time.
527                    mp_pin_out_unread(f->ppins[0],
528                                      MAKE_FRAME(MP_FRAME_AUDIO, input));
529                    input = NULL;
530                }
531            }
532
533            MP_VERBOSE(p, "format change, reinitializing resampler\n");
534
535            p->in_rate_user = in_rate;
536            p->in_format = in_format;
537            p->in_channels = in_channels;
538            p->out_rate = out_rate;
539            p->out_format = out_format;
540            p->out_channels = out_channels;
541
542            if (!configure_lavrr(p, true)) {
543                talloc_free(input);
544                return;
545            }
546
547            if (!input) {
548                // continue filtering next time
549                mp_filter_internal_mark_progress(f);
550                return;
551            }
552        }
553
554        p->input = input;
555    }
556
557    int new_rate = rate_from_speed(p->in_rate_user, p->speed);
558    bool exact_rate = new_rate == p->in_rate;
559    bool use_comp = fabs(new_rate / (double)p->in_rate - 1) <= 0.01;
560    // If we've never used compensation, avoid setting it - even if it's in
561    // theory a NOP, libswresample will enable resampling. _If_ we're
562    // resampling, we might have to disable previously enabled compensation.
563    if (exact_rate && !p->is_resampling)
564        use_comp = false;
565    if (p->avrctx && use_comp) {
566        AVRational r =
567            av_d2q(p->speed * p->in_rate_user / p->in_rate, INT_MAX / 2);
568        // Essentially, swr_set_compensation() does 2 things:
569        // - adjust output sample rate by sample_delta/compensation_distance
570        // - reset the adjustment after compensation_distance output samples
571        // Increase the compensation_distance to avoid undesired reset
572        // semantics - we want to keep the ratio for the whole frame we're
573        // feeding it, until the next filter() call.
574        int mult = INT_MAX / 2 / MPMAX(MPMAX(abs(r.num), abs(r.den)), 1);
575        r = (AVRational){ r.num * mult, r.den * mult };
576        if (r.den == r.num)
577            r = (AVRational){0}; // fully disable
578        if (swr_set_compensation(p->avrctx, r.den - r.num, r.den) >= 0) {
579            exact_rate = true;
580            p->is_resampling = true; // libswresample can auto-enable it
581        }
582    }
583
584    if (!exact_rate) {
585        // Before reconfiguring, drain the audio that is still buffered
586        // in the resampler.
587        struct mp_frame out = filter_resample_output(p, NULL);
588        bool need_drain = !!out.type;
589        if (need_drain)
590            mp_pin_in_write(f->ppins[1], out);
591        // Reinitialize resampler.
592        configure_lavrr(p, false);
593        // If we've written output, we must continue filtering next time.
594        if (need_drain)
595            return;
596    }
597
598    struct mp_frame out = filter_resample_output(p, p->input);
599
600    if (out.type) {
601        mp_pin_in_write(f->ppins[1], out);
602        if (!p->input)
603            mp_pin_out_repeat_eof(f->ppins[0]);
604    } else if (p->input) {
605        mp_filter_internal_mark_progress(f); // try to consume more input
606    } else {
607        mp_pin_in_write(f->ppins[1], MP_EOF_FRAME);
608    }
609
610    if (p->input && !mp_aframe_get_size(p->input))
611        TA_FREEP(&p->input);
612}

621static bool command(struct mp_filter *f, struct mp_filter_command *cmd)

650struct mp_swresample *mp_swresample_create(struct mp_filter *parent,
651                                           struct mp_resample_opts *opts)
652{
653    struct mp_filter *f = mp_filter_create(parent, &swresample_filter);
654    if (!f)
655        return NULL;
656
657    mp_filter_add_pin(f, MP_PIN_IN, "in");
658    mp_filter_add_pin(f, MP_PIN_OUT, "out");
659
660    struct priv *p = f->priv;
661    p->public.f = f;
662    p->public.speed = 1.0;
663    p->cmd_speed = 1.0;
664    p->log = f->log;
665
666    if (opts) {
667        p->opts = talloc_dup(p, opts);
668        p->opts->avopts = mp_dup_str_array(p, p->opts->avopts);
669    } else {
670        p->opts = mp_get_config_group(p, f->global, &resample_conf);
671    }
672
673    p->reorder_buffer = mp_aframe_pool_create(p);
674    p->out_pool = mp_aframe_pool_create(p);
675
676    return &p->public;
677}
