238static int decoder_list_help(struct mp_log *log, const m_option_t *opt,
239                             struct bstr name)
240{
241    if (strcmp(opt->name, "ad") == 0) {
242        struct mp_decoder_list *list = audio_decoder_list();
243        mp_print_decoders(log, MSGL_INFO, "Audio decoders:", list);
244        talloc_free(list);
245        return M_OPT_EXIT;
246    }
247    if (strcmp(opt->name, "vd") == 0) {
248        struct mp_decoder_list *list = video_decoder_list();
249        mp_print_decoders(log, MSGL_INFO, "Video decoders:", list);
250        talloc_free(list);
251        return M_OPT_EXIT;
252    }
253    if (strcmp(opt->name, "audio-spdif") == 0) {
254        mp_info(log, "Choices: ac3,dts-hd,dts (and possibly more)\n");
255        return M_OPT_EXIT;
256    }
257    return 1;
258}

260// Update cached values for main thread which require access to the decoder
261// thread state. Must run on/locked with decoder thread.
262static void update_cached_values(struct priv *p)
263{
264    pthread_mutex_lock(&p->cache_lock);
265
266    p->cur_hwdec = NULL;
267    if (p->decoder && p->decoder->control)
268        p->decoder->control(p->decoder->f, VDCTRL_GET_HWDEC, &p->cur_hwdec);
269
270    pthread_mutex_unlock(&p->cache_lock);
271}
272
273// Lock the decoder thread. This may synchronously wait until the decoder thread
274// is done with its current work item (such as waiting for a frame), and thus
275// may block for a while. (I.e. avoid during normal playback.)
276// If no decoder thread is running, this is a no-op, except for some debug stuff.
277static void thread_lock(struct priv *p)
278{
279    if (p->dec_dispatch)
280        mp_dispatch_lock(p->dec_dispatch);
281
282    assert(!p->dec_thread_lock);
283    p->dec_thread_lock = true;
284}
285
286// Undo thread_lock().
287static void thread_unlock(struct priv *p)
288{
289    assert(p->dec_thread_lock);
290    p->dec_thread_lock = false;
291
292    if (p->dec_dispatch)
293        mp_dispatch_unlock(p->dec_dispatch);
294}

296// This resets only the decoder. Unlike a full reset(), this doesn't imply a
297// seek reset. This distinction exists only when using timeline stuff (EDL and
298// ordered chapters). timeline stuff needs to reset the decoder state, but keep
299// some of the user-relevant state.
300static void reset_decoder(struct priv *p)
301{
302    p->first_packet_pdts = MP_NOPTS_VALUE;
303    p->start_pts = MP_NOPTS_VALUE;
304    p->codec_pts = MP_NOPTS_VALUE;
305    p->codec_dts = MP_NOPTS_VALUE;
306    p->num_codec_pts_problems = 0;
307    p->num_codec_dts_problems = 0;
308    p->has_broken_decoded_pts = 0;
309    p->packets_without_output = 0;
310    mp_frame_unref(&p->packet);
311    p->packet_fed = false;
312    p->preroll_discard = false;
313    talloc_free(p->new_segment);
314    p->new_segment = NULL;
315    p->start = p->end = MP_NOPTS_VALUE;
316
317    if (p->decoder)
318        mp_filter_reset(p->decoder->f);
319}

321static void decf_reset(struct mp_filter *f)
322{
323    struct priv *p = f->priv;
324    assert(p->decf == f);
325
326    p->pts = MP_NOPTS_VALUE;
327    p->last_format = p->fixed_format = (struct mp_image_params){0};
328
329    pthread_mutex_lock(&p->cache_lock);
330    p->pts_reset = false;
331    p->attempt_framedrops = 0;
332    p->dropped_frames = 0;
333    pthread_mutex_unlock(&p->cache_lock);
334
335    p->coverart_returned = 0;
336
337    for (int n = 0; n < p->num_reverse_queue; n++)
338        mp_frame_unref(&p->reverse_queue[n]);
339    p->num_reverse_queue = 0;
340    p->reverse_queue_byte_size = 0;
341    p->reverse_queue_complete = false;
342
343    reset_decoder(p);
344}

346int mp_decoder_wrapper_control(struct mp_decoder_wrapper *d,
347                               enum dec_ctrl cmd, void *arg)
348{
349    struct priv *p = d->f->priv;
350    int res = CONTROL_UNKNOWN;
351    if (cmd == VDCTRL_GET_HWDEC) {
352        pthread_mutex_lock(&p->cache_lock);
353        *(char **)arg = p->cur_hwdec;
354        pthread_mutex_unlock(&p->cache_lock);
355    } else {
356        thread_lock(p);
357        if (p->decoder && p->decoder->control)
358            res = p->decoder->control(p->decoder->f, cmd, arg);
359        update_cached_values(p);
360        thread_unlock(p);
361    }
362    return res;
363}

365static void decf_destroy(struct mp_filter *f)
366{
367    struct priv *p = f->priv;
368    assert(p->decf == f);
369
370    if (p->decoder) {
371        MP_DBG(f, "Uninit decoder.\n");
372        talloc_free(p->decoder->f);
373        p->decoder = NULL;
374    }
375
376    decf_reset(f);
377    mp_frame_unref(&p->decoded_coverart);
378}

394static bool reinit_decoder(struct priv *p)
395{
396    if (p->decoder)
397        talloc_free(p->decoder->f);
398    p->decoder = NULL;
399
400    reset_decoder(p);
401    p->has_broken_packet_pts = -10; // needs 10 packets to reach decision
402
403    talloc_free(p->decoder_desc);
404    p->decoder_desc = NULL;
405
406    const struct mp_decoder_fns *driver = NULL;
407    struct mp_decoder_list *list = NULL;
408    char *user_list = NULL;
409    char *fallback = NULL;
410
411    if (p->codec->type == STREAM_VIDEO) {
412        driver = &vd_lavc;
413        user_list = p->opts->video_decoders;
414        fallback = "h264";
415    } else if (p->codec->type == STREAM_AUDIO) {
416        driver = &ad_lavc;
417        user_list = p->opts->audio_decoders;
418        fallback = "aac";
419
420        pthread_mutex_lock(&p->cache_lock);
421        bool try_spdif = p->try_spdif;
422        pthread_mutex_unlock(&p->cache_lock);
423
424        if (try_spdif && p->codec->codec) {
425            struct mp_decoder_list *spdif =
426                select_spdif_codec(p->codec->codec, p->opts->audio_spdif);
427            if (spdif->num_entries) {
428                driver = &ad_spdif;
429                list = spdif;
430            } else {
431                talloc_free(spdif);
432            }
433        }
434    }
435
436    if (!list) {
437        struct mp_decoder_list *full = talloc_zero(NULL, struct mp_decoder_list);
438        if (driver)
439            driver->add_decoders(full);
440        const char *codec = p->codec->codec;
441        if (codec && strcmp(codec, "null") == 0)
442            codec = fallback;
443        list = mp_select_decoders(p->log, full, codec, user_list);
444        talloc_free(full);
445    }
446
447    mp_print_decoders(p->log, MSGL_V, "Codec list:", list);
448
449    for (int n = 0; n < list->num_entries; n++) {
450        struct mp_decoder_entry *sel = &list->entries[n];
451        MP_VERBOSE(p, "Opening decoder %s\n", sel->decoder);
452
453        p->decoder = driver->create(p->decf, p->codec, sel->decoder);
454        if (p->decoder) {
455            pthread_mutex_lock(&p->cache_lock);
456            p->decoder_desc =
457                talloc_asprintf(p, "%s (%s)", sel->decoder, sel->desc);
458            MP_VERBOSE(p, "Selected codec: %s\n", p->decoder_desc);
459            pthread_mutex_unlock(&p->cache_lock);
460            break;
461        }
462
463        MP_WARN(p, "Decoder init failed for %s\n", sel->decoder);
464    }
465
466    if (!p->decoder) {
467        MP_ERR(p, "Failed to initialize a decoder for codec '%s'.\n",
468               p->codec->codec ? p->codec->codec : "<?>");
469    }
470
471    update_cached_values(p);
472
473    talloc_free(list);
474    return !!p->decoder;
475}

559static void fix_image_params(struct priv *p,
560                             struct mp_image_params *params)
561{
562    struct mp_image_params m = *params;
563    struct mp_codec_params *c = p->codec;
564    struct dec_wrapper_opts *opts = p->opts;
565
566    MP_VERBOSE(p, "Decoder format: %s\n", mp_image_params_to_str(params));
567    p->dec_format = *params;
568
569    // While mp_image_params normally always have to have d_w/d_h set, the
570    // decoder signals unknown bitstream aspect ratio with both set to 0.
571    bool use_container = true;
572    if (opts->aspect_method == 1 && m.p_w > 0 && m.p_h > 0) {
573        MP_VERBOSE(p, "Using bitstream aspect ratio.\n");
574        use_container = false;
575    }
576
577    if (use_container && c->par_w > 0 && c->par_h) {
578        MP_VERBOSE(p, "Using container aspect ratio.\n");
579        m.p_w = c->par_w;
580        m.p_h = c->par_h;
581    }
582
583    if (opts->movie_aspect >= 0) {
584        MP_VERBOSE(p, "Forcing user-set aspect ratio.\n");
585        if (opts->movie_aspect == 0) {
586            m.p_w = m.p_h = 1;
587        } else {
588            AVRational a = av_d2q(opts->movie_aspect, INT_MAX);
589            mp_image_params_set_dsize(&m, a.num, a.den);
590        }
591    }
592
593    // Assume square pixels if no aspect ratio is set at all.
594    if (m.p_w <= 0 || m.p_h <= 0)
595        m.p_w = m.p_h = 1;
596
597    m.stereo3d = p->codec->stereo_mode;
598
599    if (opts->video_rotate < 0) {
600        m.rotate = 0;
601    } else {
602        // ffmpeg commit 535a835e51 says that frame rotate takes priority
603        if (!m.rotate)
604            m.rotate = p->codec->rotate;
605        m.rotate = (m.rotate + opts->video_rotate) % 360;
606    }
607
608    mp_colorspace_merge(&m.color, &c->color);
609
610    // Sanitize the HDR peak. Sadly necessary
611    if (!is_valid_peak(m.color.sig_peak)) {
612        MP_WARN(p, "Invalid HDR peak in stream: %f\n", m.color.sig_peak);
613        m.color.sig_peak = 0.0;
614    }
615
616    // Guess missing colorspace fields from metadata. This guarantees all
617    // fields are at least set to legal values afterwards.
618    mp_image_params_guess_csp(&m);
619
620    p->last_format = *params;
621    p->fixed_format = m;
622}

676// Return true if the current frame is outside segment range.
677static bool process_decoded_frame(struct priv *p, struct mp_frame *frame)
678{
679    if (frame->type == MP_FRAME_EOF) {
680        // if we were just draining current segment, don't propagate EOF
681        if (p->new_segment)
682            mp_frame_unref(frame);
683        return true;
684    }
685
686    bool segment_ended = false;
687
688    if (frame->type == MP_FRAME_VIDEO) {
689        struct mp_image *mpi = frame->data;
690
691        crazy_video_pts_stuff(p, mpi);
692
693        struct demux_packet *ccpkt = new_demux_packet_from_buf(mpi->a53_cc);
694        if (ccpkt) {
695            av_buffer_unref(&mpi->a53_cc);
696            ccpkt->pts = mpi->pts;
697            ccpkt->dts = mpi->dts;
698            demuxer_feed_caption(p->header, ccpkt);
699        }
700
701        // Stop hr-seek logic.
702        if (mpi->pts == MP_NOPTS_VALUE || mpi->pts >= p->start_pts)
703            p->start_pts = MP_NOPTS_VALUE;
704
705        if (mpi->pts != MP_NOPTS_VALUE) {
706            segment_ended = p->end != MP_NOPTS_VALUE && mpi->pts >= p->end;
707            if ((p->start != MP_NOPTS_VALUE && mpi->pts < p->start) ||
708                segment_ended)
709            {
710                mp_frame_unref(frame);
711                goto done;
712            }
713        }
714    } else if (frame->type == MP_FRAME_AUDIO) {
715        struct mp_aframe *aframe = frame->data;
716
717        mp_aframe_clip_timestamps(aframe, p->start, p->end);
718        double pts = mp_aframe_get_pts(aframe);
719        if (pts != MP_NOPTS_VALUE && p->start != MP_NOPTS_VALUE)
720            segment_ended = pts >= p->end;
721
722        if (mp_aframe_get_size(aframe) == 0) {
723            mp_frame_unref(frame);
724            goto done;
725        }
726    } else {
727        MP_ERR(p, "unknown frame type from decoder\n");
728    }
729
730done:
731    return segment_ended;
732}

734static void correct_video_pts(struct priv *p, struct mp_image *mpi)
735{
736    mpi->pts *= p->play_dir;
737
738    if (!p->opts->correct_pts || mpi->pts == MP_NOPTS_VALUE) {
739        double fps = p->fps > 0 ? p->fps : 25;
740
741        if (p->opts->correct_pts) {
742            if (p->has_broken_decoded_pts <= 1) {
743                MP_WARN(p, "No video PTS! Making something up. Using "
744                        "%f FPS.\n", fps);
745                if (p->has_broken_decoded_pts == 1)
746                    MP_WARN(p, "Ignoring further missing PTS warnings.\n");
747                p->has_broken_decoded_pts++;
748            }
749        }
750
751        double frame_time = 1.0f / fps;
752        double base = p->first_packet_pdts;
753        mpi->pts = p->pts;
754        if (mpi->pts == MP_NOPTS_VALUE) {
755            mpi->pts = base == MP_NOPTS_VALUE ? 0 : base;
756        } else {
757            mpi->pts += frame_time;
758        }
759    }
760
761    p->pts = mpi->pts;
762}

764static void correct_audio_pts(struct priv *p, struct mp_aframe *aframe)
765{
766    double dir = p->play_dir;
767
768    double frame_pts = mp_aframe_get_pts(aframe);
769    double frame_len = mp_aframe_duration(aframe);
770
771    if (frame_pts != MP_NOPTS_VALUE) {
772        if (dir < 0)
773            frame_pts = -(frame_pts + frame_len);
774
775        if (p->pts != MP_NOPTS_VALUE)
776            MP_STATS(p, "value %f audio-pts-err", p->pts - frame_pts);
777
778        double diff = fabs(p->pts - frame_pts);
779
780        // Attempt to detect jumps in PTS. Even for the lowest sample rates and
781        // with worst container rounded timestamp, this should be a margin more
782        // than enough.
783        if (p->pts != MP_NOPTS_VALUE && diff > 0.1) {
784            MP_WARN(p, "Invalid audio PTS: %f -> %f\n", p->pts, frame_pts);
785            if (diff >= 5) {
786                pthread_mutex_lock(&p->cache_lock);
787                p->pts_reset = true;
788                pthread_mutex_unlock(&p->cache_lock);
789            }
790        }
791
792        // Keep the interpolated timestamp if it doesn't deviate more
793        // than 1 ms from the real one. (MKV rounded timestamps.)
794        if (p->pts == MP_NOPTS_VALUE || diff > 0.001)
795            p->pts = frame_pts;
796    }
797
798    if (p->pts == MP_NOPTS_VALUE && p->header->missing_timestamps)
799        p->pts = 0;
800
801    mp_aframe_set_pts(aframe, p->pts);
802
803    if (p->pts != MP_NOPTS_VALUE)
804        p->pts += frame_len;
805}

928static void enqueue_backward_frame(struct priv *p, struct mp_frame frame)
929{
930    bool eof = frame.type == MP_FRAME_EOF;
931

936        switch (p->header->type) {
937        case STREAM_VIDEO: queue_size = opts->video_reverse_size; break;
938        case STREAM_AUDIO: queue_size = opts->audio_reverse_size; break;
939        }

941        if (p->reverse_queue_byte_size >= queue_size) {
942            MP_ERR(p, "Reversal queue overflow, discarding frame.\n");

952                        eof ? 0 : p->num_reverse_queue, frame);

957static void read_frame(struct priv *p)
958{
959    struct mp_pin *pin = p->decf->ppins[0];
960    struct mp_frame frame = {0};
961
962    if (!p->decoder || !mp_pin_in_needs_data(pin))
963        return;
964
965    if (p->decoded_coverart.type) {
966        if (p->coverart_returned == 0) {
967            frame = mp_frame_ref(p->decoded_coverart);
968            p->coverart_returned = 1;
969            goto output_frame;
970        } else if (p->coverart_returned == 1) {
971            frame = MP_EOF_FRAME;
972            p->coverart_returned = 2;

979        frame = p->reverse_queue[p->num_reverse_queue - 1];

982    }

984

986    if (!frame.type)
987        return;
988
989    pthread_mutex_lock(&p->cache_lock);
990    if (p->attached_picture && frame.type == MP_FRAME_VIDEO)
991        p->decoded_coverart = frame;
992    if (p->attempt_framedrops) {
993        int dropped = MPMAX(0, p->packets_without_output - 1);
994        p->attempt_framedrops = MPMAX(0, p->attempt_framedrops - dropped);
995        p->dropped_frames += dropped;
996    }
997    pthread_mutex_unlock(&p->cache_lock);
998
999    if (p->decoded_coverart.type) {
1000        mp_filter_internal_mark_progress(p->decf);
1001        return;
1002    }
1003
1004    p->packets_without_output = 0;
1005
1006    if (p->preroll_discard && frame.type != MP_FRAME_EOF) {
1007        double ts = mp_frame_get_pts(frame);
1008        if (ts == MP_NOPTS_VALUE) {
1009            mp_frame_unref(&frame);
1010            mp_filter_internal_mark_progress(p->decf);
1011            return;
1012        }
1013        p->preroll_discard = false;
1014    }
1015
1016    bool segment_ended = process_decoded_frame(p, &frame);
1017
1018    if (p->play_dir < 0 && frame.type) {
1019        enqueue_backward_frame(p, frame);
1020        frame = MP_NO_FRAME;
1021    }
1022
1023    // If there's a new segment, start it as soon as we're drained/finished.
1024    if (segment_ended && p->new_segment) {
1025        struct demux_packet *new_segment = p->new_segment;
1026        p->new_segment = NULL;
1027
1028        reset_decoder(p);
1029
1030        if (new_segment->segmented) {
1031            if (p->codec != new_segment->codec) {
1032                p->codec = new_segment->codec;
1033                if (!mp_decoder_wrapper_reinit(&p->public))
1034                    mp_filter_internal_mark_failed(p->decf);
1035            }
1036
1037            p->start = new_segment->start;
1038            p->end = new_segment->end;
1039        }
1040
1041        p->reverse_queue_byte_size = 0;
1042        p->reverse_queue_complete = p->num_reverse_queue > 0;
1043
1044        p->packet = MAKE_FRAME(MP_FRAME_PACKET, new_segment);
1045        mp_filter_internal_mark_progress(p->decf);
1046    }
1047
1048    if (!frame.type) {
1049        mp_filter_internal_mark_progress(p->decf); // make it retry
1050        return;
1051    }
1052
1053output_frame:
1054    process_output_frame(p, frame);
1055    mp_pin_in_write(pin, frame);
1056}

1266void lavc_process(struct mp_filter *f, struct lavc_state *state,
1267                  int (*send)(struct mp_filter *f, struct demux_packet *pkt),
1268                  int (*receive)(struct mp_filter *f, struct mp_frame *res))
1269{
1270    if (!mp_pin_in_needs_data(f->ppins[1]))
1271        return;
1272
1273    struct mp_frame frame = {0};
1274    int ret_recv = receive(f, &frame);
1275    if (frame.type) {
1276        state->eof_returned = false;
1277        mp_pin_in_write(f->ppins[1], frame);
1278    } else if (ret_recv == AVERROR_EOF) {
1279        if (!state->eof_returned)
1280            mp_pin_in_write(f->ppins[1], MP_EOF_FRAME);
1281        state->eof_returned = true;
1282        state->packets_sent = false;
1283    } else if (ret_recv == AVERROR(EAGAIN)) {
1284        // Need to feed a packet.
1285        frame = mp_pin_out_read(f->ppins[0]);
1286        struct demux_packet *pkt = NULL;
1287        if (frame.type == MP_FRAME_PACKET) {
1288            pkt = frame.data;
1289        } else if (frame.type != MP_FRAME_EOF) {
1290            if (frame.type) {
1291                MP_ERR(f, "unexpected frame type\n");
1292                mp_frame_unref(&frame);
1293                mp_filter_internal_mark_failed(f);
1294            }
1295            return;
1296        } else if (!state->packets_sent) {
1297            // EOF only; just return it, without requiring send/receive to
1298            // pass it through properly.
1299            mp_pin_in_write(f->ppins[1], MP_EOF_FRAME);
1300            return;
1301        }
1302        int ret_send = send(f, pkt);
1303        if (ret_send == AVERROR(EAGAIN)) {
1304            // Should never happen, but can happen with broken decoders.
1305            MP_WARN(f, "could not consume packet\n");
1306            mp_pin_out_unread(f->ppins[0], frame);
1307            mp_filter_wakeup(f);
1308            return;
1309        }
1310        state->packets_sent = true;
1311        talloc_free(pkt);
1312        mp_filter_internal_mark_progress(f);
1313    } else {
1314        // Decoding error, or hwdec fallback recovery. Just try again.
1315        mp_filter_internal_mark_progress(f);
1316    }
