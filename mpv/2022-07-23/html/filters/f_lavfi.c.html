151static void add_pad(struct lavfi *c, int dir, int index, AVFilterContext *filter,
152                    int filter_pad, const char *name, bool first_init)
153{
154    if (c->failed)
155        return;
156
157    enum AVMediaType avmt;
158    if (dir == MP_PIN_IN) {
159        avmt = avfilter_pad_get_type(filter->input_pads, filter_pad);
160    } else {
161        avmt = avfilter_pad_get_type(filter->output_pads, filter_pad);
162    }
163    int type;
164    switch (avmt) {
165    case AVMEDIA_TYPE_VIDEO: type = MP_FRAME_VIDEO; break;
166    case AVMEDIA_TYPE_AUDIO: type = MP_FRAME_AUDIO; break;
167    default:
168        MP_FATAL(c, "unknown media type\n");
169        c->failed = true;
170        return;
171    }
172
173    // For anonymous pads, just make something up. libavfilter allows duplicate
174    // pad names (while we don't), so we check for collisions along with normal
175    // duplicate pads below.
176    char tmp[80];
177    const char *dir_string = dir == MP_PIN_IN ? "in" : "out";
178    if (name) {
179        if (c->direct_filter) {
180            // libavfilter has this very unpleasant thing that filter labels
181            // don't have to be unique - in particular, both input and output
182            // are usually named "default". With direct filters, the user has
183            // no chance to provide better names, so do something to resolve it.
184            snprintf(tmp, sizeof(tmp), "%s_%s", name, dir_string);
185            name = tmp;
186        }
187    } else {
188        snprintf(tmp, sizeof(tmp), "%s%d", dir_string, index);
189        name = tmp;
190    }
191
192    struct lavfi_pad *p = NULL;
193    for (int n = 0; n < c->num_all_pads; n++) {
194        if (strcmp(c->all_pads[n]->name, name) == 0) {
195            p = c->all_pads[n];
196            break;
197        }
198    }
199
200    if (p) {
201        // Graph recreation case: reassociate an existing pad.
202        if (p->filter) {
203            // Collision due to duplicate names.
204            MP_FATAL(c, "more than one pad with label '%s'\n", name);
205            c->failed = true;
206            return;
207        }
208        if (p->dir != dir || p->type != type) {
209            // libavfilter graph parser behavior not deterministic.
210            MP_FATAL(c, "pad '%s' changed type or direction\n", name);
211            c->failed = true;
212            return;
213        }
214    } else {
215        if (!first_init) {
216            MP_FATAL(c, "filter pad '%s' got added later?\n", name);
217            c->failed = true;
218            return;
219        }
220        p = talloc_zero(c, struct lavfi_pad);
221        p->main = c;
222        p->dir = dir;
223        p->name = talloc_strdup(p, name);
224        p->type = type;
225        p->pin_index = -1;
226        p->metadata = talloc_zero(p, struct mp_tags);
227        if (p->dir == MP_PIN_IN)
228            MP_TARRAY_APPEND(c, c->in_pads, c->num_in_pads, p);
229        if (p->dir == MP_PIN_OUT)
230            MP_TARRAY_APPEND(c, c->out_pads, c->num_out_pads, p);
231        MP_TARRAY_APPEND(c, c->all_pads, c->num_all_pads, p);
232    }
233    p->filter = filter;
234    p->filter_pad = filter_pad;
235}

251// Parse the user-provided filter graph, and populate the unlinked filter pads.
252static void precreate_graph(struct lavfi *c, bool first_init)
253{
254    assert(!c->graph);
255
256    c->failed = false;
257
258    c->graph = avfilter_graph_alloc();
259    if (!c->graph)
260        abort();
261
262    if (mp_set_avopts(c->log, c->graph, c->graph_opts) < 0)
263        goto error;
264
265    if (c->direct_filter) {
266        AVFilterContext *filter = avfilter_graph_alloc_filter(c->graph,
267                            avfilter_get_by_name(c->graph_string), "filter");
268        if (!filter) {
269            MP_FATAL(c, "filter '%s' not found or failed to allocate\n",
270                     c->graph_string);
271            goto error;
272        }
273
274        if (mp_set_avopts_pos(c->log, filter, filter->priv,
275                              c->direct_filter_opts) < 0)
276            goto error;
277
278        if (avfilter_init_str(filter, NULL) < 0) {
279            MP_FATAL(c, "filter failed to initialize\n");
280            goto error;
281        }
282
283        add_pads_direct(c, MP_PIN_IN, filter, filter->input_pads,
284                        filter->nb_inputs, first_init);
285        add_pads_direct(c, MP_PIN_OUT, filter, filter->output_pads,
286                        filter->nb_outputs, first_init);
287    } else {
288        AVFilterInOut *in = NULL, *out = NULL;
289        if (avfilter_graph_parse2(c->graph, c->graph_string, &in, &out) < 0) {
290            MP_FATAL(c, "parsing the filter graph failed\n");
291            goto error;
292        }
293        add_pads(c, MP_PIN_IN, in, first_init);
294        add_pads(c, MP_PIN_OUT, out, first_init);
295        avfilter_inout_free(&in);
296        avfilter_inout_free(&out);
297    }
298
299    for (int n = 0; n < c->num_all_pads; n++)
300        c->failed |= !c->all_pads[n]->filter;
301
302    if (c->failed)
303        goto error;
304
305    return;
306
307error:
308    free_graph(c);
309    c->failed = true;
310    return;
311}

313// Ensure to send EOF to each input pad, so the graph can be drained properly.
314static void send_global_eof(struct lavfi *c)
315{
316    for (int n = 0; n < c->num_in_pads; n++) {
317        struct lavfi_pad *pad = c->in_pads[n];
318        if (!pad->buffer || pad->buffer_is_eof)
319            continue;
320
321        if (av_buffersrc_add_frame(pad->buffer, NULL) < 0)
322            MP_FATAL(c, "could not send EOF to filter\n");
323
324        pad->buffer_is_eof = true;
325    }
326}

342           a->w == b->w && a->h && b->h &&
343           a->params.p_w == b->params.p_w && a->params.p_h == b->params.p_h &&
344           a->nominal_fps == b->nominal_fps;

346static bool is_format_ok(struct mp_frame a, struct mp_frame b)
347{
348    if (a.type == b.type && a.type == MP_FRAME_VIDEO)
349        return is_vformat_ok(a.data, b.data);
350    if (a.type == b.type && a.type == MP_FRAME_AUDIO)
351        return is_aformat_ok(a.data, b.data);
352    return false;
353}

483        } else if (pad->type == MP_FRAME_VIDEO) {
484            struct mp_image *fmt = pad->in_fmt.data;
485            params->format = imgfmt2pixfmt(fmt->imgfmt);
486            params->width = fmt->w;
487            params->height = fmt->h;
488            params->sample_aspect_ratio.num = fmt->params.p_w;
489            params->sample_aspect_ratio.den = fmt->params.p_h;
490            params->hw_frames_ctx = fmt->hwctx;
491            params->frame_rate = av_d2q(fmt->nominal_fps, 1000000);
492            filter_name = "buffer";
493        } else {
494            MP_ASSERT_UNREACHABLE();
495        }
496
497        params->time_base = pad->timebase;
498
499        const AVFilter *filter = avfilter_get_by_name(filter_name);
500        if (filter) {
501            char name[256];
502            snprintf(name, sizeof(name), "mpv_src_%s", pad->name);
503
504            pad->buffer = avfilter_graph_alloc_filter(c->graph, filter, name);
505        }
506        if (!pad->buffer) {
507            av_free(params);
508            goto error;
509        }
510
511        int ret = av_buffersrc_parameters_set(pad->buffer, params);
512        av_free(params);
513        if (ret < 0)
514            goto error;
515
516        if (avfilter_init_str(pad->buffer, NULL) < 0)
517            goto error;
518
519        if (avfilter_link(pad->buffer, 0, pad->filter, pad->filter_pad) < 0)
520            goto error;
521    }
522
523    return true;
524error:
525    MP_FATAL(c, "could not initialize filter pads\n");
526    c->failed = true;
527    return false;
528}

531{
532    MP_DBG(c, "Filter graph:\n");
533    char *s = avfilter_graph_dump(c->graph, NULL);

539// Initialize the graph if all inputs have formats set. If it's already
540// initialized, or can't be initialized yet, do nothing.
541static void init_graph(struct lavfi *c)
542{
543    assert(!c->initialized);
544
545    if (!c->graph)
546        precreate_graph(c, false);
547
548    if (init_pads(c)) {
549        struct mp_stream_info *info = mp_filter_find_stream_info(c->f);
550        if (info && info->hwdec_devs) {
551            struct mp_hwdec_ctx *hwdec = hwdec_devices_get_first(info->hwdec_devs);
552            for (int n = 0; n < c->graph->nb_filters; n++) {
553                AVFilterContext *filter = c->graph->filters[n];
554                if (hwdec && hwdec->av_device_ref)
555                    filter->hw_device_ctx = av_buffer_ref(hwdec->av_device_ref);
556            }
557        }
558
559        // And here the actual libavfilter initialization happens.
560        if (avfilter_graph_config(c->graph, NULL) < 0) {
561            MP_FATAL(c, "failed to configure the filter graph\n");
562            free_graph(c);
563            c->failed = true;
564            return;
565        }
566
567        // The timebase is available after configuring.
568        for (int n = 0; n < c->num_out_pads; n++) {
569            struct lavfi_pad *pad = c->out_pads[n];
570
571            pad->timebase = pad->buffer->inputs[0]->time_base;
572        }
573
574        c->initialized = true;
575
576        if (!c->direct_filter) // (output uninteresting for direct filters)
577            dump_graph(c);
578    }
579}

581static bool feed_input_pads(struct lavfi *c)
582{
583    bool progress = false;
584    bool was_draining = c->draining_recover;
585
586    assert(c->initialized);
587
588    for (int n = 0; n < c->num_in_pads; n++) {
589        struct lavfi_pad *pad = c->in_pads[n];
590
591        bool requested = av_buffersrc_get_nb_failed_requests(pad->buffer) > 0;
592
593        // Always request a frame after EOF so that we can know if the EOF state
594        // changes (e.g. for sparse streams with midstream EOF).
595        requested |= pad->buffer_is_eof;
596
597        if (requested)
598            read_pad_input(c, pad);
599
600        if (!pad->pending.type || c->draining_recover)
601            continue;
602
603        if (pad->buffer_is_eof) {
604            MP_WARN(c, "eof state changed on %s\n", pad->name);
605            c->draining_recover = true;
606            send_global_eof(c);
607            continue;
608        }
609
610        if (pad->pending.type == MP_FRAME_AUDIO && !c->warned_nospeed) {
611            struct mp_aframe *aframe = pad->pending.data;
612            if (mp_aframe_get_speed(aframe) != 1.0) {
613                MP_ERR(c, "speed changing filters before libavfilter are not "
614                       "supported and can cause desyncs\n");
615                c->warned_nospeed = true;
616            }
617        }
618
619        AVFrame *frame = mp_frame_to_av(pad->pending, &pad->timebase);
620        bool eof = pad->pending.type == MP_FRAME_EOF;
621
622        if (c->emulate_audio_pts && pad->pending.type == MP_FRAME_AUDIO) {
623            struct mp_aframe *aframe = pad->pending.data;
624            c->in_pts = mp_aframe_end_pts(aframe);
625            frame->pts = c->in_samples; // timebase is 1/sample_rate
626            c->in_samples += frame->nb_samples;
627        }
628
629        mp_frame_unref(&pad->pending);
630
631        if (!frame && !eof) {
632            MP_FATAL(c, "out of memory or unsupported format\n");
633            continue;
634        }
635
636        pad->buffer_is_eof = !frame;
637
638        if (av_buffersrc_add_frame(pad->buffer, frame) < 0)
639            MP_FATAL(c, "could not pass frame to filter\n");
640        av_frame_free(&frame);
641
642        for (int i = 0; i < c->num_out_pads; i++)
643            c->out_pads[i]->got_eagain = false;
644
645        progress = true;
646    }
647
648    if (!was_draining && c->draining_recover)
649        progress = true;
650
651    return progress;
652}
653
654// Some filters get stuck and return EAGAIN forever if they did not get any
655// input (i.e. we send only EOF as input). "dynaudnorm" is known to be affected.
656static bool check_stuck_eagain_on_eof_bug(struct lavfi *c)
657{
658    for (int n = 0; n < c->num_in_pads; n++) {
659        if (!c->in_pads[n]->buffer_is_eof)
660            return false;
661    }
662
663    for (int n = 0; n < c->num_out_pads; n++) {
664        struct lavfi_pad *pad = c->out_pads[n];
665
666        if (!pad->buffer_is_eof && !pad->got_eagain)
667            return false;
668    }
669
670    MP_WARN(c, "Filter is stuck. This is a FFmpeg bug. Treating as EOF.\n");
671    return true;
672}
673
674static bool read_output_pads(struct lavfi *c)
675{
676    bool progress = false;
677
678    assert(c->initialized);
679
680    for (int n = 0; n < c->num_out_pads; n++) {
681        struct lavfi_pad *pad = c->out_pads[n];
682
683        if (!mp_pin_in_needs_data(pad->pin))
684            continue;
685
686        assert(pad->buffer);
687
688        int r = AVERROR_EOF;
689        if (!pad->buffer_is_eof)
690            r = av_buffersink_get_frame_flags(pad->buffer, c->tmp_frame, 0);
691
692        pad->got_eagain = r == AVERROR(EAGAIN);
693        if (pad->got_eagain) {
694            if (check_stuck_eagain_on_eof_bug(c))
695                r = AVERROR_EOF;
696        } else {
697            for (int i = 0; i < c->num_out_pads; i++)
698                c->out_pads[i]->got_eagain = false;
699        }
700
701        if (r >= 0) {
702            mp_tags_copy_from_av_dictionary(pad->metadata, c->tmp_frame->metadata);
703            struct mp_frame frame =
704                mp_frame_from_av(pad->type, c->tmp_frame, &pad->timebase);
705            if (c->emulate_audio_pts && frame.type == MP_FRAME_AUDIO) {
706                AVFrame *avframe = c->tmp_frame;
707                struct mp_aframe *aframe = frame.data;
708                double in_time = c->in_samples * av_q2d(c->in_pads[0]->timebase);
709                double out_time = avframe->pts * av_q2d(pad->timebase);
710                mp_aframe_set_pts(aframe, c->in_pts +
711                    (c->in_pts != MP_NOPTS_VALUE ? (out_time - in_time) : 0));
712            }
713            if (frame.type == MP_FRAME_VIDEO) {
714                struct mp_image *vframe = frame.data;
715                vframe->nominal_fps =
716                    av_q2d(av_buffersink_get_frame_rate(pad->buffer));
717            }
718            av_frame_unref(c->tmp_frame);
719            if (frame.type) {
720                mp_pin_in_write(pad->pin, frame);
721            } else {
722                MP_ERR(c, "could not use filter output\n");
723                mp_frame_unref(&frame);
724            }
725            progress = true;
726        } else if (r == AVERROR(EAGAIN)) {
727            // We expect that libavfilter will request input on one of the
728            // input pads (via av_buffersrc_get_nb_failed_requests()).
729        } else if (r == AVERROR_EOF) {
730            if (!c->draining_recover && !pad->buffer_is_eof)
731                mp_pin_in_write(pad->pin, MP_EOF_FRAME);
732            if (!pad->buffer_is_eof)
733                progress = true;
734            pad->buffer_is_eof = true;
735        } else {
736            // Real error - ignore it.
737            MP_ERR(c, "error on filtering (%d)\n", r);
738        }
739    }
740
741    return progress;
742}

744static void lavfi_process(struct mp_filter *f)
745{
746    struct lavfi *c = f->priv;
747
748    if (!c->initialized)
749        init_graph(c);
750
751    while (c->initialized) {
752        bool a = read_output_pads(c);
753        bool b = feed_input_pads(c);
754        if (!a && !b)
755            break;
756    }
757
758    // Start over on format changes or EOF draining.
759    if (c->draining_recover) {
760        // Wait until all outputs got EOF.
761        bool all_eof = true;
762        for (int n = 0; n < c->num_out_pads; n++)
763            all_eof &= c->out_pads[n]->buffer_is_eof;
764
765        if (all_eof) {
766            MP_VERBOSE(c, "recovering all eof\n");
767            free_graph(c);
768            mp_filter_internal_mark_progress(c->f);
769        }
770    }
771
772    if (c->failed)
773        mp_filter_internal_mark_failed(c->f);
774}

794static bool lavfi_command(struct mp_filter *f, struct mp_filter_command *cmd)
795{
796    struct lavfi *c = f->priv;
797
798    if (!c->initialized)
799        return false;
800
801    switch (cmd->type) {
802    case MP_FILTER_COMMAND_TEXT: {
803        return avfilter_graph_send_command(c->graph, "all", cmd->cmd, cmd->arg,
804                                           &(char){0}, 0, 0) >= 0;
805    }
806    case MP_FILTER_COMMAND_GET_META: {
807        // We can worry later about what it should do to multi output filters.
808        if (c->num_out_pads < 1)
809            return false;
810        struct mp_tags **ptags = cmd->res;
811        *ptags = mp_tags_dup(NULL, c->out_pads[0]->metadata);
812        return true;
813    }
814    default:
815        return false;
816    }
817}

846static struct mp_lavfi *do_init(struct lavfi *c)
847{
848    precreate_graph(c, true);
849
850    if (c->failed)
851        goto error;
852
853    for (int n = 0; n < c->num_in_pads + c->num_out_pads; n++) {
854        // First add input pins to satisfy order for bidir graph types.
855        struct lavfi_pad *pad =
856            n < c->num_in_pads ? c->in_pads[n] : c->out_pads[n - c->num_in_pads];
857
858        pad->pin_index = c->f->num_pins;
859        pad->pin = mp_filter_add_pin(c->f, pad->dir, pad->name);
860
861        if (c->force_type && c->force_type != pad->type) {
862            MP_FATAL(c, "mismatching media type\n");
863            goto error;
864        }
865    }
866
867    if (c->force_bidir) {
868        if (c->f->num_pins != 2) {
869            MP_FATAL(c, "exactly 2 pads required\n");
870            goto error;
871        }
872        if (mp_pin_get_dir(c->f->ppins[0]) != MP_PIN_OUT ||
873            mp_pin_get_dir(c->f->ppins[1]) != MP_PIN_IN)
874        {
875            MP_FATAL(c, "1 input and 1 output pad required\n");
876            goto error;
877        }
878    }
879
880    return &c->public;
881
882error:
883    talloc_free(c->f);
884    return NULL;
885}

954// Does it have exactly one video input and one video output?
955static bool is_usable(const AVFilter *filter, int media_type)
956{
957#if LIBAVFILTER_VERSION_INT >= AV_VERSION_INT(8, 3, 0)
958    int nb_inputs  = avfilter_filter_pad_count(filter, 0),
959        nb_outputs = avfilter_filter_pad_count(filter, 1);
960#else
961    int nb_inputs  = avfilter_pad_count(filter->inputs),
962        nb_outputs = avfilter_pad_count(filter->outputs);
963#endif
964    return nb_inputs == 1 && nb_outputs == 1 &&
965           avfilter_pad_get_type(filter->inputs, 0) == media_type &&
966           avfilter_pad_get_type(filter->outputs, 0) == media_type;
967}

970{

976{
977    mp_info(log, "Available libavfilter filters:\n");
978    void *iter = NULL;

1015#define NSTR(s) ((s) ? (s) : "")
1016
1017void print_lavfi_help(struct mp_log *log, const char *name, int media_type)
1018{
1019    const AVFilter *f = avfilter_get_by_name(name);
1020    if (!f) {
1021        mp_err(log, "Filter '%s' not found.\n", name);
1022        return;
1023    }
1024    if (!is_usable(f, media_type)) {
1025        mp_err(log, "Filter '%s' is not usable in this context (wrong media \n"
1026               "types or wrong number of inputs/outputs).\n", name);
1027    }
1028    mp_info(log, "Options:\n\n");
1029    const AVClass *class = f->priv_class;
1030    // av_opt_next() requires this for some retarded incomprehensible reason.
1031    const AVClass **c = &class;
1032    int offset= -1;
1033    int count = 0;
1034    for (const AVOption *o = av_opt_next(c, 0); o; o = av_opt_next(c, o)) {
1035        // This is how libavfilter (at the time) decided to assign positional
1036        // options (called "shorthand" in the libavfilter code). So we
1037        // duplicate it exactly.
1038        if (o->type == AV_OPT_TYPE_CONST || o->offset == offset)
1039            continue;
1040        offset = o->offset;
1041
1042        const char *t = get_avopt_type_name(o->type);
1043        char *tstr = t ? mp_tprintf(30, "<%s>", t) : "?";
1044        mp_info(log, " %-10s %-12s %s\n", o->name, tstr, NSTR(o->help));
1045
1046        const AVOption *sub = o;
1047        while (1) {
1048            sub = av_opt_next(c, sub);
1049            if (!sub || sub->type != AV_OPT_TYPE_CONST)
1050                break;
1051            mp_info(log, " %3s%-23s %s\n", "", sub->name, NSTR(sub->help));
1052        }
1053
1054        count++;
1055    }
1056    mp_info(log, "\nTotal: %d options\n", count);
1057}
