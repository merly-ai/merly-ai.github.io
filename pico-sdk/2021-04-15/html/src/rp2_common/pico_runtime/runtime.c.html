37// this is called for each thread since they have their own MPU
38void runtime_install_stack_guard(void *stack_bottom) {
39    // this is called b4 runtime_init is complete, so beware printf or assert
40
41    // make sure no one is using the MPU yet
42    if (mpu_hw->ctrl) {
43        // Note that it would be tempting to change this to a panic, but it happens so early, printing is not a good idea
44        __breakpoint();
45    }
46
47    uintptr_t addr = (uintptr_t) stack_bottom;
48    // the minimum we can protect is 32 bytes on a 32 byte boundary, so round up which will
49    // just shorten the valid stack range a tad
50    addr = (addr + 31u) & ~31u;
51
52    // mask is 1 bit per 32 bytes of the 256 byte range... clear the bit for the segment we want
53    uint32_t subregion_select = 0xffu ^ (1u << ((addr >> 5u) & 7u));
54    mpu_hw->ctrl = 5; // enable mpu with background default map
55    mpu_hw->rbar = (addr & (uint)~0xff) | 0x8 | 0;
56    mpu_hw->rasr = 1 // enable region
57                   | (0x7 << 1) // size 2^(7 + 1) = 256
58                   | (subregion_select << 8)
59                   | 0x10000000; // XN = disable instruction fetch; no other bits means no permissions
60}

62void runtime_init(void) {
63    // Reset all peripherals to put system into a known state,
64    // - except for QSPI pads and the XIP IO bank, as this is fatal if running from flash
65    // - and the PLLs, as this is fatal if clock muxing has not been reset on this boot
66    // - and USB, syscfg, as this disturbs USB-to-SWD on core 1
67    reset_block(~(
68            RESETS_RESET_IO_QSPI_BITS |
69            RESETS_RESET_PADS_QSPI_BITS |
70            RESETS_RESET_PLL_USB_BITS |
71            RESETS_RESET_USBCTRL_BITS |
72            RESETS_RESET_SYSCFG_BITS |
73            RESETS_RESET_PLL_SYS_BITS
74    ));
75
76    // Remove reset from peripherals which are clocked only by clk_sys and
77    // clk_ref. Other peripherals stay in reset until we've configured clocks.
78    unreset_block_wait(RESETS_RESET_BITS & ~(
79            RESETS_RESET_ADC_BITS |
80            RESETS_RESET_RTC_BITS |
81            RESETS_RESET_SPI0_BITS |
82            RESETS_RESET_SPI1_BITS |
83            RESETS_RESET_UART0_BITS |
84            RESETS_RESET_UART1_BITS |
85            RESETS_RESET_USBCTRL_BITS
86    ));
87
88    // pre-init runs really early since we need it even for memcpy and divide!
89    // (basically anything in aeabi that uses bootrom)
90
91    // Start and end points of the constructor list,
92    // defined by the linker script.
93    extern void (*__preinit_array_start)(void);
94    extern void (*__preinit_array_end)(void);
95
96    // Call each function in the list.
97    // We have to take the address of the symbols, as __preinit_array_start *is*
98    // the first function pointer, not the address of it.
99    for (void (**p)(void) = &__preinit_array_start; p < &__preinit_array_end; ++p) {
100        (*p)();
101    }
102
103    // After calling preinit we have enough runtime to do the exciting maths
104    // in clocks_init
105    clocks_init();
106
107    // Peripheral clocks should now all be running
108    unreset_block_wait(RESETS_RESET_BITS);
109
110#if !PICO_IE_26_29_UNCHANGED_ON_RESET
111    // after resetting BANK0 we should disable IE on 26-29
112    hw_clear_alias(padsbank0_hw)->io[26] = hw_clear_alias(padsbank0_hw)->io[27] =
113            hw_clear_alias(padsbank0_hw)->io[28] = hw_clear_alias(padsbank0_hw)->io[29] = PADS_BANK0_GPIO0_IE_BITS;
114#endif
115
116    extern mutex_t __mutex_array_start;
117    extern mutex_t __mutex_array_end;
118
119    // the first function pointer, not the address of it.
120    for (mutex_t *m = &__mutex_array_start; m < &__mutex_array_end; m++) {
121        mutex_init(m);
122    }
123
124#if !(PICO_NO_RAM_VECTOR_TABLE || PICO_NO_FLASH)
125    __builtin_memcpy(ram_vector_table, (uint32_t *) scb_hw->vtor, sizeof(ram_vector_table));
126    scb_hw->vtor = (uintptr_t) ram_vector_table;
127#endif
128
129#ifndef NDEBUG
130    if (__get_current_exception()) {
131        // crap; started in exception handler
132        __asm ("bkpt #0");
133    }
134#endif
135
136#if PICO_USE_STACK_GUARDS
137    // install core0 stack guard
138    extern char __StackBottom;
139    runtime_install_stack_guard(&__StackBottom);
140#endif
141
142    spin_locks_reset();
143    irq_init_priorities();
144    alarm_pool_init_default();
145
146    // Start and end points of the constructor list,
147    // defined by the linker script.
148    extern void (*__init_array_start)(void);
149    extern void (*__init_array_end)(void);
150
151    // Call each function in the list.
152    // We have to take the address of the symbols, as __init_array_start *is*
153    // the first function pointer, not the address of it.
154    for (void (**p)(void) = &__init_array_start; p < &__init_array_end; ++p) {
155        (*p)();
156    }
157
158}
