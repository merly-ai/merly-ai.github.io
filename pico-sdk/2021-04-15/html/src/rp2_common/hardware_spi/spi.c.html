11static inline void spi_reset(spi_inst_t *spi) {
12    invalid_params_if(SPI, spi != spi0 && spi != spi1);
13    reset_block(spi == spi0 ? RESETS_RESET_SPI0_BITS : RESETS_RESET_SPI1_BITS);
14}
15
16static inline void spi_unreset(spi_inst_t *spi) {
17    invalid_params_if(SPI, spi != spi0 && spi != spi1);
18    unreset_block_wait(spi == spi0 ? RESETS_RESET_SPI0_BITS : RESETS_RESET_SPI1_BITS);
19}

21void spi_init(spi_inst_t *spi, uint baudrate) {
22    spi_reset(spi);
23    spi_unreset(spi);
24
25    (void) spi_set_baudrate(spi, baudrate);
26    spi_set_format(spi, 8, SPI_CPOL_0, SPI_CPHA_0, SPI_MSB_FIRST);
27    // Always enable DREQ signals -- harmless if DMA is not listening
28    hw_set_bits(&spi_get_hw(spi)->dmacr, SPI_SSPDMACR_TXDMAE_BITS | SPI_SSPDMACR_RXDMAE_BITS);
29    spi_set_format(spi, 8, SPI_CPOL_0, SPI_CPHA_0, SPI_MSB_FIRST);
30
31    // Finally enable the SPI
32    hw_set_bits(&spi_get_hw(spi)->cr1, SPI_SSPCR1_SSE_BITS);
33}

35void spi_deinit(spi_inst_t *spi) {
36    hw_clear_bits(&spi_get_hw(spi)->cr1, SPI_SSPCR1_SSE_BITS);
37    hw_clear_bits(&spi_get_hw(spi)->dmacr, SPI_SSPDMACR_TXDMAE_BITS | SPI_SSPDMACR_RXDMAE_BITS);
38    spi_reset(spi);

41uint spi_set_baudrate(spi_inst_t *spi, uint baudrate) {
42    uint freq_in = clock_get_hz(clk_peri);
43    uint prescale, postdiv;
44    invalid_params_if(SPI, baudrate > freq_in);
45
46    // Find smallest prescale value which puts output frequency in range of
47    // post-divide. Prescale is an even number from 2 to 254 inclusive.
48    for (prescale = 2; prescale <= 254; prescale += 2) {
49        if (freq_in < (prescale + 2) * 256 * (uint64_t) baudrate)
50            break;
51    }
52    invalid_params_if(SPI, prescale > 254); // Frequency too low
53
54    // Find largest post-divide which makes output <= baudrate. Post-divide is
55    // an integer in the range 1 to 256 inclusive.
56    for (postdiv = 256; postdiv > 1; --postdiv) {
57        if (
 Recommendations (Experimental)  R1: (freq_in + (prescale * (postdiv - 1)) > xx_baudrate)
R2: (freq_in < (prescale * (postdiv - 1)) > xx_baudrate)
R3: (freq_in < (prescale * (postdiv - 1)) > xx_baudrate)
R4: (freq_in <= (prescale * (postdiv - 1)) > xx_baudrate)
R5: (freq_in >= (prescale * (postdiv - 1)) > xx_baudrate)
 Score  Issue:  1 unfamiliar pattern(s) detected 
Cost: 25.05, Total Score: 1025.05
Class: complex_max
freq_in / (prescale * (postdiv - 1)) > baudrate)
58            break;
59    }
60
61    spi_get_hw(spi)->cpsr = prescale;
62    hw_write_masked(&spi_get_hw(spi)->cr0, (postdiv - 1) << SPI_SSPCR0_SCR_LSB, SPI_SSPCR0_SCR_BITS);
63
64    // Return the frequency we were able to achieve
65    return freq_in / (prescale * postdiv);
66}

92// Write len bytes directly from src to the SPI, and discard any data received back
93int __not_in_flash_func(spi_write_blocking)(spi_inst_t *spi, const uint8_t *src, size_t len) {
94    invalid_params_if(SPI, 0 > (int)len);
95    // Write to TX FIFO whilst ignoring RX, then clean up afterward. When RX
96    // is full, PL022 inhibits RX pushes, and sets a sticky flag on
97    // push-on-full, but continues shifting. Safe if SSPIMSC_RORIM is not set.
98    for (size_t i = 0; i < len; ++i) {
99        while (!spi_is_writable(spi))
100            tight_loop_contents();
101        spi_get_hw(spi)->dr = (uint32_t)src[i];
102    }
103    // Drain RX FIFO, then wait for shifting to finish (which may be *after*
104    // TX FIFO drains), then drain RX FIFO again
105    while (spi_is_readable(spi))
106        (void)spi_get_hw(spi)->dr;
107    while (spi_get_hw(spi)->sr & SPI_SSPSR_BSY_BITS)
108        tight_loop_contents();
109    while (spi_is_readable(spi))
110        (void)spi_get_hw(spi)->dr;
111
112    // Don't leave overrun flag set
113    spi_get_hw(spi)->icr = SPI_SSPICR_RORIC_BITS;
114
115    return (int)len;
116}
