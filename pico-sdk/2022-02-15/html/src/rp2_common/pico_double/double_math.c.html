85static inline double dldexp(double x,int de) {
86    ui64 ix=double2ui64(x),iy;
87    int e;
88    e=dgetexp(x);
89    if(e==0||e==0x7ff) return x;
90    e+=de;
91    if(e<=0) iy=ix&0x8000000000000000ULL; // signed zero for underflow
92    else if(e>=0x7ff) iy=(ix&0x8000000000000000ULL)|0x7ff0000000000000ULL; // signed infinity on overflow
93    else iy=ix+((ui64)de<<52);
94    return ui642double(iy);
95}

120static inline int disint(double x) {
121    ui64 ix=double2ui64(x),m;
122    int e=dgetexp(x);
123    if(e==0) return 1;       // 0 is an integer
124    e-=0x3ff;                // remove exponent bias
125    if(e<0) return 0;        // |x|<1
126    e=52-e;                  // bit position in mantissa with significance 1
127    if(e<=0) return 1;       // |x| large, so must be an integer
128    m=(1ULL<<e)-1;           // mask for bits of significance <1
129    if(ix&m) return 0;       // not an integer
130    return 1;
131}

133static inline int disoddint(double x) {
134    ui64 ix=double2ui64(x),m;
135    int e=dgetexp(x);
136    e-=0x3ff;                // remove exponent bias
137    if(e<0) return 0;        // |x|<1; 0 is not odd
138    e=52-e;                  // bit position in mantissa with significance 1
139    if(e<0) return 0;        // |x| large, so must be even
140    m=(1ULL<<e)-1;           // mask for bits of significance <1 (if any)
141    if(ix&m) return 0;       // not an integer

195double WRAPPER_FUNC(round)(double x) {
196    check_nan_d1(x);
197    ui64 ix=double2ui64(x),m;
198    int e=dgetexp(x);
199    e-=0x3ff;                // remove exponent bias
200    if(e<-1) {               // |x|<0.5
201        ix&=0x8000000000000000ULL;
202        return ui642double(ix);
203    }
204    if(e==-1) {              // 0.5<=|x|<1
205        ix&=0x8000000000000000ULL;
206        ix|=0x3ff0000000000000ULL;        // 1
207        return ui642double(ix);
208    }

210    if(e<=0) return x;       // |x| large, so must be an integer
211    m=1ULL<<(e-1);           // mask for bit of significance 0.5
212    ix+=m;

214    ix&=~m;

218double WRAPPER_FUNC(floor)(double x) {
219    check_nan_d1(x);
220    ui64 ix=double2ui64(x),m;
221    int e=dgetexp(x);
222    if(e==0) {               // x==0
223        ix&=0x8000000000000000ULL;
224        return ui642double(ix);
225    }
226    e-=0x3ff;                // remove exponent bias
227    if(e<0) {                // |x|<1, not zero
228        if(disneg(x)) return -1;
229        return PZERO;
230    }
231    e=52-e;                  // bit position in mantissa with significance 1
232    if(e<=0) return x;       // |x| large, so must be an integer
233    m=(1ULL<<e)-1;           // mask for bit of significance <1
234    if(disneg(x)) ix+=m;     // add 1- to magnitude if negative
235    ix&=~m;                  // truncate
236    return ui642double(ix);

351// general power, x>0, finite
352static double dpow_1(double x,double y) {
353    int a,b,c;
354    double t,rt,u,v,v0,v1,w,ry;
355    a=dgetexp(x)-0x3ff;
356    u=log2(dldexp(x,-a)); // now log_2 x = a+u
357    if(u>0.5) u-=1,a++;              // |u|<=~0.5
358    if(a==0) return exp2(u*y);
359    // here |log_2 x| >~0.5
360    if(y>= 4096) { // then easily over/underflows
361        if(a<0) return 0;
362        return PINF;
363    }
364    if(y<=-4096) { // then easily over/underflows
365        if(a<0) return PINF;
366        return 0;
367    }
368    ry=round(y);
369    v=y-ry;
370    v0=dldexp(round(ldexp(v,26)),-26);
371    v1=v-v0;
372    b=(int)ry; // guaranteed to fit in an int; y=b+v0+v1
373    // now the result is exp2( (a+u) * (b+v0+v1) )
374    c=a*b;      // integer
375    t=a*v0;
376    rt=round(t);
377    c+=(int)rt;
378    w=t-rt;
379    t=a*v1;
380    w+=t;
381    t=u*b;
382    rt=round(t);
383    c+=(int)rt;
384    w+=t-rt;
385    w+=u*v;
386    return dldexp(exp2(w),c);
387}

472double WRAPPER_FUNC(pow)(double x,double y) {
473    _Pragma("GCC diagnostic push")
474    _Pragma("GCC diagnostic ignored \"-Wfloat-equal\"")
475
476    if(x==1.0||diszero(y)) return 1;
477    check_nan_d2(x, y);
478    if(x==-1.0&&disinf(y)) return 1;
479    _Pragma("GCC diagnostic pop")
480
481    if(diszero(x)) {
482        if(!disneg(y)) {
483            if(disoddint(y)) return x;
484            else             return 0;
485        }
486        if(disoddint(y)) return dcopysign(PINF,x);
487        return PINF;
488    }
489    if(dispinf(x)) {
490        if(disneg(y)) return 0;
491        else          return PINF;
