21//-----------------------------------------------------------------------------
22// Infrastructure for reentering XIP mode after exiting for programming (take
23// a copy of boot2 before XIP exit). Calling boot2 as a function works because
24// it accepts a return vector in LR (and doesn't trash r4-r7). Bootrom passes
25// NULL in LR, instructing boot2 to enter flash vector table's reset handler.
26
27#if !PICO_NO_FLASH
28
29#define BOOT2_SIZE_WORDS 64
30
31static uint32_t boot2_copyout[BOOT2_SIZE_WORDS];
32static bool boot2_copyout_valid = false;
33
34static void __no_inline_not_in_flash_func(flash_init_boot2_copyout)(void) {
35    if (boot2_copyout_valid)
36        return;
37    for (int i = 0; i < BOOT2_SIZE_WORDS; ++i)
38        boot2_copyout[i] = ((uint32_t *)XIP_BASE)[i];
39    __compiler_memory_barrier();
40    boot2_copyout_valid = true;
41}
42
43static void __no_inline_not_in_flash_func(flash_enable_xip_via_boot2)(void) {
44    ((void (*)(void))boot2_copyout+1)();
45}
46
47#else
48
49static void __no_inline_not_in_flash_func(flash_init_boot2_copyout)(void) {}
50
51static void __no_inline_not_in_flash_func(flash_enable_xip_via_boot2)(void) {
52    // Set up XIP for 03h read on bus access (slow but generic)
53    void (*flash_enter_cmd_xip)(void) = (void(*)(void))rom_func_lookup(rom_table_code('C', 'X'));
54    assert(flash_enter_cmd_xip);
55    flash_enter_cmd_xip();
56}
57
58#endif

61// Actual flash programming shims (work whether or not PICO_NO_FLASH==1)
62
63void __no_inline_not_in_flash_func(flash_range_erase)(uint32_t flash_offs, size_t count) {
64#ifdef PICO_FLASH_SIZE_BYTES
65    hard_assert(flash_offs + count <= PICO_FLASH_SIZE_BYTES);
66#endif
67    invalid_params_if(FLASH, flash_offs & (FLASH_SECTOR_SIZE - 1));
68    invalid_params_if(FLASH, count & (FLASH_SECTOR_SIZE - 1));
69    void (*connect_internal_flash)(void) = (void(*)(void))rom_func_lookup(rom_table_code('I', 'F'));
70    void (*flash_exit_xip)(void) = (void(*)(void))rom_func_lookup(rom_table_code('E', 'X'));
71    void (*flash_range_erase)(uint32_t, size_t, uint32_t, uint8_t) =
72        (void(*)(uint32_t, size_t, uint32_t, uint8_t))rom_func_lookup(rom_table_code('R', 'E'));
73    void (*flash_flush_cache)(void) = (void(*)(void))rom_func_lookup(rom_table_code('F', 'C'));
74    assert(connect_internal_flash && flash_exit_xip && flash_range_erase && flash_flush_cache);
75    flash_init_boot2_copyout();
76
77    // No flash accesses after this point
78    __compiler_memory_barrier();
79
80    connect_internal_flash();
81    flash_exit_xip();
82    flash_range_erase(flash_offs, count, FLASH_BLOCK_SIZE, FLASH_BLOCK_ERASE_CMD);
83    flash_flush_cache(); // Note this is needed to remove CSn IO force as well as cache flushing
84    flash_enable_xip_via_boot2();
85}

87void __no_inline_not_in_flash_func(flash_range_program)(uint32_t flash_offs, const uint8_t *data, size_t count) {
88#ifdef PICO_FLASH_SIZE_BYTES
89    hard_assert(flash_offs + count <= PICO_FLASH_SIZE_BYTES);
90#endif
91    invalid_params_if(FLASH, flash_offs & (FLASH_PAGE_SIZE - 1));
92    invalid_params_if(FLASH, count & (FLASH_PAGE_SIZE - 1));
93    void (*connect_internal_flash)(void) = (void(*)(void))rom_func_lookup(rom_table_code('I', 'F'));
94    void (*flash_exit_xip)(void) = (void(*)(void))rom_func_lookup(rom_table_code('E', 'X'));
95    void (*flash_range_program)(uint32_t, const uint8_t*, size_t) =
96        (void(*)(uint32_t, const uint8_t*, size_t))rom_func_lookup(rom_table_code('R', 'P'));
97    void (*flash_flush_cache)(void) = (void(*)(void))rom_func_lookup(rom_table_code('F', 'C'));
98    assert(connect_internal_flash && flash_exit_xip && flash_range_program && flash_flush_cache);
99    flash_init_boot2_copyout();
100
101    __compiler_memory_barrier();
102
103    connect_internal_flash();
104    flash_exit_xip();
105    flash_range_program(flash_offs, data, count);
106    flash_flush_cache(); // Note this is needed to remove CSn IO force as well as cache flushing
107    flash_enable_xip_via_boot2();
108}

114// Bitbanging the chip select using IO overrides, in case RAM-resident IRQs
115// are still running, and the FIFO bottoms out. (the bootrom does the same)
116static void __no_inline_not_in_flash_func(flash_cs_force)(bool high) {
117    uint32_t field_val = high ?
118        IO_QSPI_GPIO_QSPI_SS_CTRL_OUTOVER_VALUE_HIGH :
119        IO_QSPI_GPIO_QSPI_SS_CTRL_OUTOVER_VALUE_LOW;
120    hw_write_masked(&ioqspi_hw->io[1].ctrl,
121        field_val << IO_QSPI_GPIO_QSPI_SS_CTRL_OUTOVER_LSB,
122        IO_QSPI_GPIO_QSPI_SS_CTRL_OUTOVER_BITS
123    );
124}

126void __no_inline_not_in_flash_func(flash_do_cmd)(const uint8_t *txbuf, uint8_t *rxbuf, size_t count) {
127    void (*connect_internal_flash)(void) = (void(*)(void))rom_func_lookup(rom_table_code('I', 'F'));
128    void (*flash_exit_xip)(void) = (void(*)(void))rom_func_lookup(rom_table_code('E', 'X'));
129    void (*flash_flush_cache)(void) = (void(*)(void))rom_func_lookup(rom_table_code('F', 'C'));
130    assert(connect_internal_flash && flash_exit_xip && flash_flush_cache);
131    flash_init_boot2_copyout();
132    __compiler_memory_barrier();
133    connect_internal_flash();
134    flash_exit_xip();
135
136    flash_cs_force(0);
137    size_t tx_remaining = count;
138    size_t rx_remaining = count;
139    // We may be interrupted -- don't want FIFO to overflow if we're distracted.
140    const size_t max_in_flight = 16 - 2;
141    while (tx_remaining || rx_remaining) {
142        uint32_t flags = ssi_hw->sr;
143        bool can_put = !!(flags & SSI_SR_TFNF_BITS);
144        bool can_get = !!(flags & SSI_SR_RFNE_BITS);

161// Use standard RUID command to get a unique identifier for the flash (and
162// hence the board)

164static_assert(FLASH_UNIQUE_ID_SIZE_BYTES == FLASH_RUID_DATA_BYTES, "");
165
166void flash_get_unique_id(uint8_t *id_out) {
167#if PICO_NO_FLASH
168    __unused uint8_t *ignore = id_out;
169    panic_unsupported();
170#else
171    uint8_t txbuf[FLASH_RUID_TOTAL_BYTES] = {0};
172    uint8_t rxbuf[FLASH_RUID_TOTAL_BYTES] = {0};
173    txbuf[0] = FLASH_RUID_CMD;
174    flash_do_cmd(txbuf, rxbuf, FLASH_RUID_TOTAL_BYTES);
175    for (int i = 0; i < FLASH_RUID_DATA_BYTES; i++)
176        id_out[i] = rxbuf[i + 1 + FLASH_RUID_DUMMY_BYTES];
177#endif
