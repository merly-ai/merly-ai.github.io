16static void optbug(const struct option *opt, const char *reason)
17{
18  if (opt->long_name && opt->short_name)
19    bug("switch '%c' (--%s) %s", opt->short_name,
20        opt->long_name, reason);
21  else if (opt->long_name)
22    bug("option '%s' %s", opt->long_name, reason);
23  else
24    bug("switch '%c' %s", opt->short_name, reason);
25}

27static const char *optname(const struct option *opt, enum opt_parsed flags)
28{
29  static struct strbuf sb = STRBUF_INIT;
30
31  strbuf_reset(&sb);
32  if (flags & OPT_SHORT)
33    strbuf_addf(&sb, "switch `%c'", opt->short_name);
34  else if (flags & OPT_UNSET)
35    strbuf_addf(&sb, "option `no-%s'", opt->long_name);
36  else if (flags == OPT_LONG)
37    strbuf_addf(&sb, "option `%s'", opt->long_name);
38  else
39    BUG("optname() got unknown flags %d", flags);
40
41  return sb.buf;
42}

44static enum parse_opt_result get_arg(struct parse_opt_ctx_t *p,
45             const struct option *opt,
46             enum opt_parsed flags, const char **arg)
47{
48  if (p->opt) {
49    *arg = p->opt;
50    p->opt = NULL;
51  } else if (p->argc == 1 && (opt->flags & PARSE_OPT_LASTARG_DEFAULT)) {
52    *arg = (const char *)opt->defval;
53  } else if (p->argc > 1) {
54    p->argc--;
55    *arg = *++p->argv;
56  } else
57    return error(_("%s requires a value"), optname(opt, flags));
58  return 0;
59}

69static enum parse_opt_result opt_command_mode_error(
70  const struct option *opt,
71  const struct option *all_opts,
72  enum opt_parsed flags)
73{
74  const struct option *that;
75  struct strbuf that_name = STRBUF_INIT;
76
77  /*
78   * Find the other option that was used to set the variable
79   * already, and report that this is not compatible with it.
80   */
81  for (that = all_opts; that->type != OPTION_END; that++) {
82    if (that == opt ||
83        !(that->flags & PARSE_OPT_CMDMODE) ||
84        that->value != opt->value ||
85        that->defval != *(int *)opt->value)
86      continue;
87
88    if (that->long_name)
89      strbuf_addf(&that_name, "--%s", that->long_name);
90    else
91      strbuf_addf(&that_name, "-%c", that->short_name);
92    error(_("%s is incompatible with %s"),
93          optname(opt, flags), that_name.buf);
94    strbuf_release(&that_name);
95    return PARSE_OPT_ERROR;
96  }
97  return error(_("%s : incompatible with something else"),
98         optname(opt, flags));
99}

101static enum parse_opt_result get_value(struct parse_opt_ctx_t *p,
102               const struct option *opt,
103               const struct option *all_opts,
104               enum opt_parsed flags)
105{
106  const char *s, *arg;
107  const int unset = flags & OPT_UNSET;
108  int err;
109
110  if (unset && p->opt)
111    return error(_("%s takes no value"), optname(opt, flags));
112  if (unset && (opt->flags & PARSE_OPT_NONEG))
113    return error(_("%s isn't available"), optname(opt, flags));
114  if (!(flags & OPT_SHORT) && p->opt && (opt->flags & PARSE_OPT_NOARG))
115    return error(_("%s takes no value"), optname(opt, flags));
116
117  /*
118   * Giving the same mode option twice, although unnecessary,
119   * is not a grave error, so let it pass.
120   */
121  if ((opt->flags & PARSE_OPT_CMDMODE) &&
122      *(int *)opt->value && *(int *)opt->value != opt->defval)
123    return opt_command_mode_error(opt, all_opts, flags);
124
125  switch (opt->type) {
126  case OPTION_LOWLEVEL_CALLBACK:
127    return opt->ll_callback(p, opt, NULL, unset);
128
129  case OPTION_BIT:
130    if (unset)
131      *(int *)opt->value &= ~opt->defval;
132    else
133      *(int *)opt->value |= opt->defval;
134    return 0;
135
136  case OPTION_NEGBIT:
137    if (unset)
138      *(int *)opt->value |= opt->defval;
139    else
140      *(int *)opt->value &= ~opt->defval;
141    return 0;
142
143  case OPTION_BITOP:
144    if (unset)
145      BUG("BITOP can't have unset form");
146    *(int *)opt->value &= ~opt->extra;
147    *(int *)opt->value |= opt->defval;
148    return 0;
149
150  case OPTION_COUNTUP:
151    if (*(int *)opt->value < 0)
152      *(int *)opt->value = 0;
153    *(int *)opt->value = unset ? 0 : *(int *)opt->value + 1;
154    return 0;
155
156  case OPTION_SET_INT:
157    *(int *)opt->value = unset ? 0 : opt->defval;
158    return 0;
159
160  case OPTION_STRING:
161    if (unset)
162      *(const char **)opt->value = NULL;
163    else if (opt->flags & PARSE_OPT_OPTARG && !p->opt)
164      *(const char **)opt->value = (const char *)opt->defval;
165    else
166      return get_arg(p, opt, flags, (const char **)opt->value);
167    return 0;
168
169  case OPTION_FILENAME:
170    err = 0;
171    if (unset)
172      *(const char **)opt->value = NULL;
173    else if (opt->flags & PARSE_OPT_OPTARG && !p->opt)
174      *(const char **)opt->value = (const char *)opt->defval;
175    else
176      err = get_arg(p, opt, flags, (const char **)opt->value);
177
178    if (!err)
179      fix_filename(p->prefix, (const char **)opt->value);
180    return err;
181
182  case OPTION_CALLBACK:
183  {
184    const char *p_arg = NULL;
185    int p_unset;
186
187    if (unset)
188      p_unset = 1;
189    else if (opt->flags & PARSE_OPT_NOARG)
190      p_unset = 0;
191    else if (opt->flags & PARSE_OPT_OPTARG && !p->opt)
192      p_unset = 0;
193    else if (get_arg(p, opt, flags, &arg))
194      return -1;
195    else {
196      p_unset = 0;
197      p_arg = arg;
198    }
199    if (opt->callback)
200      return (*opt->callback)(opt, p_arg, p_unset) ? (-1) : 0;
201    else
202      return (*opt->ll_callback)(p, opt, p_arg, p_unset);
203  }
204  case OPTION_INTEGER:
205    if (unset) {
206      *(int *)opt->value = 0;
207      return 0;
208    }
209    if (opt->flags & PARSE_OPT_OPTARG && !p->opt) {
210      *(int *)opt->value = opt->defval;
211      return 0;
212    }
213    if (get_arg(p, opt, flags, &arg))
214      return -1;
215    if (!*arg)
216      return error(_("%s expects a numerical value"),
217             optname(opt, flags));
218    *(int *)opt->value = strtol(arg, (char **)&s, 10);
219    if (*s)
220      return error(_("%s expects a numerical value"),
221             optname(opt, flags));
222    return 0;
223
224  case OPTION_MAGNITUDE:
225    if (unset) {
226      *(unsigned long *)opt->value = 0;
227      return 0;
228    }
229    if (opt->flags & PARSE_OPT_OPTARG && !p->opt) {
230      *(unsigned long *)opt->value = opt->defval;
231      return 0;
232    }
233    if (get_arg(p, opt, flags, &arg))
234      return -1;
235    if (!git_parse_ulong(arg, opt->value))
236      return error(_("%s expects a non-negative integer value"
237               " with an optional k/m/g suffix"),
238             optname(opt, flags));
239    return 0;
240
241  default:
242    BUG("opt->type %d should not happen", opt->type);
243  }
244}

246static enum parse_opt_result parse_short_opt(struct parse_opt_ctx_t *p,
247               const struct option *options)
248{
249  const struct option *all_opts = options;
250  const struct option *numopt = NULL;
251
252  for (; options->type != OPTION_END; options++) {
253    if (options->short_name == *p->opt) {
254      p->opt = p->opt[1] ? p->opt + 1 : NULL;
255      return get_value(p, options, all_opts, OPT_SHORT);
256    }
257
258    /*
259     * Handle the numerical option later, explicit one-digit
260     * options take precedence over it.
261     */
262    if (options->type == OPTION_NUMBER)
263      numopt = options;
264  }
265  if (numopt && isdigit(*p->opt)) {
266    size_t len = 1;
267    char *arg;
268    int rc;
269
270    while (isdigit(p->opt[len]))
271      len++;
272    arg = xmemdupz(p->opt, len);
273    p->opt = p->opt[len] ? p->opt + len : NULL;
274    if (numopt->callback)
275      rc = (*numopt->callback)(numopt, arg, 0) ? (-1) : 0;
276    else
277      rc = (*numopt->ll_callback)(p, numopt, arg, 0);
278    free(arg);
279    return rc;
280  }
281  return PARSE_OPT_UNKNOWN;
282}

292static int is_alias(struct parse_opt_ctx_t *ctx,
293        const struct option *one_opt,
294        const struct option *another_opt)
295{
296  const char **group;
297
298  if (!ctx->alias_groups)
299    return 0;
300
301  if (!one_opt->long_name || !another_opt->long_name)
302    return 0;
303
304  for (group = ctx->alias_groups; *group; group += 3) {
305    /* it and other are from the same family? */
306    if (has_string(one_opt->long_name, group) &&
307        has_string(another_opt->long_name, group))
308      return 1;
309  }
310  return 0;
311}

313static enum parse_opt_result parse_long_opt(
314  struct parse_opt_ctx_t *p, const char *arg,
315  const struct option *options)
316{
317  const struct option *all_opts = options;
318  const char *arg_end = strchrnul(arg, '=');
319  const struct option *abbrev_option = NULL, *ambiguous_option = NULL;
320  enum opt_parsed abbrev_flags = OPT_LONG, ambiguous_flags = OPT_LONG;
321
322  for (; options->type != OPTION_END; options++) {
323    const char *rest, *long_name = options->long_name;
324    enum opt_parsed flags = OPT_LONG, opt_flags = OPT_LONG;
325
326    if (!long_name)
327      continue;
328
329again:
330    if (!skip_prefix(arg, long_name, &rest))
331      rest = NULL;
332    if (!rest) {
333      /* abbreviated? */
334      if (!(p->flags & PARSE_OPT_KEEP_UNKNOWN) &&
335          !strncmp(long_name, arg, arg_end - arg)) {
336is_abbreviated:
337        if (abbrev_option &&
338            !is_alias(p, abbrev_option, options)) {
339          /*
340           * If this is abbreviated, it is
341           * ambiguous. So when there is no
342           * exact match later, we need to
343           * error out.
344           */
345          ambiguous_option = abbrev_option;
346          ambiguous_flags = abbrev_flags;
347        }
348        if (!(flags & OPT_UNSET) && *arg_end)
349          p->opt = arg_end + 1;
350        abbrev_option = options;
351        abbrev_flags = flags ^ opt_flags;
352        continue;
353      }
354      /* negation allowed? */
355      if (options->flags & PARSE_OPT_NONEG)
356        continue;
357      /* negated and abbreviated very much? */
358      if (starts_with("no-", arg)) {
359        flags |= OPT_UNSET;
360        goto is_abbreviated;
361      }
362      /* negated? */
363      if (!starts_with(arg, "no-")) {
364        if (skip_prefix(long_name, "no-", &long_name)) {
365          opt_flags |= OPT_UNSET;
366          goto again;
367        }
368        continue;
369      }
370      flags |= OPT_UNSET;
371      if (!skip_prefix(arg + 3, long_name, &rest)) {
372        /* abbreviated and negated? */
373        if (starts_with(long_name, arg + 3))
374          goto is_abbreviated;
375        else
376          continue;
377      }
378    }
379    if (*rest) {
380      if (*rest != '=')
381        continue;
382      p->opt = rest + 1;
383    }
384    return get_value(p, options, all_opts, flags ^ opt_flags);
385  }
386
387  if (disallow_abbreviated_options && (ambiguous_option || abbrev_option))
388    die("disallowed abbreviated or ambiguous option '%.*s'",
389        (int)(arg_end - arg), arg);
390
391  if (ambiguous_option) {
392    error(_("ambiguous option: %s "
393      "(could be --%s%s or --%s%s)"),
394      arg,
395      (ambiguous_flags & OPT_UNSET) ?  "no-" : "",
396      ambiguous_option->long_name,
397      (abbrev_flags & OPT_UNSET) ?  "no-" : "",
398      abbrev_option->long_name);
399    return PARSE_OPT_HELP;
400  }
401  if (abbrev_option)
402    return get_value(p, abbrev_option, all_opts, abbrev_flags);
403  return PARSE_OPT_UNKNOWN;
404}

406static enum parse_opt_result parse_nodash_opt(struct parse_opt_ctx_t *p,
407                const char *arg,
408                const struct option *options)
409{
410  const struct option *all_opts = options;
411
412  for (; options->type != OPTION_END; options++) {
413    if (!(options->flags & PARSE_OPT_NODASH))
414      continue;
415    if (options->short_name == arg[0] && arg[1] == '\0')
416      return get_value(p, options, all_opts, OPT_SHORT);
417  }
418  return PARSE_OPT_ERROR;

421static void check_typos(const char *arg, const struct option *options)
422{
423  if (strlen(arg) < 3)
424    return;
425
426  if (starts_with(arg, "no-")) {
427    error(_("did you mean `--%s` (with two dashes)?"), arg);
428    exit(129);

430
431  for (; options->type != OPTION_END; options++) {
432    if (!options->long_name)
433      continue;
434    if (starts_with(options->long_name, arg)) {
435      error(_("did you mean `--%s` (with two dashes)?"), arg);
436      exit(129);
437    }
438  }

441static void parse_options_check(const struct option *opts)
442{
443  char short_opts[128];
444
445  memset(short_opts, '\0', sizeof(short_opts));
446  for (; opts->type != OPTION_END; opts++) {
447    if ((opts->flags & PARSE_OPT_LASTARG_DEFAULT) &&
448        (opts->flags & PARSE_OPT_OPTARG))
449      optbug(opts, "uses incompatible flags "
450             "LASTARG_DEFAULT and OPTARG");
451    if (opts->short_name) {
452      if (0x7F <= opts->short_name)
453        optbug(opts, "invalid short name");
454      else if (short_opts[opts->short_name]++)
455        optbug(opts, "short name already used");
456    }
457    if (opts->flags & PARSE_OPT_NODASH &&
458        ((opts->flags & PARSE_OPT_OPTARG) ||
459         !(opts->flags & PARSE_OPT_NOARG) ||
460         !(opts->flags & PARSE_OPT_NONEG) ||
461         opts->long_name))
462      optbug(opts, "uses feature "
463             "not supported for dashless options");
464    switch (opts->type) {
465    case OPTION_COUNTUP:
466    case OPTION_BIT:
467    case OPTION_NEGBIT:
468    case OPTION_SET_INT:
469    case OPTION_NUMBER:
470      if ((opts->flags & PARSE_OPT_OPTARG) ||
471          !(opts->flags & PARSE_OPT_NOARG))
472        optbug(opts, "should not accept an argument");
473      break;
474    case OPTION_CALLBACK:
475      if (!opts->callback && !opts->ll_callback)
476        optbug(opts, "OPTION_CALLBACK needs one callback");
477      else if (opts->callback && opts->ll_callback)
478        optbug(opts, "OPTION_CALLBACK can't have two callbacks");
479      break;
480    case OPTION_LOWLEVEL_CALLBACK:
481      if (!opts->ll_callback)
482        optbug(opts, "OPTION_LOWLEVEL_CALLBACK needs a callback");
483      if (opts->callback)
484        optbug(opts, "OPTION_LOWLEVEL_CALLBACK needs no high level callback");
485      break;
486    case OPTION_ALIAS:
487      optbug(opts, "OPT_ALIAS() should not remain at this point. "
488             "Are you using parse_options_step() directly?\n"
489             "That case is not supported yet.");
490      break;
491    default:
492      ; /* ok. (usually accepts an argument) */
493    }
494    if (opts->argh &&
495        strcspn(opts->argh, " _") != strlen(opts->argh))
496      optbug(opts, "multi-word argh should use dash to separate words");
497  }
498  BUG_if_bug("invalid 'struct option'");
499}

501static void parse_options_start_1(struct parse_opt_ctx_t *ctx,
502          int argc, const char **argv, const char *prefix,
503          const struct option *options,
504          enum parse_opt_flags flags)
505{
506  ctx->argc = argc;
507  ctx->argv = argv;
508  if (!(flags & PARSE_OPT_ONE_SHOT)) {
509    ctx->argc--;
510    ctx->argv++;
511  }
512  ctx->total = ctx->argc;
513  ctx->out   = argv;
514  ctx->prefix = prefix;
515  ctx->cpidx = ((flags & PARSE_OPT_KEEP_ARGV0) != 0);
516  ctx->flags = flags;
517  if ((flags & PARSE_OPT_KEEP_UNKNOWN) &&
518      (flags & PARSE_OPT_STOP_AT_NON_OPTION) &&
519      !(flags & PARSE_OPT_ONE_SHOT))
520    BUG("STOP_AT_NON_OPTION and KEEP_UNKNOWN don't go together");
521  if ((flags & PARSE_OPT_ONE_SHOT) &&
522      (flags & PARSE_OPT_KEEP_ARGV0))
523    BUG("Can't keep argv0 if you don't have it");
524  parse_options_check(options);

536static void show_negated_gitcomp(const struct option *opts, int show_all,
537         int nr_noopts)
538{
539  int printed_dashdash = 0;
540
541  for (; opts->type != OPTION_END; opts++) {
542    int has_unset_form = 0;
543    const char *name;
544
545    if (!opts->long_name)
546      continue;
547    if (!show_all &&
548      (opts->flags & (PARSE_OPT_HIDDEN | PARSE_OPT_NOCOMPLETE)))
549      continue;
550    if (opts->flags & PARSE_OPT_NONEG)
551      continue;
552
553    switch (opts->type) {
554    case OPTION_STRING:
555    case OPTION_FILENAME:
556    case OPTION_INTEGER:
557    case OPTION_MAGNITUDE:
558    case OPTION_CALLBACK:
559    case OPTION_BIT:
560    case OPTION_NEGBIT:
561    case OPTION_COUNTUP:
562    case OPTION_SET_INT:
563      has_unset_form = 1;
564      break;
565    default:
566      break;
567    }
568    if (!has_unset_form)
569      continue;
570
571    if (skip_prefix(opts->long_name, "no-", &name)) {
572      if (nr_noopts < 0)
573        printf(" --%s", name);
574    } else if (nr_noopts >= 0) {
575      if (nr_noopts && !printed_dashdash) {
576        printf(" --");
577        printed_dashdash = 1;
578      }
579      printf(" --no-%s", opts->long_name);
580      nr_noopts++;
581    }
582  }
583}

585static int show_gitcomp(const struct option *opts, int show_all)
586{
587  const struct option *original_opts = opts;
588  int nr_noopts = 0;
589
590  for (; opts->type != OPTION_END; opts++) {
591    const char *suffix = "";
592
593    if (!opts->long_name)
594      continue;
595    if (!show_all &&
596      (opts->flags & (PARSE_OPT_HIDDEN | PARSE_OPT_NOCOMPLETE | PARSE_OPT_FROM_ALIAS)))
597      continue;
598
599    switch (opts->type) {
600    case OPTION_GROUP:
601      continue;
602    case OPTION_STRING:
603    case OPTION_FILENAME:
604    case OPTION_INTEGER:
605    case OPTION_MAGNITUDE:
606    case OPTION_CALLBACK:
607      if (opts->flags & PARSE_OPT_NOARG)
608        break;
609      if (opts->flags & PARSE_OPT_OPTARG)
610        break;
611      if (opts->flags & PARSE_OPT_LASTARG_DEFAULT)
612        break;
613      suffix = "=";
614      break;
615    default:
616      break;
617    }
618    if (opts->flags & PARSE_OPT_COMP_ARG)
619      suffix = "=";
620    if (starts_with(opts->long_name, "no-"))
621      nr_noopts++;
622    printf(" --%s%s", opts->long_name, suffix);
623  }
624  show_negated_gitcomp(original_opts, show_all, -1);
625  show_negated_gitcomp(original_opts, show_all, nr_noopts);
626  fputc('\n', stdout);
627  return PARSE_OPT_COMPLETE;
628}

630/*
631 * Scan and may produce a new option[] array, which should be used
632 * instead of the original 'options'.
633 *
634 * Right now this is only used to preprocess and substitute
635 * OPTION_ALIAS.
636 *
637 * The returned options should be freed using free_preprocessed_options.
638 */
639static struct option *preprocess_options(struct parse_opt_ctx_t *ctx,
640           const struct option *options)
641{
642  struct option *newopt;
643  int i, nr, alias;
644  int nr_aliases = 0;
645
646  for (nr = 0; options[nr].type != OPTION_END; nr++) {
647    if (options[nr].type == OPTION_ALIAS)
648      nr_aliases++;
649  }
650
651  if (!nr_aliases)
652    return NULL;
653
654  ALLOC_ARRAY(newopt, nr + 1);
655  COPY_ARRAY(newopt, options, nr + 1);
656
657  /* each alias has two string pointers and NULL */
658  CALLOC_ARRAY(ctx->alias_groups, 3 * (nr_aliases + 1));
659
660  for (alias = 0, i = 0; i < nr; i++) {
661    int short_name;
662    const char *long_name;
663    const char *source;
664    struct strbuf help = STRBUF_INIT;
665    int j;
666
667    if (newopt[i].type != OPTION_ALIAS)
668      continue;
669
670    short_name = newopt[i].short_name;
671    long_name = newopt[i].long_name;
672    source = newopt[i].value;
673
674    if (!long_name)
675      BUG("An alias must have long option name");
676    strbuf_addf(&help, _("alias of --%s"), source);
677
678    for (j = 0; j < nr; j++) {
679      const char *name = options[j].long_name;
680
681      if (!name || strcmp(name, source))
682        continue;
683
684      if (options[j].type == OPTION_ALIAS)
685        BUG("No please. Nested aliases are not supported.");
686
687      memcpy(newopt + i, options + j, sizeof(*newopt));
688      newopt[i].short_name = short_name;
689      newopt[i].long_name = long_name;
690      newopt[i].help = strbuf_detach(&help, NULL);
691      newopt[i].flags |= PARSE_OPT_FROM_ALIAS;
692      break;
693    }
694
695    if (j == nr)
696      BUG("could not find source option '%s' of alias '%s'",
697          source, newopt[i].long_name);
698    ctx->alias_groups[alias * 3 + 0] = newopt[i].long_name;
699    ctx->alias_groups[alias * 3 + 1] = options[j].long_name;
700    ctx->alias_groups[alias * 3 + 2] = NULL;
701    alias++;
702  }
703
704  return newopt;
705}

707static void free_preprocessed_options(struct option *options)
708{
709  int i;
710
711  if (!options)
712    return;
713
714  for (i = 0; options[i].type != OPTION_END; i++) {
715    if (options[i].flags & PARSE_OPT_FROM_ALIAS)
716      free((void *)options[i].help);
717  }
718  free(options);
719}

721static enum parse_opt_result usage_with_options_internal(struct parse_opt_ctx_t *,
722               const char * const *,
723               const struct option *,
724               int, int);
725
726enum parse_opt_result parse_options_step(struct parse_opt_ctx_t *ctx,
727           const struct option *options,
728           const char * const usagestr[])
729{
730  int internal_help = !(ctx->flags & PARSE_OPT_NO_INTERNAL_HELP);
731
732  /* we must reset ->opt, unknown short option leave it dangling */
733  ctx->opt = NULL;
734
735  for (; ctx->argc; ctx->argc--, ctx->argv++) {
736    const char *arg = ctx->argv[0];
737
738    if (ctx->flags & PARSE_OPT_ONE_SHOT &&
739        ctx->argc != ctx->total)
740      break;
741
742    if (*arg != '-' || !arg[1]) {
743      if (parse_nodash_opt(ctx, arg, options) == 0)
744        continue;
745      if (ctx->flags & PARSE_OPT_STOP_AT_NON_OPTION)
746        return PARSE_OPT_NON_OPTION;
747      ctx->out[ctx->cpidx++] = ctx->argv[0];
748      continue;
749    }
750
751    /* lone -h asks for help */
752    if (internal_help && ctx->total == 1 && !strcmp(arg + 1, "h"))
753      goto show_usage;
754
755    /*
756     * lone --git-completion-helper and --git-completion-helper-all
757     * are asked by git-completion.bash
758     */
759    if (ctx->total == 1 && !strcmp(arg, "--git-completion-helper"))
760      return show_gitcomp(options, 0);
761    if (ctx->total == 1 && !strcmp(arg, "--git-completion-helper-all"))
762      return show_gitcomp(options, 1);
763
764    if (arg[1] != '-') {
765      ctx->opt = arg + 1;
766      switch (parse_short_opt(ctx, options)) {
767      case PARSE_OPT_ERROR:
768        return PARSE_OPT_ERROR;
769      case PARSE_OPT_UNKNOWN:
770        if (ctx->opt)
771          check_typos(arg + 1, options);
772        if (internal_help && *ctx->opt == 'h')
773          goto show_usage;
774        goto unknown;
775      case PARSE_OPT_NON_OPTION:
776      case PARSE_OPT_HELP:
777      case PARSE_OPT_COMPLETE:
778        BUG("parse_short_opt() cannot return these");
779      case PARSE_OPT_DONE:
780        break;
781      }
782      if (ctx->opt)
783        check_typos(arg + 1, options);
784      while (ctx->opt) {
785        switch (parse_short_opt(ctx, options)) {
786        case PARSE_OPT_ERROR:
787          return PARSE_OPT_ERROR;
788        case PARSE_OPT_UNKNOWN:
789          if (internal_help && *ctx->opt == 'h')
790            goto show_usage;
791
792          /* fake a short option thing to hide the fact that we may have
793           * started to parse aggregated stuff
794           *
795           * This is leaky, too bad.
796           */
797          ctx->argv[0] = xstrdup(ctx->opt - 1);
798          *(char *)ctx->argv[0] = '-';
799          goto unknown;
800        case PARSE_OPT_NON_OPTION:
801        case PARSE_OPT_COMPLETE:
802        case PARSE_OPT_HELP:
803          BUG("parse_short_opt() cannot return these");
804        case PARSE_OPT_DONE:
805          break;
806        }
807      }
808      continue;
809    }
810
811    if (!arg[2] /* "--" */ ||
812        !strcmp(arg + 2, "end-of-options")) {
813      if (!(ctx->flags & PARSE_OPT_KEEP_DASHDASH)) {
814        ctx->argc--;
815        ctx->argv++;
816      }
817      break;
818    }
819
820    if (internal_help && !strcmp(arg + 2, "help-all"))
821      return usage_with_options_internal(ctx, usagestr, options, 1, 0);
822    if (internal_help && !strcmp(arg + 2, "help"))
823      goto show_usage;
824    switch (parse_long_opt(ctx, arg + 2, options)) {
825    case PARSE_OPT_ERROR:
826      return PARSE_OPT_ERROR;
827    case PARSE_OPT_UNKNOWN:
828      goto unknown;
829    case PARSE_OPT_HELP:
830      goto show_usage;
831    case PARSE_OPT_NON_OPTION:
832    case PARSE_OPT_COMPLETE:
833      BUG("parse_long_opt() cannot return these");
834    case PARSE_OPT_DONE:
835      break;
836    }
837    continue;
838unknown:
839    if (ctx->flags & PARSE_OPT_ONE_SHOT)
840      break;
841    if (!(ctx->flags & PARSE_OPT_KEEP_UNKNOWN))
842      return PARSE_OPT_UNKNOWN;
843    ctx->out[ctx->cpidx++] = ctx->argv[0];
844    ctx->opt = NULL;
845  }
846  return PARSE_OPT_DONE;
847
848 show_usage:
849  return usage_with_options_internal(ctx, usagestr, options, 0, 0);
850}
851
852int parse_options_end(struct parse_opt_ctx_t *ctx)
853{
854  if (ctx->flags & PARSE_OPT_ONE_SHOT)
855    return ctx->total - ctx->argc;
856
857  MOVE_ARRAY(ctx->out + ctx->cpidx, ctx->argv, ctx->argc);
858  ctx->out[ctx->cpidx + ctx->argc] = NULL;
859  return ctx->cpidx + ctx->argc;
860}
861
862int parse_options(int argc, const char **argv,
863      const char *prefix,
864      const struct option *options,
865      const char * const usagestr[],
866      enum parse_opt_flags flags)
867{
868  struct parse_opt_ctx_t ctx;
869  struct option *real_options;
870
871  disallow_abbreviated_options =
872    git_env_bool("GIT_TEST_DISALLOW_ABBREVIATED_OPTIONS", 0);
873
874  memset(&ctx, 0, sizeof(ctx));
875  real_options = preprocess_options(&ctx, options);
876  if (real_options)
877    options = real_options;
878  parse_options_start_1(&ctx, argc, argv, prefix, options, flags);
879  switch (parse_options_step(&ctx, options, usagestr)) {
880  case PARSE_OPT_HELP:
881  case PARSE_OPT_ERROR:
882    exit(129);
883  case PARSE_OPT_COMPLETE:
884    exit(0);
885  case PARSE_OPT_NON_OPTION:
886  case PARSE_OPT_DONE:
887    break;
888  case PARSE_OPT_UNKNOWN:
889    if (ctx.argv[0][1] == '-') {
890      error(_("unknown option `%s'"), ctx.argv[0] + 2);
891    } else if (isascii(*ctx.opt)) {
892      error(_("unknown switch `%c'"), *ctx.opt);
893    } else {
894      error(_("unknown non-ascii option in string: `%s'"),
895            ctx.argv[0]);
896    }
897    usage_with_options(usagestr, options);
898  }
899
900  precompose_argv_prefix(argc, argv, NULL);
901  free_preprocessed_options(real_options);
902  free(ctx.alias_groups);
903  return parse_options_end(&ctx);
904}

906static int usage_argh(const struct option *opts, FILE *outfile)
907{
908  const char *s;
909  int literal = (opts->flags & PARSE_OPT_LITERAL_ARGHELP) ||
910    !opts->argh || !!strpbrk(opts->argh, "()<>[]|");
911  if (opts->flags & PARSE_OPT_OPTARG)
912    if (opts->long_name)
913      s = literal ? "[=%s]" : "[=<%s>]";
914    else
915      s = literal ? "[%s]" : "[<%s>]";
916  else
917    s = literal ? " %s" : " <%s>";
918  return utf8_fprintf(outfile, s, opts->argh ? _(opts->argh) : _("..."));
919}
920
921#define USAGE_OPTS_WIDTH 24
922#define USAGE_GAP         2
923
924static enum parse_opt_result usage_with_options_internal(struct parse_opt_ctx_t *ctx,
925               const char * const *usagestr,
926               const struct option *opts,
927               int full, int err)
928{
929  FILE *outfile = err ? stderr : stdout;
930  int need_newline;
931
932  const char *usage_prefix = _("usage: %s");
933  /*
934   * The translation could be anything, but we can count on
935   * msgfmt(1)'s --check option to have asserted that "%s" is in
936   * the translation. So compute the length of the "usage: "
937   * part. We are assuming that the translator wasn't overly
938   * clever and used e.g. "%1$s" instead of "%s", there's only
939   * one "%s" in "usage_prefix" above, so there's no reason to
940   * do so even with a RTL language.
941   */
942  size_t usage_len = strlen(usage_prefix) - strlen("%s");
943  /*
944   * TRANSLATORS: the colon here should align with the
945   * one in "usage: %s" translation.
946   */
947  const char *or_prefix = _("   or: %s");
948  /*
949   * TRANSLATORS: You should only need to translate this format
950   * string if your language is a RTL language (e.g. Arabic,
951   * Hebrew etc.), not if it's a LTR language (e.g. German,
952   * Russian, Chinese etc.).
953   *
954   * When a translated usage string has an embedded "\n" it's
955   * because options have wrapped to the next line. The line
956   * after the "\n" will then be padded to align with the
957   * command name, such as N_("git cmd [opt]\n<8
958   * spaces>[opt2]"), where the 8 spaces are the same length as
959   * "git cmd ".
960   *
961   * This format string prints out that already-translated
962   * line. The "%*s" is whitespace padding to account for the
963   * padding at the start of the line that we add in this
964   * function. The "%s" is a line in the (hopefully already
965   * translated) N_() usage string, which contained embedded
966   * newlines before we split it up.
967   */
968  const char *usage_continued = _("%*s%s");
969  const char *prefix = usage_prefix;
970  int saw_empty_line = 0;
971
972  if (!usagestr)
973    return PARSE_OPT_HELP;
974
975  if (!err && ctx && ctx->flags & PARSE_OPT_SHELL_EVAL)
976    fprintf(outfile, "cat <<\\EOF\n");
977
978  while (*usagestr) {
979    const char *str = _(*usagestr++);
980    struct string_list list = STRING_LIST_INIT_DUP;
981    unsigned int j;
982
983    if (!saw_empty_line && !*str)
984      saw_empty_line = 1;
985
986    string_list_split(&list, str, '\n', -1);
987    for (j = 0; j < list.nr; j++) {
988      const char *line = list.items[j].string;
989
990      if (saw_empty_line && *line)
991        fprintf_ln(outfile, _("    %s"), line);
992      else if (saw_empty_line)
993        fputc('\n', outfile);
994      else if (!j)
995        fprintf_ln(outfile, prefix, line);
996      else
997        fprintf_ln(outfile, usage_continued,
998             (int)usage_len, "", line);
999    }
1000    string_list_clear(&list, 0);
1001
1002    prefix = or_prefix;
1003  }
1004
1005  need_newline = 1;
1006
1007  for (; opts->type != OPTION_END; opts++) {
1008    size_t pos;
1009    int pad;
1010
1011    if (opts->type == OPTION_GROUP) {
1012      fputc('\n', outfile);
1013      need_newline = 0;
1014      if (*opts->help)
1015        fprintf(outfile, "%s\n", _(opts->help));
1016      continue;
1017    }
1018    if (!full && (opts->flags & PARSE_OPT_HIDDEN))
1019      continue;
1020
1021    if (need_newline) {
1022      fputc('\n', outfile);
1023      need_newline = 0;
1024    }
1025
1026    pos = fprintf(outfile, "    ");
1027    if (opts->short_name) {
1028      if (opts->flags & PARSE_OPT_NODASH)
1029        pos += fprintf(outfile, "%c", opts->short_name);
1030      else
1031        pos += fprintf(outfile, "-%c", opts->short_name);
1032    }
1033    if (opts->long_name && opts->short_name)
1034      pos += fprintf(outfile, ", ");
1035    if (opts->long_name)
1036      pos += fprintf(outfile, "--%s", opts->long_name);
1037    if (opts->type == OPTION_NUMBER)
1038      pos += utf8_fprintf(outfile, _("-NUM"));
1039
1040    if ((opts->flags & PARSE_OPT_LITERAL_ARGHELP) ||
1041        !(opts->flags & PARSE_OPT_NOARG))
1042      pos += usage_argh(opts, outfile);
1043
1044    if (pos <= USAGE_OPTS_WIDTH)
1045      pad = USAGE_OPTS_WIDTH - pos;
1046    else {
1047      fputc('\n', outfile);
1048      pad = USAGE_OPTS_WIDTH;
1049    }
1050    if (opts->type == OPTION_ALIAS) {
1051      fprintf(outfile, "%*s", pad + USAGE_GAP, "");
1052      fprintf_ln(outfile, _("alias of --%s"),
1053           (const char *)opts->value);
1054      continue;
1055    }
1056    fprintf(outfile, "%*s%s\n", pad + USAGE_GAP, "", _(opts->help));
1057  }
1058  fputc('\n', outfile);
1059
1060  if (!err && ctx && ctx->flags & PARSE_OPT_SHELL_EVAL)
1061    fputs("EOF\n", outfile);
1062
1063  return PARSE_OPT_HELP;
1064}

1094void die_for_incompatible_opt4(int opt1, const char *opt1_name,
1095             int opt2, const char *opt2_name,
1096             int opt3, const char *opt3_name,

1102  if (opt1)
