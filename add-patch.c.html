311static int parse_range(const char **p,
312           unsigned long *offset, unsigned long *count)
313{
314  char *pend;
315
316  *offset = strtoul(*p, &pend, 10);
317  if (pend == *p)
318    return -1;
319  if (*pend != ',') {
320    *count = 1;

328static int parse_hunk_header(struct add_p_state *s, struct hunk *hunk)
329{
330  struct hunk_header *header = &hunk->header;
331  const char *line = s->plain.buf + hunk->start, *p = line;
332  char *eol = memchr(p, '\n', s->plain.len - hunk->start);
333
334  if (!eol)
335    eol = s->plain.buf + s->plain.len;
336
337  if (!skip_prefix(p, "@@ -", &p) ||
338      parse_range(&p, &header->old_offset, &header->old_count) < 0 ||
339      !skip_prefix(p, " +", &p) ||
340      parse_range(&p, &header->new_offset, &header->new_count) < 0 ||
341      !skip_prefix(p, " @@", &p))
342    return error(_("could not parse hunk header '%.*s'"),
343           (int)(eol - line), line);
344
345  hunk->start = eol - s->plain.buf + (*eol == '\n');
346  header->extra_start = p - s->plain.buf;
347  header->extra_end = hunk->start;
348
349  if (!s->colored.len) {
350    header->colored_extra_start = header->colored_extra_end = 0;
351    return 0;
352  }
353
354  /* Now find the extra text in the colored diff */
355  line = s->colored.buf + hunk->colored_start;
356  eol = memchr(line, '\n', s->colored.len - hunk->colored_start);
357  if (!eol)
358    eol = s->colored.buf + s->colored.len;
359  p = memmem(line, eol - line, "@@ -", 4);
360  if (!p)
361    return error(_("could not parse colored hunk header '%.*s'"),
362           (int)(eol - line), line);
363  p = memmem(p + 4, eol - p - 4, " @@", 3);
364  if (!p)
365    return error(_("could not parse colored hunk header '%.*s'"),
366           (int)(eol - line), line);
367  hunk->colored_start = eol - s->colored.buf + (*eol == '\n');
368  header->colored_extra_start = p + 3 - s->colored.buf;
369  header->colored_extra_end = hunk->colored_start;
370
371  return 0;
372}

397{

415          !strcmp("HEAD", s->revision) &&
416          get_oid("HEAD", &oid) ?
417          empty_tree_oid_hex() : s->revision);
418  }
419  color_arg_index = args.nr;
420  /* Use `--no-color` explicitly, just in case `diff.color = always`. */
421  strvec_pushl(&args, "--no-color", "-p", "--", NULL);
422  for (i = 0; i < ps->nr; i++)
423    strvec_push(&args, ps->items[i].original);
424
425  setup_child_process(s, &cp, NULL);
426  strvec_pushv(&cp.args, args.v);
427  res = capture_command(&cp, plain, 0);
428  if (res) {
429    strvec_clear(&args);
430    return error(_("could not parse diff"));
431  }
432  if (!plain->len) {
433    strvec_clear(&args);
434    return 0;
435  }
436  strbuf_complete_line(plain);
437
438  if (want_color_fd(1, -1)) {
439    struct child_process colored_cp = CHILD_PROCESS_INIT;
440    const char *diff_filter = s->s.interactive_diff_filter;
441
442    setup_child_process(s, &colored_cp, NULL);
443    xsnprintf((char *)args.v[color_arg_index], 8, "--color");
444    strvec_pushv(&colored_cp.args, args.v);
445    colored = &s->colored;
446    res = capture_command(&colored_cp, colored, 0);
447    strvec_clear(&args);
448    if (res)
449      return error(_("could not parse colored diff"));
450
451    if (diff_filter) {
452      struct child_process filter_cp = CHILD_PROCESS_INIT;
453
454      setup_child_process(s, &filter_cp,
455              diff_filter, NULL);
456      filter_cp.git_cmd = 0;
457      filter_cp.use_shell = 1;
458      strbuf_reset(&s->buf);
459      if (pipe_command(&filter_cp,
460           colored->buf, colored->len,
461           &s->buf, colored->len,
462           NULL, 0) < 0)
463        return error(_("failed to run '%s'"),
464               diff_filter);
465      strbuf_swap(colored, &s->buf);
466    }
467
468    strbuf_complete_line(colored);
469    colored_p = colored->buf;
470    colored_pend = colored_p + colored->len;
471  }
472  strvec_clear(&args);
473
474  /* parse files and hunks */
475  p = plain->buf;
476  pend = p + plain->len;
477  while (p != pend) {
478    char *eol = memchr(p, '\n', pend - p);
479    const char *deleted = NULL, *mode_change = NULL;
480
481    if (!eol)
482      eol = pend;
483
484    if (starts_with(p, "diff ")) {
485      complete_file(marker, hunk);
486      ALLOC_GROW_BY(s->file_diff, s->file_diff_nr, 1,
487           file_diff_alloc);
488      file_diff = s->file_diff + s->file_diff_nr - 1;
489      hunk = &file_diff->head;
490      hunk->start = p - plain->buf;
491      if (colored_p)
492        hunk->colored_start = colored_p - colored->buf;
493      marker = '\0';
494    } else if (p == plain->buf)
495      BUG("diff starts with unexpected line:\n"
496          "%.*s\n", (int)(eol - p), p);
497    else if (file_diff->deleted)
498      ; /* keep the rest of the file in a single "hunk" */
499    else if (starts_with(p, "@@ ") ||
500       (hunk == &file_diff->head &&
501        (skip_prefix(p, "deleted file", &deleted)))) {
502      if (marker == '-' || marker == '+')
503        /*
504         * Should not happen; previous hunk did not end
505         * in a context line? Handle it anyway.
506         */
507        hunk->splittable_into++;
508
509      ALLOC_GROW_BY(file_diff->hunk, file_diff->hunk_nr, 1,
510           file_diff->hunk_alloc);
511      hunk = file_diff->hunk + file_diff->hunk_nr - 1;
512
513      hunk->start = p - plain->buf;
514      if (colored)
515        hunk->colored_start = colored_p - colored->buf;
516
517      if (deleted)
518        file_diff->deleted = 1;
519      else if (parse_hunk_header(s, hunk) < 0)
520        return -1;
521
522      /*
523       * Start counting into how many hunks this one can be
524       * split
525       */
526      marker = *p;
527    } else if (hunk == &file_diff->head &&
528         starts_with(p, "new file")) {
529      file_diff->added = 1;
530    } else if (hunk == &file_diff->head &&
531         skip_prefix(p, "old mode ", &mode_change) &&
532         is_octal(mode_change, eol - mode_change)) {
533      if (file_diff->mode_change)
534        BUG("double mode change?\n\n%.*s",
535            (int)(eol - plain->buf), plain->buf);
536      if (file_diff->hunk_nr)
537        BUG("mode change in the middle?\n\n%.*s",
538            (int)(eol - plain->buf), plain->buf);
539
540      /*
541       * Do *not* change `hunk`: the mode change pseudo-hunk
542       * is _part of_ the header "hunk".
543       */
544      file_diff->mode_change = 1;
545      ALLOC_GROW_BY(file_diff->hunk, file_diff->hunk_nr, 1,
546           file_diff->hunk_alloc);
547      file_diff->hunk->start = p - plain->buf;
548      if (colored_p)
549        file_diff->hunk->colored_start =
550          colored_p - colored->buf;
551    } else if (hunk == &file_diff->head &&
552         skip_prefix(p, "new mode ", &mode_change) &&
553         is_octal(mode_change, eol - mode_change)) {
554
555      /*
556       * Extend the "mode change" pseudo-hunk to include also
557       * the "new mode" line.
558       */
559      if (!file_diff->mode_change)
560        BUG("'new mode' without 'old mode'?\n\n%.*s",
561            (int)(eol - plain->buf), plain->buf);
562      if (file_diff->hunk_nr != 1)
563        BUG("mode change in the middle?\n\n%.*s",
564            (int)(eol - plain->buf), plain->buf);
565      if (p - plain->buf != file_diff->hunk->end)
566        BUG("'new mode' does not immediately follow "
567            "'old mode'?\n\n%.*s",
568            (int)(eol - plain->buf), plain->buf);
569    } else if (hunk == &file_diff->head &&
570         starts_with(p, "Binary files "))
571      file_diff->binary = 1;
572
573    if (!!file_diff->deleted + !!file_diff->added +
574        !!file_diff->mode_change > 1)
575      BUG("diff can only contain delete *or* add *or* a "
576          "mode change?!?\n%.*s",
577          (int)(eol - (plain->buf + file_diff->head.start)),
578          plain->buf + file_diff->head.start);
579
580    if ((marker == '-' || marker == '+') && *p == ' ')
581      hunk->splittable_into++;
582    if (marker && *p != '\\')
583      marker = *p;
584
585    p = eol == pend ? pend : eol + 1;
586    hunk->end = p - plain->buf;
587
588    if (colored) {
589      char *colored_eol = memchr(colored_p, '\n',
590               colored_pend - colored_p);
591      if (colored_eol)
592        colored_p = colored_eol + 1;
593      else if (p != pend)
594        /* colored shorter than non-colored? */
595        goto mismatched_output;
596      else
597        colored_p = colored_pend;
598
599      hunk->colored_end = colored_p - colored->buf;
600    }
601
602    if (mode_change) {
603      if (file_diff->hunk_nr != 1)
604        BUG("mode change in hunk #%d???",
605            (int)file_diff->hunk_nr);
606      /* Adjust the end of the "mode change" pseudo-hunk */
607      file_diff->hunk->end = hunk->end;
608      if (colored)
609        file_diff->hunk->colored_end = hunk->colored_end;
610    }
611  }
612  complete_file(marker, hunk);
613
614  /* non-colored shorter than colored? */
615  if (colored_p != colored_pend) {
616mismatched_output:
617    error(_("mismatched output from interactive.diffFilter"));
618    advise(_("Your filter must maintain a one-to-one correspondence\n"
619       "between its input and output lines."));
620    return -1;
621  }
622
623  return 0;
624}

626static size_t find_next_line(struct strbuf *sb, size_t offset)
627{
628  char *eol;
629
630  if (offset >= sb->len)
631    BUG("looking for next line beyond buffer (%d >= %d)\n%s",
632        (int)offset, (int)sb->len, sb->buf);
633
634  eol = memchr(sb->buf + offset, '\n', sb->len - offset);
635  if (!eol)
636    return sb->len;
637  return eol - sb->buf + 1;
638}

640static void render_hunk(struct add_p_state *s, struct hunk *hunk,
641      ssize_t delta, int colored, struct strbuf *out)
642{
643  struct hunk_header *header = &hunk->header;
644
645  if (hunk->header.old_offset != 0 || hunk->header.new_offset != 0) {
646    /*
647     * Generate the hunk header dynamically, except for special
648     * hunks (such as the diff header).
649     */
650    const char *p;
651    size_t len;
652    unsigned long old_offset = header->old_offset;
653    unsigned long new_offset = header->new_offset;
654
655    if (!colored) {
656      p = s->plain.buf + header->extra_start;
657      len = header->extra_end - header->extra_start;
658    } else {
659      strbuf_addstr(out, s->s.fraginfo_color);
660      p = s->colored.buf + header->colored_extra_start;
661      len = header->colored_extra_end
662        - header->colored_extra_start;
663    }
664
665    if (s->mode->is_reverse)
666      old_offset -= delta;
667    else
668      new_offset += delta;
669
670    strbuf_addf(out, "@@ -%lu", old_offset);
671    if (header->old_count != 1)
672      strbuf_addf(out, ",%lu", header->old_count);
673    strbuf_addf(out, " +%lu", new_offset);
674    if (header->new_count != 1)
675      strbuf_addf(out, ",%lu", header->new_count);
676    strbuf_addstr(out, " @@");
677
678    if (len)
679      strbuf_add(out, p, len);
680    else if (colored)
681      strbuf_addf(out, "%s\n", s->s.reset_color);
682    else
683      strbuf_addch(out, '\n');
684  }
685
686  if (colored)
687    strbuf_add(out, s->colored.buf + hunk->colored_start,
688         hunk->colored_end - hunk->colored_start);
689  else
690    strbuf_add(out, s->plain.buf + hunk->start,
691         hunk->end - hunk->start);
692}

694static void render_diff_header(struct add_p_state *s,
695             struct file_diff *file_diff, int colored,
696             struct strbuf *out)
697{
698  /*
699   * If there was a mode change, the first hunk is a pseudo hunk that
700   * corresponds to the mode line in the header. If the user did not want
701   * to stage that "hunk", we actually have to cut it out from the header.
702   */
703  int skip_mode_change =
704    file_diff->mode_change && file_diff->hunk->use != USE_HUNK;
705  struct hunk *head = &file_diff->head, *first = file_diff->hunk;
706
707  if (!skip_mode_change) {
708    render_hunk(s, head, 0, colored, out);
709    return;
710  }
711
712  if (colored) {
713    const char *p = s->colored.buf;
714
715    strbuf_add(out, p + head->colored_start,
716          first->colored_start - head->colored_start);
717    strbuf_add(out, p + first->colored_end,
718          head->colored_end - first->colored_end);
719  } else {
720    const char *p = s->plain.buf;
721
722    strbuf_add(out, p + head->start, first->start - head->start);
723    strbuf_add(out, p + first->end, head->end - first->end);
724  }
725}

727/* Coalesce hunks again that were split */
728static int merge_hunks(struct add_p_state *s, struct file_diff *file_diff,
729           size_t *hunk_index, int use_all, struct hunk *merged)
730{
731  size_t i = *hunk_index, delta;
732  struct hunk *hunk = file_diff->hunk + i;
733  /* `header` corresponds to the merged hunk */
734  struct hunk_header *header = &merged->header, *next;
735
736  if (!use_all && hunk->use != USE_HUNK)
737    return 0;
738
739  *merged = *hunk;
740  /* We simply skip the colored part (if any) when merging hunks */
741  merged->colored_start = merged->colored_end = 0;
742
743  for (; i + 1 < file_diff->hunk_nr; i++) {
744    hunk++;
745    next = &hunk->header;
746
747    /*
748     * Stop merging hunks when:
749     *
750     * - the hunk is not selected for use, or
751     * - the hunk does not overlap with the already-merged hunk(s)
752     */
753    if ((!use_all && hunk->use != USE_HUNK) ||
754        header->new_offset >= next->new_offset + merged->delta ||
755        header->new_offset + header->new_count
756        < next->new_offset + merged->delta)
757      break;
758
759    /*
760     * If the hunks were not edited, and overlap, we can simply
761     * extend the line range.
762     */
763    if (merged->start < hunk->start && merged->end > hunk->start) {
764      merged->end = hunk->end;
765      merged->colored_end = hunk->colored_end;
766      delta = 0;
767    } else {
768      const char *plain = s->plain.buf;
769      size_t  overlapping_line_count = header->new_offset
770        + header->new_count - merged->delta
771        - next->new_offset;
772      size_t overlap_end = hunk->start;
773      size_t overlap_start = overlap_end;
774      size_t overlap_next, len, j;
775
776      /*
777       * One of the hunks was edited: the modified hunk was
778       * appended to the strbuf `s->plain`.
779       *
780       * Let's ensure that at least the last context line of
781       * the first hunk overlaps with the corresponding line
782       * of the second hunk, and then merge.
783       */
784      for (j = 0; j < overlapping_line_count; j++) {
785        overlap_next = find_next_line(&s->plain,
786                    overlap_end);
787
788        if (overlap_next > hunk->end)
789          BUG("failed to find %d context lines "
790              "in:\n%.*s",
791              (int)overlapping_line_count,
792              (int)(hunk->end - hunk->start),
793              plain + hunk->start);
794
795        if (plain[overlap_end] != ' ')
796          return error(_("expected context line "
797                   "#%d in\n%.*s"),
798                 (int)(j + 1),
799                 (int)(hunk->end
800                 - hunk->start),
801                 plain + hunk->start);
802
803        overlap_start = overlap_end;
804        overlap_end = overlap_next;
805      }
806      len = overlap_end - overlap_start;
807
808      if (len > merged->end - merged->start ||
809          memcmp(plain + merged->end - len,
810           plain + overlap_start, len))
811        return error(_("hunks do not overlap:\n%.*s\n"
812                 "\tdoes not end with:\n%.*s"),
813               (int)(merged->end - merged->start),
814               plain + merged->start,
815               (int)len, plain + overlap_start);
816
817      /*
818       * Since the start-end ranges are not adjacent, we
819       * cannot simply take the union of the ranges. To
820       * address that, we temporarily append the union of the
821       * lines to the `plain` strbuf.
822       */
823      if (merged->end != s->plain.len) {
824        size_t start = s->plain.len;
825
826        strbuf_add(&s->plain, plain + merged->start,
827             merged->end - merged->start);
828        plain = s->plain.buf;
829        merged->start = start;
830        merged->end = s->plain.len;
831      }
832
833      strbuf_add(&s->plain,
834           plain + overlap_end,
835           hunk->end - overlap_end);
836      merged->end = s->plain.len;
837      merged->splittable_into += hunk->splittable_into;
838      delta = merged->delta;
839      merged->delta += hunk->delta;
840    }
841
842    header->old_count = next->old_offset + next->old_count
843      - header->old_offset;
844    header->new_count = next->new_offset + delta
845      + next->new_count - header->new_offset;
846  }
847
848  if (i == *hunk_index)
849    return 0;
850
851  *hunk_index = i;
852  return 1;
853}

855static void reassemble_patch(struct add_p_state *s,
856           struct file_diff *file_diff, int use_all,
857           struct strbuf *out)
858{
859  struct hunk *hunk;
860  size_t save_len = s->plain.len, i;
861  ssize_t delta = 0;
862
863  render_diff_header(s, file_diff, 0, out);
864
865  for (i = file_diff->mode_change; i < file_diff->hunk_nr; i++) {
866    struct hunk merged = { 0 };
867
868    hunk = file_diff->hunk + i;
869    if (!use_all && hunk->use != USE_HUNK)
870      delta += hunk->header.old_count
871        - hunk->header.new_count;
872    else {
873      /* merge overlapping hunks into a temporary hunk */
874      if (merge_hunks(s, file_diff, &i, use_all, &merged))
875        hunk = &merged;
876
877      render_hunk(s, hunk, delta, 0, out);
878
879      /*
880       * In case `merge_hunks()` used `plain` as a scratch
881       * pad (this happens when an edited hunk had to be
882       * coalesced with another hunk).
883       */
884      strbuf_setlen(&s->plain, save_len);
885
886      delta += hunk->delta;
887    }
888  }
889}

891static int split_hunk(struct add_p_state *s, struct file_diff *file_diff,
892           size_t hunk_index)
893{
894  int colored = !!s->colored.len, first = 1;
895  struct hunk *hunk = file_diff->hunk + hunk_index;
896  size_t splittable_into;
897  size_t end, colored_end, current, colored_current = 0, context_line_count;
898  struct hunk_header remaining, *header;
899  char marker, ch;
900
901  if (hunk_index >= file_diff->hunk_nr)
902    BUG("invalid hunk index: %d (must be >= 0 and < %d)",
903        (int)hunk_index, (int)file_diff->hunk_nr);
904
905  if (hunk->splittable_into < 2)
906    return 0;
907  splittable_into = hunk->splittable_into;
908
909  end = hunk->end;
910  colored_end = hunk->colored_end;
911
912  remaining = hunk->header;
913
914  file_diff->hunk_nr += splittable_into - 1;
915  ALLOC_GROW(file_diff->hunk, file_diff->hunk_nr, file_diff->hunk_alloc);
916  if (hunk_index + splittable_into < file_diff->hunk_nr)
917    memmove(file_diff->hunk + hunk_index + splittable_into,
918      file_diff->hunk + hunk_index + 1,
919      (file_diff->hunk_nr - hunk_index - splittable_into)
920      * sizeof(*hunk));
921  hunk = file_diff->hunk + hunk_index;
922  hunk->splittable_into = 1;
923  memset(hunk + 1, 0, (splittable_into - 1) * sizeof(*hunk));
924
925  header = &hunk->header;
926  header->old_count = header->new_count = 0;
927
928  current = hunk->start;
929  if (colored)
930    colored_current = hunk->colored_start;
931  marker = '\0';
932  context_line_count = 0;
933
934  while (splittable_into > 1) {
935    ch = s->plain.buf[current];
936
937    if (!ch)
938      BUG("buffer overrun while splitting hunks");
939
940    /*
941     * Is this the first context line after a chain of +/- lines?
942     * Then record the start of the next split hunk.
943     */
944    if ((marker == '-' || marker == '+') && ch == ' ') {
945      first = 0;
946      hunk[1].start = current;
947      if (colored)
948        hunk[1].colored_start = colored_current;
949      context_line_count = 0;
950    }
951
952    /*
953     * Was the previous line a +/- one? Alternatively, is this the
954     * first line (and not a +/- one)?
955     *
956     * Then just increment the appropriate counter and continue
957     * with the next line.
958     */
959    if (marker != ' ' || (ch != '-' && ch != '+')) {
960next_hunk_line:
961      /* Comment lines are attached to the previous line */
962      if (ch == '\\')
963        ch = marker ? marker : ' ';
964
965      /* current hunk not done yet */
966      if (ch == ' ')
967        context_line_count++;
968      else if (ch == '-')
969        header->old_count++;
970      else if (ch == '+')
971        header->new_count++;
972      else
973        BUG("unhandled diff marker: '%c'", ch);
974      marker = ch;
975      current = find_next_line(&s->plain, current);
976      if (colored)
977        colored_current =
978          find_next_line(&s->colored,
979                   colored_current);
980      continue;
981    }
982
983    /*
984     * We got us the start of a new hunk!
985     *
986     * This is a context line, so it is shared with the previous
987     * hunk, if any.
988     */
989
990    if (first) {
991      if (header->old_count || header->new_count)
992        BUG("counts are off: %d/%d",
993            (int)header->old_count,
994            (int)header->new_count);
995
996      header->old_count = context_line_count;
997      header->new_count = context_line_count;
998      context_line_count = 0;
999      first = 0;
1000      goto next_hunk_line;
1001    }
1002
1003    remaining.old_offset += header->old_count;
1004    remaining.old_count -= header->old_count;
1005    remaining.new_offset += header->new_count;
1006    remaining.new_count -= header->new_count;
1007
1008    /* initialize next hunk header's offsets */
1009    hunk[1].header.old_offset =
1010      header->old_offset + header->old_count;
1011    hunk[1].header.new_offset =
1012      header->new_offset + header->new_count;
1013
1014    /* add one split hunk */
1015    header->old_count += context_line_count;
1016    header->new_count += context_line_count;
1017
1018    hunk->end = current;
1019    if (colored)
1020      hunk->colored_end = colored_current;
1021
1022    hunk++;
1023    hunk->splittable_into = 1;
1024    hunk->use = hunk[-1].use;
1025    header = &hunk->header;
1026
1027    header->old_count = header->new_count = context_line_count;
1028    context_line_count = 0;
1029
1030    splittable_into--;
1031    marker = ch;
1032  }
1033
1034  /* last hunk simply gets the rest */
1035  if (header->old_offset != remaining.old_offset)
1036    BUG("miscounted old_offset: %lu != %lu",
1037        header->old_offset, remaining.old_offset);
1038  if (header->new_offset != remaining.new_offset)
1039    BUG("miscounted new_offset: %lu != %lu",
1040        header->new_offset, remaining.new_offset);
1041  header->old_count = remaining.old_count;
1042  header->new_count = remaining.new_count;
1043  hunk->end = end;
1044  if (colored)
1045    hunk->colored_end = colored_end;
1046
1047  return 0;
1048}

1051{
1052  const char *plain = s->plain.buf;

1056    return;
