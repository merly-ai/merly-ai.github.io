24static long xdl_get_rec(xdfile_t *xdf, long ri, char const **rec) {
25
26  *rec = xdf->recs[ri]->ptr;
27
28  return xdf->recs[ri]->size;
29}

32static int xdl_emit_record(xdfile_t *xdf, long ri, char const *pre, xdemitcb_t *ecb) {
33  long size, psize = strlen(pre);
34  char const *rec;
35
36  size = xdl_get_rec(xdf, ri, &rec);
37  if (xdl_emit_diffrec(rec, size, pre, psize, ecb) < 0) {
38
39    return -1;
40  }
41
42  return 0;
43}

46/*
47 * Starting at the passed change atom, find the latest change atom to be included
48 * inside the differential hunk according to the specified configuration.
49 * Also advance xscr if the first changes must be discarded.
50 */
51xdchange_t *xdl_get_hunk(xdchange_t **xscr, xdemitconf_t const *xecfg)
52{
53  xdchange_t *xch, *xchp, *lxch;
54  long max_common = 2 * xecfg->ctxlen + xecfg->interhunkctxlen;
55  long max_ignorable = xecfg->ctxlen;
56  unsigned long ignored = 0; /* number of ignored blank lines */
57
58  /* remove ignorable changes that are too far before other changes */
59  for (xchp = *xscr; xchp && xchp->ignore; xchp = xchp->next) {
60    xch = xchp->next;
61
62    if (xch == NULL ||
63        xch->i1 - (xchp->i1 + xchp->chg1) >= max_ignorable)
64      *xscr = xch;
65  }
66
67  if (!*xscr)
68    return NULL;
69
70  lxch = *xscr;
71
72  for (xchp = *xscr, xch = xchp->next; xch; xchp = xch, xch = xch->next) {
73    long distance = xch->i1 - (xchp->i1 + xchp->chg1);
74    if (distance > max_common)
75      break;
76
77    if (distance < max_ignorable && (!xch->ignore || lxch == xchp)) {
78      lxch = xch;
79      ignored = 0;
80    } else if (distance < max_ignorable && xch->ignore) {
81      ignored += xch->chg2;
82    } else if (lxch != xchp &&
83         xch->i1 + ignored - (lxch->i1 + lxch->chg1) > max_common) {
84      break;
85    } else if (!xch->ignore) {
86      lxch = xch;
87      ignored = 0;
88    } else {
89      ignored += xch->chg2;
90    }
91  }
92
93  return lxch;
94}

97static long def_ff(const char *rec, long len, char *buf, long sz, void *priv)
98{
99  if (len > 0 &&
100      (isalpha((unsigned char)*rec) || /* identifier? */
101       *rec == '_' || /* also identifier? */
102       *rec == '$')) { /* identifiers from VMS and other esoterico */
103    if (len > sz)
104      len = sz;
105    while (0 < len && isspace((unsigned char)rec[len - 1]))
106      len--;
107    memcpy(buf, rec, len);
108    return len;
109  }
110  return -1;
111}

113static long match_func_rec(xdfile_t *xdf, xdemitconf_t const *xecfg, long ri,
114         char *buf, long sz)
115{
116  const char *rec;
117  long len = xdl_get_rec(xdf, ri, &rec);
118  if (!xecfg->find_func)
119    return def_ff(rec, len, buf, sz, xecfg->find_func_priv);
120  return xecfg->find_func(rec, len, buf, sz, xecfg->find_func_priv);
121}

123static int is_func_rec(xdfile_t *xdf, xdemitconf_t const *xecfg, long ri)
124{
125  char dummy[1];
126  return match_func_rec(xdf, xecfg, ri, dummy, sizeof(dummy)) >= 0;
127}

129struct func_line {
130  long len;
131  char buf[80];
132};
133
134static long get_func_line(xdfenv_t *xe, xdemitconf_t const *xecfg,
135        struct func_line *func_line, long start, long limit)
136{
137  long l, size, step = (start > limit) ? -1 : 1;
138  char *buf, dummy[1];
139
140  buf = func_line ? func_line->buf : dummy;
141  size = func_line ? sizeof(func_line->buf) : sizeof(dummy);
142
143  for (l = start; l != limit && 0 <= l && l < xe->xdf1.nrec; l += step) {
144    long len = match_func_rec(&xe->xdf1, xecfg, l, buf, size);
145    if (len >= 0) {
146      if (func_line)
147        func_line->len = len;
148      return l;
149    }
150  }
151  return -1;
152}

154static int is_empty_rec(xdfile_t *xdf, long ri)
155{
156  const char *rec;
157  long len = xdl_get_rec(xdf, ri, &rec);
158
159  while (len > 0 && XDL_ISSPACE(*rec)) {
160    rec++;
161    len--;
162  }
163  return !len;
164}

166int xdl_emit_diff(xdfenv_t *xe, xdchange_t *xscr, xdemitcb_t *ecb,
167      xdemitconf_t const *xecfg) {
168  long s1, s2, e1, e2, lctx;
169  xdchange_t *xch, *xche;
170  long funclineprev = -1;
171  struct func_line func_line = { 0 };
172
173  for (xch = xscr; xch; xch = xche->next) {
174    xdchange_t *xchp = xch;
175    xche = xdl_get_hunk(&xch, xecfg);
176    if (!xch)
177      break;
178
179pre_context_calculation:
180    s1 = XDL_MAX(xch->i1 - xecfg->ctxlen, 0);
181    s2 = XDL_MAX(xch->i2 - xecfg->ctxlen, 0);
182
183    if (xecfg->flags & XDL_EMIT_FUNCCONTEXT) {
184      long fs1, i1 = xch->i1;
185
186      /* Appended chunk? */
187      if (i1 >= xe->xdf1.nrec) {
188        long i2 = xch->i2;
189
190        /*
191         * We don't need additional context if
192         * a whole function was added.
193         */
194        while (i2 < xe->xdf2.nrec) {
195          if (is_func_rec(&xe->xdf2, xecfg, i2))
196            goto post_context_calculation;
197          i2++;
198        }
199
200        /*
201         * Otherwise get more context from the
202         * pre-image.
203         */
204        i1 = xe->xdf1.nrec - 1;
205      }
206
207      fs1 = get_func_line(xe, xecfg, NULL, i1, -1);
208      while (fs1 > 0 && !is_empty_rec(&xe->xdf1, fs1 - 1) &&
209             !is_func_rec(&xe->xdf1, xecfg, fs1 - 1))
210        fs1--;
211      if (fs1 < 0)
212        fs1 = 0;
213      if (fs1 < s1) {
214        s2 = XDL_MAX(s2 - (s1 - fs1), 0);
215        s1 = fs1;
216
217        /*
218         * Did we extend context upwards into an
219         * ignored change?
220         */
221        while (xchp != xch &&
222               xchp->i1 + xchp->chg1 <= s1 &&
223               xchp->i2 + xchp->chg2 <= s2)
224          xchp = xchp->next;
225
226        /* If so, show it after all. */
227        if (xchp != xch) {
228          xch = xchp;
229          goto pre_context_calculation;
230        }
231      }
232    }
233
234 post_context_calculation:
235    lctx = xecfg->ctxlen;
236    lctx = XDL_MIN(lctx, xe->xdf1.nrec - (xche->i1 + xche->chg1));
237    lctx = XDL_MIN(lctx, xe->xdf2.nrec - (xche->i2 + xche->chg2));
238
239    e1 = xche->i1 + xche->chg1 + lctx;
240    e2 = xche->i2 + xche->chg2 + lctx;
241
242    if (xecfg->flags & XDL_EMIT_FUNCCONTEXT) {
243      long fe1 = get_func_line(xe, xecfg, NULL,
244             xche->i1 + xche->chg1,
245             xe->xdf1.nrec);
246      while (fe1 > 0 && is_empty_rec(&xe->xdf1, fe1 - 1))
247        fe1--;
248      if (fe1 < 0)
249        fe1 = xe->xdf1.nrec;
250      if (fe1 > e1) {
251        e2 = XDL_MIN(e2 + (fe1 - e1), xe->xdf2.nrec);
252        e1 = fe1;
253      }
254
255      /*
256       * Overlap with next change?  Then include it
257       * in the current hunk and start over to find
258       * its new end.
259       */
260      if (xche->next) {
261        long l = XDL_MIN(xche->next->i1,
262             xe->xdf1.nrec - 1);
263        if (l - xecfg->ctxlen <= e1 ||
264            get_func_line(xe, xecfg, NULL, l, e1) < 0) {
265          xche = xche->next;
266          goto post_context_calculation;
267        }
268      }
269    }
270
271    /*
272     * Emit current hunk header.
273     */
274
275    if (xecfg->flags & XDL_EMIT_FUNCNAMES) {
276      get_func_line(xe, xecfg, &func_line,
277              s1 - 1, funclineprev);
278      funclineprev = s1 - 1;
279    }
280    if (!(xecfg->flags & XDL_EMIT_NO_HUNK_HDR) &&
281        xdl_emit_hunk_hdr(s1 + 1, e1 - s1, s2 + 1, e2 - s2,
282              func_line.buf, func_line.len, ecb) < 0)
283      return -1;
284
285    /*
286     * Emit pre-context.
287     */
288    for (; s2 < xch->i2; s2++)
289      if (xdl_emit_record(&xe->xdf2, s2, " ", ecb) < 0)
290        return -1;
291
292    for (s1 = xch->i1, s2 = xch->i2;; xch = xch->next) {
293      /*
294       * Merge previous with current change atom.
295       */
296      for (; s1 < xch->i1 && s2 < xch->i2; s1++, s2++)
297        if (xdl_emit_record(&xe->xdf2, s2, " ", ecb) < 0)
298          return -1;
299
300      /*
301       * Removes lines from the first file.
302       */
303      for (s1 = xch->i1; s1 < xch->i1 + xch->chg1; s1++)
304        if (xdl_emit_record(&xe->xdf1, s1, "-", ecb) < 0)
305          return -1;
306
307      /*
308       * Adds lines from the second file.
309       */
310      for (s2 = xch->i2; s2 < xch->i2 + xch->chg2; s2++)
311        if (xdl_emit_record(&xe->xdf2, s2, "+", ecb) < 0)
312          return -1;
313
314      if (xch == xche)
315        break;
316      s1 = xch->i1 + xch->chg1;
317      s2 = xch->i2 + xch->chg2;
318    }
319
320    /*
321     * Emit post-context.
322     */
323    for (s2 = xche->i2 + xche->chg2; s2 < e2; s2++)
324      if (xdl_emit_record(&xe->xdf2, s2, " ", ecb) < 0)
325        return -1;
326  }
327
328  return 0;
329}
