102static int scanA(struct histindex *index, int line1, int count1)
103{
104  unsigned int ptr, tbl_idx;
105  unsigned int chain_len;
106  struct record **rec_chain, *rec;
107
108  for (ptr = LINE_END(1); line1 <= ptr; ptr--) {
109    tbl_idx = TABLE_HASH(index, 1, ptr);
110    rec_chain = index->records + tbl_idx;
111    rec = *rec_chain;
112
113    chain_len = 0;
114    while (rec) {
115      if (CMP(index, 1, rec->ptr, 1, ptr)) {
116        /*
117         * ptr is identical to another element. Insert
118         * it onto the front of the existing element
119         * chain.
120         */
121        NEXT_PTR(index, ptr) = rec->ptr;
122        rec->ptr = ptr;
123        /* cap rec->cnt at MAX_CNT */
124        rec->cnt = XDL_MIN(MAX_CNT, rec->cnt + 1);
125        LINE_MAP(index, ptr) = rec;
126        goto continue_scan;
127      }
128
129      rec = rec->next;
130      chain_len++;
131    }
132
133    if (chain_len == index->max_chain_length)
134      return -1;
135
136    /*
137     * This is the first time we have ever seen this particular
138     * element in the sequence. Construct a new chain for it.
139     */
140    if (!(rec = xdl_cha_alloc(&index->rcha)))
141      return -1;
142    rec->ptr = ptr;
143    rec->cnt = 1;
144    rec->next = *rec_chain;
145    *rec_chain = rec;
146    LINE_MAP(index, ptr) = rec;
147
148continue_scan:
149    ; /* no op */
150  }
151
152  return 0;
153}

155static int try_lcs(struct histindex *index, struct region *lcs, int b_ptr,
156  int line1, int count1, int line2, int count2)
157{
158  unsigned int b_next = b_ptr + 1;
159  struct record *rec = index->records[TABLE_HASH(index, 2, b_ptr)];
160  unsigned int as, ae, bs, be, np, rc;
161  int should_break;
162
163  for (; rec; rec = rec->next) {
164    if (rec->cnt > index->cnt) {
165      if (!index->has_common)
166        index->has_common = CMP(index, 1, rec->ptr, 2, b_ptr);
167      continue;
168    }
169
170    as = rec->ptr;
171    if (!CMP(index, 1, as, 2, b_ptr))
172      continue;
173
174    index->has_common = 1;
175    for (;;) {
176      should_break = 0;
177      np = NEXT_PTR(index, as);
178      bs = b_ptr;
179      ae = as;
180      be = bs;
181      rc = rec->cnt;
182
183      while (line1 < as && line2 < bs
184        && CMP(index, 1, as - 1, 2, bs - 1)) {
185        as--;
186        bs--;
187        if (1 < rc)
188          rc = XDL_MIN(rc, CNT(index, as));
189      }
190      while (ae < LINE_END(1) && be < LINE_END(2)
191        && CMP(index, 1, ae + 1, 2, be + 1)) {
192        ae++;
193        be++;
194        if (1 < rc)
195          rc = XDL_MIN(rc, CNT(index, ae));
196      }
197
198      if (b_next <= be)
199        b_next = be + 1;
200      if (lcs->end1 - lcs->begin1 < ae - as || rc < index->cnt) {
201        lcs->begin1 = as;
202        lcs->begin2 = bs;
203        lcs->end1 = ae;
204        lcs->end2 = be;
205        index->cnt = rc;
206      }
207
208      if (np == 0)
209        break;
210
211      while (np <= ae) {
212        np = NEXT_PTR(index, np);
213        if (np == 0) {
214          should_break = 1;
215          break;
216        }
217      }
218
219      if (should_break)
220        break;
221
222      as = np;
223    }
224  }
225  return b_next;
226}

248static int find_lcs(xpparam_t const *xpp, xdfenv_t *env,
249        struct region *lcs,
250        int line1, int count1, int line2, int count2)
251{
252  int b_ptr;
253  int sz, ret = -1;
254  struct histindex index;
255
256  memset(&index, 0, sizeof(index));
257
258  index.env = env;
259  index.xpp = xpp;
260
261  index.records = NULL;
262  index.line_map = NULL;
263  /* in case of early xdl_cha_free() */
264  index.rcha.head = NULL;
265
266  index.table_bits = xdl_hashbits(count1);
267  sz = index.records_size = 1 << index.table_bits;
268  sz *= sizeof(struct record *);
269  if (!(index.records = (struct record **) xdl_malloc(sz)))
270    goto cleanup;
271  memset(index.records, 0, sz);
272
273  sz = index.line_map_size = count1;
274  sz *= sizeof(struct record *);
275  if (!(index.line_map = (struct record **) xdl_malloc(sz)))
276    goto cleanup;
277  memset(index.line_map, 0, sz);
278
279  sz = index.line_map_size;
280  sz *= sizeof(unsigned int);
281  if (!(index.next_ptrs = (unsigned int *) xdl_malloc(sz)))
282    goto cleanup;
283  memset(index.next_ptrs, 0, sz);
284
285  /* lines / 4 + 1 comes from xprepare.c:xdl_prepare_ctx() */
286  if (xdl_cha_init(&index.rcha, sizeof(struct record), count1 / 4 + 1) < 0)
287    goto cleanup;
288
289  index.ptr_shift = line1;
290  index.max_chain_length = 64;
291
292  if (scanA(&index, line1, count1))
293    goto cleanup;
294
295  index.cnt = index.max_chain_length + 1;
296
297  for (b_ptr = line2; b_ptr <= LINE_END(2); )
298    b_ptr = try_lcs(&index, lcs, b_ptr, line1, count1, line2, count2);
299
300  if (index.has_common && index.max_chain_length < index.cnt)
301    ret = 1;
302  else
303    ret = 0;
304
305cleanup:
306  free_index(&index);
307  return ret;
308}

310static int histogram_diff(xpparam_t const *xpp, xdfenv_t *env,
311  int line1, int count1, int line2, int count2)
312{
313  struct region lcs;
314  int lcs_found;
315  int result;
316redo:
317  result = -1;
318
319  if (count1 <= 0 && count2 <= 0)
320    return 0;
321
322  if (LINE_END(1) >= MAX_PTR)
323    return -1;
324
325  if (!count1) {
326    while(count2--)
327      env->xdf2.rchg[line2++ - 1] = 1;
328    return 0;
329  } else if (!count2) {
330    while(count1--)
331      env->xdf1.rchg[line1++ - 1] = 1;
332    return 0;
333  }
334
335  memset(&lcs, 0, sizeof(lcs));
336  lcs_found = find_lcs(xpp, env, &lcs, line1, count1, line2, count2);
337  if (lcs_found < 0)
338    goto out;
339  else if (lcs_found)
340    result = fall_back_to_classic_diff(xpp, env, line1, count1, line2, count2);
341  else {
342    if (lcs.begin1 == 0 && lcs.begin2 == 0) {
343      while (count1--)
344        env->xdf1.rchg[line1++ - 1] = 1;
345      while (count2--)
346        env->xdf2.rchg[line2++ - 1] = 1;
347      result = 0;
348    } else {
349      result = histogram_diff(xpp, env,
350            line1, lcs.begin1 - line1,
351            line2, lcs.begin2 - line2);
352      if (result)
353        goto out;
354      /*
355       * result = histogram_diff(xpp, env,
356       *            lcs.end1 + 1, LINE_END(1) - lcs.end1,
357       *            lcs.end2 + 1, LINE_END(2) - lcs.end2);
358       * but let's optimize tail recursion ourself:
359      */
360      count1 = LINE_END(1) - lcs.end1;
361      line1 = lcs.end1 + 1;
362      count2 = LINE_END(2) - lcs.end2;
363      line2 = lcs.end2 + 1;
364      goto redo;
365    }
366  }
367out:
368  return result;
369}
