25long xdl_bogosqrt(long n) {
26  long i;
27
28  /*
29   * Classical integer square root approximation using shifts.
30   */
31  for (i = 1; n > 0; n >>= 2)
32    i <<= 1;
33
34  return i;
35}

95void *xdl_cha_alloc(chastore_t *cha) {
96  chanode_t *ancur;
97  void *data;
98
99  if (!(ancur = cha->ancur) || ancur->icurr == cha->nsize) {
100    if (!(ancur = (chanode_t *) xdl_malloc(sizeof(chanode_t) + cha->nsize))) {
101
102      return NULL;
103    }
104    ancur->icurr = 0;
105    ancur->next = NULL;
106    if (cha->tail)
107      cha->tail->next = ancur;
108    if (!cha->head)
109      cha->head = ancur;
110    cha->tail = ancur;
111    cha->ancur = ancur;
112  }
113
114  data = (char *) ancur + sizeof(chanode_t) + ancur->icurr;
115  ancur->icurr += cha->isize;
116
117  return data;
118}

120long xdl_guess_lines(mmfile_t *mf, long sample) {
121  long nl = 0, size, tsize = 0;
122  char const *data, *cur, *top;
123
124  if ((cur = data = xdl_mmfile_first(mf, &size))) {
125    for (top = data + size; nl < sample && cur < top; ) {
126      nl++;
127      if (!(cur = memchr(cur, '\n', top - cur)))
128        cur = top;
129      else
130        cur++;
131    }
132    tsize += (long) (cur - data);
133  }
134
135  if (nl && tsize)
136    nl = xdl_mmfile_size(mf) / (tsize / nl);
137
138  return nl + 1;
139}

141int xdl_blankline(const char *line, long size, long flags)
142{
143  long i;
144
145  if (!(flags & XDF_WHITESPACE_FLAGS))
146    return (size <= 1);
147
148  for (i = 0; i < size && XDL_ISSPACE(line[i]); i++)
149    ;
150
151  return (i == size);
152}

154/*
155 * Have we eaten everything on the line, except for an optional
156 * CR at the very end?
157 */
158static int ends_with_optional_cr(const char *l, long s, long i)
159{
160  int complete = s && l[s-1] == '\n';
161
162  if (complete)
163    s--;
164  if (s == i)
165    return 1;
166  /* do not ignore CR at the end of an incomplete line */
167  if (complete && s == i + 1 && l[i] == '\r')
168    return 1;
169  return 0;
170}

172int xdl_recmatch(const char *l1, long s1, const char *l2, long s2, long flags)
173{
174  int i1, i2;
175
176  if (s1 == s2 && !memcmp(l1, l2, s1))
177    return 1;
178  if (!(flags & XDF_WHITESPACE_FLAGS))
179    return 0;
180
181  i1 = 0;
182  i2 = 0;
183
184  /*
185   * -w matches everything that matches with -b, and -b in turn
186   * matches everything that matches with --ignore-space-at-eol,
187   * which in turn matches everything that matches with --ignore-cr-at-eol.
188   *
189   * Each flavor of ignoring needs different logic to skip whitespaces
190   * while we have both sides to compare.
191   */
192  if (flags & XDF_IGNORE_WHITESPACE) {
193    goto skip_ws;
194    while (i1 < s1 && i2 < s2) {
195      if (l1[i1++] != l2[i2++])
196        return 0;
197    skip_ws:
198      while (i1 < s1 && XDL_ISSPACE(l1[i1]))
199        i1++;
200      while (i2 < s2 && XDL_ISSPACE(l2[i2]))
201        i2++;
202    }
203  } else if (flags & XDF_IGNORE_WHITESPACE_CHANGE) {
204    while (i1 < s1 && i2 < s2) {
205      if (XDL_ISSPACE(l1[i1]) && XDL_ISSPACE(l2[i2])) {
206        /* Skip matching spaces and try again */
207        while (i1 < s1 && XDL_ISSPACE(l1[i1]))
208          i1++;
209        while (i2 < s2 && XDL_ISSPACE(l2[i2]))
210          i2++;
211        continue;
212      }
213      if (l1[i1++] != l2[i2++])
214        return 0;
215    }
216  } else if (flags & XDF_IGNORE_WHITESPACE_AT_EOL) {
217    while (i1 < s1 && i2 < s2 && l1[i1] == l2[i2]) {
218      i1++;
219      i2++;
220    }
221  } else if (flags & XDF_IGNORE_CR_AT_EOL) {
222    /* Find the first difference and see how the line ends */
223    while (i1 < s1 && i2 < s2 && l1[i1] == l2[i2]) {
224      i1++;
225      i2++;
226    }
227    return (ends_with_optional_cr(l1, s1, i1) &&
228      ends_with_optional_cr(l2, s2, i2));
229  }
230
231  /*
232   * After running out of one side, the remaining side must have
233   * nothing but whitespace for the lines to match.  Note that
234   * ignore-whitespace-at-eol case may break out of the loop
235   * while there still are characters remaining on both lines.
236   */
237  if (i1 < s1) {
238    while (i1 < s1 && XDL_ISSPACE(l1[i1]))
239      i1++;
240    if (s1 != i1)
241      return 0;
242  }
243  if (i2 < s2) {
244    while (i2 < s2 && XDL_ISSPACE(l2[i2]))
245      i2++;
246    return (s2 == i2);
247  }
248  return 1;
249}

251static unsigned long xdl_hash_record_with_whitespace(char const **data,
252    char const *top, long flags) {
253  unsigned long ha = 5381;
254  char const *ptr = *data;
255  int cr_at_eol_only = (flags & XDF_WHITESPACE_FLAGS) == XDF_IGNORE_CR_AT_EOL;
256
257  for (; ptr < top && *ptr != '\n'; ptr++) {
258    if (cr_at_eol_only) {
259      /* do not ignore CR at the end of an incomplete line */
260      if (*ptr == '\r' &&
261          (ptr + 1 < top && ptr[1] == '\n'))
262        continue;
263    }
264    else if (XDL_ISSPACE(*ptr)) {
265      const char *ptr2 = ptr;
266      int at_eol;
267      while (ptr + 1 < top && XDL_ISSPACE(ptr[1])
268          && ptr[1] != '\n')
269        ptr++;
270      at_eol = (top <= ptr + 1 || ptr[1] == '\n');
271      if (flags & XDF_IGNORE_WHITESPACE)
272        ; /* already handled */
273      else if (flags & XDF_IGNORE_WHITESPACE_CHANGE
274         && !at_eol) {
275        ha += (ha << 5);
276        ha ^= (unsigned long) ' ';
277      }
278      else if (flags & XDF_IGNORE_WHITESPACE_AT_EOL
279         && !at_eol) {
280        while (ptr2 != ptr + 1) {
281          ha += (ha << 5);
282          ha ^= (unsigned long) *ptr2;
283          ptr2++;
284        }
285      }
286      continue;
287    }
288    ha += (ha << 5);
289    ha ^= (unsigned long) *ptr;
290  }
291  *data = ptr < top ? ptr + 1: ptr;
292
293  return ha;
294}

296unsigned long xdl_hash_record(char const **data, char const *top, long flags) {
297  unsigned long ha = 5381;
298  char const *ptr = *data;
299
300  if (flags & XDF_WHITESPACE_FLAGS)
301    return xdl_hash_record_with_whitespace(data, top, flags);
302
303  for (; ptr < top && *ptr != '\n'; ptr++) {
304    ha += (ha << 5);
305    ha ^= (unsigned long) *ptr;
306  }
307  *data = ptr < top ? ptr + 1: ptr;
308
309  return ha;
310}

312unsigned int xdl_hashbits(unsigned int size) {
313  unsigned int val = 1, bits = 0;
314
315  for (; val < size && bits < CHAR_BIT * sizeof(unsigned int); val <<= 1, bits++);
316  return bits ? bits: 1;
317}

320int xdl_num_out(char *out, long val) {
321  char *ptr, *str = out;
322  char buf[32];
323
324  ptr = buf + sizeof(buf) - 1;
325  *ptr = '\0';
326  if (val < 0) {
327    *--ptr = '-';
328    val = -val;
329  }
330  for (; val && ptr > buf; val /= 10)
331    *--ptr = "0123456789"[val % 10];
332  if (*ptr)
333    for (; *ptr; ptr++, str++)
334      *str = *ptr;
335  else
336    *str++ = '0';
337  *str = '\0';
338
339  return str - out;
340}

342static int xdl_format_hunk_hdr(long s1, long c1, long s2, long c2,
343             const char *func, long funclen,
344             xdemitcb_t *ecb) {
345  int nb = 0;
346  mmbuffer_t mb;
347  char buf[128];
348
349  memcpy(buf, "@@ -", 4);
350  nb += 4;
351
352  nb += xdl_num_out(buf + nb, c1 ? s1: s1 - 1);
353
354  if (c1 != 1) {
355    memcpy(buf + nb, ",", 1);
356    nb += 1;
357
358    nb += xdl_num_out(buf + nb, c1);
359  }
360
361  memcpy(buf + nb, " +", 2);
362  nb += 2;
363
364  nb += xdl_num_out(buf + nb, c2 ? s2: s2 - 1);
365
366  if (c2 != 1) {
367    memcpy(buf + nb, ",", 1);
368    nb += 1;
369
370    nb += xdl_num_out(buf + nb, c2);
371  }
372
373  memcpy(buf + nb, " @@", 3);
374  nb += 3;
375  if (func && funclen) {
376    buf[nb++] = ' ';
377    if (funclen > sizeof(buf) - nb - 1)
378      funclen = sizeof(buf) - nb - 1;
379    memcpy(buf + nb, func, funclen);
380    nb += funclen;
381  }
382  buf[nb++] = '\n';
383
384  mb.ptr = buf;
385  mb.size = nb;
386  if (ecb->out_line(ecb->priv, &mb, 1) < 0)
387    return -1;
388  return 0;
389}

404int xdl_fall_back_diff(xdfenv_t *diff_env, xpparam_t const *xpp,
405    int line1, int count1, int line2, int count2)
406{
407  /*
408   * This probably does not work outside Git, since
409   * we have a very simple mmfile structure.
410   *
411   * Note: ideally, we would reuse the prepared environment, but
412   * the libxdiff interface does not (yet) allow for diffing only
413   * ranges of lines instead of the whole files.
414   */
415  mmfile_t subfile1, subfile2;
416  xdfenv_t env;
417
418  subfile1.ptr = (char *)diff_env->xdf1.recs[line1 - 1]->ptr;
419  subfile1.size = diff_env->xdf1.recs[line1 + count1 - 2]->ptr +
420    diff_env->xdf1.recs[line1 + count1 - 2]->size - subfile1.ptr;
421  subfile2.ptr = (char *)diff_env->xdf2.recs[line2 - 1]->ptr;
422  subfile2.size = diff_env->xdf2.recs[line2 + count2 - 2]->ptr +
423    diff_env->xdf2.recs[line2 + count2 - 2]->size - subfile2.ptr;
424  if (xdl_do_diff(&subfile1, &subfile2, xpp, &env) < 0)
425    return -1;
426
427  memcpy(diff_env->xdf1.rchg + line1 - 1, env.xdf1.rchg, count1);
428  memcpy(diff_env->xdf2.rchg + line2 - 1, env.xdf2.rchg, count2);
429
430  xdl_free_env(&env);
431
432  return 0;
433}
