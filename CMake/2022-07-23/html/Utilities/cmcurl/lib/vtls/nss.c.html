307/* the longest cipher name this supports */
308#define MAX_CIPHER_LENGTH 128
309
310static SECStatus set_ciphers(struct Curl_easy *data, PRFileDesc *model,
311                             const char *cipher_list)
312{
313  unsigned int i;
314  const char *cipher;
315
316  /* use accessors to avoid dynamic linking issues after an update of NSS */
317  const PRUint16 num_implemented_ciphers = SSL_GetNumImplementedCiphers();
318  const PRUint16 *implemented_ciphers = SSL_GetImplementedCiphers();
319  if(!implemented_ciphers)
320    return SECFailure;
321
322  /* First disable all ciphers. This uses a different max value in case
323   * NSS adds more ciphers later we don't want them available by
324   * accident
325   */
326  for(i = 0; i < num_implemented_ciphers; i++) {
327    SSL_CipherPrefSet(model, implemented_ciphers[i], PR_FALSE);
328  }
329
330  cipher = cipher_list;
331
332  while(cipher && cipher[0]) {
333    const char *end;
334    char name[MAX_CIPHER_LENGTH + 1];
335    size_t len;
336    bool found = FALSE;
337    while((*cipher) && (ISSPACE(*cipher)))
338      ++cipher;
339
340    end = strpbrk(cipher, ":, ");
341    if(end)
342      len = end - cipher;
343    else
344      len = strlen(cipher);
345
346    if(len > MAX_CIPHER_LENGTH) {
347      failf(data, "Bad cipher list");
348      return SECFailure;
349    }
350    else if(len) {
351      memcpy(name, cipher, len);
352      name[len] = 0;
353
354      for(i = 0; i<NUM_OF_CIPHERS; i++) {
355        if(strcasecompare(name, cipherlist[i].name)) {
356          /* Enable the selected cipher */
357          if(SSL_CipherPrefSet(model, cipherlist[i].num, PR_TRUE) !=
358             SECSuccess) {
359            failf(data, "cipher-suite not supported by NSS: %s", name);
360            return SECFailure;
361          }
362          found = TRUE;
363          break;
364        }
365      }
366    }
367
368    if(!found && len) {
369      failf(data, "Unknown cipher: %s", name);
370      return SECFailure;
371    }
372    if(end)
373      cipher = ++end;
374    else
375      break;
376  }
377
378  return SECSuccess;
379}

382 * Return true if at least one cipher-suite is enabled. Used to determine
383 * if we need to call NSS_SetDomesticPolicy() to enable the default ciphers.

385static bool any_cipher_enabled(void)

471/* Call PK11_CreateGenericObject() with the given obj_class and filename.  If
472 * the call succeeds, append the object handle to the list of objects so that
473 * the object can be destroyed in nss_close(). */
474static CURLcode nss_create_object(struct ssl_connect_data *connssl,
475                                  CK_OBJECT_CLASS obj_class,
476                                  const char *filename, bool cacert)
477{
478  PK11SlotInfo *slot;
479  PK11GenericObject *obj;
480  CK_BBOOL cktrue = CK_TRUE;
481  CK_BBOOL ckfalse = CK_FALSE;
482  CK_ATTRIBUTE attrs[/* max count of attributes */ 4];
483  int attr_cnt = 0;
484  CURLcode result = (cacert)
485    ? CURLE_SSL_CACERT_BADFILE
486    : CURLE_SSL_CERTPROBLEM;
487
488  const int slot_id = (cacert) ? 0 : 1;
489  char *slot_name = aprintf("PEM Token #%d", slot_id);
490  struct ssl_backend_data *backend = connssl->backend;
491
492  DEBUGASSERT(backend);
493
494  if(!slot_name)
495    return CURLE_OUT_OF_MEMORY;
496
497  slot = nss_find_slot_by_name(slot_name);
498  free(slot_name);
499  if(!slot)
500    return result;
501
502  PK11_SETATTRS(attrs, attr_cnt, CKA_CLASS, &obj_class, sizeof(obj_class));
503  PK11_SETATTRS(attrs, attr_cnt, CKA_TOKEN, &cktrue, sizeof(CK_BBOOL));
504  PK11_SETATTRS(attrs, attr_cnt, CKA_LABEL, (unsigned char *)filename,
505                (CK_ULONG)strlen(filename) + 1);
506
507  if(CKO_CERTIFICATE == obj_class) {
508    CK_BBOOL *pval = (cacert) ? (&cktrue) : (&ckfalse);
509    PK11_SETATTRS(attrs, attr_cnt, CKA_TRUST, pval, sizeof(*pval));
510  }
511
512  /* PK11_CreateManagedGenericObject() was introduced in NSS 3.34 because
513   * PK11_DestroyGenericObject() does not release resources allocated by
514   * PK11_CreateGenericObject() early enough.  */
515  obj =
516#ifdef HAVE_PK11_CREATEMANAGEDGENERICOBJECT
517    PK11_CreateManagedGenericObject
518#else
519    PK11_CreateGenericObject
520#endif
521    (slot, attrs, attr_cnt, PR_FALSE);
522
523  PK11_FreeSlot(slot);
524  if(!obj)
525    return result;
526
527  if(insert_wrapped_ptr(&backend->obj_list, obj) != CURLE_OK) {
528    PK11_DestroyGenericObject(obj);
529    return CURLE_OUT_OF_MEMORY;
530  }
531
532  if(!cacert && CKO_CERTIFICATE == obj_class)
533    /* store reference to a client certificate */
534    backend->obj_clicert = obj;
535
536  return CURLE_OK;
537}

539/* Destroy the NSS object whose handle is given by ptr.  This function is
540 * a callback of Curl_llist_alloc() used by Curl_llist_destroy() to destroy
541 * NSS objects in nss_close() */
542static void nss_destroy_object(void *user, void *ptr)

545  PK11GenericObject *obj = (PK11GenericObject *) wrap->ptr;
546  (void) user;
547  PK11_DestroyGenericObject(obj);
548  free(wrap);
549}

561static CURLcode nss_load_cert(struct ssl_connect_data *ssl,
562                              const char *filename, PRBool cacert)
563{
564  CURLcode result = (cacert)
565    ? CURLE_SSL_CACERT_BADFILE
566    : CURLE_SSL_CERTPROBLEM;
567
568  /* libnsspem.so leaks memory if the requested file does not exist.  For more
569   * details, go to <https://bugzilla.redhat.com/734760>. */
570  if(is_file(filename))
571    result = nss_create_object(ssl, CKO_CERTIFICATE, filename, cacert);
572
573  if(!result && !cacert) {
574    /* we have successfully loaded a client certificate */
575    char *nickname = NULL;
576    char *n = strrchr(filename, '/');
577    if(n)
578      n++;
579
580    /* The following undocumented magic helps to avoid a SIGSEGV on call
581     * of PK11_ReadRawAttribute() from SelectClientCert() when using an
582     * immature version of libnsspem.so.  For more details, go to
583     * <https://bugzilla.redhat.com/733685>. */
584    nickname = aprintf("PEM Token #1:%s", n);
585    if(nickname) {
586      CERTCertificate *cert = PK11_FindCertFromNickname(nickname, NULL);
587      if(cert)
588        CERT_DestroyCertificate(cert);
589
590      free(nickname);
591    }
592  }
593
594  return result;
595}

597/* add given CRL to cache if it is not already there */
598static CURLcode nss_cache_crl(SECItem *crl_der)
599{
600  CERTCertDBHandle *db = CERT_GetDefaultCertDB();
601  CERTSignedCrl *crl = SEC_FindCrlByDERCert(db, crl_der, 0);
602  if(crl) {
603    /* CRL already cached */
604    SEC_DestroyCrl(crl);
605    SECITEM_FreeItem(crl_der, PR_TRUE);
606    return CURLE_OK;
607  }
608
609  /* acquire lock before call of CERT_CacheCRL() and accessing nss_crl_list */
610  PR_Lock(nss_crllock);
611
612  if(SECSuccess != CERT_CacheCRL(db, crl_der)) {
613    /* unable to cache CRL */
614    SECITEM_FreeItem(crl_der, PR_TRUE);
615    PR_Unlock(nss_crllock);
616    return CURLE_SSL_CRL_BADFILE;
617  }
618
619  /* store the CRL item so that we can free it in nss_cleanup() */
620  if(insert_wrapped_ptr(&nss_crl_list, crl_der) != CURLE_OK) {
621    if(SECSuccess == CERT_UncacheCRL(db, crl_der))
622      SECITEM_FreeItem(crl_der, PR_TRUE);
623    PR_Unlock(nss_crllock);
624    return CURLE_OUT_OF_MEMORY;
625  }
626
627  /* we need to clear session cache, so that the CRL could take effect */
628  SSL_ClearSessionCache();
629  PR_Unlock(nss_crllock);
630  return CURLE_OK;
631}

633static CURLcode nss_load_crl(const char *crlfilename)
634{
635  PRFileDesc *infile;
636  PRFileInfo  info;
637  SECItem filedata = { 0, NULL, 0 };
638  SECItem *crl_der = NULL;
639  char *body;
640
641  infile = PR_Open(crlfilename, PR_RDONLY, 0);
642  if(!infile)
643    return CURLE_SSL_CRL_BADFILE;
644
645  if(PR_SUCCESS != PR_GetOpenFileInfo(infile, &info))
646    goto fail;
647
648  if(!SECITEM_AllocItem(NULL, &filedata, info.size + /* zero ended */ 1))
649    goto fail;
650
651  if(info.size != PR_Read(infile, filedata.data, info.size))
652    goto fail;
653
654  crl_der = SECITEM_AllocItem(NULL, NULL, 0U);
655  if(!crl_der)
656    goto fail;
657
658  /* place a trailing zero right after the visible data */
659  body = (char *)filedata.data;
660  body[--filedata.len] = '\0';
661
662  body = strstr(body, "-----BEGIN");
663  if(body) {
664    /* assume ASCII */
665    char *trailer;
666    char *begin = PORT_Strchr(body, '\n');
667    if(!begin)
668      begin = PORT_Strchr(body, '\r');
669    if(!begin)
670      goto fail;
671
672    trailer = strstr(++begin, "-----END");
673    if(!trailer)
674      goto fail;
675
676    /* retrieve DER from ASCII */
677    *trailer = '\0';
678    if(ATOB_ConvertAsciiToItem(crl_der, begin))
679      goto fail;
680
681    SECITEM_FreeItem(&filedata, PR_FALSE);
682  }
683  else
684    /* assume DER */
685    *crl_der = filedata;
686
687  PR_Close(infile);
688  return nss_cache_crl(crl_der);
689
690fail:
691  PR_Close(infile);
692  SECITEM_FreeItem(crl_der, PR_TRUE);
693  SECITEM_FreeItem(&filedata, PR_FALSE);
694  return CURLE_SSL_CRL_BADFILE;
695}

697static CURLcode nss_load_key(struct Curl_easy *data, struct connectdata *conn,
698                             int sockindex, char *key_file)
699{
700  PK11SlotInfo *slot, *tmp;
701  SECStatus status;
702  CURLcode result;
703  struct ssl_connect_data *ssl = conn->ssl;
704
705  (void)sockindex; /* unused */
706
707  result = nss_create_object(ssl, CKO_PRIVATE_KEY, key_file, FALSE);
708  if(result) {
709    PR_SetError(SEC_ERROR_BAD_KEY, 0);
710    return result;
711  }
712
713  slot = nss_find_slot_by_name("PEM Token #1");
714  if(!slot)
715    return CURLE_SSL_CERTPROBLEM;
716
717  /* This will force the token to be seen as re-inserted */
718  tmp = SECMOD_WaitForAnyTokenEvent(pem_module, 0, 0);
719  if(tmp)

721  if(!PK11_IsPresent(slot)) {
722    PK11_FreeSlot(slot);
723    return CURLE_SSL_CERTPROBLEM;
724  }
725
726  status = PK11_Authenticate(slot, PR_TRUE, SSL_SET_OPTION(key_passwd));
727  PK11_FreeSlot(slot);
728
729  return (SECSuccess == status) ? CURLE_OK : CURLE_SSL_CERTPROBLEM;
730}

839/**
840 * Inform the application that the handshake is complete.
841 */
842static void HandshakeCallback(PRFileDesc *sock, void *arg)
843{
844  struct Curl_easy *data = (struct Curl_easy *)arg;
845  struct connectdata *conn = data->conn;
846  unsigned int buflenmax = 50;
847  unsigned char buf[50];
848  unsigned int buflen;
849  SSLNextProtoState state;
850
851  if(!conn->bits.tls_enable_npn && !conn->bits.tls_enable_alpn) {
852    return;
853  }
854
855  if(SSL_GetNextProto(sock, &state, buf, &buflen, buflenmax) == SECSuccess) {
856
857    switch(state) {
858#if NSSVERNUM >= 0x031a00 /* 3.26.0 */
859    /* used by NSS internally to implement 0-RTT */
860    case SSL_NEXT_PROTO_EARLY_VALUE:
861      /* fall through! */
862#endif
863    case SSL_NEXT_PROTO_NO_SUPPORT:
864    case SSL_NEXT_PROTO_NO_OVERLAP:
865      infof(data, VTLS_INFOF_NO_ALPN);
866      return;
867#ifdef SSL_ENABLE_ALPN
868    case SSL_NEXT_PROTO_SELECTED:
869      infof(data, VTLS_INFOF_ALPN_ACCEPTED_LEN_1STR, buflen, buf);
870      break;
871#endif
872    case SSL_NEXT_PROTO_NEGOTIATED:
873      infof(data, "NPN, server accepted to use %.*s", buflen, buf);
874      break;
875    }
876
877#ifdef USE_HTTP2
878    if(buflen == ALPN_H2_LENGTH &&
879       !memcmp(ALPN_H2, buf, ALPN_H2_LENGTH)) {
880      conn->negnpn = CURL_HTTP_VERSION_2;
881    }
882    else
883#endif
884    if(buflen == ALPN_HTTP_1_1_LENGTH &&
885       !memcmp(ALPN_HTTP_1_1, buf, ALPN_HTTP_1_1_LENGTH)) {
886      conn->negnpn = CURL_HTTP_VERSION_1_1;
887    }
888
889    /* This callback might get called when PR_Recv() is used within
890     * close_one() during a connection shutdown. At that point there might not
891     * be any "bundle" associated with the connection anymore.
892     */
893    if(conn->bundle)
894      Curl_multiuse_state(data, conn->negnpn == CURL_HTTP_VERSION_2 ?
895                          BUNDLE_MULTIPLEX : BUNDLE_NO_MULTIUSE);
896  }
897}

899#if NSSVERNUM >= 0x030f04 /* 3.15.4 */
900static SECStatus CanFalseStartCallback(PRFileDesc *sock, void *client_data,
901                                       PRBool *canFalseStart)
902{
903  struct Curl_easy *data = (struct Curl_easy *)client_data;
904
905  SSLChannelInfo channelInfo;
906  SSLCipherSuiteInfo cipherInfo;
907
908  SECStatus rv;
909  PRBool negotiatedExtension;
910
911  *canFalseStart = PR_FALSE;
912
913  if(SSL_GetChannelInfo(sock, &channelInfo, sizeof(channelInfo)) != SECSuccess)
914    return SECFailure;
915
916  if(SSL_GetCipherSuiteInfo(channelInfo.cipherSuite, &cipherInfo,
917                            sizeof(cipherInfo)) != SECSuccess)
918    return SECFailure;
919
920  /* Prevent version downgrade attacks from TLS 1.2, and avoid False Start for
921   * TLS 1.3 and later. See https://bugzilla.mozilla.org/show_bug.cgi?id=861310
922   */
923  if(channelInfo.protocolVersion != SSL_LIBRARY_VERSION_TLS_1_2)
924    goto end;
925
926  /* Only allow ECDHE key exchange algorithm.
927   * See https://bugzilla.mozilla.org/show_bug.cgi?id=952863 */
928  if(cipherInfo.keaType != ssl_kea_ecdh)
929    goto end;

1436        (S_ISDIR(st.st_mode))) {

1438    }
1439  }

1441  if(nspr_io_identity == PR_INVALID_IO_LAYER) {

1463
1464  initialized = 1;

1571static void close_one(struct ssl_connect_data *connssl)
1572{
1573  /* before the cleanup, check whether we are using a client certificate */
1574  struct ssl_backend_data *backend = connssl->backend;
1575  bool client_cert = true;
1576
1577  DEBUGASSERT(backend);
1578
1579  client_cert = (backend->client_nickname != NULL)
1580    || (backend->obj_clicert != NULL);
1581
1582  if(backend->handle) {
1583    char buf[32];
1584    /* Maybe the server has already sent a close notify alert.
1585       Read it to avoid an RST on the TCP connection. */
1586    (void)PR_Recv(backend->handle, buf, (int)sizeof(buf), 0,
1587                  PR_INTERVAL_NO_WAIT);
1588  }
1589
1590  free(backend->client_nickname);
1591  backend->client_nickname = NULL;
1592
1593  /* destroy all NSS objects in order to avoid failure of NSS shutdown */
1594  Curl_llist_destroy(&backend->obj_list, NULL);
1595  backend->obj_clicert = NULL;
1596
1597  if(backend->handle) {
1598    if(client_cert)
1599      /* A server might require different authentication based on the
1600       * particular path being requested by the client.  To support this
1601       * scenario, we must ensure that a connection will never reuse the
1602       * authentication data from a previous connection. */
1603      SSL_InvalidateSession(backend->handle);
1604
1605    PR_Close(backend->handle);
1606    backend->handle = NULL;
1607  }
1608}

1684static CURLcode nss_load_ca_certificates(struct Curl_easy *data,
1685                                         struct connectdata *conn,
1686                                         int sockindex)
1687{
1688  const char *cafile = SSL_CONN_CONFIG(CAfile);
1689  const char *capath = SSL_CONN_CONFIG(CApath);
1690  bool use_trust_module;
1691  CURLcode result = CURLE_OK;
1692
1693  /* treat empty string as unset */
1694  if(cafile && !cafile[0])
1695    cafile = NULL;
1696  if(capath && !capath[0])
1697    capath = NULL;
1698
1699  infof(data, " CAfile: %s", cafile ? cafile : "none");
1700  infof(data, " CApath: %s", capath ? capath : "none");
1701
1702  /* load libnssckbi.so if no other trust roots were specified */
1703  use_trust_module = !cafile && !capath;
1704
1705  PR_Lock(nss_trustload_lock);
1706  if(use_trust_module && !trust_module) {
1707    /* libnssckbi.so needed but not yet loaded --> load it! */
1708    result = nss_load_module(&trust_module, trust_library, "trust");
1709    infof(data, "%s %s", (result) ? "failed to load" : "loaded",
1710          trust_library);
1711    if(result == CURLE_FAILED_INIT)
1712      /* If libnssckbi.so is not available (or fails to load), one can still
1713         use CA certificates stored in NSS database.  Ignore the failure. */
1714      result = CURLE_OK;
1715  }
1716  else if(!use_trust_module && trust_module) {
1717    /* libnssckbi.so not needed but already loaded --> unload it! */
1718    infof(data, "unloading %s", trust_library);
1719    nss_unload_module(&trust_module);
1720  }
1721  PR_Unlock(nss_trustload_lock);
1722
1723  if(cafile)
1724    result = nss_load_cert(&conn->ssl[sockindex], cafile, PR_TRUE);
1725
1726  if(result)
1727    return result;
1728
1729  if(capath) {
1730    struct_stat st;
1731    if(stat(capath, &st) == -1)
1732      return CURLE_SSL_CACERT_BADFILE;
1733
1734    if(S_ISDIR(st.st_mode)) {
1735      PRDirEntry *entry;
1736      PRDir *dir = PR_OpenDir(capath);
1737      if(!dir)
1738        return CURLE_SSL_CACERT_BADFILE;
1739
1740      while((entry =
1741             PR_ReadDir(dir, (PRDirFlags)(PR_SKIP_BOTH | PR_SKIP_HIDDEN)))) {
1742        char *fullpath = aprintf("%s/%s", capath, entry->name);
1743        if(!fullpath) {
1744          PR_CloseDir(dir);
1745          return CURLE_OUT_OF_MEMORY;
1746        }
1747
1748        if(CURLE_OK != nss_load_cert(&conn->ssl[sockindex], fullpath, PR_TRUE))
1749          /* This is purposefully tolerant of errors so non-PEM files can
1750           * be in the same directory */
1751          infof(data, "failed to load '%s' from CURLOPT_CAPATH", fullpath);
1752
1753        free(fullpath);
1754      }
1755
1756      PR_CloseDir(dir);
1757    }
1758    else
1759      infof(data, "WARNING: CURLOPT_CAPATH not a directory (%s)", capath);
1760  }
1761
1762  return CURLE_OK;
1763}

1896                                  struct connectdata *conn, int sockindex)

1902  PRBool ssl_cbc_random_iv;
1903  curl_socket_t sockfd = conn->sock[sockindex];
1904  struct ssl_connect_data *connssl = &conn->ssl[sockindex];
1905  struct ssl_backend_data *backend = connssl->backend;
1906  CURLcode result;
1907  bool second_layer = FALSE;
1908  SSLVersionRange sslver_supported;
1909  SSLVersionRange sslver = {
1910    SSL_LIBRARY_VERSION_TLS_1_0,  /* min */
1911#ifdef SSL_LIBRARY_VERSION_TLS_1_3
1912    SSL_LIBRARY_VERSION_TLS_1_3   /* max */
1913#elif defined SSL_LIBRARY_VERSION_TLS_1_2
1914    SSL_LIBRARY_VERSION_TLS_1_2
1915#elif defined SSL_LIBRARY_VERSION_TLS_1_1
1916    SSL_LIBRARY_VERSION_TLS_1_1
1917#else
1918    SSL_LIBRARY_VERSION_TLS_1_0
1919#endif
1920  };
1921  char *snihost = Curl_ssl_snihost(data, SSL_HOST_NAME(), NULL);
1922  if(!snihost) {
1923    failf(data, "Failed to set SNI");
1924    return CURLE_SSL_CONNECT_ERROR;
1925  }
1926
1927  DEBUGASSERT(backend);
1928
1929  backend->data = data;
1930
1931  /* list of all NSS objects we need to destroy in nss_do_close() */
1932  Curl_llist_init(&backend->obj_list, nss_destroy_object);
1933
1934  PR_Lock(nss_initlock);
1935  result = nss_setup(data);
1936  if(result) {
1937    PR_Unlock(nss_initlock);
1938    goto error;
1939  }
1940
1941  PK11_SetPasswordFunc(nss_get_password);
1942
1943  result = nss_load_module(&pem_module, pem_library, "PEM");
1944  PR_Unlock(nss_initlock);
1945  if(result == CURLE_FAILED_INIT)
1946    infof(data, "WARNING: failed to load NSS PEM library %s. Using "
1947                "OpenSSL PEM certificates will not work.", pem_library);
1948  else if(result)
1949    goto error;
1950
1951  result = CURLE_SSL_CONNECT_ERROR;
1952
1953  model = PR_NewTCPSocket();
1954  if(!model)
1955    goto error;
1956  model = SSL_ImportFD(NULL, model);
1957
1958  if(SSL_OptionSet(model, SSL_SECURITY, PR_TRUE) != SECSuccess)
1959    goto error;
1960  if(SSL_OptionSet(model, SSL_HANDSHAKE_AS_SERVER, PR_FALSE) != SECSuccess)
1961    goto error;
1962  if(SSL_OptionSet(model, SSL_HANDSHAKE_AS_CLIENT, PR_TRUE) != SECSuccess)
1963    goto error;
