1783      p[len-1] = '\0';
1784      len--;

1791        p[len-1] == '.') {
1792      p[len-3] = '\0';
1793      len -= 3;
1794    }
1795    if (ll == len)
1796      break;

1799    if (p[0] == '/') {
1800      if (p[1] == '/')
1801        /* Convert '//' --> '/' */
1802        memmove(p, p+1, strlen(p+1) + 1);
1803      else if (p[1] == '.' && p[2] == '/')
1804        /* Convert '/./' --> '/' */
