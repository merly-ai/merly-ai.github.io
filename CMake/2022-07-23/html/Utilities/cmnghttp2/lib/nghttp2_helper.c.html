576int nghttp2_check_authority(const uint8_t *value, size_t len) {
577  const uint8_t *last;
578  for (last = value + len; value != last; ++value) {
579    if (!VALID_AUTHORITY_CHARS[*value]) {
580      return 0;
581    }
582  }
583  return 1;
584}

586uint8_t *nghttp2_cpymem(uint8_t *dest, const void *src, size_t len) {
587  if (len == 0) {

593  return dest + len;
594}
