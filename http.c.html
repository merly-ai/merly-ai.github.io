562static void redact_sensitive_header(struct strbuf *header)
563{
564  const char *sensitive_header;
565
566  if (trace_curl_redact &&
567      (skip_iprefix(header->buf, "Authorization:", &sensitive_header) ||
568       skip_iprefix(header->buf, "Proxy-Authorization:", &sensitive_header))) {
569    /* The first token is the type, which is OK to log */
570    while (isspace(*sensitive_header))
571      sensitive_header++;
572    while (*sensitive_header && !isspace(*sensitive_header))
573      sensitive_header++;
574    /* Everything else is opaque and possibly sensitive */
575    strbuf_setlen(header,  sensitive_header - header->buf);
576    strbuf_addstr(header, " <redacted>");
577  } else if (trace_curl_redact &&
578       skip_iprefix(header->buf, "Cookie:", &sensitive_header)) {
579    struct strbuf redacted_header = STRBUF_INIT;
580    const char *cookie;
581
582    while (isspace(*sensitive_header))
583      sensitive_header++;
584
585    cookie = sensitive_header;
586
587    while (cookie) {
588      char *equals;
589      char *semicolon = strstr(cookie, "; ");
590      if (semicolon)
591        *semicolon = 0;
592      equals = strchrnul(cookie, '=');
593      if (!equals) {
594        /* invalid cookie, just append and continue */
595        strbuf_addstr(&redacted_header, cookie);
596        continue;
597      }
598      strbuf_add(&redacted_header, cookie, equals - cookie);
599      strbuf_addstr(&redacted_header, "=<redacted>");
600      if (semicolon) {
601        /*
602         * There are more cookies. (Or, for some
603         * reason, the input string ends in "; ".)
604         */
605        strbuf_addstr(&redacted_header, "; ");
606        cookie = semicolon + strlen("; ");
607      } else {
608        cookie = NULL;
609      }
610    }
611
612    strbuf_setlen(header, sensitive_header - header->buf);
613    strbuf_addbuf(header, &redacted_header);
614  }
615}

617static void curl_dump_header(const char *text, unsigned char *ptr, size_t size, int hide_sensitive_header)
618{
619  struct strbuf out = STRBUF_INIT;
620  struct strbuf **headers, **header;
621
622  strbuf_addf(&out, "%s, %10.10ld bytes (0x%8.8lx)\n",
623    text, (long)size, (long)size);
624  trace_strbuf(&trace_curl, &out);
625  strbuf_reset(&out);
626  strbuf_add(&out, ptr, size);
627  headers = strbuf_split_max(&out, '\n', 0);
628
629  for (header = headers; *header; header++) {
630    if (hide_sensitive_header)
631      redact_sensitive_header(*header);
632    strbuf_insertstr((*header), 0, text);
633    strbuf_insertstr((*header), strlen(text), ": ");
634    strbuf_rtrim((*header));
635    strbuf_addch((*header), '\n');
636    trace_strbuf(&trace_curl, (*header));
637  }
638  strbuf_list_free(headers);
639  strbuf_release(&out);
640}

642static void curl_dump_data(const char *text, unsigned char *ptr, size_t size)
643{
644  size_t i;
645  struct strbuf out = STRBUF_INIT;
646  unsigned int width = 60;
647
648  strbuf_addf(&out, "%s, %10.10ld bytes (0x%8.8lx)\n",
649    text, (long)size, (long)size);
650  trace_strbuf(&trace_curl, &out);
651
652  for (i = 0; i < size; i += width) {
653    size_t w;
654
655    strbuf_reset(&out);
656    strbuf_addf(&out, "%s: ", text);
657    for (w = 0; (w < width) && (i + w < size); w++) {
658      unsigned char ch = ptr[i + w];
659
660      strbuf_addch(&out,
661               (ch >= 0x20) && (ch < 0x80)
662               ? ch : '.');
663    }
664    strbuf_addch(&out, '\n');
665    trace_strbuf(&trace_curl, &out);
666  }
667  strbuf_release(&out);
668}

670static int curl_trace(CURL *handle, curl_infotype type, char *data, size_t size, void *userp)
671{
672  const char *text;
673  enum { NO_FILTER = 0, DO_FILTER = 1 };
674
675  switch (type) {
676  case CURLINFO_TEXT:
677    trace_printf_key(&trace_curl, "== Info: %s", data);
678    break;
679  case CURLINFO_HEADER_OUT:
680    text = "=> Send header";
681    curl_dump_header(text, (unsigned char *)data, size, DO_FILTER);
682    break;
683  case CURLINFO_DATA_OUT:
684    if (trace_curl_data) {
685      text = "=> Send data";
686      curl_dump_data(text, (unsigned char *)data, size);
687    }
688    break;
689  case CURLINFO_SSL_DATA_OUT:
690    if (trace_curl_data) {
691      text = "=> Send SSL data";
692      curl_dump_data(text, (unsigned char *)data, size);
693    }
694    break;
695  case CURLINFO_HEADER_IN:
696    text = "<= Recv header";
697    curl_dump_header(text, (unsigned char *)data, size, NO_FILTER);
698    break;
699  case CURLINFO_DATA_IN:
700    if (trace_curl_data) {
701      text = "<= Recv data";
702      curl_dump_data(text, (unsigned char *)data, size);
703    }
704    break;
705  case CURLINFO_SSL_DATA_IN:
706    if (trace_curl_data) {
707      text = "<= Recv SSL data";
708      curl_dump_data(text, (unsigned char *)data, size);
709    }
710    break;
711
712  default:    /* we ignore unknown types by default */
713    return 0;
714  }
715  return 0;
716}

774static CURL *get_curl_handle(void)
775{
776  CURL *result = curl_easy_init();
777
778  if (!result)
779    die("curl_easy_init failed");
780
781  if (!curl_ssl_verify) {
782    curl_easy_setopt(result, CURLOPT_SSL_VERIFYPEER, 0);
783    curl_easy_setopt(result, CURLOPT_SSL_VERIFYHOST, 0);
784  } else {
785    /* Verify authenticity of the peer's certificate */
786    curl_easy_setopt(result, CURLOPT_SSL_VERIFYPEER, 1);
787    /* The name in the cert must match whom we tried to connect */
788    curl_easy_setopt(result, CURLOPT_SSL_VERIFYHOST, 2);
789  }
790
791#ifdef GIT_CURL_HAVE_CURL_HTTP_VERSION_2
792    if (curl_http_version) {
793    long opt;
794    if (!get_curl_http_version_opt(curl_http_version, &opt)) {
795      /* Set request use http version */
796      curl_easy_setopt(result, CURLOPT_HTTP_VERSION, opt);
797    }
798    }
799#endif
800
801  curl_easy_setopt(result, CURLOPT_NETRC, CURL_NETRC_OPTIONAL);
802  curl_easy_setopt(result, CURLOPT_HTTPAUTH, CURLAUTH_ANY);
803
804#ifdef CURLGSSAPI_DELEGATION_FLAG
805  if (curl_deleg) {
806    int i;
807    for (i = 0; i < ARRAY_SIZE(curl_deleg_levels); i++) {
808      if (!strcmp(curl_deleg, curl_deleg_levels[i].name)) {
809        curl_easy_setopt(result, CURLOPT_GSSAPI_DELEGATION,
810            curl_deleg_levels[i].curl_deleg_param);
811        break;
812      }
813    }
814    if (i == ARRAY_SIZE(curl_deleg_levels))
815      warning("Unknown delegation method '%s': using default",
816        curl_deleg);
817  }
818#endif
819
820  if (http_ssl_backend && !strcmp("schannel", http_ssl_backend) &&
821      !http_schannel_check_revoke) {
822#ifdef GIT_CURL_HAVE_CURLSSLOPT_NO_REVOKE
823    curl_easy_setopt(result, CURLOPT_SSL_OPTIONS, CURLSSLOPT_NO_REVOKE);
824#else
825    warning(_("CURLSSLOPT_NO_REVOKE not supported with cURL < 7.44.0"));
826#endif
827  }
828
829  if (http_proactive_auth)
830    init_curl_http_auth(result);
831
832  if (getenv("GIT_SSL_VERSION"))
833    ssl_version = getenv("GIT_SSL_VERSION");
834  if (ssl_version && *ssl_version) {
835    int i;
836    for (i = 0; i < ARRAY_SIZE(sslversions); i++) {
837      if (!strcmp(ssl_version, sslversions[i].name)) {
838        curl_easy_setopt(result, CURLOPT_SSLVERSION,
839             sslversions[i].ssl_version);
840        break;
841      }
842    }
843    if (i == ARRAY_SIZE(sslversions))
844      warning("unsupported ssl version %s: using default",
845        ssl_version);
846  }
847
848  if (getenv("GIT_SSL_CIPHER_LIST"))
849    ssl_cipherlist = getenv("GIT_SSL_CIPHER_LIST");
850  if (ssl_cipherlist != NULL && *ssl_cipherlist)
851    curl_easy_setopt(result, CURLOPT_SSL_CIPHER_LIST,
852        ssl_cipherlist);
853
854  if (ssl_cert)
855    curl_easy_setopt(result, CURLOPT_SSLCERT, ssl_cert);
856  if (has_cert_password())
857    curl_easy_setopt(result, CURLOPT_KEYPASSWD, cert_auth.password);
858  if (ssl_key)
859    curl_easy_setopt(result, CURLOPT_SSLKEY, ssl_key);
860  if (ssl_capath)
861    curl_easy_setopt(result, CURLOPT_CAPATH, ssl_capath);
862#ifdef GIT_CURL_HAVE_CURLOPT_PINNEDPUBLICKEY
863  if (ssl_pinnedkey)
864    curl_easy_setopt(result, CURLOPT_PINNEDPUBLICKEY, ssl_pinnedkey);
865#endif
866  if (http_ssl_backend && !strcmp("schannel", http_ssl_backend) &&
867      !http_schannel_use_ssl_cainfo) {
868    curl_easy_setopt(result, CURLOPT_CAINFO, NULL);
869#ifdef GIT_CURL_HAVE_CURLOPT_PROXY_CAINFO
870    curl_easy_setopt(result, CURLOPT_PROXY_CAINFO, NULL);
871#endif
872  } else if (ssl_cainfo != NULL || http_proxy_ssl_ca_info != NULL) {
873    if (ssl_cainfo)
874      curl_easy_setopt(result, CURLOPT_CAINFO, ssl_cainfo);
875#ifdef GIT_CURL_HAVE_CURLOPT_PROXY_CAINFO
876    if (http_proxy_ssl_ca_info)
877      curl_easy_setopt(result, CURLOPT_PROXY_CAINFO, http_proxy_ssl_ca_info);
878#endif
879  }
880
881  if (curl_low_speed_limit > 0 && curl_low_speed_time > 0) {
882    curl_easy_setopt(result, CURLOPT_LOW_SPEED_LIMIT,
883         curl_low_speed_limit);
884    curl_easy_setopt(result, CURLOPT_LOW_SPEED_TIME,
885         curl_low_speed_time);
886  }
887
888  curl_easy_setopt(result, CURLOPT_MAXREDIRS, 20);
889  curl_easy_setopt(result, CURLOPT_POSTREDIR, CURL_REDIR_POST_ALL);
890  curl_easy_setopt(result, CURLOPT_REDIR_PROTOCOLS,
891       get_curl_allowed_protocols(0));
892  curl_easy_setopt(result, CURLOPT_PROTOCOLS,
893       get_curl_allowed_protocols(-1));
894  if (getenv("GIT_CURL_VERBOSE"))
895    http_trace_curl_no_data();
896  setup_curl_trace(result);
897  if (getenv("GIT_TRACE_CURL_NO_DATA"))
898    trace_curl_data = 0;
899  if (!git_env_bool("GIT_TRACE_REDACT", 1))
900    trace_curl_redact = 0;
901
902  curl_easy_setopt(result, CURLOPT_USERAGENT,
903    user_agent ? user_agent : git_user_agent());
904
905  if (curl_ftp_no_epsv)
906    curl_easy_setopt(result, CURLOPT_FTP_USE_EPSV, 0);
907
908  if (curl_ssl_try)
909    curl_easy_setopt(result, CURLOPT_USE_SSL, CURLUSESSL_TRY);
910
911  /*
912   * CURL also examines these variables as a fallback; but we need to query
913   * them here in order to decide whether to prompt for missing password (cf.
914   * init_curl_proxy_auth()).
915   *
916   * Unlike many other common environment variables, these are historically
917   * lowercase only. It appears that CURL did not know this and implemented
918   * only uppercase variants, which was later corrected to take both - with
919   * the exception of http_proxy, which is lowercase only also in CURL. As
920   * the lowercase versions are the historical quasi-standard, they take
921   * precedence here, as in CURL.
922   */
923  if (!curl_http_proxy) {
924    if (http_auth.protocol && !strcmp(http_auth.protocol, "https")) {
925      var_override(&curl_http_proxy, getenv("HTTPS_PROXY"));
926      var_override(&curl_http_proxy, getenv("https_proxy"));
927    } else {
928      var_override(&curl_http_proxy, getenv("http_proxy"));
929    }
930    if (!curl_http_proxy) {
931      var_override(&curl_http_proxy, getenv("ALL_PROXY"));
932      var_override(&curl_http_proxy, getenv("all_proxy"));
933    }
934  }
935
936  if (curl_http_proxy && curl_http_proxy[0] == '\0') {
937    /*
938     * Handle case with the empty http.proxy value here to keep
939     * common code clean.
940     * NB: empty option disables proxying at all.
941     */
942    curl_easy_setopt(result, CURLOPT_PROXY, "");
943  } else if (curl_http_proxy) {
944    if (starts_with(curl_http_proxy, "socks5h"))
945      curl_easy_setopt(result,
946        CURLOPT_PROXYTYPE, CURLPROXY_SOCKS5_HOSTNAME);
947    else if (starts_with(curl_http_proxy, "socks5"))
948      curl_easy_setopt(result,
949        CURLOPT_PROXYTYPE, CURLPROXY_SOCKS5);
950    else if (starts_with(curl_http_proxy, "socks4a"))
951      curl_easy_setopt(result,
952        CURLOPT_PROXYTYPE, CURLPROXY_SOCKS4A);
953    else if (starts_with(curl_http_proxy, "socks"))
954      curl_easy_setopt(result,
955        CURLOPT_PROXYTYPE, CURLPROXY_SOCKS4);
956#ifdef GIT_CURL_HAVE_CURLOPT_PROXY_KEYPASSWD
957    else if (starts_with(curl_http_proxy, "https")) {
958      curl_easy_setopt(result, CURLOPT_PROXYTYPE, CURLPROXY_HTTPS);
959
960      if (http_proxy_ssl_cert)
961        curl_easy_setopt(result, CURLOPT_PROXY_SSLCERT, http_proxy_ssl_cert);
962
963      if (http_proxy_ssl_key)
964        curl_easy_setopt(result, CURLOPT_PROXY_SSLKEY, http_proxy_ssl_key);
965
966      if (has_proxy_cert_password())
967        curl_easy_setopt(result, CURLOPT_PROXY_KEYPASSWD, proxy_cert_auth.password);
968    }
969#endif
970    if (strstr(curl_http_proxy, "://"))
971      credential_from_url(&proxy_auth, curl_http_proxy);
972    else {
973      struct strbuf url = STRBUF_INIT;
974      strbuf_addf(&url, "http://%s", curl_http_proxy);
975      credential_from_url(&proxy_auth, url.buf);
976      strbuf_release(&url);
977    }
978
979    if (!proxy_auth.host)
980      die("Invalid proxy URL '%s'", curl_http_proxy);
981
982    curl_easy_setopt(result, CURLOPT_PROXY, proxy_auth.host);
983    var_override(&curl_no_proxy, getenv("NO_PROXY"));
984    var_override(&curl_no_proxy, getenv("no_proxy"));
985    curl_easy_setopt(result, CURLOPT_NOPROXY, curl_no_proxy);
986  }
987  init_curl_proxy_auth(result);
988
989  set_curl_keepalive(result);
990
991  return result;
992}

1340void run_active_slot(struct active_request_slot *slot)
1341{
1342  fd_set readfds;
1343  fd_set writefds;
1344  fd_set excfds;
1345  int max_fd;
1346  struct timeval select_timeout;
1347  int finished = 0;
1348
1349  slot->finished = &finished;
1350  while (!finished) {
1351    step_active_slots();
1352
1353    if (slot->in_use) {
1354      long curl_timeout;
1355      curl_multi_timeout(curlm, &curl_timeout);
1356      if (curl_timeout == 0) {
1357        continue;
1358      } else if (curl_timeout == -1) {
1359        select_timeout.tv_sec  = 0;
1360        select_timeout.tv_usec = 50000;
1361      } else {
1362        select_timeout.tv_sec  =  curl_timeout / 1000;
1363        select_timeout.tv_usec = (curl_timeout % 1000) * 1000;
1364      }
1365
1366      max_fd = -1;
1367      FD_ZERO(&readfds);
1368      FD_ZERO(&writefds);
1369      FD_ZERO(&excfds);
1370      curl_multi_fdset(curlm, &readfds, &writefds, &excfds, &max_fd);
1371
1372      /*
1373       * It can happen that curl_multi_timeout returns a pathologically
1374       * long timeout when curl_multi_fdset returns no file descriptors
1375       * to read.  See commit message for more details.
1376       */
1377      if (max_fd < 0 &&
1378          (select_timeout.tv_sec > 0 ||
1379           select_timeout.tv_usec > 50000)) {
1380        select_timeout.tv_sec  = 0;
1381        select_timeout.tv_usec = 50000;
1382      }
1383
1384      select(max_fd+1, &readfds, &writefds, &excfds, &select_timeout);
1385    }
1386  }
1387
1388  /*
1389   * The value of slot->finished we set before the loop was used
1390   * to set our "finished" variable when our request completed.
1391   *
1392   * 1. The slot may not have been reused for another requst
1393   *    yet, in which case it still has &finished.
1394   *
1395   * 2. The slot may already be in-use to serve another request,
1396   *    which can further be divided into two cases:
1397   *
1398   * (a) If call run_active_slot() hasn't been called for that
1399   *     other request, slot->finished would have been cleared
1400   *     by get_active_slot() and has NULL.
1401   *
1402   * (b) If the request did call run_active_slot(), then the
1403   *     call would have updated slot->finished at the beginning
1404   *     of this function, and with the clearing of the member
1405   *     below, we would find that slot->finished is now NULL.
1406   *
1407   * In all cases, slot->finished has no useful information to
1408   * anybody at this point.  Some compilers warn us for
1409   * attempting to smuggle a pointer that is about to become
1410   * invalid, i.e. &finished.  We clear it here to assure them.
1411   */
1412  slot->finished = NULL;
1413}

1492         char *errorstr, size_t errorlen)
1493{

1496   * FAILONERROR (to keep the server's custom error response), and should
1497   * translate the code into failure here.
1498   *
1499   * Likewise, if we see a redirect (30x code), that means we turned off

1507     * FAILONERROR it is lost, so we can give only the numeric
1508     * status code.
1509     */
1510    xsnprintf(errorstr, errorlen,
1511        "The requested URL returned error: %ld",
1512        http_code);

1601/*
1602 * Check for and extract a content-type parameter. "raw"
1603 * should be positioned at the start of the potential
1604 * parameter, with any whitespace already removed.
1605 *
1606 * "name" is the name of the parameter. The value is appended
1607 * to "out".
1608 */
1609static int extract_param(const char *raw, const char *name,
1610       struct strbuf *out)
1611{
1612  size_t len = strlen(name);
1613
1614  if (strncasecmp(raw, name, len))
1615    return -1;
1616  raw += len;
1617
1618  if (*raw != '=')
1619    return -1;
1620  raw++;
1621
1622  while (*raw && !isspace(*raw) && *raw != ';')
1623    strbuf_addch(out, *raw++);
1624  return 0;
1625}

1627/*
1628 * Extract a normalized version of the content type, with any
1629 * spaces suppressed, all letters lowercased, and no trailing ";"
1630 * or parameters.
1631 *
1632 * Note that we will silently remove even invalid whitespace. For
1633 * example, "text / plain" is specifically forbidden by RFC 2616,
1634 * but "text/plain" is the only reasonable output, and this keeps
1635 * our code simple.
1636 *
1637 * If the "charset" argument is not NULL, store the value of any
1638 * charset parameter there.
1639 *
1640 * Example:
1641 *   "TEXT/PLAIN; charset=utf-8" -> "text/plain", "utf-8"
1642 *   "text / plain" -> "text/plain"
1643 */
1644static void extract_content_type(struct strbuf *raw, struct strbuf *type,
1645         struct strbuf *charset)
1646{
1647  const char *p;
1648
1649  strbuf_reset(type);
1650  strbuf_grow(type, raw->len);
1651  for (p = raw->buf; *p; p++) {
1652    if (isspace(*p))
1653      continue;
1654    if (*p == ';') {
1655      p++;
1656      break;
1657    }
1658    strbuf_addch(type, tolower(*p));
1659  }
1660
1661  if (!charset)
1662    return;
1663
1664  strbuf_reset(charset);
1665  while (*p) {
1666    while (isspace(*p) || *p == ';')
1667      p++;
1668    if (!extract_param(p, "charset", charset))
1669      return;
1670    while (*p && !isspace(*p))
1671      p++;
1672  }
1673
1674  if (!charset->len && starts_with(type->buf, "text/"))
1675    strbuf_addstr(charset, "ISO-8859-1");
1676}

1678static void write_accept_language(struct strbuf *buf)
1679{
1680  /*
1681   * MAX_DECIMAL_PLACES must not be larger than 3. If it is larger than
1682   * that, q-value will be smaller than 0.001, the minimum q-value the
1683   * HTTP specification allows. See
1684   * http://tools.ietf.org/html/rfc7231#section-5.3.1 for q-value.
1685   */
1686  const int MAX_DECIMAL_PLACES = 3;
1687  const int MAX_LANGUAGE_TAGS = 1000;
1688  const int MAX_ACCEPT_LANGUAGE_HEADER_SIZE = 4000;
1689  char **language_tags = NULL;
1690  int num_langs = 0;
1691  const char *s = get_preferred_languages();
1692  int i;
1693  struct strbuf tag = STRBUF_INIT;
1694
1695  /* Don't add Accept-Language header if no language is preferred. */
1696  if (!s)
1697    return;
1698
1699  /*
1700   * Split the colon-separated string of preferred languages into
1701   * language_tags array.
1702   */
1703  do {
1704    /* collect language tag */
1705    for (; *s && (isalnum(*s) || *s == '_'); s++)
1706      strbuf_addch(&tag, *s == '_' ? '-' : *s);
1707
1708    /* skip .codeset, @modifier and any other unnecessary parts */
1709    while (*s && *s != ':')
1710      s++;
1711
1712    if (tag.len) {
1713      num_langs++;
1714      REALLOC_ARRAY(language_tags, num_langs);
1715      language_tags[num_langs - 1] = strbuf_detach(&tag, NULL);
1716      if (num_langs >= MAX_LANGUAGE_TAGS - 1) /* -1 for '*' */
1717        break;
1718    }
1719  } while (*s++);
1720
1721  /* write Accept-Language header into buf */
1722  if (num_langs) {
1723    int last_buf_len = 0;
1724    int max_q;
1725    int decimal_places;
1726    char q_format[32];
1727
1728    /* add '*' */
1729    REALLOC_ARRAY(language_tags, num_langs + 1);
1730    language_tags[num_langs++] = "*"; /* it's OK; this won't be freed */
1731
1732    /* compute decimal_places */
1733    for (max_q = 1, decimal_places = 0;
1734         max_q < num_langs && decimal_places <= MAX_DECIMAL_PLACES;
1735         decimal_places++, max_q *= 10)
1736      ;
1737
1738    xsnprintf(q_format, sizeof(q_format), ";q=0.%%0%dd", decimal_places);
1739
1740    strbuf_addstr(buf, "Accept-Language: ");
1741
1742    for (i = 0; i < num_langs; i++) {
1743      if (i > 0)
1744        strbuf_addstr(buf, ", ");
1745
1746      strbuf_addstr(buf, language_tags[i]);
1747
1748      if (i > 0)
1749        strbuf_addf(buf, q_format, max_q - i);
1750
1751      if (buf->len > MAX_ACCEPT_LANGUAGE_HEADER_SIZE) {
1752        strbuf_remove(buf, last_buf_len, buf->len - last_buf_len);
1753        break;
1754      }
1755
1756      last_buf_len = buf->len;
1757    }
1758  }
1759
1760  /* free language tags -- last one is a static '*' */
1761  for (i = 0; i < num_langs - 1; i++)
1762    free(language_tags[i]);
1763  free(language_tags);
1764}

1878/*
1879 * Update the "base" url to a more appropriate value, as deduced by
1880 * redirects seen when requesting a URL starting with "url".
1881 *
1882 * The "asked" parameter is a URL that we asked curl to access, and must begin
1883 * with "base".
1884 *
1885 * The "got" parameter is the URL that curl reported to us as where we ended
1886 * up.
1887 *
1888 * Returns 1 if we updated the base url, 0 otherwise.
1889 *
1890 * Our basic strategy is to compare "base" and "asked" to find the bits
1891 * specific to our request. We then strip those bits off of "got" to yield the
1892 * new base. So for example, if our base is "http://example.com/foo.git",
1893 * and we ask for "http://example.com/foo.git/info/refs", we might end up
1894 * with "https://other.example.com/foo.git/info/refs". We would want the
1895 * new URL to become "https://other.example.com/foo.git".
1896 *
1897 * Note that this assumes a sane redirect scheme. It's entirely possible
1898 * in the example above to end up at a URL that does not even end in
1899 * "info/refs".  In such a case we die. There's not much we can do, such a
1900 * scheme is unlikely to represent a real git repository, and failing to
1901 * rewrite the base opens options for malicious redirects to do funny things.
1902 */
1903static int update_url_from_redirect(struct strbuf *base,
1904            const char *asked,
1905            const struct strbuf *got)
1906{
1907  const char *tail;
1908  size_t new_len;
1909
1910  if (!strcmp(asked, got->buf))
1911    return 0;
1912
1913  if (!skip_prefix(asked, base->buf, &tail))
1914    BUG("update_url_from_redirect: %s is not a superset of %s",
1915        asked, base->buf);
1916
1917  new_len = got->len;
1918  if (!strip_suffix_mem(got->buf, &new_len, tail))
1919    die(_("unable to update url base from redirection:\n"
1920          "  asked for: %s\n"
1921          "   redirect: %s"),
1922        asked, got->buf);
1923
1924  strbuf_reset(base);
1925  strbuf_add(base, got->buf, new_len);
1926
1927  return 1;
1928}

1930static int http_request_reauth(const char *url,
1931             void *result, int target,
1932             struct http_get_options *options)
1933{
1934  int ret = http_request(url, result, target, options);
1935
1936  if (ret != HTTP_OK && ret != HTTP_REAUTH)
1937    return ret;
1938
1939  if (options && options->effective_url && options->base_url) {
1940    if (update_url_from_redirect(options->base_url,
1941               url, options->effective_url)) {
1942      credential_from_url(&http_auth, options->base_url->buf);
1943      url = options->effective_url->buf;
1944    }
1945  }
1946
1947  if (ret != HTTP_REAUTH)
1948    return ret;
1949
1950  /*
1951   * The previous request may have put cruft into our output stream; we
1952   * should clear it out before making our next request.
1953   */
1954  switch (target) {
1955  case HTTP_REQUEST_STRBUF:
1956    strbuf_reset(result);
1957    break;
1958  case HTTP_REQUEST_FILE:
1959    if (fflush(result)) {
1960      error_errno("unable to flush a file");
1961      return HTTP_START_FAILED;
1962    }
1963    rewind(result);
1964    if (ftruncate(fileno(result), 0) < 0) {
1965      error_errno("unable to truncate a file");
1966      return HTTP_START_FAILED;
1967    }
1968    break;
1969  default:
1970    BUG("Unknown http_request target");
1971  }
1972
1973  credential_fill(&http_auth);
1974
1975  return http_request(url, result, target, options);
1976}
