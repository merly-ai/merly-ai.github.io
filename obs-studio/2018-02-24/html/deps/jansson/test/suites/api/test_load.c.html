12static void file_not_found()
13{

17
18    json = json_load_file("/path/to/nonexistent/file.json", 0, &error);
19    if(json)

21    if(error.line != -1)
22        fail("json_load_file returned an invalid line number");
23
24    /* The error message is locale specific, only check the beginning
25       of the error message. */
26
27    pos = strchr(error.text, ':');
28    if(!pos)
29        fail("json_load_file returne an invalid error message");
30
31    *pos = '\0';
32
33    if(strcmp(error.text, "unable to open /path/to/nonexistent/file.json") != 0)
34        fail("json_load_file returned an invalid error message");
35}

146static void allow_nul()
147{
148    const char *text = "\"nul byte \\u0000 in string\"";
149    const char *expected = "nul byte \0 in string";
150    size_t len = 20;
151    json_t *json;
152
153    json = json_loads(text, JSON_ALLOW_NUL | JSON_DECODE_ANY, NULL);
154    if(!json || !json_is_string(json))
155        fail("unable to decode embedded NUL byte");
156
157    if(json_string_length(json) != len)
158        fail("decoder returned wrong string length");
159
160    if(memcmp(json_string_value(json), expected, len + 1))
161        fail("decoder returned wrong string content");
162
163    json_decref(json);
164}
