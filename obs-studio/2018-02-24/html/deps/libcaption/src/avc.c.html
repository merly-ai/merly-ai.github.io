33static size_t _find_emulation_prevention_byte (const uint8_t* data, size_t size)
34{
35    size_t offset = 2;
36
37    while (offset < size) {
38        if (0 == data[offset]) {
39            // 0 0 X 3 //; we know X is zero
40            offset += 1;
41        } else if (3 != data[offset]) {
42            // 0 0 X 0 0 3; we know X is not 0 and not 3
43            offset += 3;
44        } else if (0 != data[offset-1]) {
45            // 0 X 0 0 3
46            offset += 2;
47        } else if (0 != data[offset-2]) {
48            // X 0 0 3
49            offset += 1;
50        } else {
51            // 0 0 3
52            return offset;
53        }
54    }
55
56    return size;
57}

59static size_t _copy_to_rbsp (uint8_t* destData, size_t destSize, const uint8_t* sorcData, size_t sorcSize)
60{
61    size_t toCopy, totlSize = 0;
62
63    for (;;) {
64        if (destSize >= sorcSize) {
65            return 0;
66        }
67
68        // The following line IS correct! We want to look in sorcData up to destSize bytes
69        // We know destSize is smaller than sorcSize because of the previous line
70        toCopy = _find_emulation_prevention_byte (sorcData,destSize);
71        memcpy (destData, sorcData, toCopy);
72        totlSize += toCopy;
73        destData += toCopy;
74        destSize -= toCopy;
75
76        if (0 == destSize) {
77            return totlSize;
78        }
79
80        // skip the emulation prevention byte
81        totlSize += 1;
82        sorcData += toCopy + 1;
83        sorcSize -= toCopy + 1;
84    }
85
86    return 0;

89static inline size_t _find_emulated (uint8_t* data, size_t size)
90{
91    size_t offset = 2;
92
93    while (offset < size) {
94        if (3 < data[offset]) {
95            // 0 0 X; we know X is not 0, 1, 2 or 3
96            offset += 3;
97        } else if (0 != data[offset-1]) {
98            // 0 X 0 0 1
99            offset += 2;
100        } else if (0 != data[offset-2]) {
101            // X 0 0 1
102            offset += 1;
103        } else {
104            // 0 0 0, 0 0 1
105            return offset;
106        }
107    }

112size_t _copy_from_rbsp (uint8_t* data, uint8_t* payloadData, size_t payloadSize)
113{
114    size_t total = 0;
115
116    while (payloadSize) {
117        size_t bytes = _find_emulated (payloadData,payloadSize);
118
119        if (bytes > payloadSize) {
120            return 0;
121        }
122
123        memcpy (data, payloadData, bytes);
124
125        if (bytes == payloadSize) {
126            return total + bytes;
127        }
128
129        data[bytes] = 3; // insert emulation prevention byte
130        data += bytes + 1; total += bytes + 1;
131        payloadData += bytes; payloadSize -= bytes;
132    }
133
134    return total;
135}

463                    sei_encode_eia608 (sei, &cea708, eia608_from_basicna (prev_cc_data,cc_data));
464                } else if (eia608_is_westeu (cc_data)) {
465                    // extended charcters overwrite the previous charcter, so insert a dummy char thren write the extended char
466                    sei_encode_eia608 (sei, &cea708, eia608_from_basicna (prev_cc_data,eia608_from_utf8_1 (EIA608_CHAR_SPACE,DEFAULT_CHANNEL)));
467                    sei_encode_eia608 (sei, &cea708, cc_data);
468                } else {
469                    // previous was basic na, but current isnt; write previous and current
470                    sei_encode_eia608 (sei, &cea708, prev_cc_data);
471                    sei_encode_eia608 (sei, &cea708, cc_data);
472                }
473
474                prev_cc_data = 0; // previous is handled, we can forget it now
475            } else if (eia608_is_westeu (cc_data)) {
476                // extended chars overwrite the previous chars, so insert a dummy char
477                sei_encode_eia608 (sei, &cea708, eia608_from_utf8_1 (EIA608_CHAR_SPACE,DEFAULT_CHANNEL));
478                sei_encode_eia608 (sei, &cea708, cc_data);
479            } else if (eia608_is_basicna (cc_data)) {
480                prev_cc_data = cc_data;
481            } else {
482                sei_encode_eia608 (sei, &cea708, cc_data);
483            }
484
485            if (eia608_is_specialna (cc_data)) {
486                // specialna are treated as controll charcters. Duplicated controll charcters are discarded
487                // So we for a resume after a specialna as a noop to break repetition detection
488                // TODO only do this if the same charcter is repeated
489                sei_encode_eia608 (sei, &cea708, eia608_control_command (eia608_control_resume_caption_loading, DEFAULT_CHANNEL));
490            }
491        }
492
493        if (0 != prev_cc_data) {
494            sei_encode_eia608 (sei, &cea708, prev_cc_data);
495        }
496    }
497
498    sei_encode_eia608 (sei, &cea708, 0); // flush
499    sei->dts = frame->timestamp; // assumes in order frames
500    // sei_dump (sei);
501    return 1;
502}

504static int avc_is_start_code (const uint8_t* data, int size, int* len)
505{
506    if (3 > size) {
507        return -1;
508    }
509
510    if (1 < data[2]) {
511        return 3;
512    }
513
514    if (0 != data[1]) {
515        return 2;
516    }
517
518    if (0 == data[0]) {
519        if (1 == data[2]) {
520            *len = 3;
521            return 0;
522        }
523
524        if (4 <= size && 1 == data[3]) {
525            *len = 4;
526            return 0;
527        }
528    }
529
530    return 1;
531}

534static int avc_find_start_code (const uint8_t* data, int size, int* len)
535{
536    int pos = 0;
537
538    for (;;) {
539        // is pos pointing to a start code?
540        int isc = avc_is_start_code (data + pos, size - pos, len);
541
542        if (0 < isc) {
543            pos += isc;
544        } else if (0 > isc) {
545            // No start code found
546            return isc;
547        } else {
548            // Start code found at pos
549            return pos;
550        }
551    }
552}

555static int avc_find_start_code_increnental (const uint8_t* data, int size, int prev_size, int* len)
556{
557    int offset = (3 <= prev_size) ? (prev_size - 3) : 0;
558    int pos = avc_find_start_code (data + offset, size - offset, len);
559
560    if (0 <= pos) {
561        return pos + offset;
562    }
563
564    return pos;
565}

572int avcnalu_parse_annexb (avcnalu_t* nalu, const uint8_t** data, size_t* size)
573{
574    int scpos, sclen;
575    int new_size = (int) (nalu->size + (*size));
576
577    if (new_size > MAX_NALU_SIZE) {
578        (*size) = nalu->size = 0;
579        return LIBCAPTION_ERROR;
580    }
581
582    memcpy (&nalu->data[nalu->size], (*data), (*size));
583    scpos = avc_find_start_code_increnental (&nalu->data[0], new_size, (int) nalu->size, &sclen);
584
585    if (0<=scpos) {
586        (*data) += (scpos - nalu->size) + sclen;
587        (*size) -= (scpos - nalu->size) + sclen;
588        nalu->size = scpos;
589        return 0 < nalu->size ? LIBCAPTION_READY : LIBCAPTION_OK;
590    } else {
591        (*size) = 0;
592        nalu->size = new_size;
593        return LIBCAPTION_OK;
594    }
