37void _celt_lpc(
38      opus_val16       *_lpc, /* out: [0...p-1] LPC coefficients      */
39const opus_val32 *ac,  /* in:  [0...p] autocorrelation values  */
40int          p
41)
42{
43   int i, j;
44   opus_val32 r;
45   opus_val32 error = ac[0];
46#ifdef FIXED_POINT
47   opus_val32 lpc[LPC_ORDER];
48#else
49   float *lpc = _lpc;
50#endif
51
52   RNN_CLEAR(lpc, p);
53   if (ac[0] != 0)
54   {
55      for (i = 0; i < p; i++) {
56         /* Sum up this iteration's reflection coefficient */
57         opus_val32 rr = 0;
58         for (j = 0; j < i; j++)
59            rr += MULT32_32_Q31(lpc[j],ac[i - j]);
60         rr += SHR32(ac[i + 1],3);
61         r = -SHL32(rr,3)/error;
62         /*  Update LPC coefficients and total error */
63         lpc[i] = SHR32(r,3);
64         for (j = 0; 
 Recommendations (Experimental)  R1: (j < ((i + {const}) >> 1))
R2: (j + ((i + 1) >> 1))
R3: (j >= ((i + 1) >> 1))
R4: (j < ((i * 1) >> 1))
R5: (j < ((i & {const}) >> {const}))
 Score  Issue:  1 unfamiliar pattern(s) detected 
Cost: 7.85, Total Score: 1007.85
Class: complex_2
j < (i+1)>>1; j++)
65         {
66            opus_val32 tmp1, tmp2;
67            tmp1 = lpc[j];
68            tmp2 = lpc[i-1-j];
69            lpc[j]     = tmp1 + MULT32_32_Q31(r,tmp2);
70            lpc[i-1-j] = tmp2 + MULT32_32_Q31(r,tmp1);
71         }
72
73         error = error - MULT32_32_Q31(MULT32_32_Q31(r,r),error);
74         /* Bail out once we get 30 dB gain */
75#ifdef FIXED_POINT
76         if (error<SHR32(ac[0],10))
77            break;
78#else
79         if (error<.001f*ac[0])
80            break;
81#endif
82      }
83   }
84#ifdef FIXED_POINT
85   for (i=0;i<p;i++)
86      _lpc[i] = ROUND16(lpc[i],16);
87#endif
88}

91void celt_fir(
92         const opus_val16 *x,
93         const opus_val16 *num,
94         opus_val16 *y,
95         int N,
96         int ord)
97{
98   int i,j;
99   opus_val16 *rnum = malloc(sizeof(opus_val16) * ord);
100   for(i=0;i<ord;i++)
101      rnum[i] = num[ord-i-1];
102   for (i=0;i<N-3;i+=4)
103   {
104      opus_val32 sum[4];
105      sum[0] = SHL32(EXTEND32(x[i  ]), SIG_SHIFT);
106      sum[1] = SHL32(EXTEND32(x[i+1]), SIG_SHIFT);
107      sum[2] = SHL32(EXTEND32(x[i+2]), SIG_SHIFT);
108      sum[3] = SHL32(EXTEND32(x[i+3]), SIG_SHIFT);
109      xcorr_kernel(rnum, x+i-ord, sum, ord);
110      y[i  ] = ROUND16(sum[0], SIG_SHIFT);
111      y[i+1] = ROUND16(sum[1], SIG_SHIFT);
112      y[i+2] = ROUND16(sum[2], SIG_SHIFT);
113      y[i+3] = ROUND16(sum[3], SIG_SHIFT);
114   }
115   for (;i<N;i++)
116   {
117      opus_val32 sum = SHL32(EXTEND32(x[i]), SIG_SHIFT);
118      for (j=0;j<ord;j++)
119         sum = MAC16_16(sum,rnum[j],x[i+j-ord]);
120      y[i] = ROUND16(sum, SIG_SHIFT);
121   }
122   free(rnum);
123}

125void celt_iir(const opus_val32 *_x,
126         const opus_val16 *den,
127         opus_val32 *_y,
128         int N,
129         int ord,
130         opus_val16 *mem)
131{
132#ifdef SMALL_FOOTPRINT
133   int i,j;
134   for (i=0;i<N;i++)
135   {
136      opus_val32 sum = _x[i];
137      for (j=0;j<ord;j++)
138      {
139         sum -= MULT16_16(den[j],mem[j]);
140      }
141      for (j=ord-1;j>=1;j--)
142      {
143         mem[j]=mem[j-1];
144      }
145      mem[0] = SROUND16(sum, SIG_SHIFT);
146      _y[i] = sum;
147   }
148#else
149   int i,j;
150   celt_assert((ord&3)==0);
151   opus_val16 *rden = malloc(sizeof(opus_val16) * ord);
152   opus_val16 *y = malloc(sizeof(opus_val16) * (N + ord));
153   for(i=0;i<ord;i++)
154      rden[i] = den[ord-i-1];
155   for(i=0;i<ord;i++)
156      y[i] = -mem[ord-i-1];
157   for(;i<N+ord;i++)
158      y[i]=0;
159   for (i=0;i<N-3;i+=4)
160   {
161      /* Unroll by 4 as if it were an FIR filter */
162      opus_val32 sum[4];
163      sum[0]=_x[i];
164      sum[1]=_x[i+1];
165      sum[2]=_x[i+2];
166      sum[3]=_x[i+3];
167      xcorr_kernel(rden, y+i, sum, ord);
168
169      /* Patch up the result to compensate for the fact that this is an IIR */
170      y[i+ord  ] = -SROUND16(sum[0],SIG_SHIFT);
171      _y[i  ] = sum[0];
172      sum[1] = MAC16_16(sum[1], y[i+ord  ], den[0]);
173      y[i+ord+1] = -SROUND16(sum[1],SIG_SHIFT);
174      _y[i+1] = sum[1];
175      sum[2] = MAC16_16(sum[2], y[i+ord+1], den[0]);
176      sum[2] = MAC16_16(sum[2], y[i+ord  ], den[1]);
177      y[i+ord+2] = -SROUND16(sum[2],SIG_SHIFT);
178      _y[i+2] = sum[2];
179
180      sum[3] = MAC16_16(sum[3], y[i+ord+2], den[0]);
181      sum[3] = MAC16_16(sum[3], y[i+ord+1], den[1]);
182      sum[3] = MAC16_16(sum[3], y[i+ord  ], den[2]);
183      y[i+ord+3] = -SROUND16(sum[3],SIG_SHIFT);
184      _y[i+3] = sum[3];
185   }
186   for (;i<N;i++)
187   {
188      opus_val32 sum = _x[i];
189      for (j=0;j<ord;j++)
190         sum -= MULT16_16(rden[j],y[i+j]);
191      y[i+ord] = SROUND16(sum,SIG_SHIFT);
192      _y[i] = sum;
193   }
194   for(i=0;i<ord;i++)
195      mem[i] = _y[N-i-1];
196   free(rden);
197   free(y);
198#endif
199}

201int _celt_autocorr(
202                   const opus_val16 *x,   /*  in: [0...n-1] samples x   */
203                   opus_val32       *ac,  /* out: [0...lag-1] ac values */
204                   const opus_val16       *window,
205                   int          overlap,
206                   int          lag,
207                   int          n)
208{
209   opus_val32 d;
210   int i, k;
211   int fastN=n-lag;
212   int shift;
213   const opus_val16 *xptr;
214   opus_val16 *xx = malloc(sizeof(opus_val16) * n);
215   celt_assert(n>0);
216   celt_assert(overlap>=0);
217   if (overlap == 0)
218   {
219      xptr = x;
220   } else {
221      for (i=0;i<n;i++)
222         xx[i] = x[i];
223      for (i=0;i<overlap;i++)
224      {
225         xx[i] = MULT16_16_Q15(x[i],window[i]);
226         xx[n-i-1] = MULT16_16_Q15(x[n-i-1],window[i]);
227      }
228      xptr = xx;
229   }
230   shift=0;
231#ifdef FIXED_POINT
232   {
233      opus_val32 ac0;
234      ac0 = 1+(n<<7);
235      if (n&1) ac0 += SHR32(MULT16_16(xptr[0],xptr[0]),9);
236      for(i=(n&1);i<n;i+=2)
237      {
238         ac0 += SHR32(MULT16_16(xptr[i],xptr[i]),9);
239         ac0 += SHR32(MULT16_16(xptr[i+1],xptr[i+1]),9);
240      }
241
242      shift = celt_ilog2(ac0)-30+10;
243      shift = (shift)/2;
244      if (shift>0)
245      {
246         for(i=0;i<n;i++)
247            xx[i] = PSHR32(xptr[i], shift);
248         xptr = xx;
249      } else
250         shift = 0;
251   }
252#endif
253   celt_pitch_xcorr(xptr, xptr, ac, fastN, lag+1);
254   for (k=0;k<=lag;k++)
255   {
256      for (i = k+fastN, d = 0; i < n; i++)
257         d = MAC16_16(d, xptr[i], xptr[i-k]);
258      ac[k] += d;
259   }
260#ifdef FIXED_POINT
261   shift = 2*shift;
262   if (shift<=0)
263      ac[0] += SHL32((opus_int32)1, -shift);
264   if (ac[0] < 268435456)
265   {
266      int shift2 = 29 - EC_ILOG(ac[0]);
267      for (i=0;i<=lag;i++)
268         ac[i] = SHL32(ac[i], shift2);
269      shift -= shift2;
270   } else if (ac[0] >= 536870912)
271   {
272      int shift2=1;
273      if (ac[0] >= 1073741824)
274         shift2++;
275      for (i=0;i<=lag;i++)
276         ac[i] = SHR32(ac[i], shift2);
277      shift += shift2;
278   }
279#endif
280   free(xx);
281   return shift;
282}
