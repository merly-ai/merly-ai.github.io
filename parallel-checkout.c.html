35void get_parallel_checkout_configs(int *num_workers, int *threshold)
36{
37  char *env_workers = getenv("GIT_TEST_CHECKOUT_WORKERS");
38
39  if (env_workers && *env_workers) {
40    if (strtol_i(env_workers, 10, num_workers)) {
41      die(_("invalid value for '%s': '%s'"),
42          "GIT_TEST_CHECKOUT_WORKERS", env_workers);
43    }
44    if (*num_workers < 1)
45      *num_workers = online_cpus();
46
47    *threshold = 0;
48    return;
49  }
50
51  if (git_config_get_int("checkout.workers", num_workers))
52    *num_workers = DEFAULT_NUM_WORKERS;
53  else if (*num_workers < 1)
54    *num_workers = online_cpus();
55
56  if (git_config_get_int("checkout.thresholdForParallelism", threshold))
57    *threshold = DEFAULT_THRESHOLD_FOR_PARALLELISM;

77static int is_eligible_for_parallel_checkout(const struct cache_entry *ce,
78               const struct conv_attrs *ca)
79{
80  enum conv_attrs_classification c;
81  size_t packed_item_size;
82
83  /*
84   * Symlinks cannot be checked out in parallel as, in case of path
85   * collision, they could racily replace leading directories of other
86   * entries being checked out. Submodules are checked out in child
87   * processes, which have their own parallel checkout queues.
88   */
89  if (!S_ISREG(ce->ce_mode))
90    return 0;
91
92  packed_item_size = sizeof(struct pc_item_fixed_portion) + ce->ce_namelen +
93    (ca->working_tree_encoding ? strlen(ca->working_tree_encoding) : 0);
94
95  /*
96   * The amount of data we send to the workers per checkout item is
97   * typically small (75~300B). So unless we find an insanely huge path
98   * of 64KB, we should never reach the 65KB limit of one pkt-line. If
99   * that does happen, we let the sequential code handle the item.
100   */
101  if (packed_item_size > LARGE_PACKET_DATA_MAX)
102    return 0;
103
104  c = classify_conv_attrs(ca);
105  switch (c) {
106  case CA_CLASS_INCORE:
107    return 1;
108
109  case CA_CLASS_INCORE_FILTER:
110    /*
111     * It would be safe to allow concurrent instances of
112     * single-file smudge filters, like rot13, but we should not
113     * assume that all filters are parallel-process safe. So we
114     * don't allow this.
115     */
116    return 0;
117
118  case CA_CLASS_INCORE_PROCESS:
119    /*
120     * The parallel queue and the delayed queue are not compatible,
121     * so they must be kept completely separated. And we can't tell
122     * if a long-running process will delay its response without
123     * actually asking it to perform the filtering. Therefore, this
124     * type of filter is not allowed in parallel checkout.
125     *
126     * Furthermore, there should only be one instance of the
127     * long-running process filter as we don't know how it is
128     * managing its own concurrency. So, spreading the entries that
129     * requisite such a filter among the parallel workers would
130     * require a lot more inter-process communication. We would
131     * probably have to designate a single process to interact with
132     * the filter and send all the necessary data to it, for each
133     * entry.
134     */
135    return 0;
136
137  case CA_CLASS_STREAMABLE:
138    return 1;
139
140  default:
141    BUG("unsupported conv_attrs classification '%d'", c);
142  }
143}

145int enqueue_checkout(struct cache_entry *ce, struct conv_attrs *ca)
146{
147  struct parallel_checkout_item *pc_item;
148
149  if (parallel_checkout.status != PC_ACCEPTING_ENTRIES ||
150      !is_eligible_for_parallel_checkout(ce, ca))
151    return -1;
152
153  ALLOC_GROW(parallel_checkout.items, parallel_checkout.nr + 1,
154       parallel_checkout.alloc);
155
156  pc_item = &parallel_checkout.items[parallel_checkout.nr];
157  pc_item->ce = ce;
158  memcpy(&pc_item->ca, ca, sizeof(pc_item->ca));
159  pc_item->status = PC_ITEM_PENDING;
160  pc_item->id = parallel_checkout.nr;
161  parallel_checkout.nr++;
162
163  return 0;
164}

180static int handle_results(struct checkout *state)
181{
182  int ret = 0;
183  size_t i;
184  int have_pending = 0;
185
186  /*
187   * We first update the successfully written entries with the collected
188   * stat() data, so that they can be found by mark_colliding_entries(),
189   * in the next loop, when necessary.
190   */
191  for (i = 0; i < parallel_checkout.nr; i++) {
192    struct parallel_checkout_item *pc_item = &parallel_checkout.items[i];
193    if (pc_item->status == PC_ITEM_WRITTEN)
194      update_ce_after_write(state, pc_item->ce, &pc_item->st);
195  }
196
197  for (i = 0; i < parallel_checkout.nr; i++) {
198    struct parallel_checkout_item *pc_item = &parallel_checkout.items[i];
199
200    switch(pc_item->status) {
201    case PC_ITEM_WRITTEN:
202      /* Already handled */
203      break;
204    case PC_ITEM_COLLIDED:
205      /*
206       * The entry could not be checked out due to a path
207       * collision with another entry. Since there can only
208       * be one entry of each colliding group on the disk, we
209       * could skip trying to check out this one and move on.
210       * However, this would leave the unwritten entries with
211       * null stat() fields on the index, which could
212       * potentially slow down subsequent operations that
213       * require refreshing it: git would not be able to
214       * trust st_size and would have to go to the filesystem
215       * to see if the contents match (see ie_modified()).
216       *
217       * Instead, let's pay the overhead only once, now, and
218       * call checkout_entry_ca() again for this file, to
219       * have its stat() data stored in the index. This also
220       * has the benefit of adding this entry and its
221       * colliding pair to the collision report message.
222       * Additionally, this overwriting behavior is consistent
223       * with what the sequential checkout does, so it doesn't
224       * add any extra overhead.
225       */
226      ret |= checkout_entry_ca(pc_item->ce, &pc_item->ca,
227             state, NULL, NULL);
228      advance_progress_meter();
229      break;
230    case PC_ITEM_PENDING:
231      have_pending = 1;
232      /* fall through */
233    case PC_ITEM_FAILED:
234      ret = -1;
235      break;
236    default:
237      BUG("unknown checkout item status in parallel checkout");
238    }
239  }
240
241  if (have_pending)
242    error("parallel checkout finished with pending entries");
243
244  return ret;
245}

256static int write_pc_item_to_fd(struct parallel_checkout_item *pc_item, int fd,
257             const char *path)
258{
259  int ret;
260  struct stream_filter *filter;
261  struct strbuf buf = STRBUF_INIT;
262  char *blob;
263  size_t size;
264  ssize_t wrote;
265
266  /* Sanity check */
267  assert(is_eligible_for_parallel_checkout(pc_item->ce, &pc_item->ca));
268
269  filter = get_stream_filter_ca(&pc_item->ca, &pc_item->ce->oid);
270  if (filter) {
271    if (stream_blob_to_fd(fd, &pc_item->ce->oid, filter, 1)) {
272      /* On error, reset fd to try writing without streaming */
273      if (reset_fd(fd, path))
274        return -1;
275    } else {
276      return 0;
277    }
278  }
279
280  blob = read_blob_entry(pc_item->ce, &size);
281  if (!blob)
282    return error("cannot read object %s '%s'",
283           oid_to_hex(&pc_item->ce->oid), pc_item->ce->name);
284
285  /*
286   * checkout metadata is used to give context for external process
287   * filters. Files requiring such filters are not eligible for parallel
288   * checkout, so pass NULL. Note: if that changes, the metadata must also
289   * be passed from the main process to the workers.
290   */
291  ret = convert_to_working_tree_ca(&pc_item->ca, pc_item->ce->name,
292           blob, size, &buf, NULL);
293
294  if (ret) {
295    size_t newsize;
296    free(blob);
297    blob = strbuf_detach(&buf, &newsize);
298    size = newsize;
299  }
300
301  wrote = write_in_full(fd, blob, size);
302  free(blob);
303  if (wrote < 0)
304    return error("unable to write file '%s'", path);
305
306  return 0;
307}

321void write_pc_item(struct parallel_checkout_item *pc_item,
322       struct checkout *state)
323{
324  unsigned int mode = (pc_item->ce->ce_mode & 0100) ? 0777 : 0666;
325  int fd = -1, fstat_done = 0;
326  struct strbuf path = STRBUF_INIT;
327  const char *dir_sep;
328
329  strbuf_add(&path, state->base_dir, state->base_dir_len);
330  strbuf_add(&path, pc_item->ce->name, pc_item->ce->ce_namelen);
331
332  dir_sep = find_last_dir_sep(path.buf);
333
334  /*
335   * The leading dirs should have been already created by now. But, in
336   * case of path collisions, one of the dirs could have been replaced by
337   * a symlink (checked out after we enqueued this entry for parallel
338   * checkout). Thus, we must check the leading dirs again.
339   */
340  if (dir_sep && !has_dirs_only_path(path.buf, dir_sep - path.buf,
341             state->base_dir_len)) {
342    pc_item->status = PC_ITEM_COLLIDED;
343    trace2_data_string("pcheckout", NULL, "collision/dirname", path.buf);
344    goto out;
345  }
346
347  fd = open(path.buf, O_WRONLY | O_CREAT | O_EXCL, mode);
348
349  if (fd < 0) {
350    if (errno == EEXIST || errno == EISDIR) {
351      /*
352       * Errors which probably represent a path collision.
353       * Suppress the error message and mark the item to be
354       * retried later, sequentially. ENOTDIR and ENOENT are
355       * also interesting, but the above has_dirs_only_path()
356       * call should have already caught these cases.
357       */
358      pc_item->status = PC_ITEM_COLLIDED;
359      trace2_data_string("pcheckout", NULL,
360             "collision/basename", path.buf);
361    } else {
362      error_errno("failed to open file '%s'", path.buf);
363      pc_item->status = PC_ITEM_FAILED;
364    }
365    goto out;
366  }
367
368  if (write_pc_item_to_fd(pc_item, fd, path.buf)) {
369    /* Error was already reported. */
370    pc_item->status = PC_ITEM_FAILED;
371    close_and_clear(&fd);
372    unlink(path.buf);
373    goto out;
374  }
375
376  fstat_done = fstat_checkout_output(fd, state, &pc_item->st);
377
378  if (close_and_clear(&fd)) {
379    error_errno("unable to close file '%s'", path.buf);
380    pc_item->status = PC_ITEM_FAILED;
381    goto out;
382  }
383
384  if (state->refresh_cache && !fstat_done && lstat(path.buf, &pc_item->st) < 0) {
385    error_errno("unable to stat just-written file '%s'",  path.buf);
386    pc_item->status = PC_ITEM_FAILED;
387    goto out;
388  }
389
390  pc_item->status = PC_ITEM_WRITTEN;
391
392out:
393  strbuf_release(&path);
394}

396static void send_one_item(int fd, struct parallel_checkout_item *pc_item)
397{
398  size_t len_data;
399  char *data, *variant;
400  struct pc_item_fixed_portion *fixed_portion;
401  const char *working_tree_encoding = pc_item->ca.working_tree_encoding;
402  size_t name_len = pc_item->ce->ce_namelen;
403  size_t working_tree_encoding_len = working_tree_encoding ?
404             strlen(working_tree_encoding) : 0;
405
406  /*
407   * Any changes in the calculation of the message size must also be made
408   * in is_eligible_for_parallel_checkout().
409   */
410  len_data = sizeof(struct pc_item_fixed_portion) + name_len +
411       working_tree_encoding_len;
412
413  data = xmalloc(len_data);
414
415  fixed_portion = (struct pc_item_fixed_portion *)data;
416  fixed_portion->id = pc_item->id;
417  fixed_portion->ce_mode = pc_item->ce->ce_mode;
418  fixed_portion->crlf_action = pc_item->ca.crlf_action;
419  fixed_portion->ident = pc_item->ca.ident;
420  fixed_portion->name_len = name_len;
421  fixed_portion->working_tree_encoding_len = working_tree_encoding_len;
422  /*
423   * We pad the unused bytes in the hash array because, otherwise,
424   * Valgrind would complain about passing uninitialized bytes to a
425   * write() syscall. The warning doesn't represent any real risk here,
426   * but it could hinder the detection of actual errors.
427   */
428  oidcpy_with_padding(&fixed_portion->oid, &pc_item->ce->oid);
429
430  variant = data + sizeof(*fixed_portion);
431  if (working_tree_encoding_len) {
432    memcpy(variant, working_tree_encoding, working_tree_encoding_len);
433    variant += working_tree_encoding_len;
434  }
435  memcpy(variant, pc_item->ce->name, name_len);
436
437  packet_write(fd, data, len_data);
438
439  free(data);
440}

442static void send_batch(int fd, size_t start, size_t nr)
443{
444  size_t i;
445  sigchain_push(SIGPIPE, SIG_IGN);
446  for (i = 0; i < nr; i++)
447    send_one_item(fd, &parallel_checkout.items[start + i]);
448  packet_flush(fd);
449  sigchain_pop(SIGPIPE);
450}

452static struct pc_worker *setup_workers(struct checkout *state, int num_workers)
453{
454  struct pc_worker *workers;
455  int i, workers_with_one_extra_item;
456  size_t base_batch_size, batch_beginning = 0;
457
458  ALLOC_ARRAY(workers, num_workers);
459
460  for (i = 0; i < num_workers; i++) {
461    struct child_process *cp = &workers[i].cp;
462
463    child_process_init(cp);
464    cp->git_cmd = 1;
465    cp->in = -1;
466    cp->out = -1;
467    cp->clean_on_exit = 1;
468    strvec_push(&cp->args, "checkout--worker");
469    if (state->base_dir_len)
470      strvec_pushf(&cp->args, "--prefix=%s", state->base_dir);
471    if (start_command(cp))
472      die("failed to spawn checkout worker");
473  }
474
475  base_batch_size = parallel_checkout.nr / num_workers;
476  workers_with_one_extra_item = parallel_checkout.nr % num_workers;
477
478  for (i = 0; i < num_workers; i++) {
479    struct pc_worker *worker = &workers[i];
480    size_t batch_size = base_batch_size;
481
482    /* distribute the extra work evenly */
483    if (i < workers_with_one_extra_item)
484      batch_size++;
485
486    send_batch(worker->cp.in, batch_beginning, batch_size);
487    worker->next_item_to_complete = batch_beginning;
488    worker->nr_items_to_complete = batch_size;
489
490    batch_beginning += batch_size;
491  }
492
493  return workers;
494}

496static void finish_workers(struct pc_worker *workers, int num_workers)
497{
498  int i;
499
500  /*
501   * Close pipes before calling finish_command() to let the workers
502   * exit asynchronously and avoid spending extra time on wait().
503   */
504  for (i = 0; i < num_workers; i++) {
505    struct child_process *cp = &workers[i].cp;
506    if (cp->in >= 0)
507      close(cp->in);
508    if (cp->out >= 0)
509      close(cp->out);
510  }
511
512  for (i = 0; i < num_workers; i++) {
513    int rc = finish_command(&workers[i].cp);
514    if (rc > 128) {
515      /*
516       * For a normal non-zero exit, the worker should have
517       * already printed something useful to stderr. But a
518       * death by signal should be mentioned to the user.
519       */
520      error("checkout worker %d died of signal %d", i, rc - 128);
521    }
522  }
523
524  free(workers);
525}

527static inline void assert_pc_item_result_size(int got, int exp)
528{
529  if (got != exp)
530    BUG("wrong result size from checkout worker (got %dB, exp %dB)",
531        got, exp);
532}
533
534static void parse_and_save_result(const char *buffer, int len,
535          struct pc_worker *worker)
536{
537  struct pc_item_result *res;
538  struct parallel_checkout_item *pc_item;
539  struct stat *st = NULL;
540
541  if (len < PC_ITEM_RESULT_BASE_SIZE)
542    BUG("too short result from checkout worker (got %dB, exp >=%dB)",
543        len, (int)PC_ITEM_RESULT_BASE_SIZE);
544
545  res = (struct pc_item_result *)buffer;
546
547  /*
548   * Worker should send either the full result struct on success, or
549   * just the base (i.e. no stat data), otherwise.
550   */
551  if (res->status == PC_ITEM_WRITTEN) {
552    assert_pc_item_result_size(len, (int)sizeof(struct pc_item_result));
553    st = &res->st;
554  } else {
555    assert_pc_item_result_size(len, (int)PC_ITEM_RESULT_BASE_SIZE);
556  }
557
558  if (!worker->nr_items_to_complete)
559    BUG("received result from supposedly finished checkout worker");
560  if (res->id != worker->next_item_to_complete)
561    BUG("unexpected item id from checkout worker (got %"PRIuMAX", exp %"PRIuMAX")",
562        (uintmax_t)res->id, (uintmax_t)worker->next_item_to_complete);
563
564  worker->next_item_to_complete++;
565  worker->nr_items_to_complete--;
566
567  pc_item = &parallel_checkout.items[res->id];
568  pc_item->status = res->status;
569  if (st)
570    pc_item->st = *st;
571
572  if (res->status != PC_ITEM_COLLIDED)
573    advance_progress_meter();
574}
575
576static void gather_results_from_workers(struct pc_worker *workers,
577          int num_workers)
578{
579  int i, active_workers = num_workers;
580  struct pollfd *pfds;
581
582  CALLOC_ARRAY(pfds, num_workers);
583  for (i = 0; i < num_workers; i++) {
584    pfds[i].fd = workers[i].cp.out;
585    pfds[i].events = POLLIN;
586  }
587
588  while (active_workers) {
589    int nr = poll(pfds, num_workers, -1);
590
591    if (nr < 0) {
592      if (errno == EINTR)
593        continue;
594      die_errno("failed to poll checkout workers");
595    }
596
597    for (i = 0; i < num_workers && nr > 0; i++) {
598      struct pc_worker *worker = &workers[i];
599      struct pollfd *pfd = &pfds[i];
600
601      if (!pfd->revents)
602        continue;
603
604      if (pfd->revents & POLLIN) {
605        int len = packet_read(pfd->fd, packet_buffer,
606                  sizeof(packet_buffer), 0);
607
608        if (len < 0) {
609          BUG("packet_read() returned negative value");
610        } else if (!len) {
611          pfd->fd = -1;
612          active_workers--;
613        } else {
614          parse_and_save_result(packet_buffer,
615                    len, worker);
616        }
617      } else if (pfd->revents & POLLHUP) {
618        pfd->fd = -1;
619        active_workers--;
620      } else if (pfd->revents & (POLLNVAL | POLLERR)) {
621        die("error polling from checkout worker");
622      }
623
624      nr--;
625    }
626  }
627
628  free(pfds);
629}
630
631static void write_items_sequentially(struct checkout *state)
632{
633  size_t i;
634
635  for (i = 0; i < parallel_checkout.nr; i++) {
636    struct parallel_checkout_item *pc_item = &parallel_checkout.items[i];
637    write_pc_item(pc_item, state);
638    if (pc_item->status != PC_ITEM_COLLIDED)
639      advance_progress_meter();
640  }
641}
642
643int run_parallel_checkout(struct checkout *state, int num_workers, int threshold,
644        struct progress *progress, unsigned int *progress_cnt)
645{
646  int ret;
647
648  if (parallel_checkout.status != PC_ACCEPTING_ENTRIES)
649    BUG("cannot run parallel checkout: uninitialized or already running");
650
651  parallel_checkout.status = PC_RUNNING;
652  parallel_checkout.progress = progress;
653  parallel_checkout.progress_cnt = progress_cnt;
654
655  if (parallel_checkout.nr < num_workers)
656    num_workers = parallel_checkout.nr;
657
658  if (num_workers <= 1 || parallel_checkout.nr < threshold) {
659    write_items_sequentially(state);
660  } else {
661    struct pc_worker *workers = setup_workers(state, num_workers);
662    gather_results_from_workers(workers, num_workers);
663    finish_workers(workers, num_workers);
664  }
665
666  ret = handle_results(state);
667
668  finish_parallel_checkout();
669  return ret;
670}
