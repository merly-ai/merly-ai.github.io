253static gid_t
254a_gid(char *s)
255{
256  struct group *gr;
257  char *gname;
258  gid_t gid;
259
260  if ((gr = getgrnam(s)) != NULL)
261    gid = gr->gr_gid;
262  else {
263    for (gname = s; *s && isdigit(*s); ++s);
264    if (!*s)
265      gid = atoi(gname);
266    else
267      errx(EX_NOUSER, "unknown group id: %s", gname);
268  }
269  return (gid);
270}

339  mount_tmpfs_parseargs(argc, argv, &args, &mntflags,
