71            resplen = sizeof(s->macaddr.a);

73                             MEMTXATTRS_UNSPECIFIED);
74            break;
75        case 0x00010004: /* Get board serial */
76            qemu_log_mask(LOG_UNIMP,
77                          "bcm2835_property: 0x%08x get board serial NYI\n",
78                          tag);
79            resplen = 8;
80            break;
81        case 0x00010005: /* Get ARM memory */
82            /* base */
83            stl_le_phys(&s->dma_as, value + 12, 0);
84            /* size */

86            resplen = 8;

97             * and we'll just claim we set it to the desired state

261            offset = ldl_le_phys(&s->dma_as, value + 12);

263            n = 0;

267                            s->fbdev->vcram_base + ((offset + n) << 2), color);
268                n++;
269            }
270            stl_le_phys(&s->dma_as, value + 12, 0);
271            resplen = 4;
272            break;
273
274        case 0x00060001: /* Get DMA channels */
275            /* channels 2-5 */
276            stl_le_phys(&s->dma_as, value + 12, 0x003C);
277            resplen = 4;
278            break;
279
280        case 0x00050001: /* Get command line */
281            resplen = 0;
282            break;
283
284        default:
285            qemu_log_mask(LOG_UNIMP,
286                          "bcm2835_property: unhandled tag 0x%08x\n", tag);
287            break;
288        }
289
290        trace_bcm2835_mbox_property(tag, bufsize, resplen);
291        if (tag == 0) {
292            break;
293        }
294
295        stl_le_phys(&s->dma_as, value + 8, (1 << 31) | resplen);

297    }

299    /* Reconfigure framebuffer if required */
300    if (fbconfig_updated) {
301        bcm2835_fb_reconfigure(s->fbdev, &fbconfig);
302    }
303
304    /* Buffer response code */
305    stl_le_phys(&s->dma_as, s->addr + 4, (1 << 31));
306}

308static uint64_t bcm2835_property_read(void *opaque, hwaddr offset,
309                                      unsigned size)
310{
311    BCM2835PropertyState *s = opaque;
312    uint32_t res = 0;
313
314    switch (offset) {
315    case MBOX_AS_DATA:
316        res = MBOX_CHAN_PROPERTY | s->addr;
317        s->pending = false;
318        qemu_set_irq(s->mbox_irq, 0);
319        break;
320
321    case MBOX_AS_PENDING:
322        res = s->pending;
323        break;
324
325    default:
326        qemu_log_mask(LOG_GUEST_ERROR, "%s: Bad offset %"HWADDR_PRIx"\n",
327                      __func__, offset);
328        return 0;
329    }
330
331    return res;
332}

334static void bcm2835_property_write(void *opaque, hwaddr offset,
335                                   uint64_t value, unsigned size)
336{
337    BCM2835PropertyState *s = opaque;
338
339    switch (offset) {
340    case MBOX_AS_DATA:
341        /* bcm2835_mbox should check our pending status before pushing */
342        assert(!s->pending);
343        s->pending = true;
344        bcm2835_property_mbox_push(s, value);
345        qemu_set_irq(s->mbox_irq, 1);
346        break;
347
348    default:
349        qemu_log_mask(LOG_GUEST_ERROR, "%s: Bad offset %"HWADDR_PRIx"\n",
350                      __func__, offset);
351        return;
352    }
353}

392static void bcm2835_property_realize(DeviceState *dev, Error **errp)
393{
394    BCM2835PropertyState *s = BCM2835_PROPERTY(dev);
395    Object *obj;
396
397    obj = object_property_get_link(OBJECT(dev), "fb", &error_abort);
398    s->fbdev = BCM2835_FB(obj);
399
400    obj = object_property_get_link(OBJECT(dev), "dma-mr", &error_abort);
401    s->dma_mr = MEMORY_REGION(obj);
402    address_space_init(&s->dma_as, s->dma_mr, TYPE_BCM2835_PROPERTY "-memory");
403
404    /* TODO: connect to MAC address of USB NIC device, once we emulate it */
405    qemu_macaddr_default_if_unset(&s->macaddr);
406
407    bcm2835_property_reset(dev);
408}
