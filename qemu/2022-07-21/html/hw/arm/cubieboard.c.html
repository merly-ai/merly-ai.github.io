29static void cubieboard_init(MachineState *machine)
30{
31    AwA10State *a10;
32    Error *err = NULL;
33    DriveInfo *di;
34    BlockBackend *blk;
35    BusState *bus;
36    DeviceState *carddev;
37
38    /* BIOS is not supported by this board */
39    if (machine->firmware) {
40        error_report("BIOS not supported for this machine");
41        exit(1);
42    }
43
44    /* This board has fixed size RAM (512MiB or 1GiB) */

machine->ram_size != 512 * MiB &&
46        machine->ram_size != 1 * GiB) {
47        error_report("This machine can only be used with 512MiB or 1GiB RAM");
48        exit(1);
49    }
50
51    /* Only allow Cortex-A8 for this board */
52    if (strcmp(machine->cpu_type, ARM_CPU_TYPE_NAME("cortex-a8")) != 0) {
53        error_report("This board can only be used with cortex-a8 CPU");
54        exit(1);
55    }
56
57    a10 = AW_A10(object_new(TYPE_AW_A10));
58    object_property_add_child(OBJECT(machine), "soc", OBJECT(a10));
59    object_unref(OBJECT(a10));
60
61    if (!object_property_set_int(OBJECT(&a10->emac), "phy-addr", 1, &err)) {
62        error_reportf_err(err, "Couldn't set phy address: ");
63        exit(1);
64    }
65
66    if (!object_property_set_int(OBJECT(&a10->timer), "clk0-freq", 32768,
67                                 &err)) {
68        error_reportf_err(err, "Couldn't set clk0 frequency: ");
69        exit(1);
70    }
71
72    if (!object_property_set_int(OBJECT(&a10->timer), "clk1-freq", 24000000,
73                                 &err)) {
74        error_reportf_err(err, "Couldn't set clk1 frequency: ");
75        exit(1);
76    }
77
78    if (!qdev_realize(DEVICE(a10), NULL, &err)) {
79        error_reportf_err(err, "Couldn't realize Allwinner A10: ");
80        exit(1);
81    }
82
83    /* Retrieve SD bus */
84    di = drive_get(IF_SD, 0, 0);
85    blk = di ? blk_by_legacy_dinfo(di) : NULL;
86    bus = qdev_get_child_bus(DEVICE(a10), "sd-bus");
87
88    /* Plug in SD card */
89    carddev = qdev_new(TYPE_SD_CARD);
90    qdev_prop_set_drive_err(carddev, "drive", blk, &error_fatal);
91    qdev_realize_and_unref(carddev, bus, &error_fatal);
92
93    memory_region_add_subregion(get_system_memory(), AW_A10_SDRAM_BASE,
94                                machine->ram);
95
96    /* TODO create and connect IDE devices for ide_drive_get() */
97
98    cubieboard_binfo.ram_size = machine->ram_size;
99    arm_load_kernel(&a10->cpu, machine, &cubieboard_binfo);
100}

102static void cubieboard_machine_init(MachineClass *mc)
103{
104    mc->desc = "cubietech cubieboard (Cortex-A8)";
105    mc->default_cpu_type = ARM_CPU_TYPE_NAME("cortex-a8");
106    mc->default_ram_size = 1 * GiB;
107    mc->init = cubieboard_init;
108    mc->block_default_type = IF_IDE;
109    mc->units_per_default_bus = 1;
110    mc->ignore_memory_transaction_failures = true;
111    mc->default_ram_id = "cubieboard.ram";
112}
