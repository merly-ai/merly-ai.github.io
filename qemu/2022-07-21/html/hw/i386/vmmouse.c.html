68static void vmmouse_get_data(uint32_t *data)
69{
70    X86CPU *cpu = X86_CPU(current_cpu);
71    CPUX86State *env = &cpu->env;
72
73    data[0] = env->regs[R_EAX]; data[1] = env->regs[R_EBX];
74    data[2] = env->regs[R_ECX]; data[3] = env->regs[R_EDX];
75    data[4] = env->regs[R_ESI]; data[5] = env->regs[R_EDI];
76}

78static void vmmouse_set_data(const uint32_t *data)
79{
80    X86CPU *cpu = X86_CPU(current_cpu);
81    CPUX86State *env = &cpu->env;
82
83    env->regs[R_EAX] = data[0]; env->regs[R_EBX] = data[1];
84    env->regs[R_ECX] = data[2]; env->regs[R_EDX] = data[3];
85    env->regs[R_ESI] = data[4]; env->regs[R_EDI] = data[5];
86}

88static uint32_t vmmouse_get_status(VMMouseState *s)
89{
90    DPRINTF("vmmouse_get_status()\n");
91    return (s->status << 16) | s->nb_queue;
92}

94static void vmmouse_mouse_event(void *opaque, int x, int y, int dz, int buttons_state)
95{
96    VMMouseState *s = opaque;
97    int buttons = 0;
98
99    if (s->nb_queue > (VMMOUSE_QUEUE_SIZE - 4))
100        return;
101
102    DPRINTF("vmmouse_mouse_event(%d, %d, %d, %d)\n",
103            x, y, dz, buttons_state);
104
105    if ((buttons_state & MOUSE_EVENT_LBUTTON))
106        buttons |= 0x20;
107    if ((buttons_state & MOUSE_EVENT_RBUTTON))
108        buttons |= 0x10;
109    if ((buttons_state & MOUSE_EVENT_MBUTTON))
110        buttons |= 0x08;
111
112    if (s->absolute) {
113        x <<= 1;
114        y <<= 1;
115    }
116
117    s->queue[s->nb_queue++] = buttons;
118    s->queue[s->nb_queue++] = x;
119    s->queue[s->nb_queue++] = y;
120    s->queue[s->nb_queue++] = dz;
121
122    /* need to still generate PS2 events to notify driver to
123       read from queue */
124    i8042_isa_mouse_fake_event(s->i8042);
125}

127static void vmmouse_remove_handler(VMMouseState *s)
128{
129    if (s->entry) {
130        qemu_remove_mouse_event_handler(s->entry);
131        s->entry = NULL;
132    }
133}

135static void vmmouse_update_handler(VMMouseState *s, int absolute)
136{
137    if (s->status != 0) {
138        return;
139    }
140    if (s->absolute != absolute) {
141        s->absolute = absolute;
142        vmmouse_remove_handler(s);
143    }
144    if (s->entry == NULL) {
145        s->entry = qemu_add_mouse_event_handler(vmmouse_mouse_event,
146                                                s, s->absolute,
147                                                "vmmouse");
148        qemu_activate_mouse_event_handler(s->entry);
149    }
150}

152static void vmmouse_read_id(VMMouseState *s)
153{
154    DPRINTF("vmmouse_read_id()\n");
155
156    if (s->nb_queue == VMMOUSE_QUEUE_SIZE)
157        return;
158
159    s->queue[s->nb_queue++] = VMMOUSE_VERSION;
160    s->status = 0;
161    vmmouse_update_handler(s, s->absolute);
162}

204static uint32_t vmmouse_ioport_read(void *opaque, uint32_t addr)
205{
206    VMMouseState *s = opaque;
207    uint32_t data[6];
208    uint16_t command;
209
210    vmmouse_get_data(data);
211
212    command = data[2] & 0xFFFF;
213
214    switch (command) {
215    case VMPORT_CMD_VMMOUSE_STATUS:
216        data[0] = vmmouse_get_status(s);
217        break;
218    case VMPORT_CMD_VMMOUSE_COMMAND:
219        switch (data[1]) {
220        case VMMOUSE_DISABLE:
221            vmmouse_disable(s);
222            break;
223        case VMMOUSE_READ_ID:
224            vmmouse_read_id(s);
225            break;
226        case VMMOUSE_REQUEST_RELATIVE:
227            vmmouse_request_relative(s);
228            break;
229        case VMMOUSE_REQUEST_ABSOLUTE:
230            vmmouse_request_absolute(s);
231            break;
232        default:
233            printf("vmmouse: unknown command %x\n", data[1]);
234            break;
235        }
236        break;
237    case VMPORT_CMD_VMMOUSE_DATA:
238        vmmouse_data(s, data, data[1]);
239        break;
240    default:
241        printf("vmmouse: unknown command %x\n", command);
242        break;
243    }
244
245    vmmouse_set_data(data);
246    return data[0];
247}

283static void vmmouse_realizefn(DeviceState *dev, Error **errp)
284{
285    VMMouseState *s = VMMOUSE(dev);
286
287    DPRINTF("vmmouse_init\n");
288
289    if (!s->i8042) {
290        error_setg(errp, "'i8042' link is not set");
291        return;
292    }
293    if (!object_resolve_path_type("", TYPE_VMPORT, NULL)) {
294        error_setg(errp, "vmmouse needs a machine with vmport");
295        return;
296    }
297
298    vmport_register(VMPORT_CMD_VMMOUSE_STATUS, vmmouse_ioport_read, s);
299    vmport_register(VMPORT_CMD_VMMOUSE_COMMAND, vmmouse_ioport_read, s);
300    vmport_register(VMPORT_CMD_VMMOUSE_DATA, vmmouse_ioport_read, s);
301}
