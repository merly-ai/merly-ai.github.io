49 * Built-in low-levels
50 */
51static enum ll_merge_result ll_binary_merge(const struct ll_merge_driver *drv_unused,
52         mmbuffer_t *result,
53         const char *path,
54         mmfile_t *orig, const char *orig_name,
55         mmfile_t *src1, const char *name1,
56         mmfile_t *src2, const char *name2,
57         const struct ll_merge_options *opts,
58         int marker_size)
59{
60  enum ll_merge_result ret;
61  mmfile_t *stolen;
62  assert(opts);
63
64  /*
65   * The tentative merge result is the common ancestor for an
66   * internal merge.  For the final merge, it is "ours" by
67   * default but -Xours/-Xtheirs can tweak the choice.
68   */
69  if (opts->virtual_ancestor) {
70    stolen = orig;
71    ret = LL_MERGE_OK;
72  } else {
73    switch (opts->variant) {
74    default:
75      ret = LL_MERGE_BINARY_CONFLICT;
76      stolen = src1;
77      break;
78    case XDL_MERGE_FAVOR_OURS:
79      ret = LL_MERGE_OK;
80      stolen = src1;
81      break;
82    case XDL_MERGE_FAVOR_THEIRS:
83      ret = LL_MERGE_OK;
84      stolen = src2;
85      break;
86    }
87  }
88
89  result->ptr = stolen->ptr;
90  result->size = stolen->size;
91  stolen->ptr = NULL;
92
93  return ret;
94}

96static enum ll_merge_result ll_xdl_merge(const struct ll_merge_driver *drv_unused,
97      mmbuffer_t *result,
98      const char *path,
99      mmfile_t *orig, const char *orig_name,
100      mmfile_t *src1, const char *name1,
101      mmfile_t *src2, const char *name2,
102      const struct ll_merge_options *opts,
103      int marker_size)
104{
105  enum ll_merge_result ret;
106  xmparam_t xmp;
107  int status;
108  assert(opts);
109
110  if (orig->size > MAX_XDIFF_SIZE ||
111      src1->size > MAX_XDIFF_SIZE ||
112      src2->size > MAX_XDIFF_SIZE ||
113      buffer_is_binary(orig->ptr, orig->size) ||
114      buffer_is_binary(src1->ptr, src1->size) ||
115      buffer_is_binary(src2->ptr, src2->size)) {
116    return ll_binary_merge(drv_unused, result,
117               path,
118               orig, orig_name,
119               src1, name1,
120               src2, name2,
121               opts, marker_size);
122  }
123
124  memset(&xmp, 0, sizeof(xmp));
125  xmp.level = XDL_MERGE_ZEALOUS;
126  xmp.favor = opts->variant;
127  xmp.xpp.flags = opts->xdl_opts;
128  if (git_xmerge_style >= 0)
129    xmp.style = git_xmerge_style;
130  if (marker_size > 0)
131    xmp.marker_size = marker_size;
132  xmp.ancestor = orig_name;
133  xmp.file1 = name1;
134  xmp.file2 = name2;
135  status = xdl_merge(orig, src1, src2, &xmp, result);
136  ret = (status > 0) ? LL_MERGE_CONFLICT : status;
137  return ret;
138}

168static void create_temp(mmfile_t *src, char *path, size_t len)
169{
170  int fd;
171
172  xsnprintf(path, len, ".merge_file_XXXXXX");
173  fd = xmkstemp(path);
174  if (write_in_full(fd, src->ptr, src->size) < 0)
175    die_errno("unable to write temp-file");
176  close(fd);
177}

179/*
180 * User defined low-level merge driver support.
181 */
182static enum ll_merge_result ll_ext_merge(const struct ll_merge_driver *fn,
183      mmbuffer_t *result,
184      const char *path,
185      mmfile_t *orig, const char *orig_name,
186      mmfile_t *src1, const char *name1,
187      mmfile_t *src2, const char *name2,
188      const struct ll_merge_options *opts,
189      int marker_size)
190{
191  char temp[4][50];
192  struct strbuf cmd = STRBUF_INIT;
193  struct strbuf_expand_dict_entry dict[6];
194  struct strbuf path_sq = STRBUF_INIT;
195  const char *args[] = { NULL, NULL };
196  int status, fd, i;
197  struct stat st;
198  enum ll_merge_result ret;
199  assert(opts);
200
201  sq_quote_buf(&path_sq, path);
202  dict[0].placeholder = "O"; dict[0].value = temp[0];
203  dict[1].placeholder = "A"; dict[1].value = temp[1];
204  dict[2].placeholder = "B"; dict[2].value = temp[2];
205  dict[3].placeholder = "L"; dict[3].value = temp[3];
206  dict[4].placeholder = "P"; dict[4].value = path_sq.buf;
207  dict[5].placeholder = NULL; dict[5].value = NULL;
208
209  if (!fn->cmdline)
210    die("custom merge driver %s lacks command line.", fn->name);
211
212  result->ptr = NULL;
213  result->size = 0;
214  create_temp(orig, temp[0], sizeof(temp[0]));
215  create_temp(src1, temp[1], sizeof(temp[1]));
216  create_temp(src2, temp[2], sizeof(temp[2]));
217  xsnprintf(temp[3], sizeof(temp[3]), "%d", marker_size);
218
219  strbuf_expand(&cmd, fn->cmdline, strbuf_expand_dict_cb, &dict);
220
221  args[0] = cmd.buf;
222  status = run_command_v_opt(args, RUN_USING_SHELL);
223  fd = open(temp[1], O_RDONLY);
224  if (fd < 0)
225    goto bad;
226  if (fstat(fd, &st))
227    goto close_bad;
228  result->size = st.st_size;
229  result->ptr = xmallocz(result->size);
230  if (read_in_full(fd, result->ptr, result->size) != result->size) {
231    FREE_AND_NULL(result->ptr);
232    result->size = 0;
233  }
234 close_bad:
235  close(fd);
236 bad:
237  for (i = 0; i < 3; i++)
238    unlink_or_warn(temp[i]);
239  strbuf_release(&cmd);
240  strbuf_release(&path_sq);
241  ret = (status > 0) ? LL_MERGE_CONFLICT : status;
242  return ret;
243}

246 * merge.default and merge.driver configuration items
247 */
248static struct ll_merge_driver *ll_user_merge, **ll_user_merge_tail;
249static const char *default_ll_merge;
250
251static int read_merge_config(const char *var, const char *value, void *cb)
252{
253  struct ll_merge_driver *fn;
254  const char *key, *name;
255  size_t namelen;
256
257  if (!strcmp(var, "merge.default"))
258    return git_config_string(&default_ll_merge, var, value);
259
260  /*
261   * We are not interested in anything but "merge.<name>.variable";
262   * especially, we do not want to look at variables such as
263   * "merge.summary", "merge.tool", and "merge.verbosity".
264   */
265  if (parse_config_key(var, "merge", &name, &namelen, &key) < 0 || !name)
266    return 0;
267
268  /*
269   * Find existing one as we might be processing merge.<name>.var2
270   * after seeing merge.<name>.var1.
271   */
272  for (fn = ll_user_merge; fn; fn = fn->next)
273    if (!strncmp(fn->name, name, namelen) && !fn->name[namelen])
274      break;
275  if (!fn) {
276    CALLOC_ARRAY(fn, 1);
277    fn->name = xmemdupz(name, namelen);
278    fn->fn = ll_ext_merge;
279    *ll_user_merge_tail = fn;
280    ll_user_merge_tail = &(fn->next);
281  }
282
283  if (!strcmp("name", key))
284    return git_config_string(&fn->description, var, value);
285
286  if (!strcmp("driver", key)) {
287    if (!value)
288      return error("%s: lacks value", var);
289    /*
290     * merge.<name>.driver specifies the command line:
291     *
292     *  command-line
293     *
294     * The command-line will be interpolated with the following
295     * tokens and is given to the shell:
296     *
297     *    %O - temporary file name for the merge base.
298     *    %A - temporary file name for our version.
299     *    %B - temporary file name for the other branches' version.
300     *    %L - conflict marker length
301     *    %P - the original path (safely quoted for the shell)
302     *
303     * The external merge driver should write the results in the
304     * file named by %A, and signal that it has done with zero exit
305     * status.
306     */
307    fn->cmdline = xstrdup(value);
308    return 0;
309  }
310
311  if (!strcmp("recursive", key))
312    return git_config_string(&fn->recursive, var, value);
313
314  return 0;
315}

317static void initialize_ll_merge(void)
318{
319  if (ll_user_merge_tail)
320    return;
321  ll_user_merge_tail = &ll_user_merge;
322  git_config(read_merge_config, NULL);
323}

325static const struct ll_merge_driver *find_ll_merge_driver(const char *merge_attr)
326{
327  struct ll_merge_driver *fn;
328  const char *name;
329  int i;
330
331  initialize_ll_merge();
332
333  if (ATTR_TRUE(merge_attr))
334    return &ll_merge_drv[LL_TEXT_MERGE];
335  else if (ATTR_FALSE(merge_attr))
336    return &ll_merge_drv[LL_BINARY_MERGE];
337  else if (ATTR_UNSET(merge_attr)) {
338    if (!default_ll_merge)
339      return &ll_merge_drv[LL_TEXT_MERGE];
340    else
341      name = default_ll_merge;
342  }
343  else
344    name = merge_attr;
345
346  for (fn = ll_user_merge; fn; fn = fn->next)
347    if (!strcmp(fn->name, name))
348      return fn;
349
350  for (i = 0; i < ARRAY_SIZE(ll_merge_drv); i++)
351    if (!strcmp(ll_merge_drv[i].name, name))
352      return &ll_merge_drv[i];
353
354  /* default to the 3-way */
355  return &ll_merge_drv[LL_TEXT_MERGE];
356}

358static void normalize_file(mmfile_t *mm, const char *path, struct index_state *istate)
359{
360  struct strbuf strbuf = STRBUF_INIT;
361  if (renormalize_buffer(istate, path, mm->ptr, mm->size, &strbuf)) {
362    free(mm->ptr);
363    mm->size = strbuf.len;
364    mm->ptr = strbuf_detach(&strbuf, NULL);
365  }
366}

368enum ll_merge_result ll_merge(mmbuffer_t *result_buf,
369       const char *path,
370       mmfile_t *ancestor, const char *ancestor_label,
371       mmfile_t *ours, const char *our_label,
372       mmfile_t *theirs, const char *their_label,
373       struct index_state *istate,
374       const struct ll_merge_options *opts)
375{
376  struct attr_check *check = load_merge_attributes();
377  static const struct ll_merge_options default_opts;
378  const char *ll_driver_name = NULL;
379  int marker_size = DEFAULT_CONFLICT_MARKER_SIZE;
380  const struct ll_merge_driver *driver;
381
382  if (!opts)
383    opts = &default_opts;
384
385  if (opts->renormalize) {
386    normalize_file(ancestor, path, istate);
387    normalize_file(ours, path, istate);
388    normalize_file(theirs, path, istate);
389  }
390
391  git_check_attr(istate, path, check);
392  ll_driver_name = check->items[0].value;
393  if (check->items[1].value) {
394    marker_size = atoi(check->items[1].value);
395    if (marker_size <= 0)
396      marker_size = DEFAULT_CONFLICT_MARKER_SIZE;
397  }
398  driver = find_ll_merge_driver(ll_driver_name);
399
400  if (opts->virtual_ancestor) {
401    if (driver->recursive)
402      driver = find_ll_merge_driver(driver->recursive);
403  }
404  if (opts->extra_marker_size) {
405    marker_size += opts->extra_marker_size;
406  }
407  return driver->fn(driver, result_buf, path, ancestor, ancestor_label,
408        ours, our_label, theirs, their_label,
409        opts, marker_size);
410}

412int ll_merge_marker_size(struct index_state *istate, const char *path)
413{
414  static struct attr_check *check;
415  int marker_size = DEFAULT_CONFLICT_MARKER_SIZE;
416
417  if (!check)
418    check = attr_check_initl("conflict-marker-size", NULL);
419  git_check_attr(istate, path, check);
420  if (check->items[0].value) {
421    marker_size = atoi(check->items[0].value);
422    if (marker_size <= 0)
423      marker_size = DEFAULT_CONFLICT_MARKER_SIZE;
424  }
425  return marker_size;
426}
