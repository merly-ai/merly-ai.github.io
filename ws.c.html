8static struct whitespace_rule {
9  const char *rule_name;
10  unsigned rule_bits;
11  unsigned loosens_error:1,
12    exclude_default:1;
13} whitespace_rule_names[] = {
14  { "trailing-space", WS_TRAILING_SPACE, 0 },
15  { "space-before-tab", WS_SPACE_BEFORE_TAB, 0 },
16  { "indent-with-non-tab", WS_INDENT_WITH_NON_TAB, 0 },
17  { "cr-at-eol", WS_CR_AT_EOL, 1 },
18  { "blank-at-eol", WS_BLANK_AT_EOL, 0 },
19  { "blank-at-eof", WS_BLANK_AT_EOF, 0 },
20  { "tab-in-indent", WS_TAB_IN_INDENT, 0, 1 },
21};

23unsigned parse_whitespace_rule(const char *string)
24{
25  unsigned rule = WS_DEFAULT_RULE;
26
27  while (string) {
28    int i;
29    size_t len;
30    const char *ep;
31    int negated = 0;
32
33    string = string + strspn(string, ", \t\n\r");
34    ep = strchrnul(string, ',');
35    len = ep - string;
36
37    if (*string == '-') {
38      negated = 1;
39      string++;
40      len--;
41    }
42    if (!len)
43      break;
44    for (i = 0; i < ARRAY_SIZE(whitespace_rule_names); i++) {
45      if (strncmp(whitespace_rule_names[i].rule_name,
46            string, len))
47        continue;
48      if (negated)
49        rule &= ~whitespace_rule_names[i].rule_bits;
50      else
51        rule |= whitespace_rule_names[i].rule_bits;
52      break;
53    }
54    if (strncmp(string, "tabwidth=", 9) == 0) {
55      unsigned tabwidth = atoi(string + 9);
56      if (0 < tabwidth && tabwidth < 0100) {
57        rule &= ~WS_TAB_WIDTH_MASK;
58        rule |= tabwidth;
59      }
60      else
61        warning("tabwidth %.*s out of range",
62          (int)(len - 9), string + 9);
63    }
64    string = ep;
65  }
66
67  if (rule & WS_TAB_IN_INDENT && rule & WS_INDENT_WITH_NON_TAB)
68    die("cannot enforce both tab-in-indent and indent-with-non-tab");
69  return rule;
70}

72unsigned whitespace_rule(struct index_state *istate, const char *pathname)
73{
74  static struct attr_check *attr_whitespace_rule;
75  const char *value;
76
77  if (!attr_whitespace_rule)
78    attr_whitespace_rule = attr_check_initl("whitespace", NULL);
79
80  git_check_attr(istate, pathname, attr_whitespace_rule);
81  value = attr_whitespace_rule->items[0].value;
82  if (ATTR_TRUE(value)) {
83    /* true (whitespace) */
84    unsigned all_rule = ws_tab_width(whitespace_rule_cfg);
85    int i;
86    for (i = 0; i < ARRAY_SIZE(whitespace_rule_names); i++)
87      if (!whitespace_rule_names[i].loosens_error &&
88          !whitespace_rule_names[i].exclude_default)
89        all_rule |= whitespace_rule_names[i].rule_bits;
90    return all_rule;
91  } else if (ATTR_FALSE(value)) {
92    /* false (-whitespace) */
93    return ws_tab_width(whitespace_rule_cfg);
94  } else if (ATTR_UNSET(value)) {
95    /* reset to default (!whitespace) */
96    return whitespace_rule_cfg;
97  } else {
98    /* string */
99    return parse_whitespace_rule(value);
100  }
101}

103/* The returned string should be freed by the caller. */
104char *whitespace_error_string(unsigned ws)
105{
106  struct strbuf err = STRBUF_INIT;
107  if ((ws & WS_TRAILING_SPACE) == WS_TRAILING_SPACE)
108    strbuf_addstr(&err, "trailing whitespace");
109  else {
110    if (ws & WS_BLANK_AT_EOL)
111      strbuf_addstr(&err, "trailing whitespace");
112    if (ws & WS_BLANK_AT_EOF) {
113      if (err.len)
114        strbuf_addstr(&err, ", ");
115      strbuf_addstr(&err, "new blank line at EOF");
116    }
117  }
118  if (ws & WS_SPACE_BEFORE_TAB) {
119    if (err.len)
120      strbuf_addstr(&err, ", ");
121    strbuf_addstr(&err, "space before tab in indent");
122  }
123  if (ws & WS_INDENT_WITH_NON_TAB) {
124    if (err.len)
125      strbuf_addstr(&err, ", ");
126    strbuf_addstr(&err, "indent with spaces");
127  }
128  if (ws & WS_TAB_IN_INDENT) {
129    if (err.len)
130      strbuf_addstr(&err, ", ");
131    strbuf_addstr(&err, "tab in indent");
132  }
133  return strbuf_detach(&err, NULL);
134}

136/* If stream is non-NULL, emits the line after checking. */
137static unsigned ws_check_emit_1(const char *line, int len, unsigned ws_rule,
138        FILE *stream, const char *set,
139        const char *reset, const char *ws)
140{
141  unsigned result = 0;
142  int written = 0;
143  int trailing_whitespace = -1;
144  int trailing_newline = 0;
145  int trailing_carriage_return = 0;
146  int i;
147
148  /* Logic is simpler if we temporarily ignore the trailing newline. */
149  if (len > 0 && line[len - 1] == '\n') {
150    trailing_newline = 1;
151    len--;
152  }
153  if ((ws_rule & WS_CR_AT_EOL) &&
154      len > 0 && line[len - 1] == '\r') {
155    trailing_carriage_return = 1;
156    len--;
157  }
158
159  /* Check for trailing whitespace. */
160  if (ws_rule & WS_BLANK_AT_EOL) {
161    for (i = len - 1; i >= 0; i--) {
162      if (isspace(line[i])) {
163        trailing_whitespace = i;
164        result |= WS_BLANK_AT_EOL;
165      }
166      else
167        break;
168    }
169  }
170
171  if (trailing_whitespace == -1)
172    trailing_whitespace = len;
173
174  /* Check indentation */
175  for (i = 0; i < trailing_whitespace; i++) {
176    if (line[i] == ' ')
177      continue;
178    if (line[i] != '\t')
179      break;
180    if ((ws_rule & WS_SPACE_BEFORE_TAB) && written < i) {
181      result |= WS_SPACE_BEFORE_TAB;
182      if (stream) {
183        fputs(ws, stream);
184        fwrite(line + written, i - written, 1, stream);
185        fputs(reset, stream);
186        fwrite(line + i, 1, 1, stream);
187      }
188    } else if (ws_rule & WS_TAB_IN_INDENT) {
189      result |= WS_TAB_IN_INDENT;
190      if (stream) {
191        fwrite(line + written, i - written, 1, stream);
192        fputs(ws, stream);
193        fwrite(line + i, 1, 1, stream);
194        fputs(reset, stream);
195      }
196    } else if (stream) {
197      fwrite(line + written, i - written + 1, 1, stream);
198    }
199    written = i + 1;
200  }
201
202  /* Check for indent using non-tab. */
203  if ((ws_rule & WS_INDENT_WITH_NON_TAB) && i - written >= ws_tab_width(ws_rule)) {
204    result |= WS_INDENT_WITH_NON_TAB;
205    if (stream) {
206      fputs(ws, stream);
207      fwrite(line + written, i - written, 1, stream);
208      fputs(reset, stream);
209    }
210    written = i;
211  }
212
213  if (stream) {
214    /*
215     * Now the rest of the line starts at "written".
216     * The non-highlighted part ends at "trailing_whitespace".
217     */
218
219    /* Emit non-highlighted (middle) segment. */
220    if (trailing_whitespace - written > 0) {
221      fputs(set, stream);
222      fwrite(line + written,
223          trailing_whitespace - written, 1, stream);
224      fputs(reset, stream);
225    }
226
227    /* Highlight errors in trailing whitespace. */
228    if (trailing_whitespace != len) {
229      fputs(ws, stream);
230      fwrite(line + trailing_whitespace,
231          len - trailing_whitespace, 1, stream);
232      fputs(reset, stream);
233    }
234    if (trailing_carriage_return)
235      fputc('\r', stream);
236    if (trailing_newline)
237      fputc('\n', stream);
238  }
239  return result;
240}

242void ws_check_emit(const char *line, int len, unsigned ws_rule,
243       FILE *stream, const char *set,
244       const char *reset, const char *ws)
245{
246  (void)ws_check_emit_1(line, len, ws_rule, stream, set, reset, ws);
247}

249unsigned ws_check(const char *line, int len, unsigned ws_rule)
250{
251  return ws_check_emit_1(line, len, ws_rule, NULL, NULL, NULL, NULL);
252}

254int ws_blank_line(const char *line, int len, unsigned ws_rule)
255{
256  /*
257   * We _might_ want to treat CR differently from other
258   * whitespace characters when ws_rule has WS_CR_AT_EOL, but
259   * for now we just use this stupid definition.
260   */
261  while (len-- > 0) {
262    if (!isspace(*line))
263      return 0;
264    line++;
265  }
266  return 1;
267}

269/* Copy the line onto the end of the strbuf while fixing whitespaces */
270void ws_fix_copy(struct strbuf *dst, const char *src, int len, unsigned ws_rule, int *error_count)
271{
272  /*
273   * len is number of bytes to be copied from src, starting
274   * at src.  Typically src[len-1] is '\n', unless this is
275   * the incomplete last line.
276   */
277  int i;
278  int add_nl_to_tail = 0;
279  int add_cr_to_tail = 0;
280  int fixed = 0;
281  int last_tab_in_indent = -1;
282  int last_space_in_indent = -1;
283  int need_fix_leading_space = 0;
284
285  /*
286   * Strip trailing whitespace
287   */
288  if (ws_rule & WS_BLANK_AT_EOL) {
289    if (0 < len && src[len - 1] == '\n') {
290      add_nl_to_tail = 1;
291      len--;
292      if (0 < len && src[len - 1] == '\r') {
293        add_cr_to_tail = !!(ws_rule & WS_CR_AT_EOL);
294        len--;
295      }
296    }
297    if (0 < len && isspace(src[len - 1])) {
298      while (0 < len && isspace(src[len-1]))
299        len--;
300      fixed = 1;
301    }
302  }
303
304  /*
305   * Check leading whitespaces (indent)
306   */
307  for (i = 0; i < len; i++) {
308    char ch = src[i];
309    if (ch == '\t') {
310      last_tab_in_indent = i;
311      if ((ws_rule & WS_SPACE_BEFORE_TAB) &&
312          0 <= last_space_in_indent)
313          need_fix_leading_space = 1;
314    } else if (ch == ' ') {
315      last_space_in_indent = i;
316      if ((ws_rule & WS_INDENT_WITH_NON_TAB) &&
317          ws_tab_width(ws_rule) <= i - last_tab_in_indent)
318        need_fix_leading_space = 1;
319    } else
320      break;
321  }
322
323  if (need_fix_leading_space) {
324    /* Process indent ourselves */
325    int consecutive_spaces = 0;
326    int last = last_tab_in_indent + 1;
327
328    if (ws_rule & WS_INDENT_WITH_NON_TAB) {
329      /* have "last" point at one past the indent */
330      if (last_tab_in_indent < last_space_in_indent)
331        last = last_space_in_indent + 1;
332      else
333        last = last_tab_in_indent + 1;
334    }
335
336    /*
337     * between src[0..last-1], strip the funny spaces,
338     * updating them to tab as needed.
339     */
340    for (i = 0; i < last; i++) {
341      char ch = src[i];
342      if (ch != ' ') {
343        consecutive_spaces = 0;
344        strbuf_addch(dst, ch);
345      } else {
346        consecutive_spaces++;
347        if (consecutive_spaces == ws_tab_width(ws_rule)) {
348          strbuf_addch(dst, '\t');
349          consecutive_spaces = 0;
350        }
351      }
352    }
353    while (0 < consecutive_spaces--)
354      strbuf_addch(dst, ' ');
355    len -= last;
356    src += last;
357    fixed = 1;
358  } else if ((ws_rule & WS_TAB_IN_INDENT) && last_tab_in_indent >= 0) {
359    /* Expand tabs into spaces */
360    int start = dst->len;
361    int last = last_tab_in_indent + 1;
362    for (i = 0; i < last; i++) {
363      if (src[i] == '\t')
364        do {
365          strbuf_addch(dst, ' ');
366        } while ((dst->len - start) % ws_tab_width(ws_rule));
367      else
368        strbuf_addch(dst, src[i]);
369    }
370    len -= last;
371    src += last;
372    fixed = 1;
373  }
374
375  strbuf_add(dst, src, len);
376  if (add_cr_to_tail)
377    strbuf_addch(dst, '\r');
378  if (add_nl_to_tail)
379    strbuf_addch(dst, '\n');
380  if (fixed && error_count)
381    (*error_count)++;
382}
