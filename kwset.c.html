51/* Balanced tree of edges and labels leaving a given trie node. */
52struct tree
53{
54  struct tree *llink;    /* Left link; MUST be first field. */
55  struct tree *rlink;    /* Right link (to larger labels). */
56  struct trie *trie;    /* Trie node pointed to by this edge. */
57  unsigned char label;    /* Label on this edge. */
58  char balance;      /* Difference in depths of subtrees. */
59};
60
61/* Node of a trie representing a set of reversed keywords. */
62struct trie
63{
64  unsigned int accepting;  /* Word index of accepted word, or zero. */
65  struct tree *links;    /* Tree of edges leaving this node. */
66  struct trie *parent;    /* Parent of this node. */
67  struct trie *next;    /* List of all trie nodes in level order. */
68  struct trie *fail;    /* Aho-Corasick failure function. */
69  int depth;      /* Depth of this node from the root. */
70  int shift;      /* Shift function for search failures. */
71  int maxshift;      /* Max shift of self and descendants. */
72};
73
74/* Structure returned opaquely to the caller, containing everything. */
75struct kwset
76{
77  struct obstack obstack;  /* Obstack for node allocation. */
78  int words;      /* Number of words in the trie. */
79  struct trie *trie;    /* The trie itself. */
80  int mind;      /* Minimum depth of an accepting node. */
81  int maxd;      /* Maximum depth of any node. */
82  unsigned char delta[NCHAR];  /* Delta table for rapid search. */
83  struct trie *next[NCHAR];  /* Table of children of the root. */
84  char *target;      /* Target string if there's only one. */
85  int mind2;      /* Used in Boyer-Moore search for one string. */
86  unsigned char const *trans;  /* Character translation table. */
87};

89/* Allocate and initialize a keyword set object, returning an opaque
90   pointer to it.  Return NULL if memory is not available. */
91kwset_t
92kwsalloc (unsigned char const *trans)
93{
94  struct kwset *kwset;
95
96  kwset = (struct kwset *) xmalloc(sizeof (struct kwset));
97
98  obstack_init(&kwset->obstack);
99  kwset->words = 0;
100  kwset->trie
101    = (struct trie *) obstack_alloc(&kwset->obstack, sizeof (struct trie));
102  if (!kwset->trie)
103    {
104      kwsfree((kwset_t) kwset);
105      return NULL;
106    }
107  kwset->trie->accepting = 0;
108  kwset->trie->links = NULL;
109  kwset->trie->parent = NULL;
110  kwset->trie->next = NULL;
111  kwset->trie->fail = NULL;
112  kwset->trie->depth = 0;
113  kwset->trie->shift = 0;
114  kwset->mind = INT_MAX;
115  kwset->maxd = -1;
116  kwset->target = NULL;
117  kwset->trans = trans;
118
119  return (kwset_t) kwset;
120}

122/* This upper bound is valid for CHAR_BIT >= 4 and
123   exact for CHAR_BIT in { 4..11, 13, 15, 17, 19 }. */
124#define DEPTH_SIZE (CHAR_BIT + CHAR_BIT/2)
125
126/* Add the given string to the contents of the keyword set.  Return NULL
127   for success, an error message otherwise. */
128const char *
129kwsincr (kwset_t kws, char const *text, size_t len)
130{
131  struct kwset *kwset;
132  register struct trie *trie;
133  register unsigned char label;
134  register struct tree *link;
135  register int depth;
136  struct tree *links[DEPTH_SIZE];
137  enum { L, R } dirs[DEPTH_SIZE];
138  struct tree *t, *r, *l, *rl, *lr;
139
140  kwset = (struct kwset *) kws;
141  trie = kwset->trie;
142  text += len;
143
144  /* Descend the trie (built of reversed keywords) character-by-character,
145     installing new nodes when necessary. */
146  while (len--)
147    {
148      label = kwset->trans ? kwset->trans[U(*--text)] : *--text;
149
150      /* Descend the tree of outgoing links for this trie node,
151   looking for the current character and keeping track
152   of the path followed. */
153      link = trie->links;
154      links[0] = (struct tree *) &trie->links;
155      dirs[0] = L;
156      depth = 1;
157
158      while (link && label != link->label)
159  {
160    links[depth] = link;
161    if (label < link->label)
162      dirs[depth++] = L, link = link->llink;
163    else
164      dirs[depth++] = R, link = link->rlink;
165  }
166
167      /* The current character doesn't have an outgoing link at
168   this trie node, so build a new trie node and install
169   a link in the current trie node's tree. */
170      if (!link)
171  {
172    link = (struct tree *) obstack_alloc(&kwset->obstack,
173                 sizeof (struct tree));
174    if (!link)
175      return "memory exhausted";
176    link->llink = NULL;
177    link->rlink = NULL;
178    link->trie = (struct trie *) obstack_alloc(&kwset->obstack,
179                 sizeof (struct trie));
180    if (!link->trie)
181      {
182        obstack_free(&kwset->obstack, link);
183        return "memory exhausted";
184      }
185    link->trie->accepting = 0;
186    link->trie->links = NULL;
187    link->trie->parent = trie;
188    link->trie->next = NULL;
189    link->trie->fail = NULL;
190    link->trie->depth = trie->depth + 1;
191    link->trie->shift = 0;
192    link->label = label;
193    link->balance = 0;
194
195    /* Install the new tree node in its parent. */
196    if (dirs[--depth] == L)
197      links[depth]->llink = link;
198    else
199      links[depth]->rlink = link;
200
201    /* Back up the tree fixing the balance flags. */
202    while (depth && !links[depth]->balance)
203      {
204        if (dirs[depth] == L)
205    --links[depth]->balance;
206        else
207    ++links[depth]->balance;
208        --depth;
209      }
210
211    /* Rebalance the tree by pointer rotations if necessary. */
212    if (depth && ((dirs[depth] == L && --links[depth]->balance)
213      || (dirs[depth] == R && ++links[depth]->balance)))
214      {
215        switch (links[depth]->balance)
216    {
217    case (char) -2:
218      switch (dirs[depth + 1])
219        {
220        case L:
221          r = links[depth], t = r->llink, rl = t->rlink;
222          t->rlink = r, r->llink = rl;
223          t->balance = r->balance = 0;
224          break;
225        case R:
226          r = links[depth], l = r->llink, t = l->rlink;
227          rl = t->rlink, lr = t->llink;
228          t->llink = l, l->rlink = lr, t->rlink = r, r->llink = rl;
229          l->balance = t->balance != 1 ? 0 : -1;
230          r->balance = t->balance != (char) -1 ? 0 : 1;
231          t->balance = 0;
232          break;
233        default:
234          abort ();
235        }
236      break;
237    case 2:
238      switch (dirs[depth + 1])
239        {
240        case R:
241          l = links[depth], t = l->rlink, lr = t->llink;
242          t->llink = l, l->rlink = lr;
243          t->balance = l->balance = 0;
244          break;
245        case L:
246          l = links[depth], r = l->rlink, t = r->llink;
247          lr = t->llink, rl = t->rlink;
248          t->llink = l, l->rlink = lr, t->rlink = r, r->llink = rl;
249          l->balance = t->balance != 1 ? 0 : -1;
250          r->balance = t->balance != (char) -1 ? 0 : 1;
251          t->balance = 0;
252          break;
253        default:
254          abort ();
255        }
256      break;
257    default:
258      abort ();
259    }
260
261        if (dirs[depth - 1] == L)
262    links[depth - 1]->llink = t;
263        else
264    links[depth - 1]->rlink = t;
265      }
266  }
267
268      trie = link->trie;
269    }
270
271  /* Mark the node we finally reached as accepting, encoding the
272     index number of this word in the keyword set so far. */
273  if (!trie->accepting)
274    trie->accepting = 1 + 2 * kwset->words;
275  ++kwset->words;
276
277  /* Keep track of the longest and shortest string of the keyword set. */
278  if (trie->depth < kwset->mind)
279    kwset->mind = trie->depth;
280  if (trie->depth > kwset->maxd)
281    kwset->maxd = trie->depth;
282
283  return NULL;
284}

286/* Enqueue the trie nodes referenced from the given tree in the
287   given queue. */
288static void
289enqueue (struct tree *tree, struct trie **last)
290{
291  if (!tree)
292    return;
293  enqueue(tree->llink, last);
294  enqueue(tree->rlink, last);
295  (*last) = (*last)->next = tree->trie;
296}

298/* Compute the Aho-Corasick failure function for the trie nodes referenced
299   from the given tree, given the failure function for their parent as
300   well as a last resort failure node. */
301static void
302treefails (register struct tree const *tree, struct trie const *fail,
303     struct trie *recourse)
304{
305  register struct tree *link;
306
307  if (!tree)
308    return;
309
310  treefails(tree->llink, fail, recourse);
311  treefails(tree->rlink, fail, recourse);
312
313  /* Find, in the chain of fails going back to the root, the first
314     node that has a descendant on the current label. */
315  while (fail)
316    {
317      link = fail->links;
318      while (link && tree->label != link->label)
319  if (tree->label < link->label)
320    link = link->llink;
321  else
322    link = link->rlink;
323      if (link)
324  {
325    tree->trie->fail = link->trie;
326    return;
327  }
328      fail = fail->fail;
329    }
330
331  tree->trie->fail = recourse;
332}

334/* Set delta entries for the links of the given tree such that
335   the preexisting delta value is larger than the current depth. */
336static void
337treedelta (register struct tree const *tree,
338     register unsigned int depth,
339     unsigned char delta[])
340{
341  if (!tree)
342    return;
343  treedelta(tree->llink, depth, delta);
344  treedelta(tree->rlink, depth, delta);
345  if (depth < delta[tree->label])
346    delta[tree->label] = depth;
347}

349/* Return true if A has every label in B. */
350static int
351hasevery (register struct tree const *a, register struct tree const *b)
352{
353  if (!b)
354    return 1;
355  if (!hasevery(a, b->llink))
356    return 0;
357  if (!hasevery(a, b->rlink))
358    return 0;
359  while (a && b->label != a->label)
360    if (b->label < a->label)
361      a = a->llink;
362    else
363      a = a->rlink;
364  return !!a;
365}

367/* Compute a vector, indexed by character code, of the trie nodes
368   referenced from the given tree. */
369static void
370treenext (struct tree const *tree, struct trie *next[])
371{
372  if (!tree)
373    return;
374  treenext(tree->llink, next);
375  treenext(tree->rlink, next);
376  next[tree->label] = tree->trie;
377}

379/* Compute the shift for each trie node, as well as the delta
380   table and next cache for the given keyword set. */
381const char *
382kwsprep (kwset_t kws)
383{
384  register struct kwset *kwset;
385  register int i;
386  register struct trie *curr;
387  register unsigned char const *trans;
388  unsigned char delta[NCHAR];
389
390  kwset = (struct kwset *) kws;
391
392  /* Initial values for the delta table; will be changed later.  The
393     delta entry for a given character is the smallest depth of any
394     node at which an outgoing edge is labeled by that character. */
395  memset(delta, kwset->mind < UCHAR_MAX ? kwset->mind : UCHAR_MAX, NCHAR);
396
397  /* Check if we can use the simple boyer-moore algorithm, instead
398     of the hairy commentz-walter algorithm. */
399  if (kwset->words == 1 && kwset->trans == NULL)
400    {
401      char c;
402
403      /* Looking for just one string.  Extract it from the trie. */
404      kwset->target = obstack_alloc(&kwset->obstack, kwset->mind);
405      if (!kwset->target)
406  return "memory exhausted";
407      for (i = kwset->mind - 1, curr = kwset->trie; i >= 0; --i)
408  {
409    kwset->target[i] = curr->links->label;
410    curr = curr->links->trie;
411  }
412      /* Build the Boyer Moore delta.  Boy that's easy compared to CW. */
413      for (i = 0; i < kwset->mind; ++i)
414  delta[U(kwset->target[i])] = kwset->mind - (i + 1);
415      /* Find the minimal delta2 shift that we might make after
416   a backwards match has failed. */
417      c = kwset->target[kwset->mind - 1];
418      for (i = kwset->mind - 2; i >= 0; --i)
419  if (kwset->target[i] == c)
420    break;
421      kwset->mind2 = kwset->mind - (i + 1);
422    }
423  else
424    {
425      register struct trie *fail;
426      struct trie *last, *next[NCHAR];
427
428      /* Traverse the nodes of the trie in level order, simultaneously
429   computing the delta table, failure function, and shift function. */
430      for (curr = last = kwset->trie; curr; curr = curr->next)
431  {
432    /* Enqueue the immediate descendants in the level order queue. */
433    enqueue(curr->links, &last);
434
435    curr->shift = kwset->mind;
436    curr->maxshift = kwset->mind;
437
438    /* Update the delta table for the descendants of this node. */
439    treedelta(curr->links, curr->depth, delta);
440
441    /* Compute the failure function for the descendants of this node. */
442    treefails(curr->links, curr->fail, kwset->trie);
443
444    /* Update the shifts at each node in the current node's chain
445       of fails back to the root. */
446    for (fail = curr->fail; fail; fail = fail->fail)
447      {
448        /* If the current node has some outgoing edge that the fail
449     doesn't, then the shift at the fail should be no larger
450     than the difference of their depths. */
451        if (!hasevery(fail->links, curr->links))
452    if (curr->depth - fail->depth < fail->shift)
453      fail->shift = curr->depth - fail->depth;
454
455        /* If the current node is accepting then the shift at the
456     fail and its descendants should be no larger than the
457     difference of their depths. */
458        if (curr->accepting && fail->maxshift > curr->depth - fail->depth)
459    fail->maxshift = curr->depth - fail->depth;
460      }
461  }
462
463      /* Traverse the trie in level order again, fixing up all nodes whose
464   shift exceeds their inherited maxshift. */
465      for (curr = kwset->trie->next; curr; curr = curr->next)
466  {
467    if (curr->maxshift > curr->parent->maxshift)
468      curr->maxshift = curr->parent->maxshift;
469    if (curr->shift > curr->maxshift)
470      curr->shift = curr->maxshift;
471  }
472
473      /* Create a vector, indexed by character code, of the outgoing links
474   from the root node. */
475      for (i = 0; i < NCHAR; ++i)
476  next[i] = NULL;
477      treenext(kwset->trie->links, next);
478
479      if ((trans = kwset->trans))
480  for (i = 0; i < NCHAR; ++i)
481    kwset->next[i] = next[U(trans[i])];
482      else
483  COPY_ARRAY(kwset->next, next, NCHAR);
484    }
485
486  /* Fix things up for any translation table. */
487  if ((trans = kwset->trans))
488    for (i = 0; i < NCHAR; ++i)
489      kwset->delta[i] = delta[U(trans[i])];
490  else
491    memcpy(kwset->delta, delta, NCHAR);
492
493  return NULL;
494}

496/* Fast boyer-moore search. */
497static size_t
498bmexec (kwset_t kws, char const *text, size_t size)
499{
500  struct kwset const *kwset;
501  register unsigned char const *d1;
502  register char const *ep, *sp, *tp;
503  register int d, gc, i, len, md2;
504
505  kwset = (struct kwset const *) kws;
506  len = kwset->mind;
507
508  if (len == 0)
509    return 0;
510  if (len > size)
511    return -1;
512  if (len == 1)
513    {
514      tp = memchr (text, kwset->target[0], size);
515      return tp ? tp - text : -1;
516    }
517
518  d1 = kwset->delta;
519  sp = kwset->target + len;
520  gc = U(sp[-2]);
521  md2 = kwset->mind2;
522  tp = text + len;
523
524  /* Significance of 12: 1 (initial offset) + 10 (skip loop) + 1 (md2). */
525  if (size > 12 * len)
526    /* 11 is not a bug, the initial offset happens only once. */
527    for (ep = text + size - 11 * len;;)
528      {
529  while (tp <= ep)
530    {
531      d = d1[U(tp[-1])], tp += d;
532      d = d1[U(tp[-1])], tp += d;
533      if (d == 0)
534        goto found;
535      d = d1[U(tp[-1])], tp += d;
536      d = d1[U(tp[-1])], tp += d;
537      d = d1[U(tp[-1])], tp += d;
538      if (d == 0)
539        goto found;
540      d = d1[U(tp[-1])], tp += d;
541      d = d1[U(tp[-1])], tp += d;
542      d = d1[U(tp[-1])], tp += d;
543      if (d == 0)
544        goto found;
545      d = d1[U(tp[-1])], tp += d;
546      d = d1[U(tp[-1])], tp += d;
547    }
548  break;
549      found:
550  if (U(tp[-2]) == gc)
551    {
552      for (i = 3; i <= len && U(tp[-i]) == U(sp[-i]); ++i)
553        ;
554      if (i > len)
555        return tp - len - text;
556    }
557  tp += md2;
558      }
559
560  /* Now we have only a few characters left to search.  We
561     carefully avoid ever producing an out-of-bounds pointer. */
562  ep = text + size;
563  d = d1[U(tp[-1])];
564  while (d <= ep - tp)
565    {
566      d = d1[U((tp += d)[-1])];
567      if (d != 0)
568  continue;
569      if (U(tp[-2]) == gc)
570  {
571    for (i = 3; i <= len && U(tp[-i]) == U(sp[-i]); ++i)
572      ;
573    if (i > len)
574      return tp - len - text;
575  }
576      d = md2;
577    }
578
579  return -1;
580}
581
582/* Hairy multiple string search. */
583static size_t
584cwexec (kwset_t kws, char const *text, size_t len, struct kwsmatch *kwsmatch)
585{
586  struct kwset const *kwset;
587  struct trie * const *next;
588  struct trie const *trie;
589  struct trie const *accept;
590  char const *beg, *lim, *mch, *lmch;
591  register unsigned char c;
592  register unsigned char const *delta;
593  register int d;
594  register char const *end, *qlim;
595  register struct tree const *tree;
596  register unsigned char const *trans;
597
598  accept = NULL;
599
600  /* Initialize register copies and look for easy ways out. */
601  kwset = (struct kwset *) kws;
602  if (len < kwset->mind)
603    return -1;
604  next = kwset->next;
605  delta = kwset->delta;
606  trans = kwset->trans;
607  lim = text + len;
608  end = text;
609  if ((d = kwset->mind) != 0)
610    mch = NULL;
611  else
612    {
613      mch = text, accept = kwset->trie;
614      goto match;
615    }
616
617  if (len >= 4 * kwset->mind)
618    qlim = lim - 4 * kwset->mind;
619  else
620    qlim = NULL;
621
622  while (lim - end >= d)
623    {
624      if (qlim && end <= qlim)
625  {
626    end += d - 1;
627    while ((d = delta[c = *end]) && end < qlim)
628      {
629        end += d;
630        end += delta[U(*end)];
631        end += delta[U(*end)];
632      }
633    ++end;
634  }
635      else
636  d = delta[c = (end += d)[-1]];
637      if (d)
638  continue;
639      beg = end - 1;
640      trie = next[c];
641      if (trie->accepting)
642  {
643    mch = beg;
644    accept = trie;
645  }
646      d = trie->shift;
647      while (beg > text)
648  {
649    c = trans ? trans[U(*--beg)] : *--beg;
650    tree = trie->links;
651    while (tree && c != tree->label)
652      if (c < tree->label)
653        tree = tree->llink;
654      else
655        tree = tree->rlink;
656    if (tree)
657      {
658        trie = tree->trie;
659        if (trie->accepting)
660    {
661      mch = beg;
662      accept = trie;
663    }
664      }
665    else
666      break;
667    d = trie->shift;
668  }
669      if (mch)
670  goto match;
671    }
672  return -1;
673
674 match:
675  /* Given a known match, find the longest possible match anchored
676     at or before its starting point.  This is nearly a verbatim
677     copy of the preceding main search loops. */
678  if (lim - mch > kwset->maxd)
679    lim = mch + kwset->maxd;
680  lmch = NULL;
681  d = 1;
682  while (lim - end >= d)
683    {
684      if ((d = delta[c = (end += d)[-1]]) != 0)
685  continue;
686      beg = end - 1;
687      if (!(trie = next[c]))
688  {
689    d = 1;
690    continue;
691  }
692      if (trie->accepting && beg <= mch)
693  {
694    lmch = beg;
695    accept = trie;
696  }
697      d = trie->shift;
698      while (beg > text)
699  {
700    c = trans ? trans[U(*--beg)] : *--beg;
701    tree = trie->links;
702    while (tree && c != tree->label)
703      if (c < tree->label)
704        tree = tree->llink;
705      else
706        tree = tree->rlink;
707    if (tree)
708      {
709        trie = tree->trie;
710        if (trie->accepting && beg <= mch)
711    {
712      lmch = beg;
713      accept = trie;
714    }
715      }
716    else
717      break;
718    d = trie->shift;
719  }
720      if (lmch)
721  {
722    mch = lmch;
723    goto match;
724  }
725      if (!d)
726  d = 1;
727    }
728
729  if (kwsmatch)
730    {
731      kwsmatch->index = accept->accepting / 2;
732      kwsmatch->offset[0] = mch - text;
733      kwsmatch->size[0] = accept->depth;
734    }
735  return mch - text;
736}
737
738/* Search through the given text for a match of any member of the
739   given keyword set.  Return a pointer to the first character of
740   the matching substring, or NULL if no match is found.  If FOUNDLEN
741   is non-NULL store in the referenced location the length of the
742   matching substring.  Similarly, if FOUNDIDX is non-NULL, store
743   in the referenced location the index number of the particular
744   keyword matched. */
745size_t
746kwsexec (kwset_t kws, char const *text, size_t size,
747   struct kwsmatch *kwsmatch)
748{
749  struct kwset const *kwset = (struct kwset *) kws;
750  if (kwset->words == 1 && kwset->trans == NULL)
751    {
752      size_t ret = bmexec (kws, text, size);
753      if (kwsmatch != NULL && ret != (size_t) -1)
754  {
755    kwsmatch->index = 0;
756    kwsmatch->offset[0] = ret;
757    kwsmatch->size[0] = kwset->mind;
758  }
759      return ret;
760    }
761  else
762    return cwexec(kws, text, size, kwsmatch);
763}
