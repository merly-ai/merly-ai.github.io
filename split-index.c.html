16int read_link_extension(struct index_state *istate,
17       const void *data_, unsigned long sz)
18{
19  const unsigned char *data = data_;
20  struct split_index *si;
21  int ret;
22
23  if (sz < the_hash_algo->rawsz)
24    return error("corrupt link extension (too short)");
25  si = init_split_index(istate);
26  oidread(&si->base_oid, data);
27  data += the_hash_algo->rawsz;
28  sz -= the_hash_algo->rawsz;
29  if (!sz)
30    return 0;
31  si->delete_bitmap = ewah_new();
32  ret = ewah_read_mmap(si->delete_bitmap, data, sz);
33  if (ret < 0)
34    return error("corrupt delete bitmap in link extension");
35  data += ret;
36  sz -= ret;
37  si->replace_bitmap = ewah_new();
38  ret = ewah_read_mmap(si->replace_bitmap, data, sz);
39  if (ret < 0)
40    return error("corrupt replace bitmap in link extension");
41  if (ret != sz)
42    return error("garbage at the end of link extension");
43  return 0;

58static void mark_base_index_entries(struct index_state *base)
59{

121static void replace_entry(size_t pos, void *data)
122{
123  struct index_state *istate = data;
124  struct split_index *si = istate->split_index;
125  struct cache_entry *dst, *src;
126
127  if (pos >= istate->cache_nr)
128    die("position for replacement %d exceeds base index size %d",
129        (int)pos, istate->cache_nr);
130  if (si->nr_replacements >= si->saved_cache_nr)
131    die("too many replacements (%d vs %d)",
132        si->nr_replacements, si->saved_cache_nr);
133  dst = istate->cache[pos];
134  if (dst->ce_flags & CE_REMOVE)
135    die("entry %d is marked as both replaced and deleted",
136        (int)pos);
137  src = si->saved_cache[si->nr_replacements];
138  if (ce_namelen(src))
139    die("corrupt link extension, entry %d should have "
140        "zero length name", (int)pos);
141  src->index = pos + 1;
142  src->ce_flags |= CE_UPDATE_IN_BASE;
143  src->ce_namelen = dst->ce_namelen;
144  copy_cache_entry(dst, src);
145  discard_cache_entry(src);
146  si->nr_replacements++;
147}

149void merge_base_index(struct index_state *istate)
150{
151  struct split_index *si = istate->split_index;
152  unsigned int i;
153
154  mark_base_index_entries(si->base);
155
156  si->saved_cache      = istate->cache;
157  si->saved_cache_nr  = istate->cache_nr;
158  istate->cache_nr    = si->base->cache_nr;
159  istate->cache      = NULL;
160  istate->cache_alloc = 0;
161  ALLOC_GROW(istate->cache, istate->cache_nr, istate->cache_alloc);
162  COPY_ARRAY(istate->cache, si->base->cache, istate->cache_nr);
163
164  si->nr_deletions = 0;
165  si->nr_replacements = 0;
166  ewah_each_bit(si->replace_bitmap, replace_entry, istate);
167  ewah_each_bit(si->delete_bitmap, mark_entry_for_delete, istate);
168  if (si->nr_deletions)
169    remove_marked_cache_entries(istate, 0);
170
171  for (i = si->nr_replacements; i < si->saved_cache_nr; i++) {
172    if (!ce_namelen(si->saved_cache[i]))
173      die("corrupt link extension, entry %d should "
174          "have non-zero length name", i);
175    add_index_entry(istate, si->saved_cache[i],
176        ADD_CACHE_OK_TO_ADD |
177        ADD_CACHE_KEEP_CACHE_TREE |
178        /*
179         * we may have to replay what
180         * merge-recursive.c:update_stages()
181         * does, which has this flag on
182         */
183        ADD_CACHE_SKIP_DFCHECK);
184    si->saved_cache[i] = NULL;
185  }
186
187  ewah_free(si->delete_bitmap);
188  ewah_free(si->replace_bitmap);
189  FREE_AND_NULL(si->saved_cache);
190  si->delete_bitmap  = NULL;
191  si->replace_bitmap = NULL;
192  si->saved_cache_nr = 0;
193}

195/*
196 * Compare most of the fields in two cache entries, i.e. all except the
197 * hashmap_entry and the name.
198 */
199static int compare_ce_content(struct cache_entry *a, struct cache_entry *b)
200{
201  const unsigned int ondisk_flags = CE_STAGEMASK | CE_VALID |
202            CE_EXTENDED_FLAGS;
203  unsigned int ce_flags = a->ce_flags;
204  unsigned int base_flags = b->ce_flags;
205  int ret;
206
207  /* only on-disk flags matter */
208  a->ce_flags &= ondisk_flags;
209  b->ce_flags &= ondisk_flags;
210  ret = memcmp(&a->ce_stat_data, &b->ce_stat_data,
211         offsetof(struct cache_entry, name) -
212         offsetof(struct cache_entry, oid)) ||
213    !oideq(&a->oid, &b->oid);
214  a->ce_flags = ce_flags;
215  b->ce_flags = base_flags;
216
217  return ret;
218}

220void prepare_to_write_split_index(struct index_state *istate)
221{
222  struct split_index *si = init_split_index(istate);
223  struct cache_entry **entries = NULL, *ce;
224  int i, nr_entries = 0, nr_alloc = 0;
225
226  si->delete_bitmap = ewah_new();
227  si->replace_bitmap = ewah_new();
228
229  if (si->base) {
230    /* Go through istate->cache[] and mark CE_MATCHED to
231     * entry with positive index. We'll go through
232     * base->cache[] later to delete all entries in base
233     * that are not marked with either CE_MATCHED or
234     * CE_UPDATE_IN_BASE. If istate->cache[i] is a
235     * duplicate, deduplicate it.
236     */
237    for (i = 0; i < istate->cache_nr; i++) {
238      struct cache_entry *base;
239      ce = istate->cache[i];
240      if (!ce->index) {
241        /*
242         * During simple update index operations this
243         * is a cache entry that is not present in
244         * the shared index.  It will be added to the
245         * split index.
246         *
247         * However, it might also represent a file
248         * that already has a cache entry in the
249         * shared index, but a new index has just
250         * been constructed by unpack_trees(), and
251         * this entry now refers to different content
252         * than what was recorded in the original
253         * index, e.g. during 'read-tree -m HEAD^' or
254         * 'checkout HEAD^'.  In this case the
255         * original entry in the shared index will be
256         * marked as deleted, and this entry will be
257         * added to the split index.
258         */
259        continue;
260      }
261      if (ce->index > si->base->cache_nr) {
262        BUG("ce refers to a shared ce at %d, which is beyond the shared index size %d",
263            ce->index, si->base->cache_nr);
264      }
265      ce->ce_flags |= CE_MATCHED; /* or "shared" */
266      base = si->base->cache[ce->index - 1];
267      if (ce == base) {
268        /* The entry is present in the shared index. */
269        if (ce->ce_flags & CE_UPDATE_IN_BASE) {
270          /*
271           * Already marked for inclusion in
272           * the split index, either because
273           * the corresponding file was
274           * modified and the cached stat data
275           * was refreshed, or because there
276           * is already a replacement entry in
277           * the split index.
278           * Nothing more to do here.
279           */
280        } else if (!ce_uptodate(ce) &&
281             is_racy_timestamp(istate, ce)) {
282          /*
283           * A racily clean cache entry stored
284           * only in the shared index: it must
285           * be added to the split index, so
286           * the subsequent do_write_index()
287           * can smudge its stat data.
288           */
289          ce->ce_flags |= CE_UPDATE_IN_BASE;
290        } else {
291          /*
292           * The entry is only present in the
293           * shared index and it was not
294           * refreshed.
295           * Just leave it there.
296           */
297        }
298        continue;
299      }
300      if (ce->ce_namelen != base->ce_namelen ||
301          strcmp(ce->name, base->name)) {
302        ce->index = 0;
303        continue;
304      }
305      /*
306       * This is the copy of a cache entry that is present
307       * in the shared index, created by unpack_trees()
308       * while it constructed a new index.
309       */
310      if (ce->ce_flags & CE_UPDATE_IN_BASE) {
311        /*
312         * Already marked for inclusion in the split
313         * index, either because the corresponding
314         * file was modified and the cached stat data
315         * was refreshed, or because the original
316         * entry already had a replacement entry in
317         * the split index.
318         * Nothing to do.
319         */
320      } else if (!ce_uptodate(ce) &&
321           is_racy_timestamp(istate, ce)) {
322        /*
323         * A copy of a racily clean cache entry from
324         * the shared index.  It must be added to
325         * the split index, so the subsequent
326         * do_write_index() can smudge its stat data.
327         */
328        ce->ce_flags |= CE_UPDATE_IN_BASE;
329      } else {
330        /*
331         * Thoroughly compare the cached data to see
332         * whether it should be marked for inclusion
333         * in the split index.
334         *
335         * This comparison might be unnecessary, as
336         * code paths modifying the cached data do
337         * set CE_UPDATE_IN_BASE as well.
338         */
339        if (compare_ce_content(ce, base))
340          ce->ce_flags |= CE_UPDATE_IN_BASE;
341      }
342      discard_cache_entry(base);
343      si->base->cache[ce->index - 1] = ce;
344    }
345    for (i = 0; i < si->base->cache_nr; i++) {
346      ce = si->base->cache[i];
347      if ((ce->ce_flags & CE_REMOVE) ||
348          !(ce->ce_flags & CE_MATCHED))
349        ewah_set(si->delete_bitmap, i);
350      else if (ce->ce_flags & CE_UPDATE_IN_BASE) {
351        ewah_set(si->replace_bitmap, i);
352        ce->ce_flags |= CE_STRIP_NAME;
353        ALLOC_GROW(entries, nr_entries+1, nr_alloc);
354        entries[nr_entries++] = ce;
355      }
356      if (is_null_oid(&ce->oid))
357        istate->drop_cache_tree = 1;
358    }
359  }
360
361  for (i = 0; i < istate->cache_nr; i++) {
362    ce = istate->cache[i];
363    if ((!si->base || !ce->index) && !(ce->ce_flags & CE_REMOVE)) {
364      assert(!(ce->ce_flags & CE_STRIP_NAME));
365      ALLOC_GROW(entries, nr_entries+1, nr_alloc);
366      entries[nr_entries++] = ce;
367    }
368    ce->ce_flags &= ~CE_MATCHED;
369  }
370
371  /*
372   * take cache[] out temporarily, put entries[] in its place
373   * for writing
374   */
375  si->saved_cache = istate->cache;
376  si->saved_cache_nr = istate->cache_nr;
377  istate->cache = entries;
378  istate->cache_nr = nr_entries;
379}

395{
396  struct split_index *si = istate->split_index;

445void remove_split_index(struct index_state *istate)
446{
447  if (istate->split_index) {
448    if (istate->split_index->base) {
449      /*
450       * When removing the split index, we need to move
451       * ownership of the mem_pool associated with the
452       * base index to the main index. There may be cache entries
453       * allocated from the base's memory pool that are shared with
454       * the_index.cache[].
455       */
456      mem_pool_combine(istate->ce_mem_pool,
457           istate->split_index->base->ce_mem_pool);
458
459      /*
460       * The split index no longer owns the mem_pool backing
461       * its cache array. As we are discarding this index,
462       * mark the index as having no cache entries, so it
463       * will not attempt to clean up the cache entries or
464       * validate them.
465       */
466      istate->split_index->base->cache_nr = 0;
467    }
468
469    /*
470     * We can discard the split index because its
471     * memory pool has been incorporated into the
472     * memory pool associated with the the_index.
473     */
474    discard_split_index(istate);
475
476    istate->cache_changed |= SOMETHING_CHANGED;
477  }
