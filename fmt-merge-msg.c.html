78/*

82static struct merge_parent *find_merge_parent(struct merge_parents *table,
83                struct object_id *given,
84                struct object_id *commit)
85{
86  int i;
87  for (i = 0; i < table->nr; i++) {

89      continue;

92    return &table->item[i];
93  }

110static int handle_line(char *line, struct merge_parents *merge_parents)
111{
112  int i, len = strlen(line);
113  struct origin_data *origin_data;
114  char *src;
115  const char *origin, *tag_name;
116  char *to_free = NULL;
117  struct src_data *src_data;
118  struct string_list_item *item;
119  int pulling_head = 0;
120  struct object_id oid;
121  const unsigned hexsz = the_hash_algo->hexsz;
122
123  if (len < hexsz + 3 || line[hexsz] != '\t')
124    return 1;
125
126  if (starts_with(line + hexsz + 1, "not-for-merge"))
127    return 0;
128
129  if (line[hexsz + 1] != '\t')
130    return 2;
131
132  i = get_oid_hex(line, &oid);
133  if (i)
134    return 3;
135
136  if (!find_merge_parent(merge_parents, &oid, NULL))
137    return 0; /* subsumed by other parents */
138
139  CALLOC_ARRAY(origin_data, 1);
140  oidcpy(&origin_data->oid, &oid);
141
142  if (line[len - 1] == '\n')
143    line[len - 1] = 0;
144  line += hexsz + 2;
145
146  /*
147   * At this point, line points at the beginning of comment e.g.
148   * "branch 'frotz' of git://that/repository.git".
149   * Find the repository name and point it with src.
150   */
151  src = strstr(line, " of ");
152  if (src) {
153    *src = 0;
154    src += 4;
155    pulling_head = 0;
156  } else {
157    src = line;
158    pulling_head = 1;
159  }
160
161  item = unsorted_string_list_lookup(&srcs, src);
162  if (!item) {
163    item = string_list_append(&srcs, src);
164    item->util = xcalloc(1, sizeof(struct src_data));
165    init_src_data(item->util);
166  }
167  src_data = item->util;
168
169  if (pulling_head) {
170    origin = src;
171    src_data->head_status |= 1;
172  } else if (skip_prefix(line, "branch ", &origin)) {
173    origin_data->is_local_branch = 1;
174    string_list_append(&src_data->branch, origin);
175    src_data->head_status |= 2;
176  } else if (skip_prefix(line, "tag ", &tag_name)) {
177    origin = line;
178    string_list_append(&src_data->tag, tag_name);
179    src_data->head_status |= 2;
180  } else if (skip_prefix(line, "remote-tracking branch ", &origin)) {
181    string_list_append(&src_data->r_branch, origin);
182    src_data->head_status |= 2;
183  } else {
184    origin = src;
185    string_list_append(&src_data->generic, line);
186    src_data->head_status |= 2;
187  }
188
189  if (!strcmp(".", src) || !strcmp(src, origin)) {
190    int len = strlen(origin);
191    if (origin[0] == '\'' && origin[len - 1] == '\'')
192      origin = to_free = xmemdupz(origin + 1, len - 2);
193  } else
194    origin = to_free = xstrfmt("%s of %s", origin, src);
195  if (strcmp(".", src))
196    origin_data->is_local_branch = 0;
197  string_list_append(&origins, origin)->util = origin_data;
198  free(to_free);
199  return 0;
200}

239static void record_person_from_buf(int which, struct string_list *people,
240           const char *buffer)
241{
242  char *name_buf, *name, *name_end;
243  struct string_list_item *elem;
244  const char *field;
245
246  field = (which == 'a') ? "\nauthor " : "\ncommitter ";
247  name = strstr(buffer, field);
248  if (!name)
249    return;
250  name += strlen(field);
251  name_end = strchrnul(name, '<');
252  if (*name_end)
253    name_end--;
254  while (isspace(*name_end) && name <= name_end)
255    name_end--;
256  if (name_end < name)
257    return;

259
260  elem = string_list_lookup(people, name_buf);
261  if (!elem) {
262    elem = string_list_insert(people, name_buf);
263    elem->util = (void *)0;
264  }
265  elem->util = (void*)(util_as_integral(elem) + 1);
266  free(name_buf);

338static void shortlog(const char *name,
339         struct origin_data *origin_data,
340         struct commit *head,
341         struct rev_info *rev,
342         struct fmt_merge_msg_opts *opts,
343         struct strbuf *out)
344{
345  int i, count = 0;
346  struct commit *commit;
347  struct object *branch;
348  struct string_list subjects = STRING_LIST_INIT_DUP;
349  struct string_list authors = STRING_LIST_INIT_DUP;
350  struct string_list committers = STRING_LIST_INIT_DUP;
351  int flags = UNINTERESTING | TREESAME | SEEN | SHOWN | ADDED;
352  struct strbuf sb = STRBUF_INIT;
353  const struct object_id *oid = &origin_data->oid;
354  int limit = opts->shortlog_len;
355
356  branch = deref_tag(the_repository, parse_object(the_repository, oid),
357         oid_to_hex(oid),
358         the_hash_algo->hexsz);
359  if (!branch || branch->type != OBJ_COMMIT)
360    return;
361
362  setup_revisions(0, NULL, rev, NULL);
363  add_pending_object(rev, branch, name);
364  add_pending_object(rev, &head->object, "^HEAD");
365  head->object.flags |= UNINTERESTING;
366  if (prepare_revision_walk(rev))
367    die("revision walk setup failed");
368  while ((commit = get_revision(rev)) != NULL) {
369    struct pretty_print_context ctx = {0};
370
371    if (commit->parents && commit->parents->next) {
372      /* do not list a merge but count committer */
373      if (opts->credit_people)
374        record_person('c', &committers, commit);
375      continue;
376    }
377    if (!count && opts->credit_people)
378      /* the 'tip' committer */
379      record_person('c', &committers, commit);
380    if (opts->credit_people)
381      record_person('a', &authors, commit);
382    count++;
383    if (subjects.nr > limit)
384      continue;
385
386    format_commit_message(commit, "%s", &sb, &ctx);
387    strbuf_ltrim(&sb);
388
389    if (!sb.len)
390      string_list_append(&subjects,
391             oid_to_hex(&commit->object.oid));
392    else
393      string_list_append_nodup(&subjects,
394             strbuf_detach(&sb, NULL));
395  }
396
397  if (opts->credit_people)
398    add_people_info(out, &authors, &committers);
399  if (count > limit)
400    strbuf_addf(out, "\n* %s: (%d commits)\n", name, count);
401  else
402    strbuf_addf(out, "\n* %s:\n", name);
403
404  if (origin_data->is_local_branch && use_branch_desc)
405    add_branch_desc(out, name);
406
407  for (i = 0; i < subjects.nr; i++)
408    if (i >= limit)
409      strbuf_addstr(out, "  ...\n");
410    else
411      strbuf_addf(out, "  %s\n", subjects.items[i].string);
412
413  clear_commit_marks((struct commit *)branch, flags);
414  clear_commit_marks(head, flags);
415  free_commit_list(rev->commits);
416  rev->commits = NULL;
417  rev->pending.nr = 0;
418
419  string_list_clear(&authors, 0);
420  string_list_clear(&committers, 0);
421  string_list_clear(&subjects, 0);
422}

539        strbuf_addstr(&sig, "gpg verification failed.\n");
540      else
541        strbuf_addstr(&sig, sigc.output);
542    }
543
544    if (!tag_number++) {
545      fmt_tag_signature(&tagbuf, &sig, buf, len);
546      first_tag = i;
547    } else {
548      if (tag_number == 2) {
549        struct strbuf tagline = STRBUF_INIT;
550        strbuf_addch(&tagline, '\n');
551        strbuf_add_commented_lines(&tagline,
552            origins.items[first_tag].string,
553            strlen(origins.items[first_tag].string));
554        strbuf_insert(&tagbuf, 0, tagline.buf,
555                tagline.len);
556        strbuf_release(&tagline);
557      }
558      strbuf_addch(&tagbuf, '\n');
559      strbuf_add_commented_lines(&tagbuf,
560          origins.items[i].string,
561          strlen(origins.items[i].string));
562      fmt_tag_signature(&tagbuf, &sig, buf, len);
563    }
564    strbuf_release(&payload);
565    strbuf_release(&sig);
566    signature_check_clear(&sigc);
567  next:
568    free(origbuf);
569  }
570  if (tagbuf.len) {
571    strbuf_addch(out, '\n');
572    strbuf_addbuf(out, &tagbuf);
573  }
574  strbuf_release(&tagbuf);
575}

577static void find_merge_parents(struct merge_parents *result,
578             struct strbuf *in, struct object_id *head)
579{
580  struct commit_list *parents;
581  struct commit *head_commit;
582  int pos = 0, i, j;
583
584  parents = NULL;
585  while (pos < in->len) {
586    int len;
587    char *p = in->buf + pos;
588    char *newline = strchr(p, '\n');
589    const char *q;
590    struct object_id oid;
591    struct commit *parent;
592    struct object *obj;
593
594    len = newline ? newline - p : strlen(p);
595    pos += len + !!newline;
596
597    if (parse_oid_hex(p, &oid, &q) ||
598        q[0] != '\t' ||
599        q[1] != '\t')
600      continue; /* skip not-for-merge */
601    /*
602     * Do not use get_merge_parent() here; we do not have
603     * "name" here and we do not want to contaminate its
604     * util field yet.
605     */
606    obj = parse_object(the_repository, &oid);
607    parent = (struct commit *)peel_to_type(NULL, 0, obj, OBJ_COMMIT);
608    if (!parent)
609      continue;
610    commit_list_insert(parent, &parents);
611    add_merge_parent(result, &obj->oid, &parent->object.oid);
612  }
613  head_commit = lookup_commit(the_repository, head);
614  if (head_commit)
615    commit_list_insert(head_commit, &parents);
616  reduce_heads_replace(&parents);
617
618  while (parents) {
619    struct commit *cmit = pop_commit(&parents);
620    for (i = 0; i < result->nr; i++)
621      if (oideq(&result->item[i].commit, &cmit->object.oid))
622        result->item[i].used = 1;
623  }
624
625  for (i = j = 0; i < result->nr; i++) {
626    if (result->item[i].used) {
627      if (i != j)
628        result->item[j] = result->item[i];
629      j++;
630    }
631  }
632  result->nr = j;
633}

636int fmt_merge_msg(struct strbuf *in, struct strbuf *out,
637      struct fmt_merge_msg_opts *opts)
638{
639  int i = 0, pos = 0;
640  struct object_id head_oid;
641  const char *current_branch;
642  void *current_branch_to_free;
643  struct merge_parents merge_parents;
644
645  if (!suppress_dest_pattern_seen) {
646    string_list_append(&suppress_dest_patterns, "main");
647    string_list_append(&suppress_dest_patterns, "master");
648  }
649
650  memset(&merge_parents, 0, sizeof(merge_parents));
651
652  /* learn the commit that we merge into and the current branch name */
653  current_branch = current_branch_to_free =
654    resolve_refdup("HEAD", RESOLVE_REF_READING, &head_oid, NULL);
655  if (!current_branch)
656    die("No current branch");
657
658  if (opts->into_name)
659    current_branch = opts->into_name;
660  else if (starts_with(current_branch, "refs/heads/"))
661    current_branch += 11;
662
663  find_merge_parents(&merge_parents, in, &head_oid);
664
665  /* get a line */
666  while (pos < in->len) {
667    int len;
668    char *newline, *p = in->buf + pos;
669
670    newline = strchr(p, '\n');
671    len = newline ? newline - p : strlen(p);
672    pos += len + !!newline;
673    i++;
674    p[len] = 0;
675    if (handle_line(p, &merge_parents))
676      die("error in line %d: %.*s", i, len, p);
677  }
678
679  if (opts->add_title && srcs.nr)
680    fmt_merge_msg_title(out, current_branch);
681
682  if (origins.nr)
683    fmt_merge_msg_sigs(out);
684
685  if (opts->shortlog_len) {
686    struct commit *head;
687    struct rev_info rev;
688
689    head = lookup_commit_or_die(&head_oid, "HEAD");
690    repo_init_revisions(the_repository, &rev, NULL);
691    rev.commit_format = CMIT_FMT_ONELINE;
692    diff_merges_suppress(&rev);
693    rev.limited = 1;
694
695    strbuf_complete_line(out);
696
697    for (i = 0; i < origins.nr; i++)
698      shortlog(origins.items[i].string,
699         origins.items[i].util,
700         head, &rev, opts, out);
701    release_revisions(&rev);
702  }
703
704  strbuf_complete_line(out);
705  free(current_branch_to_free);
706  free(merge_parents.item);
707  return 0;
708}
