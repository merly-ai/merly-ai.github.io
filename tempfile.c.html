1/*
1 * State diagram and cleanup
2 * -------------------------
3 *
4 * If the program exits while a temporary file is active, we want to
5 * make sure that we remove it. This is done by remembering the active
6 * temporary files in a linked list, `tempfile_list`. An `atexit(3)`
7 * handler and a signal handler are registered, to clean up any active
8 * temporary files.
9 *
10 * Because the signal handler can run at any time, `tempfile_list` and
11 * the `tempfile` objects that comprise it must be kept in
12 * self-consistent states at all times.
13 *
14 * The possible states of a `tempfile` object are as follows:
15 *
16 * - Uninitialized. In this state the object's `on_list` field must be
17 *   zero but the rest of its contents need not be initialized. As
18 *   soon as the object is used in any way, it is irrevocably
19 *   registered in `tempfile_list`, and `on_list` is set.
20 *
21 * - Active, file open (after `create_tempfile()` or
22 *   `reopen_tempfile()`). In this state:
23 *
24 *   - the temporary file exists
25 *   - `active` is set
26 *   - `filename` holds the filename of the temporary file
27 *   - `fd` holds a file descriptor open for writing to it
28 *   - `fp` holds a pointer to an open `FILE` object if and only if
29 *     `fdopen_tempfile()` has been called on the object
30 *   - `owner` holds the PID of the process that created the file
31 *
32 * - Active, file closed (after `close_tempfile_gently()`). Same
33 *   as the previous state, except that the temporary file is closed,
34 *   `fd` is -1, and `fp` is `NULL`.
35 *
36 * - Inactive (after `delete_tempfile()`, `rename_tempfile()`, or a
37 *   failed attempt to create a temporary file). In this state:
38 *
39 *   - `active` is unset
40 *   - `filename` is empty (usually, though there are transitory
41 *     states in which this condition doesn't hold). Client code should
42 *     *not* rely on the filename being empty in this state.
43 *   - `fd` is -1 and `fp` is `NULL`
44 *   - the object is removed from `tempfile_list` (but could be used again)
45 *
46 * A temporary file is owned by the process that created it. The
47 * `tempfile` has an `owner` field that records the owner's PID. This
48 * field is used to prevent a forked process from deleting a temporary
49 * file created by its parent.
50 */

56static VOLATILE_LIST_HEAD(tempfile_list);

58static void remove_template_directory(struct tempfile *tempfile,
59              int in_signal_handler)
60{
61  if (tempfile->directorylen > 0 &&
62      tempfile->directorylen < tempfile->filename.len &&
63      tempfile->filename.buf[tempfile->directorylen] == '/') {
64    strbuf_setlen(&tempfile->filename, tempfile->directorylen);
65    if (in_signal_handler)
66      rmdir(tempfile->filename.buf);
67    else
68      rmdir_or_warn(tempfile->filename.buf);
69  }
70}
71
72static void remove_tempfiles(int in_signal_handler)
73{
74  pid_t me = getpid();
75  volatile struct volatile_list_head *pos;
76
77  list_for_each(pos, &tempfile_list) {
78    struct tempfile *p = list_entry(pos, struct tempfile, list);
79
80    if (!is_tempfile_active(p) || p->owner != me)
81      continue;
82
83    if (p->fd >= 0)
84      close(p->fd);
85
86    if (in_signal_handler)
87      unlink(p->filename.buf);
88    else
89      unlink_or_warn(p->filename.buf);
90    remove_template_directory(p, in_signal_handler);
91
92    p->active = 0;
93  }
94}
95
96static void remove_tempfiles_on_exit(void)
97{
98  remove_tempfiles(0);
99}
100
101static void remove_tempfiles_on_signal(int signo)
102{
103  remove_tempfiles(1);
104  sigchain_pop(signo);
105  raise(signo);
106}

108static struct tempfile *new_tempfile(void)
109{
110  struct tempfile *tempfile = xmalloc(sizeof(*tempfile));
111  tempfile->fd = -1;
112  tempfile->fp = NULL;
113  tempfile->active = 0;
114  tempfile->owner = 0;
115  INIT_LIST_HEAD(&tempfile->list);
116  strbuf_init(&tempfile->filename, 0);
117  tempfile->directorylen = 0;
118  return tempfile;
119}
120
121static void activate_tempfile(struct tempfile *tempfile)
122{
123  static int initialized;
124
125  if (is_tempfile_active(tempfile))
126    BUG("activate_tempfile called for active object");
127
128  if (!initialized) {
129    sigchain_push_common(remove_tempfiles_on_signal);
130    atexit(remove_tempfiles_on_exit);
131    initialized = 1;
132  }
133
134  volatile_list_add(&tempfile->list, &tempfile_list);
135  tempfile->owner = getpid();
136  tempfile->active = 1;
137}
138
139static void deactivate_tempfile(struct tempfile *tempfile)
140{
141  tempfile->active = 0;
142  strbuf_release(&tempfile->filename);
143  volatile_list_del(&tempfile->list);
144  free(tempfile);
145}

147/* Make sure errno contains a meaningful value on error */
148struct tempfile *create_tempfile_mode(const char *path, int mode)
149{
150  struct tempfile *tempfile = new_tempfile();
151
152  strbuf_add_absolute_path(&tempfile->filename, path);
153  tempfile->fd = open(tempfile->filename.buf,
154          O_RDWR | O_CREAT | O_EXCL | O_CLOEXEC, mode);
155  if (O_CLOEXEC && tempfile->fd < 0 && errno == EINVAL)
156    /* Try again w/o O_CLOEXEC: the kernel might not support it */
157    tempfile->fd = open(tempfile->filename.buf,
158            O_RDWR | O_CREAT | O_EXCL, mode);
159  if (tempfile->fd < 0) {
160    deactivate_tempfile(tempfile);
161    return NULL;
162  }
163  activate_tempfile(tempfile);
164  if (adjust_shared_perm(tempfile->filename.buf)) {
165    int save_errno = errno;
166    error("cannot fix permission bits on %s", tempfile->filename.buf);
167    delete_tempfile(&tempfile);
168    errno = save_errno;
169    return NULL;
170  }
171
172  return tempfile;
173}

216struct tempfile *mks_tempfile_dt(const char *directory_template,
217         const char *filename)
218{
219  struct tempfile *tempfile;
220  const char *tmpdir;
221  struct strbuf sb = STRBUF_INIT;
222  int fd;
223  size_t directorylen;
224
225  if (!ends_with(directory_template, "XXXXXX")) {
226    errno = EINVAL;
227    return NULL;
228  }
229
230  tmpdir = getenv("TMPDIR");
231  if (!tmpdir)
232    tmpdir = "/tmp";
233
234  strbuf_addf(&sb, "%s/%s", tmpdir, directory_template);
235  directorylen = sb.len;
236  if (!mkdtemp(sb.buf)) {
237    int orig_errno = errno;
238    strbuf_release(&sb);
239    errno = orig_errno;
240    return NULL;
241  }
242
243  strbuf_addf(&sb, "/%s", filename);
244  fd = open(sb.buf, O_CREAT | O_EXCL | O_RDWR, 0600);
245  if (fd < 0) {
246    int orig_errno = errno;
247    strbuf_setlen(&sb, directorylen);
248    rmdir(sb.buf);
249    strbuf_release(&sb);
250    errno = orig_errno;
251    return NULL;
252  }
253
254  tempfile = new_tempfile();
255  strbuf_swap(&tempfile->filename, &sb);
256  tempfile->directorylen = directorylen;
257  tempfile->fd = fd;
258  activate_tempfile(tempfile);
259  return tempfile;
260}

277FILE *fdopen_tempfile(struct tempfile *tempfile, const char *mode)
278{
279  if (!is_tempfile_active(tempfile))
280    BUG("fdopen_tempfile() called for inactive object");
281  if (tempfile->fp)
282    BUG("fdopen_tempfile() called for open object");
283
284  tempfile->fp = fdopen(tempfile->fd, mode);
285  return tempfile->fp;
286}

309int close_tempfile_gently(struct tempfile *tempfile)
310{
311  int fd;
312  FILE *fp;
313  int err;
314
315  if (!is_tempfile_active(tempfile) || tempfile->fd < 0)
316    return 0;
317
318  fd = tempfile->fd;
319  fp = tempfile->fp;
320  tempfile->fd = -1;
321  if (fp) {
322    tempfile->fp = NULL;
323    if (ferror(fp)) {
324      err = -1;
325      if (!fclose(fp))
326        errno = EIO;
327    } else {
328      err = fclose(fp);
329    }
330  } else {
331    err = close(fd);
332  }
333
334  return err ? -1 : 0;
335}

337int reopen_tempfile(struct tempfile *tempfile)
338{
339  if (!is_tempfile_active(tempfile))
340    BUG("reopen_tempfile called for an inactive object");
341  if (0 <= tempfile->fd)
342    BUG("reopen_tempfile called for an open object");
343  tempfile->fd = open(tempfile->filename.buf, O_WRONLY|O_TRUNC);
344  return tempfile->fd;
345}

347int rename_tempfile(struct tempfile **tempfile_p, const char *path)
348{
349  struct tempfile *tempfile = *tempfile_p;
350
351  if (!is_tempfile_active(tempfile))
352    BUG("rename_tempfile called for inactive object");
353
354  if (close_tempfile_gently(tempfile)) {
355    delete_tempfile(tempfile_p);
356    return -1;
357  }
358
359  if (rename(tempfile->filename.buf, path)) {
360    int save_errno = errno;
361    delete_tempfile(tempfile_p);
362    errno = save_errno;
363    return -1;
364  }
365
366  deactivate_tempfile(tempfile);
367  *tempfile_p = NULL;
368  return 0;
369}
370
371void delete_tempfile(struct tempfile **tempfile_p)
372{
373  struct tempfile *tempfile = *tempfile_p;
374
375  if (!is_tempfile_active(tempfile))
376    return;
377
378  close_tempfile_gently(tempfile);
379  unlink_or_warn(tempfile->filename.buf);
380  remove_template_directory(tempfile, 0);
381  deactivate_tempfile(tempfile);
382  *tempfile_p = NULL;
383}
