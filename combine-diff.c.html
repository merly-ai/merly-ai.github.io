30static struct combine_diff_path *intersect_paths(
31  struct combine_diff_path *curr,
32  int n,
33  int num_parent,
34  int combined_all_paths)
35{
36  struct diff_queue_struct *q = &diff_queued_diff;
37  struct combine_diff_path *p, **tail = &curr;
38  int i, j, cmp;
39
40  if (!n) {
41    for (i = 0; i < q->nr; i++) {
42      int len;
43      const char *path;
44      if (diff_unmodified_pair(q->queue[i]))
45        continue;
46      path = q->queue[i]->two->path;
47      len = strlen(path);
48      p = xmalloc(combine_diff_path_size(num_parent, len));
49      p->path = (char *) &(p->parent[num_parent]);
50      memcpy(p->path, path, len);
51      p->path[len] = 0;
52      p->next = NULL;
53      memset(p->parent, 0,
54             sizeof(p->parent[0]) * num_parent);
55
56      oidcpy(&p->oid, &q->queue[i]->two->oid);
57      p->mode = q->queue[i]->two->mode;
58      oidcpy(&p->parent[n].oid, &q->queue[i]->one->oid);
59      p->parent[n].mode = q->queue[i]->one->mode;
60      p->parent[n].status = q->queue[i]->status;
61
62      if (combined_all_paths &&
63          filename_changed(p->parent[n].status)) {
64        strbuf_init(&p->parent[n].path, 0);
65        strbuf_addstr(&p->parent[n].path,
66                q->queue[i]->one->path);
67      }
68      *tail = p;
69      tail = &p->next;
70    }
71    return curr;
72  }
73
74  /*
75   * paths in curr (linked list) and q->queue[] (array) are
76   * both sorted in the tree order.
77   */
78  i = 0;
79  while ((p = *tail) != NULL) {
80    cmp = ((i >= q->nr)
81           ? -1 : compare_paths(p, q->queue[i]->two));
82
83    if (cmp < 0) {
84      /* p->path not in q->queue[]; drop it */
85      *tail = p->next;
86      for (j = 0; j < num_parent; j++)
87        if (combined_all_paths &&
88            filename_changed(p->parent[j].status))
89          strbuf_release(&p->parent[j].path);
90      free(p);
91      continue;
92    }
93
94    if (cmp > 0) {
95      /* q->queue[i] not in p->path; skip it */
96      i++;
97      continue;
98    }
99
100    oidcpy(&p->parent[n].oid, &q->queue[i]->one->oid);
101    p->parent[n].mode = q->queue[i]->one->mode;
102    p->parent[n].status = q->queue[i]->status;
103    if (combined_all_paths &&
104        filename_changed(p->parent[n].status))
105      strbuf_addstr(&p->parent[n].path,
106              q->queue[i]->one->path);
107
108    tail = &p->next;
109    i++;
110  }
111  return curr;
112}

145static int match_string_spaces(const char *line1, int len1,
146             const char *line2, int len2,
147             long flags)
148{
149  if (flags & XDF_WHITESPACE_FLAGS) {
150    for (; len1 > 0 && XDL_ISSPACE(line1[len1 - 1]); len1--);
151    for (; len2 > 0 && XDL_ISSPACE(line2[len2 - 1]); len2--);
152  }
153
154  if (!(flags & (XDF_IGNORE_WHITESPACE | XDF_IGNORE_WHITESPACE_CHANGE)))
155    return (len1 == len2 && !memcmp(line1, line2, len1));
156
157  while (len1 > 0 && len2 > 0) {
158    len1--;
159    len2--;
160    if (XDL_ISSPACE(line1[len1]) || XDL_ISSPACE(line2[len2])) {
161      if ((flags & XDF_IGNORE_WHITESPACE_CHANGE) &&
162          (!XDL_ISSPACE(line1[len1]) || !XDL_ISSPACE(line2[len2])))
163        return 0;
164
165      for (; len1 > 0 && XDL_ISSPACE(line1[len1]); len1--);
166      for (; len2 > 0 && XDL_ISSPACE(line2[len2]); len2--);
167    }
168    if (line1[len1] != line2[len2])
169      return 0;
170  }
171
172  if (flags & XDF_IGNORE_WHITESPACE) {
173    /* Consume remaining spaces */
174    for (; len1 > 0 && XDL_ISSPACE(line1[len1 - 1]); len1--);
175    for (; len2 > 0 && XDL_ISSPACE(line2[len2 - 1]); len2--);
176  }
177
178  /* We matched full line1 and line2 */
179  if (!len1 && !len2)
180    return 1;
181
182  return 0;
183}

187/* Coalesce new lines into base by finding LCS */
188static struct lline *coalesce_lines(struct lline *base, int *lenbase,
189            struct lline *newline, int lennew,
190            unsigned long parent, long flags)
191{
192  int **lcs;
193  enum coalesce_direction **directions;
194  struct lline *baseend, *newend = NULL;
195  int i, j, origbaselen = *lenbase;
196
197  if (!newline)
198    return base;
199
200  if (!base) {
201    *lenbase = lennew;
202    return newline;
203  }
204
205  /*
206   * Coalesce new lines into base by finding the LCS
207   * - Create the table to run dynamic programming
208   * - Compute the LCS
209   * - Then reverse read the direction structure:
210   *   - If we have MATCH, assign parent to base flag, and consume
211   *   both baseend and newend
212   *   - Else if we have BASE, consume baseend
213   *   - Else if we have NEW, insert newend lline into base and
214   *   consume newend
215   */
216  CALLOC_ARRAY(lcs, st_add(origbaselen, 1));
217  CALLOC_ARRAY(directions, st_add(origbaselen, 1));
218  for (i = 0; i < origbaselen + 1; i++) {
219    CALLOC_ARRAY(lcs[i], st_add(lennew, 1));
220    CALLOC_ARRAY(directions[i], st_add(lennew, 1));
221    directions[i][0] = BASE;
222  }
223  for (j = 1; j < lennew + 1; j++)
224    directions[0][j] = NEW;
225
226  for (i = 1, baseend = base; i < origbaselen + 1; i++) {
227    for (j = 1, newend = newline; j < lennew + 1; j++) {
228      if (match_string_spaces(baseend->line, baseend->len,
229            newend->line, newend->len, flags)) {
230        lcs[i][j] = lcs[i - 1][j - 1] + 1;
231        directions[i][j] = MATCH;
232      } else if (lcs[i][j - 1] >= lcs[i - 1][j]) {
233        lcs[i][j] = lcs[i][j - 1];
234        directions[i][j] = NEW;
235      } else {
236        lcs[i][j] = lcs[i - 1][j];
237        directions[i][j] = BASE;
238      }
239      if (newend->next)
240        newend = newend->next;
241    }
242    if (baseend->next)
243      baseend = baseend->next;
244  }
245
246  for (i = 0; i < origbaselen + 1; i++)
247    free(lcs[i]);
248  free(lcs);
249
250  /* At this point, baseend and newend point to the end of each lists */
251  i--;
252  j--;
253  while (i != 0 || j != 0) {
254    if (directions[i][j] == MATCH) {
255      baseend->parent_map |= 1<<parent;
256      baseend = baseend->prev;
257      newend = newend->prev;
258      i--;
259      j--;
260    } else if (directions[i][j] == NEW) {
261      struct lline *lline;
262
263      lline = newend;
264      /* Remove lline from new list and update newend */
265      if (lline->prev)
266        lline->prev->next = lline->next;
267      else
268        newline = lline->next;
269      if (lline->next)
270        lline->next->prev = lline->prev;
271
272      newend = lline->prev;
273      j--;
274
275      /* Add lline to base list */
276      if (baseend) {
277        lline->next = baseend->next;
278        lline->prev = baseend;
279        if (lline->prev)
280          lline->prev->next = lline;
281      }
282      else {
283        lline->next = base;
284        base = lline;
285      }
286      (*lenbase)++;
287
288      if (lline->next)
289        lline->next->prev = lline;
290
291    } else {
292      baseend = baseend->prev;
293      i--;
294    }
295  }
296
297  newend = newline;
298  while (newend) {
299    struct lline *lline = newend;
300    newend = newend->next;
301    free(lline);
302  }
303
304  for (i = 0; i < origbaselen + 1; i++)
305    free(directions[i]);
306  free(directions);
307
308  return base;
309}

341static void append_lost(struct sline *sline, int n, const char *line, int len)
342{
343  struct lline *lline;
344  unsigned long this_mask = (1UL<<n);
345  if (line[len-1] == '\n')
346    len--;
347
348  FLEX_ALLOC_MEM(lline, line, line, len);
349  lline->len = len;
350  lline->next = NULL;
351  lline->prev = sline->plost.lost_tail;
352  if (lline->prev)
353    lline->prev->next = lline;
354  else
355    sline->plost.lost_head = lline;
356  sline->plost.lost_tail = lline;
357  sline->plost.len++;
358  lline->parent_map = this_mask;
359}

371static void consume_hunk(void *state_,
372       long ob, long on,
373       long nb, long nn,
374       const char *funcline, long funclen)
375{
376  struct combine_diff_state *state = state_;
377
378  state->ob = ob;
379  state->on = on;
380  state->nb = nb;
381  state->nn = nn;
382  state->lno = state->nb;
383  if (state->nn == 0) {
384    /* @@ -X,Y +N,0 @@ removed Y lines
385     * that would have come *after* line N
386     * in the result.  Our lost buckets hang
387     * to the line after the removed lines,
388     *
389     * Note that this is correct even when N == 0,
390     * in which case the hunk removes the first
391     * line in the file.
392     */
393    state->lost_bucket = &state->sline[state->nb];
394    if (!state->nb)
395      state->nb = 1;
396  } else {
397    state->lost_bucket = &state->sline[state->nb-1];
398  }
399  if (!state->sline[state->nb-1].p_lno)
400    CALLOC_ARRAY(state->sline[state->nb - 1].p_lno,
401           state->num_parent);
402  state->sline[state->nb-1].p_lno[state->n] = state->ob;
403}

405static int consume_line(void *state_, char *line, unsigned long len)
406{
407  struct combine_diff_state *state = state_;
408  if (!state->lost_bucket)
409    return 0; /* not in any hunk yet */
410  switch (line[0]) {
411  case '-':
412    append_lost(state->lost_bucket, state->n, line+1, len-1);
413    break;
414  case '+':
415    state->sline[state->lno-1].flag |= state->nmask;
416    state->lno++;
417    break;
418  }
419  return 0;
420}

422static void combine_diff(struct repository *r,
423       const struct object_id *parent, unsigned int mode,
424       mmfile_t *result_file,
425       struct sline *sline, unsigned int cnt, int n,
426       int num_parent, int result_deleted,
427       struct userdiff_driver *textconv,
428       const char *path, long flags)
429{
430  unsigned int p_lno, lno;
431  unsigned long nmask = (1UL << n);
432  xpparam_t xpp;
433  xdemitconf_t xecfg;
434  mmfile_t parent_file;
435  struct combine_diff_state state;
436  unsigned long sz;
437
438  if (result_deleted)
439    return; /* result deleted */
440
441  parent_file.ptr = grab_blob(r, parent, mode, &sz, textconv, path);
442  parent_file.size = sz;
443  memset(&xpp, 0, sizeof(xpp));
444  xpp.flags = flags;
445  memset(&xecfg, 0, sizeof(xecfg));
446  memset(&state, 0, sizeof(state));
447  state.nmask = nmask;
448  state.sline = sline;
449  state.lno = 1;
450  state.num_parent = num_parent;
451  state.n = n;
452
453  if (xdi_diff_outf(&parent_file, result_file, consume_hunk,
454        consume_line, &state, &xpp, &xecfg))
455    die("unable to generate combined diff for %s",
456        oid_to_hex(parent));
457  free(parent_file.ptr);
458
459  /* Assign line numbers for this parent.
460   *
461   * sline[lno].p_lno[n] records the first line number
462   * (counting from 1) for parent N if the final hunk display
463   * started by showing sline[lno] (possibly showing the lost
464   * lines attached to it first).
465   */
466  for (lno = 0,  p_lno = 1; lno <= cnt; lno++) {
467    struct lline *ll;
468    sline[lno].p_lno[n] = p_lno;
469
470    /* Coalesce new lines */
471    if (sline[lno].plost.lost_head) {
472      struct sline *sl = &sline[lno];
473      sl->lost = coalesce_lines(sl->lost, &sl->lenlost,
474              sl->plost.lost_head,
475              sl->plost.len, n, flags);
476      sl->plost.lost_head = sl->plost.lost_tail = NULL;
477      sl->plost.len = 0;
478    }
479
480    /* How many lines would this sline advance the p_lno? */
481    ll = sline[lno].lost;
482    while (ll) {
483      if (ll->parent_map & nmask)
484        p_lno++; /* '-' means parent had it */
485      ll = ll->next;
486    }
487    if (lno < cnt && !(sline[lno].flag & nmask))
488      p_lno++; /* no '+' means parent had it */
489  }
490  sline[lno].p_lno[n] = p_lno; /* trailer */
491}

508{

522             unsigned long mark,
523             unsigned long i,
524             unsigned long cnt,
525             int look_for_uninteresting)
526{
527  /* We have examined up to i-1 and are about to look at i.
528   * Find next interesting or uninteresting line.  Here,
529   * "interesting" does not mean interesting(), but marked by
530   * the give_context() function below (i.e. it includes context
531   * lines that are not interesting to interesting() function
532   * that are surrounded by interesting() ones.
533   */
534  while (i <= cnt)
535    if (look_for_uninteresting
536        ? !(sline[i].flag & mark)
537        : (sline[i].flag & mark))
538      return i;
539    else
540      i++;
541  return i;
542}
543
544static int give_context(struct sline *sline, unsigned long cnt, int num_parent)
545{
546  unsigned long all_mask = (1UL<<num_parent) - 1;
547  unsigned long mark = (1UL<<num_parent);
548  unsigned long no_pre_delete = (2UL<<num_parent);
549  unsigned long i;
550
551  /* Two groups of interesting lines may have a short gap of
552   * uninteresting lines.  Connect such groups to give them a
553   * bit of context.
554   *
555   * We first start from what the interesting() function says,
556   * and mark them with "mark", and paint context lines with the
557   * mark.  So interesting() would still say false for such context
558   * lines but they are treated as "interesting" in the end.
559   */
560  i = find_next(sline, mark, 0, cnt, 0);
561  if (cnt < i)
562    return 0;
563
564  while (i <= cnt) {
565    unsigned long j = (context < i) ? (i - context) : 0;
566    unsigned long k;
567
568    /* Paint a few lines before the first interesting line. */
569    while (j < i) {
570      if (!(sline[j].flag & mark))
571        sline[j].flag |= no_pre_delete;
572      sline[j++].flag |= mark;
573    }
574
575  again:
576    /* we know up to i is to be included.  where does the
577     * next uninteresting one start?
578     */
579    j = find_next(sline, mark, i, cnt, 1);
580    if (cnt < j)
581      break; /* the rest are all interesting */
582
583    /* lookahead context lines */
584    k = find_next(sline, mark, j, cnt, 0);
585    j = adjust_hunk_tail(sline, all_mask, i, j);
586
587    if (k < j + context) {
588      /* k is interesting and [j,k) are not, but
589       * paint them interesting because the gap is small.
590       */
591      while (j < k)
592        sline[j++].flag |= mark;
593      i = k;
594      goto again;
595    }
596
597    /* j is the first uninteresting line and there is
598     * no overlap beyond it within context lines.  Paint
599     * the trailing edge a bit.
600     */
601    i = k;
602    k = (j + context < cnt+1) ? j + context : cnt+1;
603    while (j < k)
604      sline[j++].flag |= mark;
605  }
606  return 1;
607}
608
609static int make_hunks(struct sline *sline, unsigned long cnt,
610           int num_parent, int dense)
611{
612  unsigned long all_mask = (1UL<<num_parent) - 1;
613  unsigned long mark = (1UL<<num_parent);
614  unsigned long i;
615  int has_interesting = 0;
616
617  for (i = 0; i <= cnt; i++) {
618    if (interesting(&sline[i], all_mask))
619      sline[i].flag |= mark;
620    else
621      sline[i].flag &= ~mark;
622  }
623  if (!dense)
624    return give_context(sline, cnt, num_parent);
625
626  /* Look at each hunk, and if we have changes from only one
627   * parent, or the changes are the same from all but one
628   * parent, mark that uninteresting.
629   */
630  i = 0;
631  while (i <= cnt) {
632    unsigned long j, hunk_begin, hunk_end;
633    unsigned long same_diff;
634    while (i <= cnt && !(sline[i].flag & mark))
635      i++;
636    if (cnt < i)
637      break; /* No more interesting hunks */
638    hunk_begin = i;
639    for (j = i + 1; j <= cnt; j++) {
640      if (!(sline[j].flag & mark)) {
641        /* Look beyond the end to see if there
642         * is an interesting line after this
643         * hunk within context span.
644         */
645        unsigned long la; /* lookahead */
646        int contin = 0;
647        la = adjust_hunk_tail(sline, all_mask,
648                 hunk_begin, j);
649        la = (la + context < cnt + 1) ?
650          (la + context) : cnt + 1;
651        while (la && j <= --la) {
652          if (sline[la].flag & mark) {
653            contin = 1;
654            break;
655          }
656        }
657        if (!contin)
658          break;
659        j = la;
660      }
661    }
662    hunk_end = j;
663
664    /* [i..hunk_end) are interesting.  Now is it really
665     * interesting?  We check if there are only two versions
666     * and the result matches one of them.  That is, we look
667     * at:
668     *   (+) line, which records lines added to which parents;
669     *       this line appears in the result.
670     *   (-) line, which records from what parents the line
671     *       was removed; this line does not appear in the result.
672     * then check the set of parents the result has difference
673     * from, from all lines.  If there are lines that has
674     * different set of parents that the result has differences
675     * from, that means we have more than two versions.
676     *
677     * Even when we have only two versions, if the result does
678     * not match any of the parents, the it should be considered
679     * interesting.  In such a case, we would have all '+' line.
680     * After passing the above "two versions" test, that would
681     * appear as "the same set of parents" to be "all parents".
682     */
683    same_diff = 0;
684    has_interesting = 0;
685    for (j = i; j < hunk_end && !has_interesting; j++) {
686      unsigned long this_diff = sline[j].flag & all_mask;
687      struct lline *ll = sline[j].lost;
688      if (this_diff) {
689        /* This has some changes.  Is it the
690         * same as others?
691         */
692        if (!same_diff)
693          same_diff = this_diff;
694        else if (same_diff != this_diff) {
695          has_interesting = 1;
696          break;
697        }
698      }
699      while (ll && !has_interesting) {
700        /* Lost this line from these parents;
701         * who are they?  Are they the same?
702         */
703        this_diff = ll->parent_map;
704        if (!same_diff)
705          same_diff = this_diff;
706        else if (same_diff != this_diff) {
707          has_interesting = 1;
708        }
709        ll = ll->next;
710      }
711    }
712
713    if (!has_interesting && same_diff != all_mask) {
714      /* This hunk is not that interesting after all */
715      for (j = hunk_begin; j < hunk_end; j++)
716        sline[j].flag &= ~mark;
717    }
718    i = hunk_end;
719  }
720
721  has_interesting = give_context(sline, cnt, num_parent);
722  return has_interesting;
723}

754static void dump_sline(struct sline *sline, const char *line_prefix,
755           unsigned long cnt, int num_parent,
756           int use_color, int result_deleted)
757{
758  unsigned long mark = (1UL<<num_parent);
759  unsigned long no_pre_delete = (2UL<<num_parent);
760  int i;
761  unsigned long lno = 0;
762  const char *c_frag = diff_get_color(use_color, DIFF_FRAGINFO);
763  const char *c_func = diff_get_color(use_color, DIFF_FUNCINFO);
764  const char *c_new = diff_get_color(use_color, DIFF_FILE_NEW);
765  const char *c_old = diff_get_color(use_color, DIFF_FILE_OLD);
766  const char *c_context = diff_get_color(use_color, DIFF_CONTEXT);
767  const char *c_reset = diff_get_color(use_color, DIFF_RESET);
768
769  if (result_deleted)
770    return; /* result deleted */
771
772  while (1) {
773    unsigned long hunk_end;
774    unsigned long rlines;
775    const char *hunk_comment = NULL;
776    unsigned long null_context = 0;
777
778    while (lno <= cnt && !(sline[lno].flag & mark)) {
779      if (hunk_comment_line(sline[lno].bol))
780        hunk_comment = sline[lno].bol;
781      lno++;
782    }
783    if (cnt < lno)
784      break;
785    else {
786      for (hunk_end = lno + 1; hunk_end <= cnt; hunk_end++)
787        if (!(sline[hunk_end].flag & mark))
788          break;
789    }
790    rlines = hunk_end - lno;
791    if (cnt < hunk_end)
792      rlines--; /* pointing at the last delete hunk */
793
794    if (!context) {
795      /*
796       * Even when running with --unified=0, all
797       * lines in the hunk needs to be processed in
798       * the loop below in order to show the
799       * deletion recorded in lost_head.  However,
800       * we do not want to show the resulting line
801       * with all blank context markers in such a
802       * case.  Compensate.
803       */
804      unsigned long j;
805      for (j = lno; j < hunk_end; j++)
806        if (!(sline[j].flag & (mark-1)))
807          null_context++;
808      rlines -= null_context;
809    }
810
811    printf("%s%s", line_prefix, c_frag);
812    for (i = 0; i <= num_parent; i++) putchar(combine_marker);
813    for (i = 0; i < num_parent; i++)
814      show_parent_lno(sline, lno, hunk_end, i, null_context);
815    printf(" +%lu,%lu ", lno+1, rlines);
816    for (i = 0; i <= num_parent; i++) putchar(combine_marker);
817
818    if (hunk_comment) {
819      int comment_end = 0;
820      for (i = 0; i < 40; i++) {
821        int ch = hunk_comment[i] & 0xff;
822        if (!ch || ch == '\n')
823          break;
824        if (!isspace(ch))
825            comment_end = i;
826      }
827      if (comment_end)
828        printf("%s%s %s%s", c_reset,
829                c_context, c_reset,
830                c_func);
831      for (i = 0; i < comment_end; i++)
832        putchar(hunk_comment[i]);
833    }
834
835    printf("%s\n", c_reset);
836    while (lno < hunk_end) {
837      struct lline *ll;
838      int j;
839      unsigned long p_mask;
840      struct sline *sl = &sline[lno++];
841      ll = (sl->flag & no_pre_delete) ? NULL : sl->lost;
842      while (ll) {
843        printf("%s%s", line_prefix, c_old);
844        for (j = 0; j < num_parent; j++) {
845          if (ll->parent_map & (1UL<<j))
846            putchar('-');
847          else
848            putchar(' ');
849        }
850        show_line_to_eol(ll->line, -1, c_reset);
851        ll = ll->next;
852      }
853      if (cnt < lno)
854        break;
855      p_mask = 1;
856      fputs(line_prefix, stdout);
857      if (!(sl->flag & (mark-1))) {
858        /*
859         * This sline was here to hang the
860         * lost lines in front of it.
861         */
862        if (!context)
863          continue;
864        fputs(c_context, stdout);
865      }
866      else
867        fputs(c_new, stdout);
868      for (j = 0; j < num_parent; j++) {
869        if (p_mask & sl->flag)
870          putchar('+');
871        else
872          putchar(' ');
873        p_mask <<= 1;
874      }
875      show_line_to_eol(sl->bol, sl->len, c_reset);
876    }
877  }
878}

880static void reuse_combine_diff(struct sline *sline, unsigned long cnt,
881             int i, int j)
882{
883  /* We have already examined parent j and we know parent i
884   * and parent j are the same, so reuse the combined result
885   * of parent j for parent i.
886   */
887  unsigned long lno, imask, jmask;
888  imask = (1UL<<i);
889  jmask = (1UL<<j);
890
891  for (lno = 0; lno <= cnt; lno++) {
892    struct lline *ll = sline->lost;
893    sline->p_lno[i] = sline->p_lno[j];
894    while (ll) {
895      if (ll->parent_map & jmask)
896        ll->parent_map |= imask;
897      ll = ll->next;
898    }
899    if (sline->flag & jmask)
900      sline->flag |= imask;
901    sline++;
902  }
903  /* the overall size of the file (sline[cnt]) */
904  sline->p_lno[i] = sline->p_lno[j];
905}

924static void show_combined_header(struct combine_diff_path *elem,
925         int num_parent,
926         struct rev_info *rev,
927         const char *line_prefix,
928         int mode_differs,
929         int show_file_header)
930{
931  struct diff_options *opt = &rev->diffopt;
932  int abbrev = opt->flags.full_index ? the_hash_algo->hexsz : DEFAULT_ABBREV;
933  const char *a_prefix = opt->a_prefix ? opt->a_prefix : "a/";
934  const char *b_prefix = opt->b_prefix ? opt->b_prefix : "b/";
935  const char *c_meta = diff_get_color_opt(opt, DIFF_METAINFO);
936  const char *c_reset = diff_get_color_opt(opt, DIFF_RESET);
937  const char *abb;
938  int added = 0;
939  int deleted = 0;
940  int i;
941  int dense = rev->dense_combined_merges;
942
943  if (rev->loginfo && !rev->no_commit_id)
944    show_log(rev);
945
946  dump_quoted_path(dense ? "diff --cc " : "diff --combined ",
947       "", elem->path, line_prefix, c_meta, c_reset);
948  printf("%s%sindex ", line_prefix, c_meta);
949  for (i = 0; i < num_parent; i++) {
950    abb = find_unique_abbrev(&elem->parent[i].oid,
951           abbrev);
952    printf("%s%s", i ? "," : "", abb);
953  }
954  abb = find_unique_abbrev(&elem->oid, abbrev);
955  printf("..%s%s\n", abb, c_reset);
956
957  if (mode_differs) {
958    deleted = !elem->mode;
959
960    /* We say it was added if nobody had it */
961    added = !deleted;
962    for (i = 0; added && i < num_parent; i++)
963      if (elem->parent[i].status !=
964          DIFF_STATUS_ADDED)
965        added = 0;
966    if (added)
967      printf("%s%snew file mode %06o",
968             line_prefix, c_meta, elem->mode);
969    else {
970      if (deleted)
971        printf("%s%sdeleted file ",
972               line_prefix, c_meta);
973      printf("mode ");
974      for (i = 0; i < num_parent; i++) {
975        printf("%s%06o", i ? "," : "",
976               elem->parent[i].mode);
977      }
978      if (elem->mode)
979        printf("..%06o", elem->mode);
980    }
981    printf("%s\n", c_reset);
982  }
983
984  if (!show_file_header)
985    return;
986
987  if (rev->combined_all_paths) {
988    for (i = 0; i < num_parent; i++) {
989      char *path = filename_changed(elem->parent[i].status)
990        ? elem->parent[i].path.buf : elem->path;
991      if (elem->parent[i].status == DIFF_STATUS_ADDED)
992        dump_quoted_path("--- ", "", "/dev/null",
993             line_prefix, c_meta, c_reset);
994      else
995        dump_quoted_path("--- ", a_prefix, path,
996             line_prefix, c_meta, c_reset);
997    }
998  } else {
999    if (added)
1000      dump_quoted_path("--- ", "", "/dev/null",
1001           line_prefix, c_meta, c_reset);
1002    else
1003      dump_quoted_path("--- ", a_prefix, elem->path,
1004           line_prefix, c_meta, c_reset);
1005  }
1006  if (deleted)
1007    dump_quoted_path("+++ ", "", "/dev/null",

1207             line_prefix, mode_differs, 1);

1209         opt->use_color, result_deleted);

1211  free(result);
1212
1213  for (lno = 0; lno < cnt; lno++) {

1217        struct lline *tmp = ll;

1500  if (opt->ignore_regex_nr)
1501    die("combined diff and '%s' cannot be used together",
1502        "--ignore-matching-lines");
1503  if (opt->close_file)
1504    die("combined diff and '%s' cannot be used together",
1505        "--output");
1506
1507  /* nothing to do, if no parents */
1508  if (!num_parent)
1509    return;
1510
1511  show_log_first = !!rev->loginfo && !rev->no_commit_id;
1512  needsep = 0;
1513  if (show_log_first) {
1514    show_log(rev);
1515
1516    if (rev->verbose_header && opt->output_format &&
1517        opt->output_format != DIFF_FORMAT_NO_OUTPUT &&
1518        !commit_format_is_empty(rev->commit_format))
1519      printf("%s%c", diff_line_prefix(opt),
1520             opt->line_termination);
1521  }
1522
1523  diffopts = *opt;
1524  copy_pathspec(&diffopts.pathspec, &opt->pathspec);
1525  diffopts.flags.recursive = 1;
1526  diffopts.flags.allow_external = 0;
1527
1528  /* find set of paths that everybody touches
1529   *
1530   * NOTE
1531   *
1532   * Diffcore transformations are bound to diff_filespec and logic
1533   * comparing two entries - i.e. they do not apply directly to combine
1534   * diff.
1535   *
1536   * If some of such transformations is requested - we launch generic
1537   * path scanning, which works significantly slower compared to
1538   * simultaneous all-trees-in-one-go scan in find_paths_multitree().
1539   *
1540   * TODO some of the filters could be ported to work on
1541   * combine_diff_paths - i.e. all functionality that skips paths, so in
1542   * theory, we could end up having only multitree path scanning.
1543   *
1544   * NOTE please keep this semantically in sync with diffcore_std()
1545   */
1546  need_generic_pathscan = opt->skip_stat_unmatch  ||
1547      opt->flags.follow_renames  ||
1548      opt->break_opt != -1  ||
1549      opt->detect_rename  ||
1550      (opt->pickaxe_opts &
1551       (DIFF_PICKAXE_KINDS_MASK & ~DIFF_PICKAXE_KIND_OBJFIND)) ||
1552      opt->filter;
1553
1554  if (need_generic_pathscan) {
1555    /*
1556     * NOTE generic case also handles --stat, as it computes
1557     * diff(sha1,parent_i) for all i to do the job, specifically
1558     * for parent0.
1559     */
1560    paths = find_paths_generic(oid, parents, &diffopts,
1561             rev->combined_all_paths);
1562  }
1563  else {
1564    int stat_opt;
1565    paths = find_paths_multitree(oid, parents, &diffopts);
1566
1567    if (opt->pickaxe_opts & DIFF_PICKAXE_KIND_OBJFIND)
1568      paths = combined_objfind(opt, paths, num_parent);
1569
1570    /*
1571     * show stat against the first parent even
1572     * when doing combined diff.
1573     */
1574    stat_opt = opt->output_format & STAT_FORMAT_MASK;
1575    if (stat_opt) {
1576      diffopts.output_format = stat_opt;
1577
1578      diff_tree_oid(&parents->oid[0], oid, "", &diffopts);
1579      diffcore_std(&diffopts);
1580      if (opt->orderfile)
1581        diffcore_order(opt->orderfile);
1582      diff_flush(&diffopts);
1583    }
1584  }
1585
1586  /* find out number of surviving paths */
1587  for (num_paths = 0, p = paths; p; p = p->next)
1588    num_paths++;
1589
1590  /* order paths according to diffcore_order */
1591  if (opt->orderfile && num_paths) {
1592    struct obj_order *o;
1593
1594    ALLOC_ARRAY(o, num_paths);
1595    for (i = 0, p = paths; p; p = p->next, i++)
1596      o[i].obj = p;
1597    order_objects(opt->orderfile, path_path, o, num_paths);
1598    for (i = 0; i < num_paths - 1; i++) {
1599      p = o[i].obj;
1600      p->next = o[i+1].obj;
1601    }
1602
1603    p = o[num_paths-1].obj;
1604    p->next = NULL;
1605    paths = o[0].obj;
1606    free(o);
1607  }
1608
1609
1610  if (num_paths) {
1611    if (opt->output_format & (DIFF_FORMAT_RAW |
1612            DIFF_FORMAT_NAME |
1613            DIFF_FORMAT_NAME_STATUS)) {
1614      for (p = paths; p; p = p->next)
1615        show_raw_diff(p, num_parent, rev);
1616      needsep = 1;
1617    }
1618    else if (opt->output_format & STAT_FORMAT_MASK)
1619      needsep = 1;
1620    else if (opt->output_format & DIFF_FORMAT_CALLBACK)
1621      handle_combined_callback(opt, paths, num_parent, num_paths);
1622
1623    if (opt->output_format & DIFF_FORMAT_PATCH) {
1624      if (needsep)
1625        printf("%s%c", diff_line_prefix(opt),
1626               opt->line_termination);
1627      for (p = paths; p; p = p->next)
1628        show_patch_diff(p, num_parent, 0, rev);
1629    }
1630  }
1631
1632  /* Clean things up */
1633  while (paths) {
1634    struct combine_diff_path *tmp = paths;
1635    paths = paths->next;
1636    for (i = 0; i < num_parent; i++)
1637      if (rev->combined_all_paths &&
1638          filename_changed(tmp->parent[i].status))
1639        strbuf_release(&tmp->parent[i].path);
1640    free(tmp);
1641  }
1642
1643  clear_pathspec(&diffopts.pathspec);
1644}
1645
1646void diff_tree_combined_merge(const struct commit *commit,
1647            struct rev_info *rev)
1648{
1649  struct commit_list *parent = get_saved_parents(rev, commit);
1650  struct oid_array parents = OID_ARRAY_INIT;
1651
1652  while (parent) {
1653    oid_array_append(&parents, &parent->item->object.oid);
1654    parent = parent->next;
1655  }
1656  diff_tree_combined(&commit->object.oid, &parents, rev);
1657  oid_array_clear(&parents);
1658}
