13static int append_normalized_escapes(struct strbuf *buf,
14             const char *from,
15             size_t from_len,
16             const char *esc_extra,
17             const char *esc_ok)
18{
19  /*
20   * Append to strbuf 'buf' characters from string 'from' with length
21   * 'from_len' while unescaping characters that do not need to be escaped
22   * and escaping characters that do.  The set of characters to escape
23   * (the complement of which is unescaped) starts out as the RFC 3986
24   * unsafe characters (0x00-0x1F,0x7F-0xFF," <>\"#%{}|\\^`").  If
25   * 'esc_extra' is not NULL, those additional characters will also always
26   * be escaped.  If 'esc_ok' is not NULL, those characters will be left
27   * escaped if found that way, but will not be unescaped otherwise (used
28   * for delimiters).  If a %-escape sequence is encountered that is not
29   * followed by 2 hexadecimal digits, the sequence is invalid and
30   * false (0) will be returned.  Otherwise true (1) will be returned for
31   * success.
32   *
33   * Note that all %-escape sequences will be normalized to UPPERCASE
34   * as indicated in RFC 3986.  Unless included in esc_extra or esc_ok
35   * alphanumerics and "-._~" will always be unescaped as per RFC 3986.
36   */
37
38  while (from_len) {
39    int ch = *from++;
40    int was_esc = 0;
41
42    from_len--;
43    if (ch == '%') {
44      if (from_len < 2)
45        return 0;
46      ch = hex2chr(from);
47      if (ch < 0)
48        return 0;
49      from += 2;
50      from_len -= 2;
51      was_esc = 1;
52    }
53    if ((unsigned char)ch <= 0x1F || (unsigned char)ch >= 0x7F ||
54        strchr(URL_UNSAFE_CHARS, ch) ||
55        (esc_extra && strchr(esc_extra, ch)) ||
56        (was_esc && strchr(esc_ok, ch)))
57      strbuf_addf(buf, "%%%02X", (unsigned char)ch);
58    else
59      strbuf_addch(buf, ch);
60  }
61
62  return 1;
63}

73static int match_host(const struct url_info *url_info,
74          const struct url_info *pattern_info)
75{
76  const char *url = url_info->url + url_info->host_off;
77  const char *pat = pattern_info->url + pattern_info->host_off;
78  int url_len = url_info->host_len;
79  int pat_len = pattern_info->host_len;
80
81  while (url_len && pat_len) {
82    const char *url_next = end_of_token(url, '.', url_len);
83    const char *pat_next = end_of_token(pat, '.', pat_len);
84
85    if (pat_next == pat + 1 && pat[0] == '*')
86      /* wildcard matches anything */
87      ;
88    else if ((pat_next - pat) == (url_next - url) &&
89       !memcmp(url, pat, url_next - url))
90      /* the components are the same */
91      ;
92    else
93      return 0; /* found an unmatch */
94
95    if (url_next < url + url_len)
96      url_next++;
97    url_len -= url_next - url;
98    url = url_next;
99    if (pat_next < pat + pat_len)
100      pat_next++;
101    pat_len -= pat_next - pat;
102    pat = pat_next;
103  }
104
105  return (!url_len && !pat_len);
106}

108static char *url_normalize_1(const char *url, struct url_info *out_info, char allow_globs)
109{
110  /*
111   * Normalize NUL-terminated url using the following rules:
112   *
113   * 1. Case-insensitive parts of url will be converted to lower case
114   * 2. %-encoded characters that do not need to be will be unencoded
115   * 3. Characters that are not %-encoded and must be will be encoded
116   * 4. All %-encodings will be converted to upper case hexadecimal
117   * 5. Leading 0s are removed from port numbers
118   * 6. If the default port for the scheme is given it will be removed
119   * 7. A path part (including empty) not starting with '/' has one added
120   * 8. Any dot segments (. or ..) in the path are resolved and removed
121   * 9. IPv6 host literals are allowed (but not normalized or validated)
122   *
123   * The rules are based on information in RFC 3986.
124   *
125   * Please note this function requires a full URL including a scheme
126   * and host part (except for file: URLs which may have an empty host).
127   *
128   * The return value is a newly allocated string that must be freed
129   * or NULL if the url is not valid.
130   *
131   * If out_info is non-NULL, the url and err fields therein will always
132   * be set.  If a non-NULL value is returned, it will be stored in
133   * out_info->url as well, out_info->err will be set to NULL and the
134   * other fields of *out_info will also be filled in.  If a NULL value
135   * is returned, NULL will be stored in out_info->url and out_info->err
136   * will be set to a brief, translated, error message, but no other
137   * fields will be filled in.
138   *
139   * This is NOT a URL validation function.  Full URL validation is NOT
140   * performed.  Some invalid host names are passed through this function
141   * undetected.  However, most all other problems that make a URL invalid
142   * will be detected (including a missing host for non file: URLs).
143   */
144
145  size_t url_len = strlen(url);
146  struct strbuf norm;
147  size_t spanned;
148  size_t scheme_len, user_off=0, user_len=0, passwd_off=0, passwd_len=0;
149  size_t host_off=0, host_len=0, port_off=0, port_len=0, path_off, path_len, result_len;
150  const char *slash_ptr, *at_ptr, *colon_ptr, *path_start;
151  char *result;
152
153  /*
154   * Copy lowercased scheme and :// suffix, %-escapes are not allowed
155   * First character of scheme must be URL_ALPHA
156   */
157  spanned = strspn(url, URL_SCHEME_CHARS);
158  if (!spanned || !isalpha(url[0]) || spanned + 3 > url_len ||
159      url[spanned] != ':' || url[spanned+1] != '/' || url[spanned+2] != '/') {
160    if (out_info) {
161      out_info->url = NULL;
162      out_info->err = _("invalid URL scheme name or missing '://' suffix");
163    }
164    return NULL; /* Bad scheme and/or missing "://" part */
165  }
166  strbuf_init(&norm, url_len);
167  scheme_len = spanned;
168  spanned += 3;
169  url_len -= spanned;
170  while (spanned--)
171    strbuf_addch(&norm, tolower(*url++));
172
173
174  /*
175   * Copy any username:password if present normalizing %-escapes
176   */
177  at_ptr = strchr(url, '@');
178  slash_ptr = url + strcspn(url, "/?#");
179  if (at_ptr && at_ptr < slash_ptr) {
180    user_off = norm.len;
181    if (at_ptr > url) {
182      if (!append_normalized_escapes(&norm, url, at_ptr - url,
183                   "", URL_RESERVED)) {
184        if (out_info) {
185          out_info->url = NULL;
186          out_info->err = _("invalid %XX escape sequence");
187        }
188        strbuf_release(&norm);
189        return NULL;
190      }
191      colon_ptr = strchr(norm.buf + scheme_len + 3, ':');
192      if (colon_ptr) {
193        passwd_off = (colon_ptr + 1) - norm.buf;
194        passwd_len = norm.len - passwd_off;
195        user_len = (passwd_off - 1) - (scheme_len + 3);
196      } else {
197        user_len = norm.len - (scheme_len + 3);
198      }
199    }
200    strbuf_addch(&norm, '@');
201    url_len -= (++at_ptr - url);
202    url = at_ptr;
203  }
204
205
206  /*
207   * Copy the host part excluding any port part, no %-escapes allowed
208   */
209  if (!url_len || strchr(":/?#", *url)) {
210    /* Missing host invalid for all URL schemes except file */
211    if (strncmp(norm.buf, "file:", 5)) {
212      if (out_info) {
213        out_info->url = NULL;
214        out_info->err = _("missing host and scheme is not 'file:'");
215      }
216      strbuf_release(&norm);
217      return NULL;
218    }
219  } else {
220    host_off = norm.len;
221  }
222  colon_ptr = slash_ptr - 1;
223  while (colon_ptr > url && *colon_ptr != ':' && *colon_ptr != ']')
224    colon_ptr--;
225  if (*colon_ptr != ':') {
226    colon_ptr = slash_ptr;
227  } else if (!host_off && colon_ptr < slash_ptr && colon_ptr + 1 != slash_ptr) {
228    /* file: URLs may not have a port number */
229    if (out_info) {
230      out_info->url = NULL;
231      out_info->err = _("a 'file:' URL may not have a port number");
232    }
233    strbuf_release(&norm);
234    return NULL;
235  }
236
237  if (allow_globs)
238    spanned = strspn(url, URL_HOST_CHARS "*");
239  else
240    spanned = strspn(url, URL_HOST_CHARS);
241
242  if (spanned < colon_ptr - url) {
243    /* Host name has invalid characters */
244    if (out_info) {
245      out_info->url = NULL;
246      out_info->err = _("invalid characters in host name");
247    }
248    strbuf_release(&norm);
249    return NULL;
250  }
251  while (url < colon_ptr) {
252    strbuf_addch(&norm, tolower(*url++));
253    url_len--;
254  }
255
256
257  /*
258   * Check the port part and copy if not the default (after removing any
259   * leading 0s); no %-escapes allowed
260   */
261  if (colon_ptr < slash_ptr) {
262    /* skip the ':' and leading 0s but not the last one if all 0s */
263    url++;
264    url += strspn(url, "0");
265    if (url == slash_ptr && url[-1] == '0')
266      url--;
267    if (url == slash_ptr) {
268      /* Skip ":" port with no number, it's same as default */
269    } else if (slash_ptr - url == 2 &&
270         !strncmp(norm.buf, "http:", 5) &&
271         !strncmp(url, "80", 2)) {
272      /* Skip http :80 as it's the default */
273    } else if (slash_ptr - url == 3 &&
274         !strncmp(norm.buf, "https:", 6) &&
275         !strncmp(url, "443", 3)) {
276      /* Skip https :443 as it's the default */
277    } else {
278      /*
279       * Port number must be all digits with leading 0s removed
280       * and since all the protocols we deal with have a 16-bit
281       * port number it must also be in the range 1..65535
282       * 0 is not allowed because that means "next available"
283       * on just about every system and therefore cannot be used
284       */
285      unsigned long pnum = 0;
286      spanned = strspn(url, URL_DIGIT);
287      if (spanned < slash_ptr - url) {
288        /* port number has invalid characters */
289        if (out_info) {
290          out_info->url = NULL;
291          out_info->err = _("invalid port number");
292        }
293        strbuf_release(&norm);
294        return NULL;
295      }
296      if (slash_ptr - url <= 5)
297        pnum = strtoul(url, NULL, 10);
298      if (pnum == 0 || pnum > 65535) {
299        /* port number not in range 1..65535 */
300        if (out_info) {
301          out_info->url = NULL;
302          out_info->err = _("invalid port number");
303        }
304        strbuf_release(&norm);
305        return NULL;
306      }
307      strbuf_addch(&norm, ':');
308      port_off = norm.len;
309      strbuf_add(&norm, url, slash_ptr - url);
310      port_len = slash_ptr - url;
311    }
312    url_len -= slash_ptr - colon_ptr;
313    url = slash_ptr;
314  }
315  if (host_off)
316    host_len = norm.len - host_off - (port_len ? port_len + 1 : 0);
317
318
319  /*
320   * Now copy the path resolving any . and .. segments being careful not
321   * to corrupt the URL by unescaping any delimiters, but do add an
322   * initial '/' if it's missing and do normalize any %-escape sequences.
323   */
324  path_off = norm.len;
325  path_start = norm.buf + path_off;
326  strbuf_addch(&norm, '/');
327  if (*url == '/') {
328    url++;
329    url_len--;
330  }
331  for (;;) {
332    const char *seg_start;
333    size_t seg_start_off = norm.len;
334    const char *next_slash = url + strcspn(url, "/?#");
335    int skip_add_slash = 0;
336
337    /*
338     * RFC 3689 indicates that any . or .. segments should be
339     * unescaped before being checked for.
340     */
341    if (!append_normalized_escapes(&norm, url, next_slash - url, "",
342                 URL_RESERVED)) {
343      if (out_info) {
344        out_info->url = NULL;
345        out_info->err = _("invalid %XX escape sequence");
346      }
347      strbuf_release(&norm);
348      return NULL;
349    }
350
351    seg_start = norm.buf + seg_start_off;
352    if (!strcmp(seg_start, ".")) {
353      /* ignore a . segment; be careful not to remove initial '/' */
354      if (seg_start == path_start + 1) {
355        strbuf_setlen(&norm, norm.len - 1);
356        skip_add_slash = 1;
357      } else {
358        strbuf_setlen(&norm, norm.len - 2);
359      }
360    } else if (!strcmp(seg_start, "..")) {
361      /*
362       * ignore a .. segment and remove the previous segment;
363       * be careful not to remove initial '/' from path
364       */
365      const char *prev_slash = norm.buf + norm.len - 3;
366      if (prev_slash == path_start) {
367        /* invalid .. because no previous segment to remove */
368        if (out_info) {
369          out_info->url = NULL;
370          out_info->err = _("invalid '..' path segment");
371        }
372        strbuf_release(&norm);
373        return NULL;
374      }
375      while (*--prev_slash != '/') {}
376      if (prev_slash == path_start) {
377        strbuf_setlen(&norm, prev_slash - norm.buf + 1);
378        skip_add_slash = 1;
379      } else {
380        strbuf_setlen(&norm, prev_slash - norm.buf);
381      }
382    }
383    url_len -= next_slash - url;
384    url = next_slash;
385    /* if the next char is not '/' done with the path */
386    if (*url != '/')
387      break;
388    url++;
389    url_len--;
390    if (!skip_add_slash)
391      strbuf_addch(&norm, '/');
392  }
393  path_len = norm.len - path_off;
394
395
396  /*
397   * Now simply copy the rest, if any, only normalizing %-escapes and
398   * being careful not to corrupt the URL by unescaping any delimiters.
399   */
400  if (*url) {
401    if (!append_normalized_escapes(&norm, url, url_len, "", URL_RESERVED)) {
402      if (out_info) {
403        out_info->url = NULL;
404        out_info->err = _("invalid %XX escape sequence");
405      }
406      strbuf_release(&norm);
407      return NULL;
408    }
409  }
410
411
412  result = strbuf_detach(&norm, &result_len);
413  if (out_info) {
414    out_info->url = result;
415    out_info->err = NULL;
416    out_info->url_len = result_len;
417    out_info->scheme_len = scheme_len;
418    out_info->user_off = user_off;
419    out_info->user_len = user_len;
420    out_info->passwd_off = passwd_off;
421    out_info->passwd_len = passwd_len;
422    out_info->host_off = host_off;
423    out_info->host_len = host_len;
424    out_info->port_off = port_off;
425    out_info->port_len = port_len;
426    out_info->path_off = path_off;
427    out_info->path_len = path_len;
428  }
429  return result;
430}

437static size_t url_match_prefix(const char *url,
438             const char *url_prefix,
439             size_t url_prefix_len)
440{
441  /*
442   * url_prefix matches url if url_prefix is an exact match for url or it
443   * is a prefix of url and the match ends on a path component boundary.
444   * Both url and url_prefix are considered to have an implicit '/' on the
445   * end for matching purposes if they do not already.
446   *
447   * url must be NUL terminated.  url_prefix_len is the length of
448   * url_prefix which need not be NUL terminated.
449   *
450   * The return value is the length of the match in characters (including
451   * the final '/' even if it's implicit) or 0 for no match.
452   *
453   * Passing NULL as url and/or url_prefix will always cause 0 to be
454   * returned without causing any faults.
455   */
456  if (!url || !url_prefix)
457    return 0;
458  if (!url_prefix_len || (url_prefix_len == 1 && *url_prefix == '/'))
459    return (!*url || *url == '/') ? 1 : 0;
460  if (url_prefix[url_prefix_len - 1] == '/')
461    url_prefix_len--;
462  if (strncmp(url, url_prefix, url_prefix_len))
463    return 0;
464  if ((strlen(url) == url_prefix_len) || (url[url_prefix_len] == '/'))
465    return url_prefix_len + 1;
466  return 0;
467}
468
469static int match_urls(const struct url_info *url,
470          const struct url_info *url_prefix,
471          struct urlmatch_item *match)
472{
473  /*
474   * url_prefix matches url if the scheme, host and port of url_prefix
475   * are the same as those of url and the path portion of url_prefix
476   * is the same as the path portion of url or it is a prefix that
477   * matches at a '/' boundary.  If url_prefix contains a user name,
478   * that must also exactly match the user name in url.
479   *
480   * If the user, host, port and path match in this fashion, the returned
481   * value is the length of the path match including any implicit
482   * final '/'.  For example, "http://me@example.com/path" is matched by
483   * "http://example.com" with a path length of 1.
484   *
485   * If there is a match and exactusermatch is not NULL, then
486   * *exactusermatch will be set to true if both url and url_prefix
487   * contained a user name or false if url_prefix did not have a
488   * user name.  If there is no match *exactusermatch is left untouched.
489   */
490  char usermatched = 0;
491  size_t pathmatchlen;
492
493  if (!url || !url_prefix || !url->url || !url_prefix->url)
494    return 0;
495
496  /* check the scheme */
497  if (url_prefix->scheme_len != url->scheme_len ||
498      strncmp(url->url, url_prefix->url, url->scheme_len))
499    return 0; /* schemes do not match */
500
501  /* check the user name if url_prefix has one */
502  if (url_prefix->user_off) {
503    if (!url->user_off || url->user_len != url_prefix->user_len ||
504        strncmp(url->url + url->user_off,
505          url_prefix->url + url_prefix->user_off,
506          url->user_len))
507      return 0; /* url_prefix has a user but it's not a match */
508    usermatched = 1;
509  }
510
511  /* check the host */
512  if (!match_host(url, url_prefix))
513    return 0; /* host names do not match */
514
515  /* check the port */
516  if (url_prefix->port_len != url->port_len ||
517      strncmp(url->url + url->port_off,
518        url_prefix->url + url_prefix->port_off, url->port_len))
519    return 0; /* ports do not match */
520
521  /* check the path */
522  pathmatchlen = url_match_prefix(
523    url->url + url->path_off,
524    url_prefix->url + url_prefix->path_off,
525    url_prefix->url_len - url_prefix->path_off);
526  if (!pathmatchlen)
527    return 0; /* paths do not match */
528
529  if (match) {
530    match->hostmatch_len = url_prefix->host_len;
531    match->pathmatch_len = pathmatchlen;
532    match->user_matched = usermatched;
533  }
534
535  return 1;
536}
537
538static int cmp_matches(const struct urlmatch_item *a,
539           const struct urlmatch_item *b)
540{
541  if (a->hostmatch_len != b->hostmatch_len)
542    return a->hostmatch_len < b->hostmatch_len ? -1 : 1;
543  if (a->pathmatch_len != b->pathmatch_len)
544    return a->pathmatch_len < b->pathmatch_len ? -1 : 1;
545  if (a->user_matched != b->user_matched)
546    return b->user_matched ? -1 : 1;
547  return 0;
548}

550int urlmatch_config_entry(const char *var, const char *value, void *cb)
551{
552  struct string_list_item *item;
553  struct urlmatch_config *collect = cb;
554  struct urlmatch_item matched = {0};
555  struct url_info *url = &collect->url;
556  const char *key, *dot;
557  struct strbuf synthkey = STRBUF_INIT;
558  int retval;
559  int (*select_fn)(const struct urlmatch_item *a, const struct urlmatch_item *b) =
560    collect->select_fn ? collect->select_fn : cmp_matches;
561
562  if (!skip_prefix(var, collect->section, &key) || *(key++) != '.') {
563    if (collect->cascade_fn)
564      return collect->cascade_fn(var, value, cb);
565    return 0; /* not interested */
566  }
567  dot = strrchr(key, '.');
568  if (dot) {
569    char *config_url, *norm_url;
570    struct url_info norm_info;
571
572    config_url = xmemdupz(key, dot - key);
573    norm_url = url_normalize_1(config_url, &norm_info, 1);
574    if (norm_url)
575      retval = match_urls(url, &norm_info, &matched);
576    else if (collect->fallback_match_fn)
577      retval = collect->fallback_match_fn(config_url,
578                  collect->cb);
579    else
580      retval = 0;
581    free(config_url);
582    free(norm_url);
583    if (!retval)
584      return 0;
585    key = dot + 1;
586  }
587
588  if (collect->key && strcmp(key, collect->key))
589    return 0;
590
591  item = string_list_insert(&collect->vars, key);
592  if (!item->util) {
593    item->util = xcalloc(1, sizeof(matched));
594  } else {
595    if (select_fn(&matched, item->util) < 0)
596       /*
597        * Our match is worse than the old one,
598        * we cannot use it.
599        */
600      return 0;
601    /* Otherwise, replace it with this one. */
602  }
603
604  memcpy(item->util, &matched, sizeof(matched));
605  strbuf_addstr(&synthkey, collect->section);
606  strbuf_addch(&synthkey, '.');
607  strbuf_addstr(&synthkey, key);
608  retval = collect->collect_fn(synthkey.buf, value, collect->cb);
609
610  strbuf_release(&synthkey);
611  return retval;
612}

614void urlmatch_config_release(struct urlmatch_config *config)
615{
616  string_list_clear(&config->vars, 1);
617}
