17#define FILTER_BUFFER (1024*16)
18
19struct filtered_istream {
20  struct git_istream *upstream;
21  struct stream_filter *filter;
22  char ibuf[FILTER_BUFFER];
23  char obuf[FILTER_BUFFER];
24  int i_end, i_ptr;
25  int o_end, o_ptr;
26  int input_finished;
27};
28
29struct git_istream {
30  open_istream_fn open;
31  close_istream_fn close;
32  read_istream_fn read;
33
34  unsigned long size; /* inflated size of full object */
35  git_zstream z;
36  enum { z_unused, z_used, z_done, z_error } z_state;
37
38  union {
39    struct {
40      char *buf; /* from read_object() */
41      unsigned long read_ptr;
42    } incore;
43
44    struct {
45      void *mapped;
46      unsigned long mapsize;
47      char hdr[32];
48      int hdr_avail;
49      int hdr_used;
50    } loose;
51
52    struct {
53      struct packed_git *pack;
54      off_t pos;
55    } in_pack;
56
57    struct filtered_istream filtered;
58  } u;
59};

86static ssize_t read_istream_filtered(struct git_istream *st, char *buf,
87             size_t sz)
88{
89  struct filtered_istream *fs = &(st->u.filtered);
90  size_t filled = 0;
91
92  while (sz) {
93    /* do we already have filtered output? */
94    if (fs->o_ptr < fs->o_end) {
95      size_t to_move = fs->o_end - fs->o_ptr;
96      if (sz < to_move)
97        to_move = sz;
98      memcpy(buf + filled, fs->obuf + fs->o_ptr, to_move);
99      fs->o_ptr += to_move;
100      sz -= to_move;
101      filled += to_move;
102      continue;
103    }
104    fs->o_end = fs->o_ptr = 0;
105
106    /* do we have anything to feed the filter with? */
107    if (fs->i_ptr < fs->i_end) {
108      size_t to_feed = fs->i_end - fs->i_ptr;
109      size_t to_receive = FILTER_BUFFER;
110      if (stream_filter(fs->filter,
111            fs->ibuf + fs->i_ptr, &to_feed,
112            fs->obuf, &to_receive))
113        return -1;
114      fs->i_ptr = fs->i_end - to_feed;
115      fs->o_end = FILTER_BUFFER - to_receive;
116      continue;
117    }
118
119    /* tell the filter to drain upon no more input */
120    if (fs->input_finished) {
121      size_t to_receive = FILTER_BUFFER;
122      if (stream_filter(fs->filter,
123            NULL, NULL,
124            fs->obuf, &to_receive))
125        return -1;
126      fs->o_end = FILTER_BUFFER - to_receive;
127      if (!fs->o_end)
128        break;
129      continue;
130    }
131    fs->i_end = fs->i_ptr = 0;
132
133    /* refill the input from the upstream */
134    if (!fs->input_finished) {
135      fs->i_end = read_istream(fs->upstream, fs->ibuf, FILTER_BUFFER);
136      if (fs->i_end < 0)
137        return -1;
138      if (fs->i_end)
139        continue;
140    }
141    fs->input_finished = 1;
142  }
143  return filled;
144}

146static struct git_istream *attach_stream_filter(struct git_istream *st,
147            struct stream_filter *filter)
148{
149  struct git_istream *ifs = xmalloc(sizeof(*ifs));
150  struct filtered_istream *fs = &(ifs->u.filtered);
151
152  ifs->close = close_istream_filtered;
153  ifs->read = read_istream_filtered;
154  fs->upstream = st;
155  fs->filter = filter;
156  fs->i_end = fs->i_ptr = 0;
157  fs->o_end = fs->o_ptr = 0;
158  fs->input_finished = 0;
159  ifs->size = -1; /* unknown */
160  return ifs;
161}

163/*****************************************************************

165 * Loose object stream

169static ssize_t read_istream_loose(struct git_istream *st, char *buf, size_t sz)
170{
171  size_t total_read = 0;
172
173  switch (st->z_state) {
174  case z_done:
175    return 0;
176  case z_error:
177    return -1;
178  default:
179    break;
180  }
181
182  if (st->u.loose.hdr_used < st->u.loose.hdr_avail) {
183    size_t to_copy = st->u.loose.hdr_avail - st->u.loose.hdr_used;
184    if (sz < to_copy)
185      to_copy = sz;
186    memcpy(buf, st->u.loose.hdr + st->u.loose.hdr_used, to_copy);
187    st->u.loose.hdr_used += to_copy;
188    total_read += to_copy;
189  }
190
191  while (total_read < sz) {
192    int status;
193
194    st->z.next_out = (unsigned char *)buf + total_read;
195    st->z.avail_out = sz - total_read;
196    status = git_inflate(&st->z, Z_FINISH);
197
198    total_read = st->z.next_out - (unsigned char *)buf;
199
200    if (status == Z_STREAM_END) {
201      git_inflate_end(&st->z);
202      st->z_state = z_done;
203      break;
204    }
205    if (status != Z_OK && (status != Z_BUF_ERROR || total_read < sz)) {
206      git_inflate_end(&st->z);
207      st->z_state = z_error;
208      return -1;
209    }
210  }
211  return total_read;
212}

214static int close_istream_loose(struct git_istream *st)
215{
216  close_deflated_stream(st);
217  munmap(st->u.loose.mapped, st->u.loose.mapsize);
218  return 0;
219}

221static int open_istream_loose(struct git_istream *st, struct repository *r,
222            const struct object_id *oid,
223            enum object_type *type)
224{
225  struct object_info oi = OBJECT_INFO_INIT;
226  oi.sizep = &st->size;
227  oi.typep = type;
228
229  st->u.loose.mapped = map_loose_object(r, oid, &st->u.loose.mapsize);
230  if (!st->u.loose.mapped)
231    return -1;
232  switch (unpack_loose_header(&st->z, st->u.loose.mapped,
233            st->u.loose.mapsize, st->u.loose.hdr,
234            sizeof(st->u.loose.hdr), NULL)) {
235  case ULHR_OK:
236    break;
237  case ULHR_BAD:
238  case ULHR_TOO_LONG:
239    goto error;
240  }
241  if (parse_loose_header(st->u.loose.hdr, &oi) < 0 || *type < 0)
242    goto error;
243
244  st->u.loose.hdr_used = strlen(st->u.loose.hdr) + 1;
245  st->u.loose.hdr_avail = st->z.total_out;
246  st->z_state = z_used;
247  st->close = close_istream_loose;
248  st->read = read_istream_loose;
249
250  return 0;
251error:
252  git_inflate_end(&st->z);
253  munmap(st->u.loose.mapped, st->u.loose.mapsize);
254  return -1;
255}

258/*****************************************************************
259 *
260 * Non-delta packed object stream
261 *
262 *****************************************************************/
263
264static ssize_t read_istream_pack_non_delta(struct git_istream *st, char *buf,
265             size_t sz)
266{
267  size_t total_read = 0;
268
269  switch (st->z_state) {
270  case z_unused:
271    memset(&st->z, 0, sizeof(st->z));
272    git_inflate_init(&st->z);
273    st->z_state = z_used;
274    break;
275  case z_done:
276    return 0;
277  case z_error:
278    return -1;
279  case z_used:
280    break;
281  }
282
283  while (total_read < sz) {
284    int status;
285    struct pack_window *window = NULL;
286    unsigned char *mapped;
287
288    mapped = use_pack(st->u.in_pack.pack, &window,
289          st->u.in_pack.pos, &st->z.avail_in);
290
291    st->z.next_out = (unsigned char *)buf + total_read;
292    st->z.avail_out = sz - total_read;
293    st->z.next_in = mapped;
294    status = git_inflate(&st->z, Z_FINISH);
295
296    st->u.in_pack.pos += st->z.next_in - mapped;
297    total_read = st->z.next_out - (unsigned char *)buf;
298    unuse_pack(&window);
299
300    if (status == Z_STREAM_END) {
301      git_inflate_end(&st->z);
302      st->z_state = z_done;
303      break;
304    }
305
306    /*
307     * Unlike the loose object case, we do not have to worry here
308     * about running out of input bytes and spinning infinitely. If
309     * we get Z_BUF_ERROR due to too few input bytes, then we'll
310     * replenish them in the next use_pack() call when we loop. If
311     * we truly hit the end of the pack (i.e., because it's corrupt
312     * or truncated), then use_pack() catches that and will die().
313     */
314    if (status != Z_OK && status != Z_BUF_ERROR) {
315      git_inflate_end(&st->z);
316      st->z_state = z_error;
317      return -1;
318    }
319  }
320  return total_read;
321}

323static int close_istream_pack_non_delta(struct git_istream *st)
324{
325  close_deflated_stream(st);
326  return 0;

329static int open_istream_pack_non_delta(struct git_istream *st,
330               struct repository *r,
331               const struct object_id *oid,
332               enum object_type *type)
333{
334  struct pack_window *window;
335  enum object_type in_pack_type;
336
337  window = NULL;
338
339  in_pack_type = unpack_object_header(st->u.in_pack.pack,
340              &window,
341              &st->u.in_pack.pos,
342              &st->size);
343  unuse_pack(&window);
344  switch (in_pack_type) {
345  default:
346    return -1; /* we do not do deltas for now */
347  case OBJ_COMMIT:
348  case OBJ_TREE:
349  case OBJ_BLOB:
350  case OBJ_TAG:
351    break;
352  }
353  st->z_state = z_unused;
354  st->close = close_istream_pack_non_delta;
355  st->read = read_istream_pack_non_delta;
356
357  return 0;
358}

373static ssize_t read_istream_incore(struct git_istream *st, char *buf, size_t sz)
374{
375  size_t read_size = sz;
376  size_t remainder = st->size - st->u.incore.read_ptr;
377
378  if (remainder <= read_size)
379    read_size = remainder;
380  if (read_size) {
381    memcpy(buf, st->u.incore.buf + st->u.incore.read_ptr, read_size);
382    st->u.incore.read_ptr += read_size;
383  }
384  return read_size;
385}

393  st->read = read_istream_incore;
394
395  return st->u.incore.buf ? 0 : -1;

398/*****************************************************************************
399 * static helpers variables and functions for users of streaming interface

401
402static int istream_source(struct git_istream *st,
403        struct repository *r,
404        const struct object_id *oid,
405        enum object_type *type)
406{
407  unsigned long size;
408  int status;
409  struct object_info oi = OBJECT_INFO_INIT;
410
411  oi.typep = type;
412  oi.sizep = &size;
413  status = oid_object_info_extended(r, oid, &oi, 0);
414  if (status < 0)
415    return status;
416
417  switch (oi.whence) {
418  case OI_LOOSE:
419    st->open = open_istream_loose;
420    return 0;
421  case OI_PACKED:
422    if (!oi.u.packed.is_delta && big_file_threshold < size) {
423      st->u.in_pack.pack = oi.u.packed.pack;
424      st->u.in_pack.pos = oi.u.packed.offset;
425      st->open = open_istream_pack_non_delta;
426      return 0;
427    }
428    /* fallthru */
429  default:
430    st->open = open_istream_incore;
431    return 0;
432  }
433}

451struct git_istream *open_istream(struct repository *r,
452         const struct object_id *oid,
453         enum object_type *type,
454         unsigned long *size,
455         struct stream_filter *filter)
456{
457  struct git_istream *st = xmalloc(sizeof(*st));
458  const struct object_id *real = lookup_replace_object(r, oid);
459  int ret = istream_source(st, r, real, type);
460
461  if (ret) {
462    free(st);
463    return NULL;
464  }
465
466  if (st->open(st, r, real, type)) {
467    if (open_istream_incore(st, r, real, type)) {
468      free(st);
469      return NULL;
470    }
471  }
472  if (filter) {
473    /* Add "&& !is_null_stream_filter(filter)" for performance */
474    struct git_istream *nst = attach_stream_filter(st, filter);
475    if (!nst) {
476      close_istream(st);
477      return NULL;
478    }
479    st = nst;
480  }
481
482  *size = st->size;
483  return st;
484}

486int stream_blob_to_fd(int fd, const struct object_id *oid, struct stream_filter *filter,
487          int can_seek)
488{
489  struct git_istream *st;
490  enum object_type type;
491  unsigned long sz;
492  ssize_t kept = 0;
493  int result = -1;
494
495  st = open_istream(the_repository, oid, &type, &sz, filter);
496  if (!st) {
497    if (filter)
498      free_stream_filter(filter);
499    return result;
500  }
501  if (type != OBJ_BLOB)
502    goto close_and_exit;
503  for (;;) {
504    char buf[1024 * 16];
505    ssize_t wrote, holeto;
506    ssize_t readlen = read_istream(st, buf, sizeof(buf));
507
508    if (readlen < 0)
509      goto close_and_exit;
510    if (!readlen)
511      break;
512    if (can_seek && sizeof(buf) == readlen) {
513      for (holeto = 0; holeto < readlen; holeto++)
514        if (buf[holeto])
515          break;
516      if (readlen == holeto) {
517        kept += holeto;
518        continue;
519      }
520    }
521
522    if (kept && lseek(fd, kept, SEEK_CUR) == (off_t) -1)
523      goto close_and_exit;
524    else
525      kept = 0;
526    wrote = write_in_full(fd, buf, readlen);
527
528    if (wrote < 0)
529      goto close_and_exit;
530  }
531  if (kept && (lseek(fd, kept - 1, SEEK_CUR) == (off_t) -1 ||
532         xwrite(fd, "", 1) != 1))
533    goto close_and_exit;
534  result = 0;
535
536 close_and_exit:
537  close_istream(st);
538  return result;
539}
