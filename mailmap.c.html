58/*
59 * On some systems (e.g. MinGW 4.0), string.h has _only_ inline
60 * definition of strcasecmp and no non-inline implementation is
61 * supplied anywhere, which is, eh, "unusual"; we cannot take an
62 * address of such a function to store it in namemap.cmp.  This is
63 * here as a workaround---do not assign strcasecmp directly to
64 * namemap.cmp until we know no systems that matter have such an
65 * "unusual" string.h.
66 */
67static int namemap_cmp(const char *a, const char *b)
68{
69  return strcasecmp(a, b);
70}

72static void add_mapping(struct string_list *map,
73      char *new_name, char *new_email,
74      char *old_name, char *old_email)
75{
76  struct mailmap_entry *me;
77  struct string_list_item *item;
78
79  if (!old_email) {
80    old_email = new_email;
81    new_email = NULL;
82  }
83
84  item = string_list_insert(map, old_email);
85  if (item->util) {
86    me = (struct mailmap_entry *)item->util;
87  } else {
88    CALLOC_ARRAY(me, 1);
89    me->namemap.strdup_strings = 1;
90    me->namemap.cmp = namemap_cmp;
91    item->util = me;
92  }
93
94  if (!old_name) {
95    debug_mm("mailmap: adding (simple) entry for '%s'\n", old_email);
96
97    /* Replace current name and new email for simple entry */
98    if (new_name) {
99      free(me->name);
100      me->name = xstrdup(new_name);
101    }
102    if (new_email) {
103      free(me->email);
104      me->email = xstrdup(new_email);
105    }
106  } else {
107    struct mailmap_info *mi = xcalloc(1, sizeof(struct mailmap_info));
108    debug_mm("mailmap: adding (complex) entry for '%s'\n", old_email);
109    mi->name = xstrdup_or_null(new_name);
110    mi->email = xstrdup_or_null(new_email);
111    string_list_insert(&me->namemap, old_name)->util = mi;
112  }
113
114  debug_mm("mailmap:  '%s' <%s> -> '%s' <%s>\n",
115     debug_str(old_name), old_email,
116     debug_str(new_name), debug_str(new_email));
117}

119static char *parse_name_and_email(char *buffer, char **name,
120          char **email, int allow_empty_email)
121{
122  char *left, *right, *nstart, *nend;
123  *name = *email = NULL;
124
125  if (!(left = strchr(buffer, '<')))
126    return NULL;
127  if (!(right = strchr(left + 1, '>')))
128    return NULL;
129  if (!allow_empty_email && (left+1 == right))
130    return NULL;
131
132  /* remove whitespace from beginning and end of name */
133  nstart = buffer;
134  while (isspace(*nstart) && nstart < left)
135    ++nstart;
136  nend = left-1;
137  while (nend > nstart && isspace(*nend))
138    --nend;
139
140  *name = (nstart <= nend ? nstart : NULL);
141  *email = left+1;
142  *(nend+1) = '\0';
143  *right++ = '\0';
144
145  return (*right == '\0' ? NULL : right);
146}

148static void read_mailmap_line(struct string_list *map, char *buffer)
149{
150  char *name1 = NULL, *email1 = NULL, *name2 = NULL, *email2 = NULL;
151
152  if (buffer[0] == '#')
153    return;
154
155  if ((name2 = parse_name_and_email(buffer, &name1, &email1, 0)))
156    parse_name_and_email(name2, &name2, &email2, 1);
157
158  if (email1)
159    add_mapping(map, name1, email1, name2, email2);
160}

163#define MAILMAP_NOFOLLOW (1<<0)

165static int read_mailmap_file(struct string_list *map, const char *filename,
166           unsigned flags)
167{
168  char buffer[1024];
169  FILE *f;
170  int fd;
171
172  if (!filename)
173    return 0;
174
175  if (flags & MAILMAP_NOFOLLOW)
176    fd = open_nofollow(filename, O_RDONLY);
177  else
178    fd = open(filename, O_RDONLY);
179
180  if (fd < 0) {
181    if (errno == ENOENT)
182      return 0;
183    return error_errno("unable to open mailmap at %s", filename);
184  }
185  f = xfdopen(fd, "r");
186
187  while (fgets(buffer, sizeof(buffer), f) != NULL)
188    read_mailmap_line(map, buffer);
189  fclose(f);
190  return 0;
191}

193static void read_mailmap_string(struct string_list *map, char *buf)
194{
195  while (*buf) {
196    char *end = strchrnul(buf, '\n');
197
198    if (*end)
199      *end++ = '\0';
200
201    read_mailmap_line(map, buf);
202    buf = end;
203  }
204}

206static int read_mailmap_blob(struct string_list *map, const char *name)
207{
208  struct object_id oid;
209  char *buf;
210  unsigned long size;
211  enum object_type type;
212
213  if (!name)
214    return 0;
215  if (get_oid(name, &oid) < 0)
216    return 0;
217
218  buf = read_object_file(&oid, &type, &size);
219  if (!buf)

222    return error("mailmap is not a blob: %s", name);

259/*
260 * Look for an entry in map that match string[0:len]; string[len]
261 * does not have to be NUL (but it could be).
262 */
263static struct string_list_item *lookup_prefix(struct string_list *map,
264                const char *string, size_t len)
265{
266  int i = string_list_find_insert_index(map, string, 1);
267  if (i < 0) {
268    /* exact match */
269    i = -1 - i;
270    if (!string[len])
271      return &map->items[i];
272    /*
273     * that map entry matches exactly to the string, including
274     * the cruft at the end beyond "len".  That is not a match
275     * with string[0:len] that we are looking for.
276     */
277  } else if (!string[len]) {
278    /*
279     * asked with the whole string, and got nothing.  No
280     * matching entry can exist in the map.
281     */
282    return NULL;
283  }
284
285  /*
286   * i is at the exact match to an overlong key, or location the
287   * overlong key would be inserted, which must come after the
288   * real location of the key if one exists.
289   */
290  while (0 <= --i && i < map->nr) {
291    int cmp = strncasecmp(map->items[i].string, string, len);
292    if (cmp < 0)
293      /*
294       * "i" points at a key definitely below the prefix;
295       * the map does not have string[0:len] in it.
296       */
297      break;
298    else if (!cmp && !map->items[i].string[len])
299      /* found it */
300      return &map->items[i];
301    /*
302     * otherwise, the string at "i" may be string[0:len]
303     * followed by a string that sorts later than string[len:];
304     * keep trying.
305     */
306  }
307  return NULL;
308}

310int map_user(struct string_list *map,
311       const char **email, size_t *emaillen,
312       const char **name, size_t *namelen)
313{
314  struct string_list_item *item;
315  struct mailmap_entry *me;
316
317  debug_mm("map_user: map '%.*s' <%.*s>\n",
318     (int)*namelen, debug_str(*name),
319     (int)*emaillen, debug_str(*email));
320
321  item = lookup_prefix(map, *email, *emaillen);
322  if (item) {
323    me = (struct mailmap_entry *)item->util;
324    if (me->namemap.nr) {
325      /*
326       * The item has multiple items, so we'll look up on
327       * name too. If the name is not found, we choose the
328       * simple entry.
329       */
330      struct string_list_item *subitem;
331      subitem = lookup_prefix(&me->namemap, *name, *namelen);
332      if (subitem)
333        item = subitem;
334    }
335  }
336  if (item) {
337    struct mailmap_info *mi = (struct mailmap_info *)item->util;
338    if (mi->name == NULL && mi->email == NULL) {
339      debug_mm("map_user:  -- (no simple mapping)\n");
340      return 0;
341    }
342    if (mi->email) {
343        *email = mi->email;
344        *emaillen = strlen(*email);
345    }
346    if (mi->name) {
347        *name = mi->name;
348        *namelen = strlen(*name);
349    }
350    debug_mm("map_user:  to '%.*s' <%.*s>\n",
351       (int)*namelen, debug_str(*name),
352       (int)*emaillen, debug_str(*email));
353    return 1;
354  }
355  debug_mm("map_user:  --\n");
356  return 0;
357}
