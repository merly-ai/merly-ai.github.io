9/*
10 * This is like mktime, but without normalization of tm_wday and tm_yday.
11 */
12time_t tm_to_time_t(const struct tm *tm)
13{
14  static const int mdays[] = {
15      0, 31, 59, 90, 120, 151, 181, 212, 243, 273, 304, 334
16  };
17  int year = tm->tm_year - 70;
18  int month = tm->tm_mon;
19  int day = tm->tm_mday;
20
21  if (year < 0 || year > 129) /* algo only works for 1970-2099 */
22    return -1;
23  if (month < 0 || month > 11) /* array bounds */
24    return -1;
25  if (month < 2 || (year + 2) % 4)
26    day--;
27  if (tm->tm_hour < 0 || tm->tm_min < 0 || tm->tm_sec < 0)
28    return -1;
29  return (year * 365 + (year + 1) / 4 + mdays[month] + day) * 24*60*60UL +
30    tm->tm_hour * 60*60 + tm->tm_min * 60 + tm->tm_sec;
31}

42static time_t gm_time_t(timestamp_t time, int tz)
43{
44  int minutes;
45
46  minutes = tz < 0 ? -tz : tz;
47  minutes = (minutes / 100)*60 + (minutes % 100);
48  minutes = tz < 0 ? -minutes : minutes;
49
50  if (minutes > 0) {
51    if (unsigned_add_overflows(time, minutes * 60))
52      die("Timestamp+tz too large: %"PRItime" +%04d",
53          time, tz);
54  } else if (time < -minutes * 60)
55    die("Timestamp before Unix epoch: %"PRItime" %04d", time, tz);
56  time += minutes * 60;
57  if (date_overflows(time))
58    die("Timestamp too large for this system: %"PRItime, time);
59  return (time_t)time;
60}
61
62/*
63 * The "tz" thing is passed in as this strange "decimal parse of tz"
64 * thing, which means that tz -0100 is passed in as the integer -100,
65 * even though it means "sixty minutes off"
66 */
67static struct tm *time_to_tm(timestamp_t time, int tz, struct tm *tm)
68{
69  time_t t = gm_time_t(time, tz);
70  return gmtime_r(&t, tm);
71}

79/*
80 * Fill in the localtime 'struct tm' for the supplied time,
81 * and return the local tz.
82 */
83static int local_time_tzoffset(time_t t, struct tm *tm)
84{
85  time_t t_local;
86  int offset, eastwest;
87
88  localtime_r(&t, tm);
89  t_local = tm_to_time_t(tm);
90  if (t_local == -1)
91    return 0; /* error; just use +0000 */
92  if (t_local < t) {
93    eastwest = -1;
94    offset = t - t_local;
95  } else {
96    eastwest = 1;
97    offset = t_local - t;
98  }
99  offset /= 60; /* in minutes */
100  offset = (offset % 60) + ((offset / 60) * 100);
101  return offset * eastwest;
102}

104/*
105 * What value of "tz" was in effect back then at "time" in the
106 * local timezone?
107 */
108static int local_tzoffset(timestamp_t time)
109{
110  struct tm tm;
111
112  if (date_overflows(time))
113    die("Timestamp too large for this system: %"PRItime, time);
114
115  return local_time_tzoffset((time_t)time, &tm);
116}

131void show_date_relative(timestamp_t time, struct strbuf *timebuf)
132{
133  struct timeval now;
134  timestamp_t diff;
135
136  get_time(&now);
137  if (now.tv_sec < time) {
138    strbuf_addstr(timebuf, _("in the future"));
139    return;
140  }
141  diff = now.tv_sec - time;
142  if (diff < 90) {
143    strbuf_addf(timebuf,
144       Q_("%"PRItime" second ago", "%"PRItime" seconds ago", diff), diff);
145    return;
146  }
147  /* Turn it into minutes */
148  diff = (diff + 30) / 60;
149  if (diff < 90) {
150    strbuf_addf(timebuf,
151       Q_("%"PRItime" minute ago", "%"PRItime" minutes ago", diff), diff);
152    return;
153  }
154  /* Turn it into hours */
155  diff = (diff + 30) / 60;
156  if (diff < 36) {
157    strbuf_addf(timebuf,
158       Q_("%"PRItime" hour ago", "%"PRItime" hours ago", diff), diff);
159    return;
160  }
161  /* We deal with number of days from here on */
162  diff = (diff + 12) / 24;
163  if (diff < 14) {
164    strbuf_addf(timebuf,
165       Q_("%"PRItime" day ago", "%"PRItime" days ago", diff), diff);
166    return;
167  }
168  /* Say weeks for the past 10 weeks or so */
169  if (diff < 70) {
170    strbuf_addf(timebuf,
171       Q_("%"PRItime" week ago", "%"PRItime" weeks ago", (diff + 3) / 7),
172       (diff + 3) / 7);
173    return;
174  }
175  /* Say months for the past 12 months or so */
176  if (diff < 365) {
177    strbuf_addf(timebuf,
178       Q_("%"PRItime" month ago", "%"PRItime" months ago", (diff + 15) / 30),
179       (diff + 15) / 30);
180    return;
181  }
182  /* Give years and months for 5 years or so */
183  if (diff < 1825) {
184    timestamp_t totalmonths = (diff * 12 * 2 + 365) / (365 * 2);
185    timestamp_t years = totalmonths / 12;
186    timestamp_t months = totalmonths % 12;
187    if (months) {
188      struct strbuf sb = STRBUF_INIT;
189      strbuf_addf(&sb, Q_("%"PRItime" year", "%"PRItime" years", years), years);
190      strbuf_addf(timebuf,
191         /* TRANSLATORS: "%s" is "<n> years" */
192         Q_("%s, %"PRItime" month ago", "%s, %"PRItime" months ago", months),
193         sb.buf, months);
194      strbuf_release(&sb);
195    } else
196      strbuf_addf(timebuf,
197         Q_("%"PRItime" year ago", "%"PRItime" years ago", years), years);
198    return;
199  }
200  /* Otherwise, just years. Centuries is probably overkill. */
201  strbuf_addf(timebuf,
202     Q_("%"PRItime" year ago", "%"PRItime" years ago", (diff + 183) / 365),
203     (diff + 183) / 365);
204}

215static void show_date_normal(struct strbuf *buf, timestamp_t time, struct tm *tm, int tz, struct tm *human_tm, int human_tz, int local)
216{
217  struct {
218    unsigned int  year:1,
219        date:1,
220        wday:1,
221        time:1,
222        seconds:1,
223        tz:1;
224  } hide = { 0 };
225
226  hide.tz = local || tz == human_tz;
227  hide.year = tm->tm_year == human_tm->tm_year;
228  if (hide.year) {
229    if (tm->tm_mon == human_tm->tm_mon) {
230      if (tm->tm_mday > human_tm->tm_mday) {
231        /* Future date: think timezones */
232      } else if (tm->tm_mday == human_tm->tm_mday) {
233        hide.date = hide.wday = 1;
234      } else if (tm->tm_mday + 5 > human_tm->tm_mday) {
235        /* Leave just weekday if it was a few days ago */
236        hide.date = 1;
237      }
238    }
239  }
240
241  /* Show "today" times as just relative times */
242  if (hide.wday) {
243    show_date_relative(time, buf);
244    return;
245  }
246
247  /*
248   * Always hide seconds for human-readable.
249   * Hide timezone if showing date.
250   * Hide weekday and time if showing year.
251   *
252   * The logic here is two-fold:
253   *  (a) only show details when recent enough to matter
254   *  (b) keep the maximum length "similar", and in check
255   */
256  if (human_tm->tm_year) {
257    hide.seconds = 1;
258    hide.tz |= !hide.date;
259    hide.wday = hide.time = !hide.year;
260  }
261
262  if (!hide.wday)
263    strbuf_addf(buf, "%.3s ", weekday_names[tm->tm_wday]);
264  if (!hide.date)
265    strbuf_addf(buf, "%.3s %d ", month_names[tm->tm_mon], tm->tm_mday);
266
267  /* Do we want AM/PM depending on locale? */
268  if (!hide.time) {
269    strbuf_addf(buf, "%02d:%02d", tm->tm_hour, tm->tm_min);
270    if (!hide.seconds)
271      strbuf_addf(buf, ":%02d", tm->tm_sec);
272  } else
273    strbuf_rtrim(buf);
274
275  if (!hide.year)
276    strbuf_addf(buf, " %d", tm->tm_year + 1900);
277
278  if (!hide.tz)
279    strbuf_addf(buf, " %+05d", tz);
280}

282const char *show_date(timestamp_t time, int tz, const struct date_mode *mode)
283{
284  struct tm *tm;
285  struct tm tmbuf = { 0 };
286  struct tm human_tm = { 0 };
287  int human_tz = -1;
288  static struct strbuf timebuf = STRBUF_INIT;
289
290  if (mode->type == DATE_UNIX) {
291    strbuf_reset(&timebuf);
292    strbuf_addf(&timebuf, "%"PRItime, time);
293    return timebuf.buf;
294  }
295
296  if (mode->type == DATE_HUMAN) {
297    struct timeval now;
298
299    get_time(&now);
300
301    /* Fill in the data for "current time" in human_tz and human_tm */
302    human_tz = local_time_tzoffset(now.tv_sec, &human_tm);
303  }
304
305  if (mode->local)
306    tz = local_tzoffset(time);
307
308  if (mode->type == DATE_RAW) {
309    strbuf_reset(&timebuf);
310    strbuf_addf(&timebuf, "%"PRItime" %+05d", time, tz);
311    return timebuf.buf;
312  }
313
314  if (mode->type == DATE_RELATIVE) {
315    strbuf_reset(&timebuf);
316    show_date_relative(time, &timebuf);
317    return timebuf.buf;
318  }
319
320  if (mode->local)
321    tm = time_to_tm_local(time, &tmbuf);
322  else
323    tm = time_to_tm(time, tz, &tmbuf);
324  if (!tm) {
325    tm = time_to_tm(0, 0, &tmbuf);
326    tz = 0;
327  }
328
329  strbuf_reset(&timebuf);
330  if (mode->type == DATE_SHORT)
331    strbuf_addf(&timebuf, "%04d-%02d-%02d", tm->tm_year + 1900,
332        tm->tm_mon + 1, tm->tm_mday);
333  else if (mode->type == DATE_ISO8601)
334    strbuf_addf(&timebuf, "%04d-%02d-%02d %02d:%02d:%02d %+05d",
335        tm->tm_year + 1900,
336        tm->tm_mon + 1,
337        tm->tm_mday,
338        tm->tm_hour, tm->tm_min, tm->tm_sec,
339        tz);
340  else if (mode->type == DATE_ISO8601_STRICT) {
341    char sign = (tz >= 0) ? '+' : '-';
342    tz = abs(tz);
343    strbuf_addf(&timebuf, "%04d-%02d-%02dT%02d:%02d:%02d%c%02d:%02d",
344        tm->tm_year + 1900,
345        tm->tm_mon + 1,
346        tm->tm_mday,
347        tm->tm_hour, tm->tm_min, tm->tm_sec,
348        sign, tz / 100, tz % 100);
349  } else if (mode->type == DATE_RFC2822)
350    strbuf_addf(&timebuf, "%.3s, %d %.3s %d %02d:%02d:%02d %+05d",
351      weekday_names[tm->tm_wday], tm->tm_mday,
352      month_names[tm->tm_mon], tm->tm_year + 1900,
353      tm->tm_hour, tm->tm_min, tm->tm_sec, tz);
354  else if (mode->type == DATE_STRFTIME)
355    strbuf_addftime(&timebuf, mode->strftime_fmt, tm, tz,
356        !mode->local);
357  else
358    show_date_normal(&timebuf, time, tm, tz, &human_tm, human_tz, mode->local);
359  return timebuf.buf;
360}

421static int match_string(const char *date, const char *str)
422{
423  int i = 0;
424
425  for (i = 0; *date; date++, str++, i++) {
426    if (*date == *str)
427      continue;
428    if (toupper(*date) == toupper(*str))
429      continue;
430    if (!isalnum(*date))
431      break;
432    return 0;
433  }
434  return i;
435}

446/*
447* Parse month, weekday, or timezone name
448*/
449static int match_alpha(const char *date, struct tm *tm, int *offset)
450{
451  int i;
452
453  for (i = 0; i < 12; i++) {
454    int match = match_string(date, month_names[i]);
455    if (match >= 3) {
456      tm->tm_mon = i;
457      return match;
458    }
459  }
460
461  for (i = 0; i < 7; i++) {
462    int match = match_string(date, weekday_names[i]);
463    if (match >= 3) {
464      tm->tm_wday = i;
465      return match;
466    }
467  }
468
469  for (i = 0; i < ARRAY_SIZE(timezone_names); i++) {
470    int match = match_string(date, timezone_names[i].name);
471    if (match >= 3 || match == strlen(timezone_names[i].name)) {
472      int off = timezone_names[i].offset;
473
474      /* This is bogus, but we like summer */
475      off += timezone_names[i].dst;
476
477      /* Only use the tz name offset if we don't have anything better */
478      if (*offset == -1)
479        *offset = 60*off;
480
481      return match;
482    }
483  }
484
485  if (match_string(date, "PM") == 2) {
486    tm->tm_hour = (tm->tm_hour % 12) + 12;
487    return 2;
488  }
489
490  if (match_string(date, "AM") == 2) {
491    tm->tm_hour = (tm->tm_hour % 12) + 0;
492    return 2;
493  }
494
495  /* BAD CRAP */
496  return skip_alpha(date);
497}

499static int set_date(int year, int month, int day, struct tm *now_tm, time_t now, struct tm *tm)
500{
501  if (month > 0 && month < 13 && day > 0 && day < 32) {
502    struct tm check = *tm;
503    struct tm *r = (now_tm ? &check : tm);
504    time_t specified;
505
506    r->tm_mon = month - 1;
507    r->tm_mday = day;
508    if (year == -1) {
509      if (!now_tm)
510        return 1;
511      r->tm_year = now_tm->tm_year;
512    }
513    else if (year >= 1970 && year < 2100)
514      r->tm_year = year - 1900;
515    else if (year > 70 && year < 100)
516      r->tm_year = year;
517    else if (year < 38)
518      r->tm_year = year + 100;
519    else
520      return -1;
521    if (!now_tm)
522      return 0;
523
524    specified = tm_to_time_t(r);
525
526    /* Be it commit time or author time, it does not make
527     * sense to specify timestamp way into the future.  Make
528     * sure it is not later than ten days from now...
529     */
530    if ((specified != -1) && (now + 10*24*3600 < specified))
531      return -1;
532    tm->tm_mon = r->tm_mon;
533    tm->tm_mday = r->tm_mday;
534    if (year != -1)
535      tm->tm_year = r->tm_year;
536    return 0;
537  }
538  return -1;
539}

541static int set_time(long hour, long minute, long second, struct tm *tm)
542{
543  /* We accept 61st second because of leap second */
544  if (0 <= hour && hour <= 24 &&
545      0 <= minute && minute < 60 &&
546      0 <= second && second <= 60) {
547    tm->tm_hour = hour;
548    tm->tm_min = minute;
549    tm->tm_sec = second;
550    return 0;
551  }
552  return -1;
553}

555static int is_date_known(struct tm *tm)
556{
557  return tm->tm_year != -1 && tm->tm_mon != -1 && tm->tm_mday != -1;
558}

560static int match_multi_number(timestamp_t num, char c, const char *date,
561            char *end, struct tm *tm, time_t now)
562{
563  struct tm now_tm;
564  struct tm *refuse_future;
565  long num2, num3;
566
567  num2 = strtol(end+1, &end, 10);
568  num3 = -1;
569  if (*end == c && isdigit(end[1]))
570    num3 = strtol(end+1, &end, 10);
571
572  /* Time? Date? */
573  switch (c) {
574  case ':':
575    if (num3 < 0)
576      num3 = 0;
577    if (set_time(num, num2, num3, tm) == 0) {
578      /*
579       * If %H:%M:%S was just parsed followed by: .<num4>
580       * Consider (& discard) it as fractional second
581       * if %Y%m%d is parsed before.
582       */
583      if (*end == '.' && isdigit(end[1]) && is_date_known(tm))
584        strtol(end + 1, &end, 10);
585      break;
586    }
587    return 0;
588
589  case '-':
590  case '/':
591  case '.':
592    if (!now)
593      now = time(NULL);
594    refuse_future = NULL;
595    if (gmtime_r(&now, &now_tm))
596      refuse_future = &now_tm;
597
598    if (num > 70) {
599      /* yyyy-mm-dd? */
600      if (set_date(num, num2, num3, NULL, now, tm) == 0)
601        break;
602      /* yyyy-dd-mm? */
603      if (set_date(num, num3, num2, NULL, now, tm) == 0)
604        break;
605    }
606    /* Our eastern European friends say dd.mm.yy[yy]
607     * is the norm there, so giving precedence to
608     * mm/dd/yy[yy] form only when separator is not '.'
609     */
610    if (c != '.' &&
611        set_date(num3, num, num2, refuse_future, now, tm) == 0)
612      break;
613    /* European dd.mm.yy[yy] or funny US dd/mm/yy[yy] */
614    if (set_date(num3, num2, num, refuse_future, now, tm) == 0)
615      break;
616    /* Funny European mm.dd.yy */
617    if (c == '.' &&
618        set_date(num3, num, num2, refuse_future, now, tm) == 0)
619      break;
620    return 0;
621  }
622  return end - date;
623}
624
625/*
626 * Have we filled in any part of the time/date yet?
627 * We just do a binary 'and' to see if the sign bit
628 * is set in all the values.
629 */
630static inline int nodate(struct tm *tm)
631{
632  return (tm->tm_year &
633    tm->tm_mon &
634    tm->tm_mday &
635    tm->tm_hour &
636    tm->tm_min &
637    tm->tm_sec) < 0;
638}
639
640/*
641 * We've seen a digit. Time? Year? Date?
642 */
643static int match_digit(const char *date, struct tm *tm, int *offset, int *tm_gmt)
644{
645  int n;
646  char *end;
647  timestamp_t num;
648
649  num = parse_timestamp(date, &end, 10);
650
651  /*
652   * Seconds since 1970? We trigger on that for any numbers with
653   * more than 8 digits. This is because we don't want to rule out
654   * numbers like 20070606 as a YYYYMMDD date.
655   */
656  if (num >= 100000000 && nodate(tm)) {
657    time_t time = num;
658    if (gmtime_r(&time, tm)) {
659      *tm_gmt = 1;
660      return end - date;
661    }
662  }
663
664  /*
665   * Check for special formats: num[-.:/]num[same]num
666   */
667  switch (*end) {
668  case ':':
669  case '.':
670  case '/':
671  case '-':
672    if (isdigit(end[1])) {
673      int match = match_multi_number(num, *end, date, end, tm, 0);
674      if (match)
675        return match;
676    }
677  }
678
679  /*
680   * None of the special formats? Try to guess what
681   * the number meant. We use the number of digits
682   * to make a more educated guess..
683   */
684  n = 0;
685  do {
686    n++;
687  } while (isdigit(date[n]));
688
689  /* 8 digits, compact style of ISO-8601's date: YYYYmmDD */
690  /* 6 digits, compact style of ISO-8601's time: HHMMSS */
691  if (n == 8 || n == 6) {
692    unsigned int num1 = num / 10000;
693    unsigned int num2 = (num % 10000) / 100;
694    unsigned int num3 = num % 100;
695    if (n == 8)
696      set_date(num1, num2, num3, NULL, time(NULL), tm);
697    else if (n == 6 && set_time(num1, num2, num3, tm) == 0 &&
698       *end == '.' && isdigit(end[1]))
699      strtoul(end + 1, &end, 10);
700    return end - date;
701  }
702
703  /* Four-digit year or a timezone? */
704  if (n == 4) {
705    if (num <= 1400 && *offset == -1) {
706      unsigned int minutes = num % 100;
707      unsigned int hours = num / 100;
708      *offset = hours*60 + minutes;
709    } else if (num > 1900 && num < 2100)
710      tm->tm_year = num - 1900;
711    return n;
712  }
713
714  /*
715   * Ignore lots of numerals. We took care of 4-digit years above.
716   * Days or months must be one or two digits.
717   */
718  if (n > 2)
719    return n;
720
721  /*
722   * NOTE! We will give precedence to day-of-month over month or
723   * year numbers in the 1-12 range. So 05 is always "mday 5",
724   * unless we already have a mday..
725   *
726   * IOW, 01 Apr 05 parses as "April 1st, 2005".
727   */
728  if (num > 0 && num < 32 && tm->tm_mday < 0) {
729    tm->tm_mday = num;
730    return n;
731  }
732
733  /* Two-digit year? */
734  if (n == 2 && tm->tm_year < 0) {
735    if (num < 10 && tm->tm_mday >= 0) {
736      tm->tm_year = num + 100;
737      return n;
738    }
739    if (num >= 70) {
740      tm->tm_year = num;
741      return n;
742    }
743  }
744
745  if (num > 0 && num < 13 && tm->tm_mon < 0)
746    tm->tm_mon = num-1;
747
748  return n;
749}

751static int match_tz(const char *date, int *offp)
752{
753  char *end;
754  int hour = strtoul(date + 1, &end, 10);
755  int n = end - (date + 1);
756  int min = 0;
757
758  if (n == 4) {
759    /* hhmm */
760    min = hour % 100;
761    hour = hour / 100;
762  } else if (n != 2) {
763    min = 99; /* random crap */
764  } else if (*end == ':') {
765    /* hh:mm? */
766    min = strtoul(end + 1, &end, 10);
767    if (end - (date + 1) != 5)
768      min = 99; /* random crap */
769  } /* otherwise we parsed "hh" */
770
771  /*
772   * Don't accept any random crap. Even though some places have
773   * offset larger than 12 hours (e.g. Pacific/Kiritimati is at
774   * UTC+14), there is something wrong if hour part is much
775   * larger than that. We might also want to check that the
776   * minutes are divisible by 15 or something too. (Offset of
777   * Kathmandu, Nepal is UTC+5:45)
778   */
779  if (min < 60 && hour < 24) {
780    int offset = hour * 60 + min;
781    if (*date == '-')
782      offset = -offset;
783    *offp = offset;
784  }
785  return end - date;
786}

788static void date_string(timestamp_t date, int offset, struct strbuf *buf)
789{
790  int sign = '+';
791
792  if (offset < 0) {
793    offset = -offset;
794    sign = '-';
795  }
796  strbuf_addf(buf, "%"PRItime" %c%02d%02d", date, sign, offset / 60, offset % 60);
797}
798
799/*
800 * Parse a string like "0 +0000" as ancient timestamp near epoch, but
801 * only when it appears not as part of any other string.
802 */
803static int match_object_header_date(const char *date, timestamp_t *timestamp, int *offset)
804{
805  char *end;
806  timestamp_t stamp;
807  int ofs;
808
809  if (*date < '0' || '9' < *date)
810    return -1;
811  stamp = parse_timestamp(date, &end, 10);
812  if (*end != ' ' || stamp == TIME_MAX || (end[1] != '+' && end[1] != '-'))
813    return -1;
814  date = end + 2;
815  ofs = strtol(date, &end, 10);
816  if ((*end != '\0' && (*end != '\n')) || end != date + 4)
817    return -1;
818  ofs = (ofs / 100) * 60 + (ofs % 100);
819  if (date[-1] == '-')
820    ofs = -ofs;
821  *timestamp = stamp;
822  *offset = ofs;
823  return 0;
824}
825
826/* Gr. strptime is crap for this; it doesn't have a way to require RFC2822
827   (i.e. English) day/month names, and it doesn't work correctly with %z. */
828int parse_date_basic(const char *date, timestamp_t *timestamp, int *offset)
829{
830  struct tm tm;
831  int tm_gmt;
832  timestamp_t dummy_timestamp;
833  int dummy_offset;
834
835  if (!timestamp)
836    timestamp = &dummy_timestamp;
837  if (!offset)
838    offset = &dummy_offset;
839
840  memset(&tm, 0, sizeof(tm));
841  tm.tm_year = -1;
842  tm.tm_mon = -1;
843  tm.tm_mday = -1;
844  tm.tm_isdst = -1;
845  tm.tm_hour = -1;
846  tm.tm_min = -1;
847  tm.tm_sec = -1;
848  *offset = -1;
849  tm_gmt = 0;
850
851  if (*date == '@' &&
852      !match_object_header_date(date + 1, timestamp, offset))
853    return 0; /* success */
854  for (;;) {
855    int match = 0;
856    unsigned char c = *date;
857
858    /* Stop at end of string or newline */
859    if (!c || c == '\n')
860      break;
861
862    if (isalpha(c))
863      match = match_alpha(date, &tm, offset);
864    else if (isdigit(c))
865      match = match_digit(date, &tm, offset, &tm_gmt);
866    else if ((c == '-' || c == '+') && isdigit(date[1]))
867      match = match_tz(date, offset);
868
869    if (!match) {
870      /* BAD CRAP */
871      match = 1;
872    }
873
874    date += match;
875  }
876
877  /* do not use mktime(), which uses local timezone, here */
878  *timestamp = tm_to_time_t(&tm);
879  if (*timestamp == -1)
880    return -1;
881
882  if (*offset == -1) {
883    time_t temp_time;
884
885    /* gmtime_r() in match_digit() may have clobbered it */
886    tm.tm_isdst = -1;
887    temp_time = mktime(&tm);
888    if ((time_t)*timestamp > temp_time) {
889      *offset = ((time_t)*timestamp - temp_time) / 60;
890    } else {
891      *offset = -(int)((temp_time - (time_t)*timestamp) / 60);
892    }
893  }
894
895  if (!tm_gmt)
896    *timestamp -= *offset * 60;
897  return 0; /* success */
898}

900int parse_expiry_date(const char *date, timestamp_t *timestamp)
901{
902  int errors = 0;
903
904  if (!strcmp(date, "never") || !strcmp(date, "false"))
905    *timestamp = 0;
906  else if (!strcmp(date, "all") || !strcmp(date, "now"))
907    /*
908     * We take over "now" here, which usually translates
909     * to the current timestamp.  This is because the user
910     * really means to expire everything that was done in
911     * the past, and by definition reflogs are the record
912     * of the past, and there is nothing from the future
913     * to be kept.
914     */
915    *timestamp = TIME_MAX;
916  else
917    *timestamp = approxidate_careful(date, &errors);
918
919  return errors;
920}

922int parse_date(const char *date, struct strbuf *result)
923{
924  timestamp_t timestamp;
925  int offset;
926  if (parse_date_basic(date, &timestamp, &offset))
927    return -1;
928  date_string(timestamp, offset, result);
929  return 0;
930}

966{
967  const char *p;
968

970  if (skip_prefix(format, "auto:", &p)) {
971    if (isatty(1) || pager_in_use())
972      format = p;
973    else
974      format = "default";
975  }
976
977  /* historical alias */
978  if (!strcmp(format, "local"))
979    format = "default-local";
980
981  mode->type = parse_date_type(format, &p);
982  mode->local = 0;
983
984  if (skip_prefix(p, "-local", &p))
985    mode->local = 1;

988    if (!skip_prefix(p, ":", &p))
989      die("date format missing colon separator: %s", format);
990    mode->strftime_fmt = xstrdup(p);
991  } else if (*p)
992    die("unknown date format %s", format);
993}

1000void datestamp(struct strbuf *out)
1001{
1002  time_t now;
1003  int offset;
1004  struct tm tm = { 0 };
1005
1006  time(&now);
1007
1008  offset = tm_to_time_t(localtime_r(&now, &tm)) - now;
1009  offset /= 60;
1010
1011  date_string(now, offset, out);
1012}

1014/*
1015 * Relative time update (eg "2 days ago").  If we haven't set the time
1016 * yet, we need to set it from current time.
1017 */
1018static time_t update_tm(struct tm *tm, struct tm *now, time_t sec)
1019{
1020  time_t n;
1021
1022  if (tm->tm_mday < 0)
1023    tm->tm_mday = now->tm_mday;
1024  if (tm->tm_mon < 0)
1025    tm->tm_mon = now->tm_mon;
1026  if (tm->tm_year < 0) {
1027    tm->tm_year = now->tm_year;
1028    if (tm->tm_mon > now->tm_mon)
1029      tm->tm_year--;
1030  }
1031
1032  n = mktime(tm) - sec;
1033  localtime_r(&n, tm);
1034  return n;
1035}

1037/*
1038 * Do we have a pending number at the end, or when
1039 * we see a new one? Let's assume it's a month day,
1040 * as in "Dec 6, 1992"
1041 */
1042static void pending_number(struct tm *tm, int *num)
1043{
1044  int number = *num;
1045
1046  if (number) {
1047    *num = 0;
1048    if (tm->tm_mday < 0 && number < 32)
1049      tm->tm_mday = number;
1050    else if (tm->tm_mon < 0 && number < 13)
1051      tm->tm_mon = number-1;
1052    else if (tm->tm_year < 0) {
1053      if (number > 1969 && number < 2100)
1054        tm->tm_year = number - 1900;
1055      else if (number > 69 && number < 100)
1056        tm->tm_year = number;
1057      else if (number < 38)
1058        tm->tm_year = 100 + number;
1059      /* We screw up for number = 00 ? */
1060    }
1061  }
1062}

1170static const char *approxidate_alpha(const char *date, struct tm *tm, struct tm *now, int *num, int *touched)
1171{
1172  const struct typelen *tl;
1173  const struct special *s;
1174  const char *end = date;
1175  int i;
1176
1177  while (isalpha(*++end))
1178    ;
1179
1180  for (i = 0; i < 12; i++) {
1181    int match = match_string(date, month_names[i]);
1182    if (match >= 3) {
1183      tm->tm_mon = i;
1184      *touched = 1;
1185      return end;
1186    }
1187  }
1188
1189  for (s = special; s->name; s++) {
1190    int len = strlen(s->name);
1191    if (match_string(date, s->name) == len) {
1192      s->fn(tm, now, num);
1193      *touched = 1;
1194      return end;
1195    }
1196  }
1197
1198  if (!*num) {
1199    for (i = 1; i < 11; i++) {
1200      int len = strlen(number_name[i]);
1201      if (match_string(date, number_name[i]) == len) {
1202        *num = i;
1203        *touched = 1;
1204        return end;
1205      }
1206    }
1207    if (match_string(date, "last") == 4) {
1208      *num = 1;
1209      *touched = 1;
1210    }
1211    return end;
1212  }
1213
1214  tl = typelen;
1215  while (tl->type) {
1216    int len = strlen(tl->type);
1217    if (match_string(date, tl->type) >= len-1) {
1218      update_tm(tm, now, tl->length * *num);
1219      *num = 0;
1220      *touched = 1;
1221      return end;
1222    }
1223    tl++;
1224  }
1225
1226  for (i = 0; i < 7; i++) {
1227    int match = match_string(date, weekday_names[i]);
1228    if (match >= 3) {
1229      int diff, n = *num -1;
1230      *num = 0;
1231
1232      diff = tm->tm_wday - i;
1233      if (diff <= 0)
1234        n++;
1235      diff += 7*n;
1236
1237      update_tm(tm, now, diff * 24 * 60 * 60);
1238      *touched = 1;
1239      return end;
1240    }
1241  }
1242
1243  if (match_string(date, "months") >= 5) {
1244    int n;
1245    update_tm(tm, now, 0); /* fill in date fields if needed */
1246    n = tm->tm_mon - *num;
1247    *num = 0;
1248    while (n < 0) {
1249      n += 12;
1250      tm->tm_year--;
1251    }
1252    tm->tm_mon = n;
1253    *touched = 1;
1254    return end;
1255  }
1256
1257  if (match_string(date, "years") >= 4) {
1258    update_tm(tm, now, 0); /* fill in date fields if needed */
1259    tm->tm_year -= *num;
1260    *num = 0;
1261    *touched = 1;
1262    return end;
1263  }
1264
1265  return end;
1266}

1268static const char *approxidate_digit(const char *date, struct tm *tm, int *num,
1269             time_t now)
1270{
1271  char *end;
1272  timestamp_t number = parse_timestamp(date, &end, 10);
1273
1274  switch (*end) {
1275  case ':':
1276  case '.':
1277  case '/':
1278  case '-':
1279    if (isdigit(end[1])) {
1280      int match = match_multi_number(number, *end, date, end,
1281                   tm, now);
1282      if (match)
1283        return date + match;
1284    }
1285  }
1286
1287  /* Accept zero-padding only for small numbers ("Dec 02", never "Dec 0002") */
1288  if (date[0] != '0' || end - date <= 2)
1289    *num = number;
1290  return end;
1291}

1293static timestamp_t approxidate_str(const char *date,
1294           const struct timeval *tv,
1295           int *error_ret)
1296{
1297  int number = 0;
1298  int touched = 0;
1299  struct tm tm, now;
1300  time_t time_sec;
1301
1302  time_sec = tv->tv_sec;
1303  localtime_r(&time_sec, &tm);
1304  now = tm;
1305
1306  tm.tm_year = -1;
1307  tm.tm_mon = -1;
1308  tm.tm_mday = -1;
1309
1310  for (;;) {
1311    unsigned char c = *date;
1312    if (!c)
1313      break;
1314    date++;
1315    if (isdigit(c)) {
1316      pending_number(&tm, &number);
1317      date = approxidate_digit(date-1, &tm, &number, time_sec);
1318      touched = 1;
1319      continue;
1320    }
1321    if (isalpha(c))
1322      date = approxidate_alpha(date-1, &tm, &now, &number, &touched);
1323  }
1324  pending_number(&tm, &number);
1325  if (!touched)
1326    *error_ret = 1;
1327  return (timestamp_t)update_tm(&tm, &now, 0);
1328}

1362int date_overflows(timestamp_t t)
1363{
1364  time_t sys;
1365
1366  /* If we overflowed our timestamp data type, that's bad... */
1367  if ((uintmax_t)t >= TIME_MAX)
1368    return 1;
1369
1370  /*
1371   * ...but we also are going to feed the result to system
1372   * functions that expect time_t, which is often "signed long".
1373   * Make sure that we fit into time_t, as well.
1374   */
1375  sys = t;
1376  return t != sys || (t < 1) != (sys < 1);
1377}
