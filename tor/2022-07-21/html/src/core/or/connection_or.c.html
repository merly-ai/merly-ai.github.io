149/** Change conn->identity_digest to digest, and add conn into
150 * the appropriate digest maps.
151 *
152 * NOTE that this function only allows two kinds of transitions: from
153 * unset identity to set identity, and from idempotent re-settings
154 * of the same identity.  It's not allowed to clear an identity or to
155 * change an identity.  Return 0 on success, and -1 if the transition
156 * is not allowed.
157 **/
158static void
159connection_or_set_identity_digest(or_connection_t *conn,
160                                  const char *rsa_digest,
161                                  const ed25519_public_key_t *ed_id)
162{
163  channel_t *chan = NULL;
164  tor_assert(conn);
165  tor_assert(rsa_digest);
166
167  if (conn->chan)
168    chan = TLS_CHAN_TO_BASE(conn->chan);
169
170  log_info(LD_HANDSHAKE, "Set identity digest for %s at %p: %s %s.",
171           connection_describe(TO_CONN(conn)),
172           conn,
173           hex_str(rsa_digest, DIGEST_LEN),
174           ed25519_fmt(ed_id));
175  log_info(LD_HANDSHAKE, "   (Previously: %s %s)",
176           hex_str(conn->identity_digest, DIGEST_LEN),
177           chan ? ed25519_fmt(&chan->ed25519_identity) : "<null>");
178
179  const int rsa_id_was_set = ! tor_digest_is_zero(conn->identity_digest);
180  const int ed_id_was_set =
181    chan && !ed25519_public_key_is_zero(&chan->ed25519_identity);
182  const int rsa_changed =
183    tor_memneq(conn->identity_digest, rsa_digest, DIGEST_LEN);
184  const int ed_changed = ed_id_was_set &&
185    (!ed_id || !ed25519_pubkey_eq(ed_id, &chan->ed25519_identity));
186
187  tor_assert(!rsa_changed || !rsa_id_was_set);
188  tor_assert(!ed_changed || !ed_id_was_set);
189
190  if (!rsa_changed && !ed_changed)
191    return;
192
193  /* If the identity was set previously, remove the old mapping. */
194  if (rsa_id_was_set) {
195    connection_or_clear_identity(conn);
196    if (chan)
197      channel_clear_identity_digest(chan);
198  }
199
200  memcpy(conn->identity_digest, rsa_digest, DIGEST_LEN);
201
202  /* If we're initializing the IDs to zero, don't add a mapping yet. */
203  if (tor_digest_is_zero(rsa_digest) &&
204      (!ed_id || ed25519_public_key_is_zero(ed_id)))
205    return;
206
207  /* Deal with channels */
208  if (chan)
209    channel_set_identity_digest(chan, rsa_digest, ed_id);
210}

307/** Helper type used to sort connection states and find the most frequent. */
308typedef struct broken_state_count_t {
309  intptr_t count;
310  const char *state;
311} broken_state_count_t;
312
313/** Helper function used to sort broken_state_count_t by frequency. */
314static int
315broken_state_count_compare(const void **a_ptr, const void **b_ptr)
316{
317  const broken_state_count_t *a = *a_ptr, *b = *b_ptr;
318  if (b->count < a->count)
319    return -1;
320  else if (b->count == a->count)
321    return 0;
322  else
323    return 1;
324}

326/** Upper limit on the number of different states to report for connection
327 * failure. */
328#define MAX_REASONS_TO_REPORT 10
329
330/** Report a list of the top states for failed OR connections at log level
331 * <b>severity</b>, in log domain <b>domain</b>. */
332void
333connection_or_report_broken_states(int severity, int domain)
334{
335  int total = 0;
336  smartlist_t *items;
337
338  if (!broken_connection_counts || disable_broken_connection_counts)
339    return;
340
341  items = smartlist_new();
342  STRMAP_FOREACH(broken_connection_counts, state, void *, countptr) {
343    broken_state_count_t *c = tor_malloc(sizeof(broken_state_count_t));
344    c->count = (intptr_t)countptr;
345    total += (int)c->count;
346    c->state = state;
347    smartlist_add(items, c);
348  } STRMAP_FOREACH_END;
349
350  smartlist_sort(items, broken_state_count_compare);
351
352  tor_log(severity, domain, "%d connections have failed%s", total,
353      smartlist_len(items) > MAX_REASONS_TO_REPORT ? ". Top reasons:" : ":");
354
355  SMARTLIST_FOREACH_BEGIN(items, const broken_state_count_t *, c) {
356    if (c_sl_idx > MAX_REASONS_TO_REPORT)
357      break;
358    tor_log(severity, domain,
359        " %d connections died in state %s", (int)c->count, c->state);
360  } SMARTLIST_FOREACH_END(c);
361
362  SMARTLIST_FOREACH(items, broken_state_count_t *, c, tor_free(c));
363  smartlist_free(items);
364}

447/** Pack the cell_t host-order structure <b>src</b> into network-order
448 * in the buffer <b>dest</b>. See tor-spec.txt for details about the
449 * wire format.
450 *
451 * Note that this function doesn't touch <b>dst</b>-\>next: the caller
452 * should set it or clear it as appropriate.
453 */
454void
455cell_pack(packed_cell_t *dst, const cell_t *src, int wide_circ_ids)
456{
457  char *dest = dst->body;
458  if (wide_circ_ids) {
459    set_uint32(dest, htonl(src->circ_id));
460    dest += 4;
461  } else {
462    /* Clear the last two bytes of dest, in case we can accidentally
463     * send them to the network somehow. */
464    memset(dest+CELL_MAX_NETWORK_SIZE-2, 0, 2);
465    set_uint16(dest, htons(src->circ_id));
466    dest += 2;
467  }
468  set_uint8(dest, src->command);
469  memcpy(dest+1, src->payload, CELL_PAYLOAD_SIZE);
470}

472/** Unpack the network-order buffer <b>src</b> into a host-order
473 * cell_t structure <b>dest</b>.

475static void

477{

483    src += 2;

485  dest->command = get_uint8(src);
486  memcpy(dest->payload, src+1, CELL_PAYLOAD_SIZE);

492var_cell_pack_header(const var_cell_t *cell, char *hdr_out, int wide_circ_ids)

509/** Allocate and return a new var_cell_t with <b>payload_len</b> bytes of
510 * payload space. */
511var_cell_t *
512var_cell_new(uint16_t payload_len)
513{

563/** Handle any new bytes that have come in on connection <b>conn</b>.
564 * If conn is in 'open' state, hand it to
565 * connection_or_process_cells_from_inbuf()
566 * (else do nothing).
567 */
568int
569connection_or_process_inbuf(or_connection_t *conn)
570{
571  int ret = 0;
572  tor_assert(conn);
573
574  switch (conn->base_.state) {
575    case OR_CONN_STATE_PROXY_HANDSHAKING:
576      ret = connection_read_proxy_handshake(TO_CONN(conn));
577
578      /* start TLS after handshake completion, or deal with error */
579      if (ret == 1) {
580        tor_assert(TO_CONN(conn)->proxy_state == PROXY_CONNECTED);
581        if (buf_datalen(conn->base_.inbuf) != 0) {
582          log_fn(LOG_PROTOCOL_WARN, LD_NET, "Found leftover (%d bytes) "
583                 "when transitioning from PROXY_HANDSHAKING state on %s: "
584                 "closing.",
585                 (int)buf_datalen(conn->base_.inbuf),
586                 connection_describe(TO_CONN(conn)));
587          connection_or_close_for_error(conn, 0);
588          return -1;
589        }
590        if (connection_tls_start_handshake(conn, 0) < 0)
591          ret = -1;
592        /* Touch the channel's active timestamp if there is one */
593        if (conn->chan)
594          channel_timestamp_active(TLS_CHAN_TO_BASE(conn->chan));
595      }
596      if (ret < 0) {
597        connection_or_close_for_error(conn, 0);
598      }
599
600      return ret;
601    case OR_CONN_STATE_TLS_SERVER_RENEGOTIATING:
602    case OR_CONN_STATE_OPEN:
603    case OR_CONN_STATE_OR_HANDSHAKING_V2:
604    case OR_CONN_STATE_OR_HANDSHAKING_V3:
605      return connection_or_process_cells_from_inbuf(conn);
606    default:
607      break; /* don't do anything */
608  }
609
610  /* This check makes sure that we don't have any data on the inbuf if we're
611   * doing our TLS handshake: if we did, they were probably put there by a
612   * SOCKS proxy trying to trick us into accepting unauthenticated data.
613   */
614  if (buf_datalen(conn->base_.inbuf) != 0) {
615    log_fn(LOG_PROTOCOL_WARN, LD_NET, "Accumulated data (%d bytes) "
616           "on non-open %s; closing.",
617           (int)buf_datalen(conn->base_.inbuf),
618           connection_describe(TO_CONN(conn)));
619    connection_or_close_for_error(conn, 0);
620    ret = -1;
621  }
622
623  return ret;
624}

672/** Connection <b>conn</b> has finished writing and has no bytes left on
673 * its outbuf.
674 *
675 * Otherwise it's in state "open": stop writing and return.
676 *
677 * If <b>conn</b> is broken, mark it for close and return -1, else
678 * return 0.
679 */
680int
681connection_or_finished_flushing(or_connection_t *conn)
682{
683  tor_assert(conn);
684  assert_connection_ok(TO_CONN(conn),0);
685
686  switch (conn->base_.state) {
687    case OR_CONN_STATE_PROXY_HANDSHAKING:
688      /* PROXY_HAPROXY gets connected by receiving an ack. */
689      if (conn->proxy_type == PROXY_HAPROXY) {
690        tor_assert(TO_CONN(conn)->proxy_state == PROXY_HAPROXY_WAIT_FOR_FLUSH);
691        IF_BUG_ONCE(buf_datalen(TO_CONN(conn)->inbuf) != 0) {
692          /* This should be impossible; we're not even reading. */
693          connection_or_close_for_error(conn, 0);
694          return -1;
695        }
696        TO_CONN(conn)->proxy_state = PROXY_CONNECTED;
697
698        if (connection_tls_start_handshake(conn, 0) < 0) {
699          /* TLS handshaking error of some kind. */
700          connection_or_close_for_error(conn, 0);
701          return -1;
702        }
703        break;
704      }
705      break;
706    case OR_CONN_STATE_OPEN:
707    case OR_CONN_STATE_OR_HANDSHAKING_V2:
708    case OR_CONN_STATE_OR_HANDSHAKING_V3:
709      break;
710    default:
711      log_err(LD_BUG,"Called in unexpected state %d.", conn->base_.state);
712      tor_fragile_assert();
713      return -1;
714  }
715
716  /* Update the channel's active timestamp if there is one */
717  if (conn->chan)
718    channel_timestamp_active(TLS_CHAN_TO_BASE(conn->chan));
719
720  return 0;
721}

759/** Called when we're about to finally unlink and free an OR connection:
760 * perform necessary accounting and cleanup */
761void
762connection_or_about_to_close(or_connection_t *or_conn)
763{
764  connection_t *conn = TO_CONN(or_conn);
765
766  /* Tell the controlling channel we're closed */
767  if (or_conn->chan) {
768    channel_closed(TLS_CHAN_TO_BASE(or_conn->chan));
769    /*
770     * NULL this out because the channel might hang around a little
771     * longer before channel_run_cleanup() gets it.
772     */
773    or_conn->chan->conn = NULL;
774    or_conn->chan = NULL;
775  }
776
777  /* Remember why we're closing this connection. */
778  if (conn->state != OR_CONN_STATE_OPEN) {
779    /* now mark things down as needed */
780    if (connection_or_nonopen_was_started_here(or_conn)) {
781      const or_options_t *options = get_options();
782      connection_or_note_state_when_broken(or_conn);
783      /* Tell the new guard API about the channel failure */
784      entry_guard_chan_failed(TLS_CHAN_TO_BASE(or_conn->chan));
785      if (conn->state >= OR_CONN_STATE_TLS_HANDSHAKING) {
786        int reason = tls_error_to_orconn_end_reason(or_conn->tls_error);
787        connection_or_event_status(or_conn, OR_CONN_EVENT_FAILED,
788                                   reason);
789        if (!authdir_mode_tests_reachability(options)) {
790          const char *warning = NULL;
791          if (reason == END_OR_CONN_REASON_TLS_ERROR && or_conn->tls) {
792            warning = tor_tls_get_last_error_msg(or_conn->tls);
793          }
794          if (warning == NULL) {
795            warning = orconn_end_reason_to_control_string(reason);
796          }
797          control_event_bootstrap_prob_or(warning, reason, or_conn);
798        }
799      }
800    }
801  } else if (conn->hold_open_until_flushed) {
802    /* We only set hold_open_until_flushed when we're intentionally
803     * closing a connection. */
804    connection_or_event_status(or_conn, OR_CONN_EVENT_CLOSED,
805                tls_error_to_orconn_end_reason(or_conn->tls_error));
806  } else if (!tor_digest_is_zero(or_conn->identity_digest)) {
807    connection_or_event_status(or_conn, OR_CONN_EVENT_CLOSED,
808                tls_error_to_orconn_end_reason(or_conn->tls_error));
809  }
810}

812/** Return 1 if identity digest <b>id_digest</b> is known to be a

825/** Set the per-conn read and write limits for <b>conn</b>. If it's a known
826 * relay, we will rely on the global read and write buckets, so give it
827 * per-conn limits that are big enough they'll never matter. But if it's
828 * not a known relay, first check if we set PerConnBwRate/Burst, then
829 * check if the consensus sets them, else default to 'big enough'.
830 *
831 * If <b>reset</b> is true, set the bucket to be full.  Otherwise, just
832 * clip the bucket if it happens to be <em>too</em> full.
833 */
834static void
835connection_or_update_token_buckets_helper(or_connection_t *conn, int reset,
836                                          const or_options_t *options)
837{
838  int rate, burst; /* per-connection rate limiting params */
839  if (connection_or_digest_is_known_relay(conn->identity_digest)) {
840    /* It's in the consensus, or we have a descriptor for it meaning it
841     * was probably in a recent consensus. It's a recognized relay:
842     * give it full bandwidth. */
843    rate = (int)options->BandwidthRate;
844    burst = (int)options->BandwidthBurst;
845  } else {
846    /* Not a recognized relay. Squeeze it down based on the suggested
847     * bandwidth parameters in the consensus, but allow local config
848     * options to override. */
849    rate = options->PerConnBWRate ? (int)options->PerConnBWRate :
850        networkstatus_get_param(NULL, "perconnbwrate",
851                                (int)options->BandwidthRate, 1, INT32_MAX);
852    burst = options->PerConnBWBurst ? (int)options->PerConnBWBurst :
853        networkstatus_get_param(NULL, "perconnbwburst",
854                                (int)options->BandwidthBurst, 1, INT32_MAX);
855  }
856
857  token_bucket_rw_adjust(&conn->bucket, rate, burst);
858  if (reset) {
859    token_bucket_rw_reset(&conn->bucket, monotime_coarse_get_stamp());
860  }
861}

932/** Check whether the identity of <b>conn</b> matches a known node.  If it
933 * does, check whether the address of conn matches the expected address, and
934 * update the connection's is_canonical flag, nickname, and address fields as
935 * appropriate. */
936static void
937connection_or_check_canonicity(or_connection_t *conn, int started_here)
938{
939  (void) started_here;
940
941  const char *id_digest = conn->identity_digest;
942  const ed25519_public_key_t *ed_id = NULL;
943  if (conn->chan)
944    ed_id = & TLS_CHAN_TO_BASE(conn->chan)->ed25519_identity;
945
946  const node_t *r = node_get_by_id(id_digest);
947  if (r &&
948      node_supports_ed25519_link_authentication(r, 1) &&
949      ! node_ed25519_id_matches(r, ed_id)) {
950    /* If this node is capable of proving an ed25519 ID,
951     * we can't call this a canonical connection unless both IDs match. */
952     r = NULL;
953  }
954
955  if (r) {
956    tor_addr_port_t node_ipv4_ap;
957    tor_addr_port_t node_ipv6_ap;
958    node_get_prim_orport(r, &node_ipv4_ap);
959    node_get_pref_ipv6_orport(r, &node_ipv6_ap);
960    if (tor_addr_eq(&conn->base_.addr, &node_ipv4_ap.addr) ||
961        tor_addr_eq(&conn->base_.addr, &node_ipv6_ap.addr)) {
962      connection_or_set_canonical(conn, 1);
963    }
964    /* Choose the correct canonical address and port. */
965    tor_addr_port_t *node_ap;
966    if (tor_addr_family(&conn->base_.addr) == AF_INET) {
967      node_ap = &node_ipv4_ap;
968    } else {
969      node_ap = &node_ipv6_ap;
970    }
971    /* Remember the canonical addr/port so our log messages will make
972       sense. */
973    tor_addr_port_copy(&conn->canonical_orport, node_ap);
974    tor_free(conn->nickname);
975    conn->nickname = tor_strdup(node_get_nickname(r));
976  } else {
977    tor_free(conn->nickname);
978    conn->nickname = tor_malloc(HEX_DIGEST_LEN+2);
979    conn->nickname[0] = '$';
980    base16_encode(conn->nickname+1, HEX_DIGEST_LEN+1,
981                  conn->identity_digest, DIGEST_LEN);
982  }
983
984  /*
985   * We have to tell channeltls.c to update the channel marks (local, in
986   * particular), since we may have changed the address.
987   */
988

1793/** <b>Conn</b> just completed its handshake. Return 0 if all is well, and
1794 * return -1 if they are lying, broken, or otherwise something is wrong.
1795 *
1796 * If we initiated this connection (<b>started_here</b> is true), make sure
1797 * the other side sent a correctly formed certificate. If I initiated the
1798 * connection, make sure it's the right relay by checking the certificate.
1799 *
1800 * Otherwise (if we _didn't_ initiate this connection), it's okay for
1801 * the certificate to be weird or absent.
1802 *
1803 * If we return 0, and the certificate is as expected, write a hash of the
1804 * identity key into <b>digest_rcvd_out</b>, which must have DIGEST_LEN
1805 * space in it.
1806 * If the certificate is invalid or missing on an incoming connection,
1807 * we return 0 and set <b>digest_rcvd_out</b> to DIGEST_LEN NUL bytes.
1808 * (If we return -1, the contents of this buffer are undefined.)
1809 *
1810 * As side effects,
1811 * 1) Set conn->circ_id_type according to tor-spec.txt.
1812 * 2) If we're an authdirserver and we initiated the connection: drop all
1813 *    descriptors that claim to be on that IP/port but that aren't
1814 *    this relay; and note that this relay is reachable.
1815 * 3) If this is a bridge and we didn't configure its identity
1816 *    fingerprint, remember the keyid we just learned.
1817 */
1818static int
1819connection_or_check_valid_tls_handshake(or_connection_t *conn,
1820                                        int started_here,
1821                                        char *digest_rcvd_out)
1822{
1823  crypto_pk_t *identity_rcvd=NULL;
1824  const or_options_t *options = get_options();
1825  int severity = server_mode(options) ? LOG_PROTOCOL_WARN : LOG_WARN;
1826  const char *conn_type = started_here ? "outgoing" : "incoming";
1827  int has_cert = 0;
1828
1829  check_no_tls_errors();
1830  has_cert = tor_tls_peer_has_cert(conn->tls);
1831  if (started_here && !has_cert) {
1832    log_info(LD_HANDSHAKE,"Tried connecting to router at %s, but it didn't "
1833             "send a cert! Closing.",
1834             connection_describe_peer(TO_CONN(conn)));
1835    return -1;
1836  } else if (!has_cert) {
1837    log_debug(LD_HANDSHAKE,"Got incoming connection with no certificate. "
1838              "That's ok.");
1839  }
1840  check_no_tls_errors();

2279    channel_mark_client(TLS_CHAN_TO_BASE(conn->chan));
2280  }
2281
2282  or_handshake_state_free(conn->handshake_state);
2283  conn->handshake_state = NULL;
2284  connection_start_reading(TO_CONN(conn));
2285
2286  return 0;

2289/** Pack <b>cell</b> into wire-format, and write it onto <b>conn</b>'s outbuf.
2290 * For cells that use or affect a circuit, this should only be called by
2291 * connection_or_flush_from_first_active_circuit().
2292 */
2293void
2294connection_or_write_cell_to_buf(const cell_t *cell, or_connection_t *conn)
2295{
2296  packed_cell_t networkcell;
2297  size_t cell_network_size = get_cell_network_size(conn->wide_circ_ids);
2298
2299  tor_assert(cell);
2300  tor_assert(conn);
2301
2302  cell_pack(&networkcell, cell, conn->wide_circ_ids);
2303
2304  /* We need to count padding cells from this non-packed code path
2305   * since they are sent via chan->write_cell() (which is not packed) */
2306  rep_hist_padding_count_write(PADDING_TYPE_TOTAL);
2307  if (cell->command == CELL_PADDING)
2308    rep_hist_padding_count_write(PADDING_TYPE_CELL);
2309
2310  connection_buf_add(networkcell.body, cell_network_size, TO_CONN(conn));
2311
2312  /* Touch the channel's active timestamp if there is one */
2313  if (conn->chan) {
2314    channel_timestamp_active(TLS_CHAN_TO_BASE(conn->chan));
2315
2316    if (TLS_CHAN_TO_BASE(conn->chan)->padding_enabled) {
2317      rep_hist_padding_count_write(PADDING_TYPE_ENABLED_TOTAL);
2318      if (cell->command == CELL_PADDING)
2319        rep_hist_padding_count_write(PADDING_TYPE_ENABLED_CELL);
2320    }
2321  }
2322
2323  if (conn->base_.state == OR_CONN_STATE_OR_HANDSHAKING_V3)
2324    or_handshake_state_record_cell(conn, conn->handshake_state, cell, 0);
2325}

2361/** Process cells from <b>conn</b>'s inbuf.
2362 *
2363 * Loop: while inbuf contains a cell, pull it off the inbuf, unpack it,
2364 * and hand it to command_process_cell().
2365 *
2366 * Always return 0.
2367 */
2368static int
2369connection_or_process_cells_from_inbuf(or_connection_t *conn)
2370{
2371  var_cell_t *var_cell;
2372
2373  /*
2374   * Note on memory management for incoming cells: below the channel layer,
2375   * we shouldn't need to consider its internal queueing/copying logic.  It
2376   * is safe to pass cells to it on the stack or on the heap, but in the
2377   * latter case we must be sure we free them later.
2378   *
2379   * The incoming cell queue code in channel.c will (in the common case)
2380   * decide it can pass them to the upper layer immediately, in which case
2381   * those functions may run directly on the cell pointers we pass here, or
2382   * it may decide to queue them, in which case it will allocate its own
2383   * buffer and copy the cell.
2384   */
2385
2386  while (1) {
2387    log_debug(LD_OR,
2388              TOR_SOCKET_T_FORMAT": starting, inbuf_datalen %d "
2389              "(%d pending in tls object).",
2390              conn->base_.s,(int)connection_get_inbuf_len(TO_CONN(conn)),
2391              tor_tls_get_pending_bytes(conn->tls));
2392    if (connection_fetch_var_cell_from_buf(conn, &var_cell)) {
2393      if (!var_cell)
2394        return 0; /* not yet. */
2395
2396      /* Touch the channel's active timestamp if there is one */
2397      if (conn->chan)
2398        channel_timestamp_active(TLS_CHAN_TO_BASE(conn->chan));
2399
2400      circuit_build_times_network_is_live(get_circuit_build_times_mutable());
2401      channel_tls_handle_var_cell(var_cell, conn);
2402      var_cell_free(var_cell);
2403    } else {
2404      const int wide_circ_ids = conn->wide_circ_ids;
2405      size_t cell_network_size = get_cell_network_size(conn->wide_circ_ids);
2406      char buf[CELL_MAX_NETWORK_SIZE];
2407      cell_t cell;
2408      if (connection_get_inbuf_len(TO_CONN(conn))
2409          < cell_network_size) /* whole response available? */
2410        return 0; /* not yet */
2411
2412      /* Touch the channel's active timestamp if there is one */
2413      if (conn->chan)
2414        channel_timestamp_active(TLS_CHAN_TO_BASE(conn->chan));
2415
2416      circuit_build_times_network_is_live(get_circuit_build_times_mutable());
2417      connection_buf_get_bytes(buf, cell_network_size, TO_CONN(conn));
2418
2419      /* retrieve cell info from buf (create the host-order struct from the
2420       * network-order string) */
2421      cell_unpack(&cell, buf, wide_circ_ids);
2422
2423      channel_tls_handle_cell(&cell, conn);
2424    }
2425  }
2426}

2429static const uint16_t or_protocol_versions[] = { 1, 2, 3, 4, 5 };
2430/** Number of versions in <b>or_protocol_versions</b>. */

2432  (int)( sizeof(or_protocol_versions)/sizeof(uint16_t) );

2435 * implementation believes it can support. */

2437is_or_protocol_version_known(uint16_t v)
2438{

2447/** Send a VERSIONS cell on <b>conn</b>, telling the other host about the
2448 * link protocol versions that this Tor can support.
2449 *
2450 * If <b>v3_plus</b>, this is part of a V3 protocol handshake, so only
2451 * allow protocol version v3 or later.  If not <b>v3_plus</b>, this is
2452 * not part of a v3 protocol handshake, so don't allow protocol v3 or
2453 * later.
2454 **/
2455int
2456connection_or_send_versions(or_connection_t *conn, int v3_plus)
2457{
2458  var_cell_t *cell;
2459  int i;
2460  int n_versions = 0;
2461  const int min_version = v3_plus ? 3 : 0;
2462  const int max_version = v3_plus ? UINT16_MAX : 2;
2463  tor_assert(conn->handshake_state &&
2464             !conn->handshake_state->sent_versions_at);
2465  cell = var_cell_new(n_or_protocol_versions * 2);
2466  cell->command = CELL_VERSIONS;
2467  for (i = 0; i < n_or_protocol_versions; ++i) {
2468    uint16_t v = or_protocol_versions[i];
2469    if (v < min_version || v > max_version)
2470      continue;
2471    set_uint16(cell->payload+(2*n_versions), htons(v));
2472    ++n_versions;
2473  }
2474  cell->payload_len = n_versions * 2;
2475
2476  connection_or_write_var_cell_to_buf(cell, conn);
2477  conn->handshake_state->sent_versions_at = time(NULL);
2478
2479  var_cell_free(cell);
2480  return 0;
2481}

2508/** Send a NETINFO cell on <b>conn</b>, telling the other server what we know
2509 * about their address, our address, and the current time. */
2510MOCK_IMPL(int,
2511connection_or_send_netinfo,(or_connection_t *conn))
2512{
2513  cell_t cell;
2514  time_t now = time(NULL);
2515  const routerinfo_t *me;
2516  int r = -1;
2517
2518  tor_assert(conn->handshake_state);
2519
2520  if (conn->handshake_state->sent_netinfo) {
2521    log_warn(LD_BUG, "Attempted to send an extra netinfo cell on a connection "
2522             "where we already sent one.");
2523    return 0;
2524  }
2525
2526  memset(&cell, 0, sizeof(cell_t));
2527  cell.command = CELL_NETINFO;
2528
2529  netinfo_cell_t *netinfo_cell = netinfo_cell_new();
2530
2531  /* Timestamp, if we're a relay. */
2532  if (public_server_mode(get_options()) || ! conn->is_outgoing)
2533    netinfo_cell_set_timestamp(netinfo_cell, (uint32_t)now);
2534
2535  /* Their address. */
2536  const tor_addr_t *remote_tor_addr = &TO_CONN(conn)->addr;
2537  /* We can safely use TO_CONN(conn)->addr here, since we no longer replace
2538   * it with a canonical address. */
2539  netinfo_addr_t *their_addr = netinfo_addr_from_tor_addr(remote_tor_addr);
2540
2541  netinfo_cell_set_other_addr(netinfo_cell, their_addr);
2542
2543  /* My address -- only include it if I'm a public relay, or if I'm a
2544   * bridge and this is an incoming connection. If I'm a bridge and this
2545   * is an outgoing connection, act like a normal client and omit it. */
2546  if ((public_server_mode(get_options()) || !conn->is_outgoing) &&
2547      (me = router_get_my_routerinfo())) {
2548    uint8_t n_my_addrs = 1 + !tor_addr_is_null(&me->ipv6_addr);
2549    netinfo_cell_set_n_my_addrs(netinfo_cell, n_my_addrs);
2550
2551    netinfo_cell_add_my_addrs(netinfo_cell,
2552                              netinfo_addr_from_tor_addr(&me->ipv4_addr));
2553
2554    if (!tor_addr_is_null(&me->ipv6_addr)) {
2555      netinfo_cell_add_my_addrs(netinfo_cell,
2556                                netinfo_addr_from_tor_addr(&me->ipv6_addr));
2557    }
2558  }
2559
2560  const char *errmsg = NULL;
2561  if ((errmsg = netinfo_cell_check(netinfo_cell))) {
2562    log_warn(LD_OR, "Failed to validate NETINFO cell with error: %s",
2563                    errmsg);
2564    goto cleanup;
2565  }
2566
2567  if (netinfo_cell_encode(cell.payload, CELL_PAYLOAD_SIZE,
2568                          netinfo_cell) < 0) {
2569    log_warn(LD_OR, "Failed generating NETINFO cell");
2570    goto cleanup;
2571  }
2572
2573  conn->handshake_state->digest_sent_data = 0;
2574  conn->handshake_state->sent_netinfo = 1;
2575  connection_or_write_cell_to_buf(&cell, conn);
2576
2577  r = 0;
2578 cleanup:
2579  netinfo_cell_free(netinfo_cell);
2580
2581  return r;
2582}
