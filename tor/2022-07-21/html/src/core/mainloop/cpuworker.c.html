81/** Initialize the cpuworker subsystem. It is OK to call this more than once
82 * during Tor's lifetime.
83 */
84void
85cpu_init(void)
86{
87  if (!replyqueue) {
88    replyqueue = replyqueue_new(0);
89  }
90  if (!threadpool) {
91    /*
92      In our threadpool implementation, half the threads are permissive and
93      half are strict (when it comes to running lower-priority tasks). So we
94      always make sure we have at least two threads, so that there will be at
95      least one thread of each kind.
96    */
97    const int n_threads = get_num_cpus(get_options()) + 1;
98    threadpool = threadpool_new(n_threads,
99                                replyqueue,
100                                worker_state_new,
101                                worker_state_free_void,
102                                NULL);
103
104    int r = threadpool_register_reply_event(threadpool, NULL);
105
106    tor_assert(r == 0);
107  }
108
109  /* Total voodoo. Can we make this more sensible? */
110  max_pending_tasks = get_num_cpus(get_options()) * 64;
111}

214/** Indexed by handshake type: how many onionskins have we processed and
215 * counted of that type? */
216static uint64_t onionskins_n_processed[MAX_ONION_HANDSHAKE_TYPE+1];
217/** Indexed by handshake type, corresponding to the onionskins counted in
218 * onionskins_n_processed: how many microseconds have we spent in cpuworkers
219 * processing that kind of onionskin? */
220static uint64_t onionskins_usec_internal[MAX_ONION_HANDSHAKE_TYPE+1];
221/** Indexed by handshake type, corresponding to onionskins counted in
222 * onionskins_n_processed: how many microseconds have we spent waiting for
223 * cpuworkers to give us answers for that kind of onionskin?
224 */
225static uint64_t onionskins_usec_roundtrip[MAX_ONION_HANDSHAKE_TYPE+1];
226
227/** If any onionskin takes longer than this, we clip them to this
228 * time. (microseconds) */
229#define MAX_BELIEVABLE_ONIONSKIN_DELAY (2*1000*1000)
230
231/** Return true iff we'd like to measure a handshake of type
232 * <b>onionskin_type</b>. Call only from the main thread. */
233static int
234should_time_request(uint16_t onionskin_type)
235{
236  /* If we've never heard of this type, we shouldn't even be here. */
237  if (onionskin_type > MAX_ONION_HANDSHAKE_TYPE)
238    return 0;
239  /* Measure the first N handshakes of each type, to ensure we have a
240   * sample */
241  if (onionskins_n_processed[onionskin_type] < 4096)
242    return 1;
243
244  /** Otherwise, measure with P=1/128.  We avoid doing this for every
245   * handshake, since the measurement itself can take a little time. */
246  return crypto_fast_rng_one_in_n(get_thread_fast_rng(), 128);
247}

249/** Return an estimate of how many microseconds we will need for a single
250 * cpuworker to process <b>n_requests</b> onionskins of type
251 * <b>onionskin_type</b>. */
252uint64_t
253estimated_usec_for_onionskins(uint32_t n_requests, uint16_t onionskin_type)
254{
255  if (onionskin_type > MAX_ONION_HANDSHAKE_TYPE) /* should be impossible */
256    return 1000 * (uint64_t)n_requests;
257  if (PREDICT_UNLIKELY(onionskins_n_processed[onionskin_type] < 100)) {
258    /* Until we have 100 data points, just assume everything takes 1 msec. */
259    return 1000 * (uint64_t)n_requests;
260  } else {
261    /* This can't overflow: we'll never have more than 500000 onionskins
262     * measured in onionskin_usec_internal, and they won't take anything near
263     * 1 sec each, and we won't have anything like 1 million queued
264     * onionskins.  But that's 5e5 * 1e6 * 1e6, which is still less than
265     * UINT64_MAX. */
266    return (onionskins_usec_internal[onionskin_type] * n_requests) /
267      onionskins_n_processed[onionskin_type];
268  }
269}

271/** Compute the absolute and relative overhead of using the cpuworker
272 * framework for onionskins of type <b>onionskin_type</b>.*/
273static int
274get_overhead_for_onionskins(uint32_t *usec_out, double *frac_out,
275                            uint16_t onionskin_type)
276{
277  uint64_t overhead;
278
279  *usec_out = 0;
280  *frac_out = 0.0;
281
282  if (onionskin_type > MAX_ONION_HANDSHAKE_TYPE) /* should be impossible */
283    return -1;
284  if (onionskins_n_processed[onionskin_type] == 0 ||
285      onionskins_usec_internal[onionskin_type] == 0 ||
286      onionskins_usec_roundtrip[onionskin_type] == 0)
287    return -1;
288
289  overhead = onionskins_usec_roundtrip[onionskin_type] -
290    onionskins_usec_internal[onionskin_type];
291
292  *usec_out = (uint32_t)(overhead / onionskins_n_processed[onionskin_type]);
293  *frac_out = ((double)overhead) / onionskins_usec_internal[onionskin_type];
294
295  return 0;
296}

298/** If we've measured overhead for onionskins of type <b>onionskin_type</b>,

300void
301cpuworker_log_onionskin_overhead(int severity, int onionskin_type,

304  uint32_t overhead;

315         "cpuworker code ",
316         onionskin_type_name, (unsigned)overhead, relative_overhead*100);
317}

421/** Implementation function for onion handshake requests. */
422static workqueue_reply_t
423cpuworker_onion_handshake_threadfn(void *state_, void *work_)
424{
425  worker_state_t *state = state_;
426  cpuworker_job_t *job = work_;
427
428  /* variables for onion processing */
429  server_onion_keys_t *onion_keys = state->onion_keys;
430  cpuworker_request_t req;
431  cpuworker_reply_t rpl;
432
433  memcpy(&req, &job->u.request, sizeof(req));
434
435  tor_assert(req.magic == CPUWORKER_REQUEST_MAGIC);
436  memset(&rpl, 0, sizeof(rpl));
437
438  const create_cell_t *cc = &req.create_cell;
439  created_cell_t *cell_out = &rpl.created_cell;
440  struct timeval tv_start = {0,0}, tv_end;
441  int n;
442  rpl.timed = req.timed;
443  rpl.started_at = req.started_at;
444  rpl.handshake_type = cc->handshake_type;
445  if (req.timed)
446    tor_gettimeofday(&tv_start);
447  n = onion_skin_server_handshake(cc->handshake_type,
448                                  cc->onionskin, cc->handshake_len,
449                                  onion_keys,
450                                  &req.circ_ns_params,
451                                  cell_out->reply,
452                                  sizeof(cell_out->reply),
453                                  rpl.keys, CPATH_KEY_MATERIAL_LEN,
454                                  rpl.rend_auth_material,
455                                  &rpl.circ_params);
456  if (n < 0) {
457    /* failure */
458    log_debug(LD_OR,"onion_skin_server_handshake failed.");
459    memset(&rpl, 0, sizeof(rpl));
460    rpl.success = 0;
461  } else {
462    /* success */
463    log_debug(LD_OR,"onion_skin_server_handshake succeeded.");
464    cell_out->handshake_len = n;
465    switch (cc->cell_type) {
466    case CELL_CREATE:
467      cell_out->cell_type = CELL_CREATED; break;
468    case CELL_CREATE2:
469      cell_out->cell_type = CELL_CREATED2; break;
470    case CELL_CREATE_FAST:
471      cell_out->cell_type = CELL_CREATED_FAST; break;
472    default:
473      tor_assert(0);
474      return WQ_RPL_SHUTDOWN;
475    }
476    rpl.success = 1;
477  }
478
479  rpl.magic = CPUWORKER_REPLY_MAGIC;
480  if (req.timed) {
481    struct timeval tv_diff;
482    int64_t usec;
483    tor_gettimeofday(&tv_end);
484    timersub(&tv_end, &tv_start, &tv_diff);
485    usec = ((int64_t)tv_diff.tv_sec)*1000000 + tv_diff.tv_usec;
486    if (usec < 0 || usec > MAX_BELIEVABLE_ONIONSKIN_DELAY)
487      rpl.n_usec = MAX_BELIEVABLE_ONIONSKIN_DELAY;
488    else
489      rpl.n_usec = (uint32_t) usec;
490  }
491
492  memcpy(&job->u.reply, &rpl, sizeof(rpl));
493
494  memwipe(&req, 0, sizeof(req));
495  memwipe(&rpl, 0, sizeof(req));
496  return WQ_RPL_REPLY;
497}
498
499/** Take pending tasks from the queue and assign them to cpuworkers. */
500static void
501queue_pending_tasks(void)
502{
503  or_circuit_t *circ;
504  create_cell_t *onionskin = NULL;
505
506  while (total_pending_tasks < max_pending_tasks) {
507    circ = onion_next_task(&onionskin);
508
509    if (!circ)
510      return;
511
512    if (assign_onionskin_to_cpuworker(circ, onionskin) < 0)
513      log_info(LD_OR,"assign_to_cpuworker failed. Ignoring.");
514  }
515}

517/** DOCDOC */
518MOCK_IMPL(workqueue_entry_t *,
519cpuworker_queue_work,(workqueue_priority_t priority,
520                      workqueue_reply_t (*fn)(void *, void *),
521                      void (*reply_fn)(void *),
522                      void *arg))
523{
524  tor_assert(threadpool);
525
526  return threadpool_queue_work_priority(threadpool,
527                                        priority,
528                                        fn,
529                                        reply_fn,
530                                        arg);
531}

533/** Try to tell a cpuworker to perform the public key operations necessary to
534 * respond to <b>onionskin</b> for the circuit <b>circ</b>.
535 *
536 * Return 0 if we successfully assign the task, or -1 on failure.
537 */
538int
539assign_onionskin_to_cpuworker(or_circuit_t *circ,
540                              create_cell_t *onionskin)
541{
542  workqueue_entry_t *queue_entry;
543  cpuworker_job_t *job;
544  cpuworker_request_t req;
545  int should_time;
546
547  tor_assert(threadpool);
548
549  if (!circ->p_chan) {
550    log_info(LD_OR,"circ->p_chan gone. Failing circ.");
551    tor_free(onionskin);
552    return -1;
553  }
554
555  if (total_pending_tasks >= max_pending_tasks) {
556    log_debug(LD_OR,"No idle cpuworkers. Queuing.");
557    if (onion_pending_add(circ, onionskin) < 0) {
558      tor_free(onionskin);
559      return -1;
560    }
561    return 0;
562  }
563
564  if (!channel_is_client(circ->p_chan))
565    rep_hist_note_circuit_handshake_assigned(onionskin->handshake_type);
566
567  should_time = should_time_request(onionskin->handshake_type);
568  memset(&req, 0, sizeof(req));
569  req.magic = CPUWORKER_REQUEST_MAGIC;
570  req.timed = should_time;
571
572  memcpy(&req.create_cell, onionskin, sizeof(create_cell_t));
573
574  tor_free(onionskin);
575
576  if (should_time)
577    tor_gettimeofday(&req.started_at);
578
579  /* Copy the current cached consensus params relevant to
580   * circuit negotiation into the CPU worker context */
581  req.circ_ns_params.cc_enabled = congestion_control_enabled();
582  req.circ_ns_params.sendme_inc_cells = congestion_control_sendme_inc();
583
584  job = tor_malloc_zero(sizeof(cpuworker_job_t));
585  job->circ = circ;
586  memcpy(&job->u.request, &req, sizeof(req));
587  memwipe(&req, 0, sizeof(req));
588
589  ++total_pending_tasks;
590  queue_entry = threadpool_queue_work_priority(threadpool,
591                                      WQ_PRI_HIGH,
592                                      cpuworker_onion_handshake_threadfn,
593                                      cpuworker_onion_handshake_replyfn,
594                                      job);
595  if (!queue_entry) {
596    log_warn(LD_BUG, "Couldn't queue work on threadpool");
597    tor_free(job);
598    return -1;
599  }
600
601  log_debug(LD_OR, "Queued task %p (qe=%p, circ=%p)",
602            job, queue_entry, job->circ);
603
604  circ->workqueue_entry = queue_entry;
605
606  return 0;
607}
608
609/** If <b>circ</b> has a pending handshake that hasn't been processed yet,
610 * remove it from the worker queue. */
611void
612cpuworker_cancel_circ_handshake(or_circuit_t *circ)
613{
614  cpuworker_job_t *job;
615  if (circ->workqueue_entry == NULL)
616    return;
617
618  job = workqueue_entry_cancel(circ->workqueue_entry);
619  if (job) {
620    /* It successfully cancelled. */
621    memwipe(job, 0xe0, sizeof(*job));
622    tor_free(job);
623    tor_assert(total_pending_tasks > 0);
624    --total_pending_tasks;
625    /* if (!job), this is done in cpuworker_onion_handshake_replyfn. */
626    circ->workqueue_entry = NULL;
627  }
