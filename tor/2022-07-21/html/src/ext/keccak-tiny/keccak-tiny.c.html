46/*** Keccak-f[1600] ***/
47static inline void keccakf(void* state) {
48  uint64_t* a = (uint64_t*)state;
49  uint64_t b[5] = {0};
50  uint64_t t = 0;
51  uint8_t x, y;
52
53  for (int i = 0; i < 24; i++) {
54    // Theta
55    FOR5(x, 1,
56         b[x] = 0;
57         FOR5(y, 5,
58              b[x] ^= a[x + y]; ))
59    FOR5(x, 1,
60         FOR5(y, 5,
61              a[y + x] ^= b[(x + 4) % 5] ^ rol(b[(x + 1) % 5], 1); ))
62    // Rho and pi
63    t = a[1];
64    x = 0;
65    REPEAT24(b[0] = a[pi[x]];
66             a[pi[x]] = rol(t, rho[x]);
67             t = b[0];
68             x++; )
69    // Chi
70    FOR5(y,
71       5,
72       FOR5(x, 1,
73            b[x] = a[y + x];)
74       FOR5(x, 1,
75            a[y + x] = b[x] ^ ((~b[(x + 1) % 5]) & b[(x + 2) % 5]); ))
76    // Iota
77    a[0] ^= RC[i];
78  }
79}
