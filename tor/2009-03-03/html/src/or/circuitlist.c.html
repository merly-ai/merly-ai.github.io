29/** A map from OR connection and circuit ID to circuit.  (Lookup performance is
30 * very important here, since we need to do it every time a cell arrives.) */
31typedef struct orconn_circid_circuit_map_t {
32  HT_ENTRY(orconn_circid_circuit_map_t) node;
33  or_connection_t *or_conn;
34  circid_t circ_id;
35  circuit_t *circuit;
36} orconn_circid_circuit_map_t;
37
38/** Helper for hash tables: compare the OR connection and circuit ID for a and
39 * b, and return less than, equal to, or greater than zero appropriately.
40 */
41static INLINE int
42_orconn_circid_entries_eq(orconn_circid_circuit_map_t *a,
43                          orconn_circid_circuit_map_t *b)
44{
45  return a->or_conn == b->or_conn && a->circ_id == b->circ_id;
46}
47
48/** Helper: return a hash based on circuit ID and the pointer value of
49 * or_conn in <b>a</b>. */
50static INLINE unsigned int
51_orconn_circid_entry_hash(orconn_circid_circuit_map_t *a)
52{
53  return (((unsigned)a->circ_id)<<8) ^ (unsigned)(uintptr_t)(a->or_conn);
54}

56/** Map from [orconn,circid] to circuit. */
57static HT_HEAD(orconn_circid_map, orconn_circid_circuit_map_t)
58     orconn_circid_circuit_map = HT_INITIALIZER();
59HT_PROTOTYPE(orconn_circid_map, orconn_circid_circuit_map_t, node,
60             _orconn_circid_entry_hash, _orconn_circid_entries_eq)
61HT_GENERATE(orconn_circid_map, orconn_circid_circuit_map_t, node,
62            _orconn_circid_entry_hash, _orconn_circid_entries_eq, 0.6,
63            malloc, realloc, free)

65/** The most recently returned entry from circuit_get_by_circid_orconn;
66 * used to improve performance when many cells arrive in a row from the
67 * same circuit.
68 */
69orconn_circid_circuit_map_t *_last_circid_orconn_ent = NULL;

71/** Implementation helper for circuit_set_{p,n}_circid_orconn: A circuit ID
72 * and/or or_connection for circ has just changed from <b>old_conn, old_id</b>
73 * to <b>conn, id</b>.  Adjust the conn,circid map as appropriate, removing
74 * the old entry (if any) and adding a new one.  If <b>active</b> is true,
75 * remove the circuit from the list of active circuits on old_conn and add it
76 * to the list of active circuits on conn.
77 * XXX "active" isn't an arg anymore */
78static void
79circuit_set_circid_orconn_helper(circuit_t *circ, int direction,
80                                 circid_t id,
81                                 or_connection_t *conn)
82{
83  orconn_circid_circuit_map_t search;
84  orconn_circid_circuit_map_t *found;
85  or_connection_t *old_conn, **conn_ptr;
86  circid_t old_id, *circid_ptr;
87  int was_active, make_active;
88
89  if (direction == CELL_DIRECTION_OUT) {
90    conn_ptr = &circ->n_conn;
91    circid_ptr = &circ->n_circ_id;
92    was_active = circ->next_active_on_n_conn != NULL;
93    make_active = circ->n_conn_cells.n > 0;
94  } else {
95    or_circuit_t *c = TO_OR_CIRCUIT(circ);
96    conn_ptr = &c->p_conn;
97    circid_ptr = &c->p_circ_id;
98    was_active = c->next_active_on_p_conn != NULL;
99    make_active = c->p_conn_cells.n > 0;
100  }
101  old_conn = *conn_ptr;
102  old_id = *circid_ptr;
103
104  if (id == old_id && conn == old_conn)
105    return;
106
107  if (_last_circid_orconn_ent &&
108      ((old_id == _last_circid_orconn_ent->circ_id &&
109        old_conn == _last_circid_orconn_ent->or_conn) ||
110       (id == _last_circid_orconn_ent->circ_id &&
111        conn == _last_circid_orconn_ent->or_conn))) {
112    _last_circid_orconn_ent = NULL;
113  }
114
115  if (old_conn) { /* we may need to remove it from the conn-circid map */
116    tor_assert(old_conn->_base.magic == OR_CONNECTION_MAGIC);
117    search.circ_id = old_id;
118    search.or_conn = old_conn;
119    found = HT_REMOVE(orconn_circid_map, &orconn_circid_circuit_map, &search);
120    if (found) {
121      tor_free(found);
122      --old_conn->n_circuits;
123    }
124    if (was_active && old_conn != conn)
125      make_circuit_inactive_on_conn(circ,old_conn);
126  }
127
128  /* Change the values only after we have possibly made the circuit inactive
129   * on the previous conn. */
130  *conn_ptr = conn;
131  *circid_ptr = id;
132
133  if (conn == NULL)
134    return;
135
136  /* now add the new one to the conn-circid map */
137  search.circ_id = id;
138  search.or_conn = conn;
139  found = HT_FIND(orconn_circid_map, &orconn_circid_circuit_map, &search);
140  if (found) {
141    found->circuit = circ;
142  } else {
143    found = tor_malloc_zero(sizeof(orconn_circid_circuit_map_t));
144    found->circ_id = id;
145    found->or_conn = conn;
146    found->circuit = circ;
147    HT_INSERT(orconn_circid_map, &orconn_circid_circuit_map, found);
148  }
149  if (make_active && old_conn != conn)
150    make_circuit_active_on_conn(circ,conn);
151
152  ++conn->n_circuits;
153}

182/** Change the state of <b>circ</b> to <b>state</b>, adding it to or removing
183 * it from lists as appropriate. */
184void
185circuit_set_state(circuit_t *circ, uint8_t state)
186{
187  tor_assert(circ);
188  if (state == circ->state)
189    return;
190  if (!circuits_pending_or_conns)
191    circuits_pending_or_conns = smartlist_create();
192  if (circ->state == CIRCUIT_STATE_OR_WAIT) {
193    /* remove from waiting-circuit list. */
194    smartlist_remove(circuits_pending_or_conns, circ);
195  }
196  if (state == CIRCUIT_STATE_OR_WAIT) {
197    /* add to waiting-circuit list. */
198    smartlist_add(circuits_pending_or_conns, circ);
199  }
200  if (state == CIRCUIT_STATE_OPEN)
201    tor_assert(!circ->n_conn_onionskin);
202  circ->state = state;
203}

205/** Add <b>circ</b> to the global list of circuits. This is called only from
206 * within circuit_new.
207 */
208static void
209circuit_add(circuit_t *circ)
210{
211  if (!global_circuitlist) { /* first one */
212    global_circuitlist = circ;
213    circ->next = NULL;
214  } else {
215    circ->next = global_circuitlist;
216    global_circuitlist = circ;
217  }

220/** Append to <b>out</b> all circuits in state OR_WAIT waiting for
221 * the given connection. */
222void
223circuit_get_all_pending_on_or_conn(smartlist_t *out, or_connection_t *or_conn)
224{
225  tor_assert(out);
226  tor_assert(or_conn);
227
228  if (!circuits_pending_or_conns)
229    return;
230
231  SMARTLIST_FOREACH_BEGIN(circuits_pending_or_conns, circuit_t *, circ) {
232    if (circ->marked_for_close)
233      continue;
234    if (!circ->n_hop)
235      continue;
236    tor_assert(circ->state == CIRCUIT_STATE_OR_WAIT);
237    if (tor_digest_is_zero(circ->n_hop->identity_digest)) {
238      /* Look at addr/port. This is an unkeyed connection. */
239      if (!tor_addr_eq(&circ->n_hop->addr, &or_conn->_base.addr) ||
240          circ->n_hop->port != or_conn->_base.port)
241        continue;
242    } else {
243      /* We expected a key. See if it's the right one. */
244      if (memcmp(or_conn->identity_digest,
245                 circ->n_hop->identity_digest, DIGEST_LEN))
246        continue;
247    }
248    smartlist_add(out, circ);
249  } SMARTLIST_FOREACH_END(circ);
250}

252/** Return the number of circuits in state OR_WAIT, waiting for the given
253 * connection. */
254int
255circuit_count_pending_on_or_conn(or_connection_t *or_conn)
256{
257  int cnt;
258  smartlist_t *sl = smartlist_create();
259  circuit_get_all_pending_on_or_conn(sl, or_conn);
260  cnt = smartlist_len(sl);
261  smartlist_free(sl);
262  log_debug(LD_CIRC,"or_conn to %s, %d pending circs",
263            or_conn->nickname ? or_conn->nickname : "NULL", cnt);
264  return cnt;
265}

267/** Detach from the global circuit list, and deallocate, all
268 * circuits that have been marked for close.
269 */
270void
271circuit_close_all_marked(void)
272{
273  circuit_t *tmp,*m;
274
275  while (global_circuitlist && global_circuitlist->marked_for_close) {
276    tmp = global_circuitlist->next;
277    circuit_free(global_circuitlist);
278    global_circuitlist = tmp;
279  }
280
281  tmp = global_circuitlist;
282  while (tmp && tmp->next) {
283    if (tmp->next->marked_for_close) {
284      m = tmp->next->next;
285      circuit_free(tmp->next);
286      tmp->next = m;
287      /* Need to check new tmp->next; don't advance tmp. */
288    } else {
289      /* Advance tmp. */
290      tmp = tmp->next;
291    }
292  }
293}

315      return buf;

422/** Deallocate space associated with circ.
423 */
424static void
425circuit_free(circuit_t *circ)
426{
427  void *mem;
428  size_t memlen;
429  tor_assert(circ);
430  if (CIRCUIT_IS_ORIGIN(circ)) {
431    origin_circuit_t *ocirc = TO_ORIGIN_CIRCUIT(circ);
432    mem = ocirc;
433    memlen = sizeof(origin_circuit_t);
434    tor_assert(circ->magic == ORIGIN_CIRCUIT_MAGIC);
435    if (ocirc->build_state) {
436      if (ocirc->build_state->chosen_exit)
437        extend_info_free(ocirc->build_state->chosen_exit);
438      if (ocirc->build_state->pending_final_cpath)
439        circuit_free_cpath_node(ocirc->build_state->pending_final_cpath);
440    }
441    tor_free(ocirc->build_state);
442
443    circuit_free_cpath(ocirc->cpath);
444    if (ocirc->intro_key)
445      crypto_free_pk_env(ocirc->intro_key);
446    if (ocirc->rend_data)
447      rend_data_free(ocirc->rend_data);
448  } else {
449    or_circuit_t *ocirc = TO_OR_CIRCUIT(circ);
450    mem = ocirc;
451    memlen = sizeof(or_circuit_t);
452    tor_assert(circ->magic == OR_CIRCUIT_MAGIC);
453
454    if (ocirc->p_crypto)
455      crypto_free_cipher_env(ocirc->p_crypto);
456    if (ocirc->p_digest)
457      crypto_free_digest_env(ocirc->p_digest);
458    if (ocirc->n_crypto)
459      crypto_free_cipher_env(ocirc->n_crypto);
460    if (ocirc->n_digest)
461      crypto_free_digest_env(ocirc->n_digest);
462
463    if (ocirc->rend_splice) {
464      or_circuit_t *other = ocirc->rend_splice;
465      tor_assert(other->_base.magic == OR_CIRCUIT_MAGIC);
466      other->rend_splice = NULL;
467    }
468
469    /* remove from map. */
470    circuit_set_p_circid_orconn(ocirc, 0, NULL);
471
472    /* Clear cell queue _after_ removing it from the map.  Otherwise our
473     * "active" checks will be violated. */
474    cell_queue_clear(&ocirc->p_conn_cells);
475  }
476
477  if (circ->n_hop)
478    extend_info_free(circ->n_hop);
479  tor_free(circ->n_conn_onionskin);
480
481  /* Remove from map. */
482  circuit_set_n_circid_orconn(circ, 0, NULL);
483
484  /* Clear cell queue _after_ removing it from the map.  Otherwise our
485   * "active" checks will be violated. */
486  cell_queue_clear(&circ->n_conn_cells);
487
488  memset(circ, 0xAA, memlen); /* poison memory */
489  tor_free(mem);
490}

492/** Deallocate space associated with the linked list <b>cpath</b>. */
493static void
494circuit_free_cpath(crypt_path_t *cpath)
495{
496  crypt_path_t *victim, *head=cpath;
497
498  if (!cpath)
499    return;
500
501  /* it's a doubly linked list, so we have to notice when we've
502   * gone through it once. */
503  while (cpath->next && cpath->next != head) {
504    victim = cpath;
505    cpath = victim->next;

513void
514circuit_free_all(void)
515{

534  }

961  for (circ=global_circuitlist; circ; circ = circ->next) {
962    if (CIRCUIT_IS_ORIGIN(circ) &&
963        !circ->marked_for_close &&
964        !circ->timestamp_dirty)
965      circuit_mark_for_close(circ, END_CIRC_REASON_FINISHED);
966  }
967}

969/** Go through the circuitlist; for each circuit that starts at us
970 * and is dirty, frob its timestamp_dirty so we won't use it for any
971 * new streams.
972 *
973 * This is useful for letting the user change pseudonyms, so new
974 * streams will not be linkable to old streams.
975 */
976void
977circuit_expire_all_dirty_circs(void)
978{
979  circuit_t *circ;
980  or_options_t *options = get_options();
981
982  for (circ=global_circuitlist; circ; circ = circ->next) {
983    if (CIRCUIT_IS_ORIGIN(circ) &&
984        !circ->marked_for_close &&
985        circ->timestamp_dirty)
986      circ->timestamp_dirty -= options->MaxCircuitDirtiness;
987  }
988}

990/** Mark <b>circ</b> to be closed next time we call
991 * circuit_close_all_marked(). Do any cleanup needed:
992 *   - If state is onionskin_pending, remove circ from the onion_pending
993 *     list.
994 *   - If circ isn't open yet: call circuit_build_failed() if we're
995 *     the origin, and in either case call circuit_rep_hist_note_result()
996 *     to note stats.
997 *   - If purpose is C_INTRODUCE_ACK_WAIT, remove the intro point we
998 *     just tried from our list of intro points for that service
999 *     descriptor.
1000 *   - Send appropriate destroys and edge_destroys for conns and
1001 *     streams attached to circ.
1002 *   - If circ->rend_splice is set (we are the midpoint of a joined
1003 *     rendezvous stream), then mark the other circuit to close as well.
1004 */
1005void
1006_circuit_mark_for_close(circuit_t *circ, int reason, int line,
1007                        const char *file)
1008{
1009  int orig_reason = reason; /* Passed to the controller */
1010  assert_circuit_ok(circ);
1011  tor_assert(line);
1012  tor_assert(file);
1013
1014  if (circ->marked_for_close) {
1015    log(LOG_WARN,LD_BUG,
1016        "Duplicate call to circuit_mark_for_close at %s:%d"
1017        " (first at %s:%d)", file, line,
1018        circ->marked_for_close_file, circ->marked_for_close);
1019    return;
1020  }
1021  if (reason == END_CIRC_AT_ORIGIN) {
1022    if (!CIRCUIT_IS_ORIGIN(circ)) {
1023      log_warn(LD_BUG, "Specified 'at-origin' non-reason for ending circuit, "
1024               "but circuit was not at origin. (called %s:%d, purpose=%d)",
1025               file, line, circ->purpose);
1026    }
1027    reason = END_CIRC_REASON_NONE;
1028  }
1029  if (CIRCUIT_IS_ORIGIN(circ)) {
1030    /* We don't send reasons when closing circuits at the origin. */
1031    reason = END_CIRC_REASON_NONE;
1032  }
1033
1034  if (reason & END_CIRC_REASON_FLAG_REMOTE)
1035    reason &= ~END_CIRC_REASON_FLAG_REMOTE;
1036
1037  if (reason < _END_CIRC_REASON_MIN || reason > _END_CIRC_REASON_MAX) {
1038    if (!(orig_reason & END_CIRC_REASON_FLAG_REMOTE))
1039      log_warn(LD_BUG, "Reason %d out of range at %s:%d", reason, file, line);
1040    reason = END_CIRC_REASON_NONE;
1041  }
1042
1043  if (circ->state == CIRCUIT_STATE_ONIONSKIN_PENDING) {
1044    onion_pending_remove(TO_OR_CIRCUIT(circ));
1045  }
1046  /* If the circuit ever became OPEN, we sent it to the reputation history
1047   * module then.  If it isn't OPEN, we send it there now to remember which
1048   * links worked and which didn't.
1049   */
1050  if (circ->state != CIRCUIT_STATE_OPEN) {
1051    if (CIRCUIT_IS_ORIGIN(circ)) {
1052      origin_circuit_t *ocirc = TO_ORIGIN_CIRCUIT(circ);
1053      circuit_build_failed(ocirc); /* take actions if necessary */
1054      circuit_rep_hist_note_result(ocirc);
1055    }
1056  }
1057  if (circ->state == CIRCUIT_STATE_OR_WAIT) {
1058    if (circuits_pending_or_conns)
1059      smartlist_remove(circuits_pending_or_conns, circ);
1060  }
1061  if (CIRCUIT_IS_ORIGIN(circ)) {
1062    control_event_circuit_status(TO_ORIGIN_CIRCUIT(circ),
1063     (circ->state == CIRCUIT_STATE_OPEN)?CIRC_EVENT_CLOSED:CIRC_EVENT_FAILED,
1064     orig_reason);
1065  }
1066  if (circ->purpose == CIRCUIT_PURPOSE_C_INTRODUCE_ACK_WAIT) {
1067    origin_circuit_t *ocirc = TO_ORIGIN_CIRCUIT(circ);
1068    tor_assert(circ->state == CIRCUIT_STATE_OPEN);
1069    tor_assert(ocirc->build_state->chosen_exit);
1070    tor_assert(ocirc->rend_data);
1071    /* treat this like getting a nack from it */
1072    log_info(LD_REND, "Failed intro circ %s to %s (awaiting ack). "
1073             "Removing from descriptor.",
1074             safe_str(ocirc->rend_data->onion_address),
1075             safe_str(build_state_get_exit_nickname(ocirc->build_state)));
1076    rend_client_remove_intro_point(ocirc->build_state->chosen_exit,
1077                                   ocirc->rend_data);
1078  }
1079  if (circ->n_conn)
1080    connection_or_send_destroy(circ->n_circ_id, circ->n_conn, reason);
1081
1082  if (! CIRCUIT_IS_ORIGIN(circ)) {
1083    or_circuit_t *or_circ = TO_OR_CIRCUIT(circ);
1084    edge_connection_t *conn;
1085    for (conn=or_circ->n_streams; conn; conn=conn->next_stream)
1086      connection_edge_destroy(or_circ->p_circ_id, conn);
1087
1088    while (or_circ->resolving_streams) {
1089      conn = or_circ->resolving_streams;
1090      or_circ->resolving_streams = conn->next_stream;
1091      if (!conn->_base.marked_for_close) {
1092        /* The client will see a DESTROY, and infer that the connections
1093         * are closing because the circuit is getting torn down.  No need
1094         * to send an end cell. */
1095        conn->edge_has_sent_end = 1;
1096        conn->end_reason = END_STREAM_REASON_DESTROY;
1097        conn->end_reason |= END_STREAM_REASON_FLAG_ALREADY_SENT_CLOSED;
1098        connection_mark_for_close(TO_CONN(conn));
1099      }
1100      conn->on_circuit = NULL;
1101    }
1102
1103    if (or_circ->p_conn)
1104      connection_or_send_destroy(or_circ->p_circ_id, or_circ->p_conn, reason);
1105  } else {
1106    origin_circuit_t *ocirc = TO_ORIGIN_CIRCUIT(circ);
1107    edge_connection_t *conn;
1108    for (conn=ocirc->p_streams; conn; conn=conn->next_stream)
1109      connection_edge_destroy(circ->n_circ_id, conn);
1110  }
1111
1112  circ->marked_for_close = line;
1113  circ->marked_for_close_file = file;
1114
1115  if (!CIRCUIT_IS_ORIGIN(circ)) {
1116    or_circuit_t *or_circ = TO_OR_CIRCUIT(circ);
1117    if (or_circ->rend_splice) {
1118      if (!or_circ->rend_splice->_base.marked_for_close) {
1119        /* do this after marking this circuit, to avoid infinite recursion. */
1120        circuit_mark_for_close(TO_CIRCUIT(or_circ->rend_splice), reason);
1121      }
1122      or_circ->rend_splice = NULL;
1123    }
1124  }
1125}

1127/** Verify that cpath layer <b>cp</b> has all of its invariants
1128 * correct. Trigger an assert if anything is invalid.
1129 */
1130void
1131assert_cpath_layer_ok(const crypt_path_t *cp)
1132{
1133//  tor_assert(cp->addr); /* these are zero for rendezvous extra-hops */
1134//  tor_assert(cp->port);
1135  tor_assert(cp);
1136  tor_assert(cp->magic == CRYPT_PATH_MAGIC);
1137  switch (cp->state)
1138    {
1139    case CPATH_STATE_OPEN:
1140      tor_assert(cp->f_crypto);
1141      tor_assert(cp->b_crypto);
1142      /* fall through */
1143    case CPATH_STATE_CLOSED:
1144      tor_assert(!cp->dh_handshake_state);
1145      break;
1146    case CPATH_STATE_AWAITING_KEYS:
1147      /* tor_assert(cp->dh_handshake_state); */
1148      break;
1149    default:
1150      log_fn(LOG_ERR, LD_BUG, "Unexpected state %d", cp->state);
1151      tor_assert(0);
1152    }
1153  tor_assert(cp->package_window >= 0);
1154  tor_assert(cp->deliver_window >= 0);
1155}

1157/** Verify that cpath <b>cp</b> has all of its invariants
1158 * correct. Trigger an assert if anything is invalid.
1159 */
1160static void
1161assert_cpath_ok(const crypt_path_t *cp)
1162{
1163  const crypt_path_t *start = cp;
1164
1165  do {
1166    assert_cpath_layer_ok(cp);
1167    /* layers must be in sequence of: "open* awaiting? closed*" */
1168    if (cp != start) {
1169      if (cp->state == CPATH_STATE_AWAITING_KEYS) {
1170        tor_assert(cp->prev->state == CPATH_STATE_OPEN);
1171      } else if (cp->state == CPATH_STATE_OPEN) {
1172        tor_assert(cp->prev->state == CPATH_STATE_OPEN);
1173      }
1174    }
1175    cp = cp->next;
1176    tor_assert(cp);
1177  } while (cp != start);
1178}

1180/** Verify that circuit <b>c</b> has all of its invariants
1181 * correct. Trigger an assert if anything is invalid.
1182 */
1183void
1184assert_circuit_ok(const circuit_t *c)
1185{
1186  edge_connection_t *conn;
1187  const or_circuit_t *or_circ = NULL;
1188  const origin_circuit_t *origin_circ = NULL;
1189
1190  tor_assert(c);
1191  tor_assert(c->magic == ORIGIN_CIRCUIT_MAGIC || c->magic == OR_CIRCUIT_MAGIC);
1192  tor_assert(c->purpose >= _CIRCUIT_PURPOSE_MIN &&
1193             c->purpose <= _CIRCUIT_PURPOSE_MAX);
1194
1195  {
1196    /* Having a separate variable for this pleases GCC 4.2 in ways I hope I
1197     * never understand. -NM. */
1198    circuit_t *nonconst_circ = (circuit_t*) c;
1199    if (CIRCUIT_IS_ORIGIN(c))
1200      origin_circ = TO_ORIGIN_CIRCUIT(nonconst_circ);
1201    else
1202      or_circ = TO_OR_CIRCUIT(nonconst_circ);
1203  }
1204
1205  if (c->n_conn) {
1206    tor_assert(!c->n_hop);
1207
1208    if (c->n_circ_id) {
1209      /* We use the _impl variant here to make sure we don't fail on marked
1210       * circuits, which would not be returned by the regular function. */
1211      circuit_t *c2 = circuit_get_by_circid_orconn_impl(c->n_circ_id,
1212                                                        c->n_conn);
1213      tor_assert(c == c2);
1214    }
1215  }
1216  if (or_circ && or_circ->p_conn) {
1217    if (or_circ->p_circ_id) {
1218      /* ibid */
1219      circuit_t *c2 = circuit_get_by_circid_orconn_impl(or_circ->p_circ_id,
1220                                                        or_circ->p_conn);
1221      tor_assert(c == c2);
1222    }
1223  }
1224#if 0 /* false now that rendezvous exits are attached to p_streams */
1225  if (origin_circ)
1226    for (conn = origin_circ->p_streams; conn; conn = conn->next_stream)
1227      tor_assert(conn->_base.type == CONN_TYPE_AP);
1228#endif
1229  if (or_circ)
1230    for (conn = or_circ->n_streams; conn; conn = conn->next_stream)
1231      tor_assert(conn->_base.type == CONN_TYPE_EXIT);
1232
1233  tor_assert(c->deliver_window >= 0);
1234  tor_assert(c->package_window >= 0);
1235  if (c->state == CIRCUIT_STATE_OPEN) {
1236    tor_assert(!c->n_conn_onionskin);
1237    if (or_circ) {
1238      tor_assert(or_circ->n_crypto);
1239      tor_assert(or_circ->p_crypto);
1240      tor_assert(or_circ->n_digest);
1241      tor_assert(or_circ->p_digest);
1242    }
1243  }
1244  if (c->state == CIRCUIT_STATE_OR_WAIT && !c->marked_for_close) {
1245    tor_assert(circuits_pending_or_conns &&
1246               smartlist_isin(circuits_pending_or_conns, c));
1247  } else {
1248    tor_assert(!circuits_pending_or_conns ||
1249               !smartlist_isin(circuits_pending_or_conns, c));
1250  }
1251  if (origin_circ && origin_circ->cpath) {
1252    assert_cpath_ok(origin_circ->cpath);
1253  }
1254  if (c->purpose == CIRCUIT_PURPOSE_REND_ESTABLISHED) {
1255    tor_assert(or_circ);
1256    if (!c->marked_for_close) {
1257      tor_assert(or_circ->rend_splice);
1258      tor_assert(or_circ->rend_splice->rend_splice == or_circ);
1259    }
1260    tor_assert(or_circ->rend_splice != or_circ);
1261  } else {
1262    tor_assert(!or_circ || !or_circ->rend_splice);
1263  }
1264}
