65/** Does the digest for this circuit indicate that this cell is for us?
66 *
67 * Update digest from the payload of cell (with the integrity part set
68 * to 0). If the integrity part is valid, return 1, else restore digest
69 * and cell to their original state and return 0.
70 */
71static int
72relay_digest_matches(crypto_digest_env_t *digest, cell_t *cell)
73{
74  char received_integrity[4], calculated_integrity[4];
75  relay_header_t rh;
76  crypto_digest_env_t *backup_digest=NULL;
77
78  backup_digest = crypto_digest_dup(digest);
79
80  relay_header_unpack(&rh, cell->payload);
81  memcpy(received_integrity, rh.integrity, 4);
82  memset(rh.integrity, 0, 4);
83  relay_header_pack(cell->payload, &rh);
84
85//  log_fn(LOG_DEBUG,"Reading digest of %u %u %u %u from relay cell.",
86//    received_integrity[0], received_integrity[1],
87//    received_integrity[2], received_integrity[3]);
88
89  crypto_digest_add_bytes(digest, cell->payload, CELL_PAYLOAD_SIZE);
90  crypto_digest_get_digest(digest, calculated_integrity, 4);
91
92  if (memcmp(received_integrity, calculated_integrity, 4)) {
93//    log_fn(LOG_INFO,"Recognized=0 but bad digest. Not recognizing.");
94// (%d vs %d).", received_integrity, calculated_integrity);
95    /* restore digest to its old form */
96    crypto_digest_assign(digest, backup_digest);
97    /* restore the relay header */
98    memcpy(rh.integrity, received_integrity, 4);
99    relay_header_pack(cell->payload, &rh);
100    crypto_free_digest_env(backup_digest);
101    return 0;
102  }
103  crypto_free_digest_env(backup_digest);
104  return 1;
105}

112 * Return -1 if the crypto fails, else return 0.

124    return -1;
125  }

236/** Do the appropriate en/decryptions for <b>cell</b> arriving on
237 * <b>circ</b> in direction <b>cell_direction</b>.
238 *
239 * If cell_direction == CELL_DIRECTION_IN:
240 *   - If we're at the origin (we're the OP), for hops 1..N,
241 *     decrypt cell. If recognized, stop.
242 *   - Else (we're not the OP), encrypt one hop. Cell is not recognized.
243 *
244 * If cell_direction == CELL_DIRECTION_OUT:
245 *   - decrypt one hop. Check if recognized.
246 *
247 * If cell is recognized, set *recognized to 1, and set
248 * *layer_hint to the hop that recognized it.
249 *
250 * Return -1 to indicate that we should mark the circuit for close,
251 * else return 0.
252 */
253static int
254relay_crypt(circuit_t *circ, cell_t *cell, cell_direction_t cell_direction,
255            crypt_path_t **layer_hint, char *recognized)
256{
257  relay_header_t rh;
258
259  tor_assert(circ);
260  tor_assert(cell);
261  tor_assert(recognized);
262  tor_assert(cell_direction == CELL_DIRECTION_IN ||
263             cell_direction == CELL_DIRECTION_OUT);
264
265  if (cell_direction == CELL_DIRECTION_IN) {
266    if (CIRCUIT_IS_ORIGIN(circ)) { /* We're at the beginning of the circuit.
267                                    * We'll want to do layered decrypts. */
268      crypt_path_t *thishop, *cpath = TO_ORIGIN_CIRCUIT(circ)->cpath;
269      thishop = cpath;
270      if (thishop->state != CPATH_STATE_OPEN) {
271        log_fn(LOG_PROTOCOL_WARN, LD_PROTOCOL,
272               "Relay cell before first created cell? Closing.");
273        return -1;
274      }
275      do { /* Remember: cpath is in forward order, that is, first hop first. */
276        tor_assert(thishop);
277
278        if (relay_crypt_one_payload(thishop->b_crypto, cell->payload, 0) < 0)
279          return -1;
280
281        relay_header_unpack(&rh, cell->payload);
282        if (rh.recognized == 0) {
283          /* it's possibly recognized. have to check digest to be sure. */
284          if (relay_digest_matches(thishop->b_digest, cell)) {
285            *recognized = 1;
286            *layer_hint = thishop;
287            return 0;
288          }
289        }
290
291        thishop = thishop->next;
292      } while (thishop != cpath && thishop->state == CPATH_STATE_OPEN);
293      log_fn(LOG_PROTOCOL_WARN, LD_OR,
294             "Incoming cell at client not recognized. Closing.");
295      return -1;
296    } else { /* we're in the middle. Just one crypt. */
297      if (relay_crypt_one_payload(TO_OR_CIRCUIT(circ)->p_crypto,
298                                  cell->payload, 1) < 0)
299        return -1;
300//      log_fn(LOG_DEBUG,"Skipping recognized check, because we're not "
301//             "the client.");
302    }
303  } else /* cell_direction == CELL_DIRECTION_OUT */ {
304    /* we're in the middle. Just one crypt. */
305
306    if (relay_crypt_one_payload(TO_OR_CIRCUIT(circ)->n_crypto,
307                                cell->payload, 0) < 0)
308      return -1;
309
310    relay_header_unpack(&rh, cell->payload);
311    if (rh.recognized == 0) {
312      /* it's possibly recognized. have to check digest to be sure. */
313      if (relay_digest_matches(TO_OR_CIRCUIT(circ)->n_digest, cell)) {
314        *recognized = 1;
315        return 0;
316      }
317    }
318  }
319  return 0;
320}

322/** Package a relay cell from an edge:
323 *  - Encrypt it to the right layer
324 *  - Append it to the appropriate cell_queue on <b>circ</b>.
325 */
326static int
327circuit_package_relay_cell(cell_t *cell, circuit_t *circ,
328                           cell_direction_t cell_direction,
329                           crypt_path_t *layer_hint)
330{
331  or_connection_t *conn; /* where to send the cell */
332
333  if (cell_direction == CELL_DIRECTION_OUT) {
334    crypt_path_t *thishop; /* counter for repeated crypts */
335    conn = circ->n_conn;
336    if (!CIRCUIT_IS_ORIGIN(circ) || !conn) {
337      log_warn(LD_BUG,"outgoing relay cell has n_conn==NULL. Dropping.");
338      return 0; /* just drop it */
339    }
340
341    relay_set_digest(layer_hint->f_digest, cell);
342
343    thishop = layer_hint;
344    /* moving from farthest to nearest hop */
345    do {
346      tor_assert(thishop);
347      /* XXXX RD This is a bug, right? */
348      log_debug(LD_OR,"crypting a layer of the relay cell.");
349      if (relay_crypt_one_payload(thishop->f_crypto, cell->payload, 1) < 0) {
350        return -1;
351      }
352
353      thishop = thishop->prev;
354    } while (thishop != TO_ORIGIN_CIRCUIT(circ)->cpath->prev);
355
356  } else { /* incoming cell */
357    or_circuit_t *or_circ;
358    if (CIRCUIT_IS_ORIGIN(circ)) {
359      /* We should never package an _incoming_ cell from the circuit
360       * origin; that means we messed up somewhere. */
361      log_warn(LD_BUG,"incoming relay cell at origin circuit. Dropping.");
362      assert_circuit_ok(circ);
363      return 0; /* just drop it */
364    }
365    or_circ = TO_OR_CIRCUIT(circ);
366    conn = or_circ->p_conn;
367    relay_set_digest(or_circ->p_digest, cell);
368    if (relay_crypt_one_payload(or_circ->p_crypto, cell->payload, 1) < 0)
369      return -1;
370  }
371  ++stats_n_relay_cells_relayed;
372
373  append_cell_to_circuit_queue(circ, conn, cell, cell_direction);
374  return 0;
375}

377/** If cell's stream_id matches the stream_id of any conn that's
378 * attached to circ, return that conn, else return NULL.
379 */
380static edge_connection_t *
381relay_lookup_conn(circuit_t *circ, cell_t *cell,
382                  cell_direction_t cell_direction, crypt_path_t *layer_hint)
383{
384  edge_connection_t *tmpconn;
385  relay_header_t rh;
386
387  relay_header_unpack(&rh, cell->payload);
388
389  if (!rh.stream_id)
390    return NULL;
391
392  /* IN or OUT cells could have come from either direction, now
393   * that we allow rendezvous *to* an OP.
394   */
395
396  if (CIRCUIT_IS_ORIGIN(circ)) {
397    for (tmpconn = TO_ORIGIN_CIRCUIT(circ)->p_streams; tmpconn;
398         tmpconn=tmpconn->next_stream) {
399      if (rh.stream_id == tmpconn->stream_id &&
400          !tmpconn->_base.marked_for_close &&
401          tmpconn->cpath_layer == layer_hint) {
402        log_debug(LD_APP,"found conn for stream %d.", rh.stream_id);
403        return tmpconn;
404      }
405    }
406  } else {
407    for (tmpconn = TO_OR_CIRCUIT(circ)->n_streams; tmpconn;
408         tmpconn=tmpconn->next_stream) {
409      if (rh.stream_id == tmpconn->stream_id &&
410          !tmpconn->_base.marked_for_close) {
411        log_debug(LD_EXIT,"found conn for stream %d.", rh.stream_id);
412        if (cell_direction == CELL_DIRECTION_OUT ||
413            connection_edge_is_rendezvous_stream(tmpconn))
414          return tmpconn;
415      }
416    }
417    for (tmpconn = TO_OR_CIRCUIT(circ)->resolving_streams; tmpconn;
418         tmpconn=tmpconn->next_stream) {
419      if (rh.stream_id == tmpconn->stream_id &&
420          !tmpconn->_base.marked_for_close) {
421        log_debug(LD_EXIT,"found conn for stream %d.", rh.stream_id);
422        return tmpconn;
423      }
424    }
425  }
426  return NULL; /* probably a begin relay cell */
427}

429/** Pack the relay_header_t host-order structure <b>src</b> into
430 * network-order in the buffer <b>dest</b>. See tor-spec.txt for details
431 * about the wire format.
432 */
433void
434relay_header_pack(char *dest, const relay_header_t *src)
435{
436  *(uint8_t*)(dest) = src->command;
437
438  set_uint16(dest+1, htons(src->recognized));
439  set_uint16(dest+3, htons(src->stream_id));
440  memcpy(dest+5, src->integrity, 4);
441  set_uint16(dest+9, htons(src->length));
442}

444/** Unpack the network-order buffer <b>src</b> into a host-order
445 * relay_header_t structure <b>dest</b>.

447void

454  memcpy(dest->integrity, src+5, 4);
455  dest->length = ntohs(get_uint16(src+9));

1714  *prevp = *nextp = NULL;
1715  assert_active_circuits_ok_paranoid(conn);
1716}

1718/** Remove all circuits from the list of circuits with pending cells on
1719 * <b>conn</b>. */
1720void
1721connection_or_unlink_all_active_circs(or_connection_t *orconn)
1722{
1723  circuit_t *head = orconn->active_circuits;
1724  circuit_t *cur = head;
1725  if (! head)
1726    return;
1727  do {
1728    circuit_t *next = *next_circ_on_conn_p(cur, orconn);
1729    *prev_circ_on_conn_p(cur, orconn) = NULL;
1730    *next_circ_on_conn_p(cur, orconn) = NULL;
1731    cur = next;
1732  } while (cur != head);
1733  orconn->active_circuits = NULL;
1734}

1736/** Block (if <b>block</b> is true) or unblock (if <b>block</b> is false)
1737 * every edge connection that is using <b>circ</b> to write to <b>orconn</b>,
1738 * and start or stop reading as appropriate. */
1739static void
1740set_streams_blocked_on_circ(circuit_t *circ, or_connection_t *orconn,
1741                            int block)
1742{
1743  edge_connection_t *edge = NULL;
1744  if (circ->n_conn == orconn) {
1745    circ->streams_blocked_on_n_conn = block;
1746    if (CIRCUIT_IS_ORIGIN(circ))
1747      edge = TO_ORIGIN_CIRCUIT(circ)->p_streams;
1748  } else {
1749    circ->streams_blocked_on_p_conn = block;
1750    tor_assert(!CIRCUIT_IS_ORIGIN(circ));
1751    edge = TO_OR_CIRCUIT(circ)->n_streams;
1752  }
1753
1754  for (; edge; edge = edge->next_stream) {
1755    connection_t *conn = TO_CONN(edge);
1756    edge->edge_blocked_on_circ = block;
1757
1758    if (!conn->read_event) {
1759      /* This connection is a placeholder for something; probably a DNS
1760       * request.  It can't actually stop or start reading.*/
1761      continue;
1762    }
1763
1764    if (block) {
1765      if (connection_is_reading(conn))
1766        connection_stop_reading(conn);
1767    } else {
1768      /* Is this right? */
1769      if (!connection_is_reading(conn))
1770        connection_start_reading(conn);
1771    }
1772  }
1773}

1775/** Pull as many cells as possible (but no more than <b>max</b>) from the
1776 * queue of the first active circuit on <b>conn</b>, and write then to
1777 * <b>conn</b>-&gt;outbuf.  Return the number of cells written.  Advance
1778 * the active circuit pointer to the next active circuit in the ring. */
1779int
1780connection_or_flush_from_first_active_circuit(or_connection_t *conn, int max,
1781                                              time_t now)
1782{
1783  int n_flushed;
1784  cell_queue_t *queue;
1785  circuit_t *circ;
1786  int streams_blocked;
1787  circ = conn->active_circuits;
1788  if (!circ) return 0;
1789  assert_active_circuits_ok_paranoid(conn);
1790  if (circ->n_conn == conn) {
1791    queue = &circ->n_conn_cells;
1792    streams_blocked = circ->streams_blocked_on_n_conn;
1793  } else {
1794    queue = &TO_OR_CIRCUIT(circ)->p_conn_cells;
1795    streams_blocked = circ->streams_blocked_on_p_conn;
1796  }
1797  tor_assert(*next_circ_on_conn_p(circ,conn));
1798
1799  for (n_flushed = 0; n_flushed < max && queue->head; ) {
1800    packed_cell_t *cell = cell_queue_pop(queue);
1801    tor_assert(*next_circ_on_conn_p(circ,conn));
1802
1803    connection_write_to_buf(cell->body, CELL_NETWORK_SIZE, TO_CONN(conn));
1804
1805    packed_cell_free(cell);
1806    ++n_flushed;
1807    if (circ != conn->active_circuits) {
1808      /* If this happens, the current circuit just got made inactive by
1809       * a call in connection_write_to_buf().  That's nothing to worry about:
1810       * circuit_make_inactive_on_conn() already advanced conn->active_circuits
1811       * for us.
1812       */
1813      assert_active_circuits_ok_paranoid(conn);
1814      goto done;
1815    }
1816  }
1817  tor_assert(*next_circ_on_conn_p(circ,conn));
1818  assert_active_circuits_ok_paranoid(conn);
1819  conn->active_circuits = *next_circ_on_conn_p(circ, conn);
1820
1821  /* Is the cell queue low enough to unblock all the streams that are waiting
1822   * to write to this circuit? */
1823  if (streams_blocked && queue->n <= CELL_QUEUE_LOWWATER_SIZE)
1824    set_streams_blocked_on_circ(circ, conn, 0); /* unblock streams */
1825
1826  /* Did we just ran out of cells on this queue? */
1827  if (queue->n == 0) {
1828    log_debug(LD_GENERAL, "Made a circuit inactive.");
1829    make_circuit_inactive_on_conn(circ, conn);
1830  }
1831 done:
1832  if (n_flushed)
1833    conn->timestamp_last_added_nonpadding = now;
1834  return n_flushed;
1835}

1837/** Add <b>cell</b> to the queue of <b>circ</b> writing to <b>orconn</b>
1838 * transmitting in <b>direction</b>. */
1839void
1840append_cell_to_circuit_queue(circuit_t *circ, or_connection_t *orconn,
1841                             cell_t *cell, cell_direction_t direction)
1842{
1843  cell_queue_t *queue;
1844  int streams_blocked;
1845  if (direction == CELL_DIRECTION_OUT) {
1846    queue = &circ->n_conn_cells;
1847    streams_blocked = circ->streams_blocked_on_n_conn;
1848  } else {
1849    or_circuit_t *orcirc = TO_OR_CIRCUIT(circ);
1850    queue = &orcirc->p_conn_cells;
1851    streams_blocked = circ->streams_blocked_on_p_conn;
1852  }
1853  if (cell->command == CELL_RELAY_EARLY && orconn->link_proto < 2) {
1854    /* V1 connections don't understand RELAY_EARLY. */
1855    cell->command = CELL_RELAY;
1856  }
1857
1858  cell_queue_append_packed_copy(queue, cell);
1859
1860  /* If we have too many cells on the circuit, we should stop reading from
1861   * the edge streams for a while. */
1862  if (!streams_blocked && queue->n >= CELL_QUEUE_HIGHWATER_SIZE)
1863    set_streams_blocked_on_circ(circ, orconn, 1); /* block streams */
1864
1865  if (queue->n == 1) {
1866    /* This was the first cell added to the queue.  We need to make this
1867     * circuit active. */
1868    log_debug(LD_GENERAL, "Made a circuit active.");
1869    make_circuit_active_on_conn(circ, orconn);
1870  }
1871
1872  if (! buf_datalen(orconn->_base.outbuf)) {
1873    /* There is no data at all waiting to be sent on the outbuf.  Add a
1874     * cell, so that we can notice when it gets flushed, flushed_some can
1875     * get called, and we can start putting more data onto the buffer then.
1876     */
1877    log_debug(LD_GENERAL, "Primed a buffer.");
1878    connection_or_flush_from_first_active_circuit(orconn, 1, approx_time());
1879  }
1880}

1882/** Append an encoded value of <b>addr</b> to <b>payload_out</b>, which must
1883 * have at least 18 bytes of free space.  The encoding is, as specified in
1884 * tor-spec.txt:
1885 *   RESOLVED_TYPE_IPV4 or RESOLVED_TYPE_IPV6  [1 byte]
1886 *   LENGTH                                    [1 byte]
1887 *   ADDRESS                                   [length bytes]
1888 * Return the number of bytes added, or -1 on error */
1889int
1890append_address_to_payload(char *payload_out, const tor_addr_t *addr)
1891{
1892  uint32_t a;
1893  switch (tor_addr_family(addr)) {
1894  case AF_INET:
1895    payload_out[0] = RESOLVED_TYPE_IPV4;
1896    payload_out[1] = 4;
1897    a = tor_addr_to_ipv4n(addr);
1898    memcpy(payload_out+2, &a, 4);
1899    return 6;
1900  case AF_INET6:
1901    payload_out[0] = RESOLVED_TYPE_IPV6;
1902    payload_out[1] = 16;
1903    memcpy(payload_out+2, tor_addr_to_in6_addr8(addr), 16);
1904    return 18;
1905  case AF_UNSPEC:
1906  default:
1907    return -1;
1908  }
1909}

1911/** Given <b>payload_len</b> bytes at <b>payload</b>, starting with an address
1912 * encoded as by append_address_to_payload(), try to decode the address into
1913 * *<b>addr_out</b>.  Return the next byte in the payload after the address on
1914 * success, or NULL on failure. */
1915const char *
1916decode_address_from_payload(tor_addr_t *addr_out, const char *payload,
1917                            int payload_len)
1918{
1919  if (payload_len < 2)
1920    return NULL;
1921  if (payload_len < 2+(uint8_t)payload[1])
1922    return NULL;
1923
1924  switch (payload[0]) {
1925  case RESOLVED_TYPE_IPV4:
1926    if (payload[1] != 4)
1927      return NULL;
1928    tor_addr_from_ipv4n(addr_out, get_uint32(payload+2));
1929    break;
1930  case RESOLVED_TYPE_IPV6:
1931    if (payload[1] != 16)
1932      return NULL;
1933    tor_addr_from_ipv6_bytes(addr_out, payload+2);
1934    break;
1935  default:
1936    tor_addr_make_unspec(addr_out);
1937    break;
1938  }
1939  return payload + 2 + (uint8_t)payload[1];
1940}

1942/** Fail with an assert if the active circuits ring on <b>orconn</b> is
1943 * corrupt.  */
1944void
1945assert_active_circuits_ok(or_connection_t *orconn)
1946{
1947  circuit_t *head = orconn->active_circuits;
1948  circuit_t *cur = head;
1949  if (! head)
1950    return;
1951  do {
1952    circuit_t *next = *next_circ_on_conn_p(cur, orconn);
1953    circuit_t *prev = *prev_circ_on_conn_p(cur, orconn);
1954    tor_assert(next);
1955    tor_assert(prev);
1956    tor_assert(*next_circ_on_conn_p(prev, orconn) == cur);
1957    tor_assert(*prev_circ_on_conn_p(next, orconn) == cur);
1958    cur = next;
1959  } while (cur != head);
1960}
