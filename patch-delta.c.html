11#include "git-compat-util.h"
12#include "delta.h"
13
14void *patch_delta(const void *src_buf, unsigned long src_size,
15      const void *delta_buf, unsigned long delta_size,
16      unsigned long *dst_size)
17{
18  const unsigned char *data, *top;
19  unsigned char *dst_buf, *out, cmd;
20  unsigned long size;
21
22  if (delta_size < DELTA_SIZE_MIN)
23    return NULL;
24
25  data = delta_buf;
26  top = (const unsigned char *) delta_buf + delta_size;
27
28  /* make sure the orig file size matches what we expect */
29  size = get_delta_hdr_size(&data, top);
30  if (size != src_size)
31    return NULL;
32
33  /* now the result size */
34  size = get_delta_hdr_size(&data, top);
35  dst_buf = xmallocz(size);
36
37  out = dst_buf;
38  while (data < top) {
39    cmd = *data++;
40    if (cmd & 0x80) {
41      unsigned long cp_off = 0, cp_size = 0;
42#define PARSE_CP_PARAM(bit, var, shift) do { \
43      if (cmd & (bit)) { \
44        if (data >= top) \
45          goto bad_length; \
46        var |= ((unsigned) *data++ << (shift)); \
47      } } while (0)
48      PARSE_CP_PARAM(0x01, cp_off, 0);
49      PARSE_CP_PARAM(0x02, cp_off, 8);
50      PARSE_CP_PARAM(0x04, cp_off, 16);
51      PARSE_CP_PARAM(0x08, cp_off, 24);
52      PARSE_CP_PARAM(0x10, cp_size, 0);
53      PARSE_CP_PARAM(0x20, cp_size, 8);
54      PARSE_CP_PARAM(0x40, cp_size, 16);
55#undef PARSE_CP_PARAM
56      if (cp_size == 0) cp_size = 0x10000;
57      if (unsigned_add_overflows(cp_off, cp_size) ||
58          cp_off + cp_size > src_size ||
59          cp_size > size)
60        goto bad_length;
61      memcpy(out, (char *) src_buf + cp_off, cp_size);
62      out += cp_size;
63      size -= cp_size;
64    } else if (cmd) {
65      if (cmd > size || cmd > top - data)
66        goto bad_length;
67      memcpy(out, data, cmd);
68      out += cmd;
69      data += cmd;
70      size -= cmd;
71    } else {
72      /*
73       * cmd == 0 is reserved for future encoding
74       * extensions. In the mean time we must fail when
75       * encountering them (might be data corruption).
76       */
77      error("unexpected delta opcode 0");
78      goto bad;
79    }
80  }
81
82  /* sanity check */
83  if (data != top || size != 0) {
84    bad_length:
85    error("delta replay has gone wild");
86    bad:
87    free(dst_buf);
88    return NULL;
89  }
90
91  *dst_size = out - dst_buf;
92  return dst_buf;
93}
