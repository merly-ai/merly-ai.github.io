76/*
77 * BUILD_ASSERT_OR_ZERO - assert a build-time dependency, as an expression.
78 * @cond: the compile-time condition which must be true.
79 *
80 * Your compile will fail if the condition isn't true, or can't be evaluated
81 * by the compiler.  This can be used in an expression: its value is "0".
82 *
83 * Example:
84 *  #define foo_to_char(foo)          \
85 *     ((char *)(foo)            \
86 *      + BUILD_ASSERT_OR_ZERO(offsetof(struct foo, string) == 0))
87 */
88#define BUILD_ASSERT_OR_ZERO(cond) \
89  (sizeof(char [1 - 2*!(cond)]) - 1)

117/*
118 * Signed integer overflow is undefined in C, so here's a helper macro
119 * to detect if the sum of two integers will overflow.
120 *
121 * Requires: a >= 0, typeof(a) equals typeof(b)
122 */
123#define signed_add_overflows(a, b) \
124    ((b) > maximum_signed_value_of_type(a) - (a))
125
126#define unsigned_add_overflows(a, b) \
127    ((b) > maximum_unsigned_value_of_type(a) - (a))

129/*
130 * Returns true if the multiplication of "a" and "b" will
131 * overflow. The types of "a" and "b" must match and must be unsigned.
132 * Note that this macro evaluates "a" twice!
133 */
134#define unsigned_mult_overflows(a, b) \
135    ((a) && (b) > maximum_unsigned_value_of_type(a) / (a))
136
137/*
138 * Returns true if the left shift of "a" by "shift" bits will
139 * overflow. The type of "a" must be unsigned.
140 */
141#define unsigned_left_shift_overflows(a, shift) \
142    ((shift) < bitsizeof(a) && \
143     (a) > maximum_unsigned_value_of_type(a) >> (shift))

452/*
453 * Do not use this function when
454 * (1) geteuid() did not say we are running as 'root', or
455 * (2) using this function will compromise the system.
456 *
457 * PORTABILITY WARNING:
458 * This code assumes uid_t is unsigned because that is what sudo does.
459 * If your uid_t type is signed and all your ids are positive then it
460 * should all work fine.
461 * If your version of sudo uses negative values for uid_t or it is
462 * buggy and return an overflowed value in SUDO_UID, then git might
463 * fail to grant access to your repository properly or even mistakenly
464 * grant access to someone else.
465 * In the unlikely scenario this happened to you, and that is how you
466 * got to this message, we would like to know about it; so sent us an
467 * email to git@vger.kernel.org indicating which platform you are
468 * using and which version of sudo, so we can improve this logic and
469 * maybe provide you with a patch that would prevent this issue again
470 * in the future.
471 */
472static inline void extract_id_from_env(const char *env, uid_t *id)
473{
474  const char *real_uid = getenv(env);
475
476  /* discard anything empty to avoid a more complex check below */
477  if (real_uid && *real_uid) {
478    char *endptr = NULL;
479    unsigned long env_id;
480
481    errno = 0;
482    /* silent overflow errors could trigger a bug here */
483    env_id = strtoul(real_uid, &endptr, 10);
484    if (!*endptr && !errno)
485      *id = env_id;
486  }
487}

489static inline int is_path_owned_by_current_uid(const char *path)
490{
491  struct stat st;
492  uid_t euid;
493
494  if (lstat(path, &st))
495    return 0;
496
497  euid = geteuid();
498  if (euid == ROOT_UID)
499  {
500    if (st.st_uid == ROOT_UID)
501      return 1;
502    else
503      extract_id_from_env("SUDO_UID", &euid);
504  }
505
506  return st.st_uid == euid;
507}

632/*
633 * If the string "str" begins with the string found in "prefix", return 1.
634 * The "out" parameter is set to "str + strlen(prefix)" (i.e., to the point in
635 * the string right after the prefix).
636 *
637 * Otherwise, return 0 and leave "out" untouched.
638 *
639 * Examples:
640 *
641 *   [extract branch name, fail if not a branch]
642 *   if (!skip_prefix(ref, "refs/heads/", &branch)
643 *  return -1;
644 *
645 *   [skip prefix if present, otherwise use whole string]
646 *   skip_prefix(name, "refs/heads/", &name);
647 */
648static inline int skip_prefix(const char *str, const char *prefix,
649            const char **out)
650{
651  do {
652    if (!*prefix) {
653      *out = str;
654      return 1;
655    }
656  } while (*str++ == *prefix++);
657  return 0;
658}
659
660/*
661 * If the string "str" is the same as the string in "prefix", then the "arg"
662 * parameter is set to the "def" parameter and 1 is returned.
663 * If the string "str" begins with the string found in "prefix" and then a
664 * "=" sign, then the "arg" parameter is set to "str + strlen(prefix) + 1"
665 * (i.e., to the point in the string right after the prefix and the "=" sign),
666 * and 1 is returned.
667 *
668 * Otherwise, return 0 and leave "arg" untouched.
669 *
670 * When we accept both a "--key" and a "--key=<val>" option, this function
671 * can be used instead of !strcmp(arg, "--key") and then
672 * skip_prefix(arg, "--key=", &arg) to parse such an option.
673 */
674int skip_to_optional_arg_default(const char *str, const char *prefix,
675         const char **arg, const char *def);
676
677static inline int skip_to_optional_arg(const char *str, const char *prefix,
678               const char **arg)
679{
680  return skip_to_optional_arg_default(str, prefix, arg, "");
681}

683/*
684 * Like skip_prefix, but promises never to read past "len" bytes of the input
685 * buffer, and returns the remaining number of bytes in "out" via "outlen".
686 */
687static inline int skip_prefix_mem(const char *buf, size_t len,
688          const char *prefix,
689          const char **out, size_t *outlen)
690{
691  size_t prefix_len = strlen(prefix);
692  if (prefix_len <= len && !memcmp(buf, prefix, prefix_len)) {
693    *out = buf + prefix_len;
694    *outlen = len - prefix_len;
695    return 1;
696  }
697  return 0;
698}
699
700/*
701 * If buf ends with suffix, return 1 and subtract the length of the suffix
702 * from *len. Otherwise, return 0 and leave *len untouched.
703 */
704static inline int strip_suffix_mem(const char *buf, size_t *len,
705           const char *suffix)
706{
707  size_t suflen = strlen(suffix);
708  if (*len < suflen || memcmp(buf + (*len - suflen), suffix, suflen))
709    return 0;
710  *len -= suflen;
711  return 1;
712}
713
714/*
715 * If str ends with suffix, return 1 and set *len to the size of the string
716 * without the suffix. Otherwise, return 0 and set *len to the size of the
717 * string.
718 *
719 * Note that we do _not_ NUL-terminate str to the new length.
720 */
721static inline int strip_suffix(const char *str, const char *suffix, size_t *len)
722{
723  *len = strlen(str);
724  return strip_suffix_mem(str, len, suffix);
725}

727static inline int ends_with(const char *str, const char *suffix)
728{
729  size_t len;
730  return strip_suffix(str, suffix, &len);
731}

1062/*
1063 * These functions help you allocate structs with flex arrays, and copy
1064 * the data directly into the array. For example, if you had:
1065 *
1066 *   struct foo {
1067 *     int bar;
1068 *     char name[FLEX_ARRAY];
1069 *   };
1070 *
1071 * you can do:
1072 *
1073 *   struct foo *f;
1074 *   FLEX_ALLOC_MEM(f, name, src, len);
1075 *
1076 * to allocate a "foo" with the contents of "src" in the "name" field.
1077 * The resulting struct is automatically zero'd, and the flex-array field
1078 * is NUL-terminated (whether the incoming src buffer was or not).
1079 *
1080 * The FLEXPTR_* variants operate on structs that don't use flex-arrays,
1081 * but do want to store a pointer to some extra data in the same allocated
1082 * block. For example, if you have:
1083 *
1084 *   struct foo {
1085 *     char *name;
1086 *     int bar;
1087 *   };
1088 *
1089 * you can do:
1090 *
1091 *   struct foo *f;
1092 *   FLEXPTR_ALLOC_STR(f, name, src);
1093 *
1094 * and "name" will point to a block of memory after the struct, which will be
1095 * freed along with the struct (but the pointer can be repointed anywhere).
1096 *
1097 * The *_STR variants accept a string parameter rather than a ptr/len
1098 * combination.
1099 *
1100 * Note that these macros will evaluate the first parameter multiple
1101 * times, and it must be assignable as an lvalue.
1102 */
1103#define FLEX_ALLOC_MEM(x, flexname, buf, len) do { \
1104  size_t flex_array_len_ = (len); \
1105  (x) = xcalloc(1, st_add3(sizeof(*(x)), flex_array_len_, 1)); \
1106  memcpy((void *)(x)->flexname, (buf), flex_array_len_); \
1107} while (0)
1108#define FLEXPTR_ALLOC_MEM(x, ptrname, buf, len) do { \
1109  size_t flex_array_len_ = (len); \
1110  (x) = xcalloc(1, st_add3(sizeof(*(x)), flex_array_len_, 1)); \
1111  memcpy((x) + 1, (buf), flex_array_len_); \
1112  (x)->ptrname = (void *)((x)+1); \
1113} while(0)
1114#define FLEX_ALLOC_STR(x, flexname, str) \
1115  FLEX_ALLOC_MEM((x), flexname, (str), strlen(str))
1116#define FLEXPTR_ALLOC_STR(x, ptrname, str) \
1117  FLEXPTR_ALLOC_MEM((x), ptrname, (str), strlen(str))
