41#define BITS_PER_LONG (sizeof(unsigned long) * 8)
42#define NBITS(x) ((((x)-1)/BITS_PER_LONG)+1)
43#define OFF(x)  ((x)%BITS_PER_LONG)
44#define BIT(x)  (1UL<<OFF(x))
45#define LONG(x) ((x)/BITS_PER_LONG)
46#define test_bit(bit, array)    ((array[LONG(bit)] >> OFF(bit)) & 1)

59static inline int abs_size_mm(const struct input_absinfo *absinfo) {
60        /* Resolution is defined to be in units/mm for ABS_X/Y */
61        return (absinfo->maximum - absinfo->minimum) / absinfo->resolution;
62}

64static void extract_info(struct udev_device *dev, const char *devpath, bool test) {
65        char width[DECIMAL_STR_MAX(int)], height[DECIMAL_STR_MAX(int)];
66        struct input_absinfo xabsinfo = {}, yabsinfo = {};
67        _cleanup_close_ int fd = -1;
68
69        fd = open(devpath, O_RDONLY|O_CLOEXEC);
70        if (fd < 0)
71                return;
72
73        if (ioctl(fd, EVIOCGABS(ABS_X), &xabsinfo) < 0 ||
74            ioctl(fd, EVIOCGABS(ABS_Y), &yabsinfo) < 0)
75                return;
76
77        if (xabsinfo.resolution <= 0 || yabsinfo.resolution <= 0)
78                return;
79
80        xsprintf(width, "%d", abs_size_mm(&xabsinfo));
81        xsprintf(height, "%d", abs_size_mm(&yabsinfo));
82
83        udev_builtin_add_property(dev, test, "ID_INPUT_WIDTH_MM", width);
84        udev_builtin_add_property(dev, test, "ID_INPUT_HEIGHT_MM", height);
85}

87/*
88 * Read a capability attribute and return bitmask.
89 * @param dev udev_device
90 * @param attr sysfs attribute name (e. g. "capabilities/key")
91 * @param bitmask: Output array which has a sizeof of bitmask_size
92 */
93static void get_cap_mask(struct udev_device *dev,
94                         struct udev_device *pdev, const char* attr,
95                         unsigned long *bitmask, size_t bitmask_size,
96                         bool test) {
97        const char *v;
98        char text[4096];
99        unsigned i;
100        char* word;
101        unsigned long val;
102
103        v = udev_device_get_sysattr_value(pdev, attr);
104        if (!v)
105                v = "";
106
107        xsprintf(text, "%s", v);
108        log_debug("%s raw kernel attribute: %s", attr, text);
109
110        memzero(bitmask, bitmask_size);
111        i = 0;
112        while ((word = strrchr(text, ' ')) != NULL) {
113                val = strtoul (word+1, NULL, 16);
114                if (i < bitmask_size/sizeof(unsigned long))
115                        bitmask[i] = val;
116                else
117                        log_debug("ignoring %s block %lX which is larger than maximum size", attr, val);
118                *word = '\0';
119                ++i;
120        }
121        val = strtoul (text, NULL, 16);
122        if (i < bitmask_size / sizeof(unsigned long))
123                bitmask[i] = val;
124        else
125                log_debug("ignoring %s block %lX which is larger than maximum size", attr, val);
126
127        if (test) {
128                /* printf pattern with the right unsigned long number of hex chars */
129                xsprintf(text, "  bit %%4u: %%0%zulX\n",
130                         2 * sizeof(unsigned long));
131                log_debug("%s decoded bit map:", attr);
132                val = bitmask_size / sizeof (unsigned long);
133                /* skip over leading zeros */
134                while (bitmask[val-1] == 0 && val > 0)
135                        --val;
136                for (i = 0; i < val; ++i) {
137                        DISABLE_WARNING_FORMAT_NONLITERAL;
138                        log_debug(text, i * BITS_PER_LONG, bitmask[i]);
139                        REENABLE_WARNING;
140                }
141        }
142}

144/* pointer devices */
145static bool test_pointers(struct udev_device *dev,
146                          const unsigned long* bitmask_ev,
147                          const unsigned long* bitmask_abs,
148                          const unsigned long* bitmask_key,
149                          const unsigned long* bitmask_rel,
150                          const unsigned long* bitmask_props,
151                          bool test) {
152        int button, axis;
153        bool has_abs_coordinates = false;
154        bool has_rel_coordinates = false;
155        bool has_mt_coordinates = false;
156        bool has_joystick_axes_or_buttons = false;
157        bool is_direct = false;
158        bool has_touch = false;
159        bool has_3d_coordinates = false;
160        bool has_keys = false;
161        bool stylus_or_pen = false;
162        bool finger_but_no_pen = false;
163        bool has_mouse_button = false;
164        bool is_mouse = false;
165        bool is_touchpad = false;
166        bool is_touchscreen = false;
167        bool is_tablet = false;
168        bool is_joystick = false;
169        bool is_accelerometer = false;
170        bool is_pointing_stick= false;
171
172        has_keys = test_bit(EV_KEY, bitmask_ev);
173        has_abs_coordinates = test_bit(ABS_X, bitmask_abs) && test_bit(ABS_Y, bitmask_abs);
174        has_3d_coordinates = has_abs_coordinates && test_bit(ABS_Z, bitmask_abs);
175        is_accelerometer = test_bit(INPUT_PROP_ACCELEROMETER, bitmask_props);
176
177        if (!has_keys && has_3d_coordinates)
178                is_accelerometer = true;
179
180        if (is_accelerometer) {
181                udev_builtin_add_property(dev, test, "ID_INPUT_ACCELEROMETER", "1");
182                return true;
183        }
184
185        is_pointing_stick = test_bit(INPUT_PROP_POINTING_STICK, bitmask_props);
186        stylus_or_pen = test_bit(BTN_STYLUS, bitmask_key) || test_bit(BTN_TOOL_PEN, bitmask_key);
187        finger_but_no_pen = test_bit(BTN_TOOL_FINGER, bitmask_key) && !test_bit(BTN_TOOL_PEN, bitmask_key);
188        for (button = BTN_MOUSE; button < BTN_JOYSTICK && !has_mouse_button; button++)
189                has_mouse_button = test_bit(button, bitmask_key);
190        has_rel_coordinates = test_bit(EV_REL, bitmask_ev) && test_bit(REL_X, bitmask_rel) && test_bit(REL_Y, bitmask_rel);
191        has_mt_coordinates = test_bit(ABS_MT_POSITION_X, bitmask_abs) && test_bit(ABS_MT_POSITION_Y, bitmask_abs);
192
193        /* unset has_mt_coordinates if devices claims to have all abs axis */
194        if (has_mt_coordinates && test_bit(ABS_MT_SLOT, bitmask_abs) && test_bit(ABS_MT_SLOT - 1, bitmask_abs))
195                has_mt_coordinates = false;
196        is_direct = test_bit(INPUT_PROP_DIRECT, bitmask_props);
197        has_touch = test_bit(BTN_TOUCH, bitmask_key);
198        /* joysticks don't necessarily have buttons; e. g.
199         * rudders/pedals are joystick-like, but buttonless; they have
200         * other fancy axes. Others have buttons only but no axes. */
201        for (button = BTN_JOYSTICK; button < BTN_DIGI && !has_joystick_axes_or_buttons; button++)
202                has_joystick_axes_or_buttons = test_bit(button, bitmask_key);
203        for (button = BTN_TRIGGER_HAPPY1; button <= BTN_TRIGGER_HAPPY40 && !has_joystick_axes_or_buttons; button++)
204                has_joystick_axes_or_buttons = test_bit(button, bitmask_key);
205        for (button = BTN_DPAD_UP; button <= BTN_DPAD_RIGHT && !has_joystick_axes_or_buttons; button++)
206                has_joystick_axes_or_buttons = test_bit(button, bitmask_key);
207        for (axis = ABS_RX; axis < ABS_PRESSURE && !has_joystick_axes_or_buttons; axis++)
208                has_joystick_axes_or_buttons = test_bit(axis, bitmask_abs);
209
210        if (has_abs_coordinates) {
211                if (stylus_or_pen)
212                        is_tablet = true;
213                else if (finger_but_no_pen && !is_direct)
214                        is_touchpad = true;
215                else if (has_mouse_button)
216                        /* This path is taken by VMware's USB mouse, which has
217                         * absolute axes, but no touch/pressure button. */
218                        is_mouse = true;
219                else if (has_touch || is_direct)
220                        is_touchscreen = true;
221                else if (has_joystick_axes_or_buttons)
222                        is_joystick = true;
223        } else if (has_joystick_axes_or_buttons) {
224                is_joystick = true;
225        }
226
227        if (has_mt_coordinates) {
228                if (stylus_or_pen)
229                        is_tablet = true;
230                else if (finger_but_no_pen && !is_direct)
231                        is_touchpad = true;
232                else if (has_touch || is_direct)
233                        is_touchscreen = true;
234        }
235
236        if (has_mouse_button &&
237            (has_rel_coordinates ||
238            !has_abs_coordinates)) /* mouse buttons and no axis */
239                is_mouse = true;
240
241        if (is_pointing_stick)
242                udev_builtin_add_property(dev, test, "ID_INPUT_POINTINGSTICK", "1");
243        if (is_mouse)
244                udev_builtin_add_property(dev, test, "ID_INPUT_MOUSE", "1");
245        if (is_touchpad)
246                udev_builtin_add_property(dev, test, "ID_INPUT_TOUCHPAD", "1");
247        if (is_touchscreen)
248                udev_builtin_add_property(dev, test, "ID_INPUT_TOUCHSCREEN", "1");
249        if (is_joystick)
250                udev_builtin_add_property(dev, test, "ID_INPUT_JOYSTICK", "1");
251        if (is_tablet)
252                udev_builtin_add_property(dev, test, "ID_INPUT_TABLET", "1");
253
254        return is_tablet || is_mouse || is_touchpad || is_touchscreen || is_joystick || is_pointing_stick;
255}

257/* key like devices */
258static bool test_key(struct udev_device *dev,
259                     const unsigned long* bitmask_ev,
260                     const unsigned long* bitmask_key,
261                     bool test) {
262        unsigned i;
263        unsigned long found;
264        unsigned long mask;
265        bool ret = false;
266
267        /* do we have any KEY_* capability? */
268        if (!test_bit(EV_KEY, bitmask_ev)) {
269                log_debug("test_key: no EV_KEY capability");
270                return false;
271        }
272
273        /* only consider KEY_* here, not BTN_* */
274        found = 0;
275        for (i = 0; i < BTN_MISC/BITS_PER_LONG; ++i) {
276                found |= bitmask_key[i];
277                log_debug("test_key: checking bit block %lu for any keys; found=%i", (unsigned long)i*BITS_PER_LONG, found > 0);
278        }
279        /* If there are no keys in the lower block, check the higher blocks */
280        if (!found) {
281                unsigned block;
282                for (block = 0; block < (sizeof(high_key_blocks) / sizeof(struct range)); ++block) {
283                        for (i = high_key_blocks[block].start; i < high_key_blocks[block].end; ++i) {
284                                if (test_bit(i, bitmask_key)) {
285                                        log_debug("test_key: Found key %x in high block", i);
286                                        found = 1;
287                                        break;
288                                }
289                        }
290                }
291        }
292
293        if (found > 0) {
294                udev_builtin_add_property(dev, test, "ID_INPUT_KEY", "1");
295                ret = true;
296        }
297
298        /* the first 32 bits are ESC, numbers, and Q to D; if we have all of
299         * those, consider it a full keyboard; do not test KEY_RESERVED, though */
300        mask = 0xFFFFFFFE;
301        if ((bitmask_key[0] & mask) == mask) {
302                udev_builtin_add_property(dev, test, "ID_INPUT_KEYBOARD", "1");
303                ret = true;
304        }
305
306        return ret;
307}

309static int builtin_input_id(struct udev_device *dev, int argc, char *argv[], bool test) {
310        struct udev_device *pdev;
311        unsigned long bitmask_ev[NBITS(EV_MAX)];
312        unsigned long bitmask_abs[NBITS(ABS_MAX)];
313        unsigned long bitmask_key[NBITS(KEY_MAX)];
314        unsigned long bitmask_rel[NBITS(REL_MAX)];
315        unsigned long bitmask_props[NBITS(INPUT_PROP_MAX)];
316        const char *sysname, *devnode;
317        bool is_pointer;
318        bool is_key;
319
320        assert(dev);
321
322        /* walk up the parental chain until we find the real input device; the
323         * argument is very likely a subdevice of this, like eventN */
324        pdev = dev;
325        while (pdev != NULL && udev_device_get_sysattr_value(pdev, "capabilities/ev") == NULL)
326                pdev = udev_device_get_parent_with_subsystem_devtype(pdev, "input", NULL);
327
328        if (pdev) {
329                /* Use this as a flag that input devices were detected, so that this
330                 * program doesn't need to be called more than once per device */
331                udev_builtin_add_property(dev, test, "ID_INPUT", "1");
332                get_cap_mask(dev, pdev, "capabilities/ev", bitmask_ev, sizeof(bitmask_ev), test);
333                get_cap_mask(dev, pdev, "capabilities/abs", bitmask_abs, sizeof(bitmask_abs), test);
334                get_cap_mask(dev, pdev, "capabilities/rel", bitmask_rel, sizeof(bitmask_rel), test);
335                get_cap_mask(dev, pdev, "capabilities/key", bitmask_key, sizeof(bitmask_key), test);
336                get_cap_mask(dev, pdev, "properties", bitmask_props, sizeof(bitmask_props), test);
337                is_pointer = test_pointers(dev, bitmask_ev, bitmask_abs,
338                                           bitmask_key, bitmask_rel,
339                                           bitmask_props, test);
340                is_key = test_key(dev, bitmask_ev, bitmask_key, test);
341                /* Some evdev nodes have only a scrollwheel */
342                if (

344                        udev_builtin_add_property(dev, test, "ID_INPUT_KEY", "1");
