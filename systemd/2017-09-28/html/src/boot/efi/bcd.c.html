108static const Key *get_subkey(const uint8_t *bcd, uint32_t bcd_len, uint32_t offset, const char *name) {
109        assert(bcd);
110        assert(name);
111
112        if (BAD_STRUCT(SubkeyFast, offset, bcd_len))
113                return NULL;
114
115        const SubkeyFast *subkey = (const SubkeyFast *) (bcd + offset);
116        if (subkey->sig != SIG_SUBKEY_FAST)
117                return NULL;
118
119        if (BAD_ARRAY(SubkeyFast, entries, offset, subkey->n_entries, bcd_len))
120                return NULL;
121
122        for (uint16_t i = 0; i < subkey->n_entries; i++) {
123                if (!strncaseeq8(name, subkey->entries[i].name_hint, sizeof(subkey->entries[i].name_hint)))
124                        continue;
125
126                const Key *key = get_key(bcd, bcd_len, subkey->entries[i].key_offset, name);
127                if (key)
128                        return key;
129        }
130
131        return NULL;
132}

134/* We use NUL as registry path separators for convenience. To start from the root, begin
135 * name with a NUL. Name must end with two NUL. The lookup depth is not restricted, so
136 * name must be properly validated before calling get_key(). */
137static const Key *get_key(const uint8_t *bcd, uint32_t bcd_len, uint32_t offset, const char *name) {
138        assert(bcd);
139        assert(name);
140
141        if (BAD_STRUCT(Key, offset, bcd_len))
142                return NULL;
143
144        const Key *key = (const Key *) (bcd + offset);
145        if (key->sig != SIG_KEY)
146                return NULL;
147
148        if (BAD_ARRAY(Key, key_name, offset, key->key_name_len, bcd_len))
149                return NULL;
150
151        if (*name) {
152                if (strncaseeq8(name, key->key_name, key->key_name_len) && strlen8(name) == key->key_name_len)
153                        name += key->key_name_len;
154                else
155                        return NULL;
156        }
157
158        name++;
159        return *name ? get_subkey(bcd, bcd_len, key->subkeys_offset, name) : key;
160}

162static const KeyValue *get_key_value(const uint8_t *bcd, uint32_t bcd_len, const Key *key, const char *name) {
163        assert(bcd);
164        assert(key);
165        assert(name);
166
167        if (key->n_key_values == 0)
168                return NULL;
169
170        if (BAD_OFFSET(key->key_values_offset, sizeof(uint32_t) * (uint64_t) key->n_key_values, bcd_len) ||
171            (uintptr_t) (bcd + key->key_values_offset) % alignof(uint32_t) != 0)
172                return NULL;
173
174        const uint32_t *key_value_list = (const uint32_t *) (bcd + key->key_values_offset);
175        for (uint32_t i = 0; i < key->n_key_values; i++) {
176                uint32_t offset = *(key_value_list + i);
177
178                if (BAD_STRUCT(KeyValue, offset, bcd_len))
179                        continue;
180
181                const KeyValue *kv = (const KeyValue *) (bcd + offset);
182                if (kv->sig != SIG_KEY_VALUE)
183                        continue;
184
185                if (BAD_ARRAY(KeyValue, name, offset, kv->name_len, bcd_len))
186                        continue;
187
188                /* If most significant bit is set, data is stored in data_offset itself, but
189                 * we are only interested in UTF16 strings. The only strings that could fit
190                 * would have just one char in it, so let's not bother with this. */
191                if (FLAGS_SET(kv->data_size, UINT32_C(1) << 31))
192                        continue;
193
194                if (BAD_OFFSET(kv->data_offset, kv->data_size, bcd_len))
195                        continue;
196
197                if (strncaseeq8(name, kv->name, kv->name_len) && strlen8(name) == kv->name_len)
198                        return kv;
199        }
200
201        return NULL;
202}

204/* The BCD store is really just a regular windows registry hive with a rather cryptic internal
205 * key structure. On a running system it gets mounted to HKEY_LOCAL_MACHINE\BCD00000000.
206 *
207 * Of interest to us are the these two keys:
208 * - \Objects\{bootmgr}\Elements\24000001
209 *   This key is the "displayorder" property and contains a value of type REG_MULTI_SZ
210 *   with the name "Element" that holds a {GUID} list (UTF16, NUL-separated).
211 * - \Objects\{GUID}\Elements\12000004
212 *   This key is the "description" property and contains a value of type REG_SZ with the
213 *   name "Element" that holds a NUL-terminated UTF16 string.
214 *
215 * The GUIDs and properties are as reported by "bcdedit.exe /v".
216 *
217 * To get a title for the BCD store we first look at the displayorder property of {bootmgr}
218 * (it always has the GUID 9dea862c-5cdd-4e70-acc1-f32b344d4795). If it contains more than
219 * one GUID, the BCD is multi-boot and we stop looking. Otherwise we take that GUID, look it
220 * up, and return its description property. */
