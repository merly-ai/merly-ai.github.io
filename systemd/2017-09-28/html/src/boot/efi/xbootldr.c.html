15static EFI_DEVICE_PATH *path_chop(EFI_DEVICE_PATH *path, EFI_DEVICE_PATH *node) {
16        assert(path);
17        assert(node);
18
19        UINTN len = (uint8_t *) node - (uint8_t *) path;
20        EFI_DEVICE_PATH *chopped = xmalloc(len + END_DEVICE_PATH_LENGTH);
21
22        memcpy(chopped, path, len);
23        SetDevicePathEndNode((EFI_DEVICE_PATH *) ((uint8_t *) chopped + len));
24
25        return chopped;
26}

28static EFI_DEVICE_PATH *path_dup(const EFI_DEVICE_PATH *dp) {
29        assert(dp);
30
31        const EFI_DEVICE_PATH *node = dp;
32        size_t size = 0;
33        while (!IsDevicePathEnd(node)) {
34                size += DevicePathNodeLength(node);
35                node = NextDevicePathNode(node);
36        }
37        size += DevicePathNodeLength(node);
38
39        EFI_DEVICE_PATH *dup = xmalloc(size);
40        memcpy(dup, dp, size);
41        return dup;
42}

44static bool verify_gpt(union GptHeaderBuffer *gpt_header_buffer, EFI_LBA lba_expected) {
45        EFI_PARTITION_TABLE_HEADER *h;
46        uint32_t crc32, crc32_saved;
47        EFI_STATUS err;
48
49        assert(gpt_header_buffer);
50
51        h = &gpt_header_buffer->gpt_header;
52
53        /* Some superficial validation of the GPT header */
54        if (memcmp(&h->Header.Signature, "EFI PART", sizeof(h->Header.Signature)) != 0)
55                return false;
56
57        if (h->Header.HeaderSize < 92 || h->Header.HeaderSize > 512)
58                return false;
59
60        if (h->Header.Revision != 0x00010000U)
61                return false;
62
63        /* Calculate CRC check */
64        crc32_saved = h->Header.CRC32;
65        h->Header.CRC32 = 0;
66        err = BS->CalculateCrc32(gpt_header_buffer, h->Header.HeaderSize, &crc32);
67        h->Header.CRC32 = crc32_saved;
68        if (err != EFI_SUCCESS || crc32 != crc32_saved)
69                return false;
70
71        if (h->MyLBA != lba_expected)
72                return false;
73
74        if (h->SizeOfPartitionEntry < sizeof(EFI_PARTITION_ENTRY))
75                return false;
76
77        if (h->NumberOfPartitionEntries <= 0 || h->NumberOfPartitionEntries > 1024)
78                return false;
79
80        /* overflow check */
81        if (h->SizeOfPartitionEntry > UINTN_MAX / h->NumberOfPartitionEntries)
82                return false;
83
84        return true;
85}

87static EFI_STATUS try_gpt(
88                EFI_BLOCK_IO_PROTOCOL *block_io,
89                EFI_LBA lba,
90                EFI_LBA *ret_backup_lba, /* May be changed even on error! */
91                HARDDRIVE_DEVICE_PATH *ret_hd) {
92
93        _cleanup_free_ EFI_PARTITION_ENTRY *entries = NULL;
94        union GptHeaderBuffer gpt;
95        EFI_STATUS err;
96        uint32_t crc32;
97        UINTN size;
98
99        assert(block_io);
100        assert(ret_hd);
101
102        /* Read the GPT header */
103        err = block_io->ReadBlocks(
104                        block_io,
105                        block_io->Media->MediaId,
106                        lba,
107                        sizeof(gpt), &gpt);
108        if (err != EFI_SUCCESS)
109                return err;
110
111        /* Indicate the location of backup LBA even if the rest of the header is corrupt. */
112        if (ret_backup_lba)
113                *ret_backup_lba = gpt.gpt_header.AlternateLBA;
114
115        if (!verify_gpt(&gpt, lba))
116                return EFI_NOT_FOUND;
117
118        /* Now load the GPT entry table */
119        size = ALIGN_TO((UINTN) gpt.gpt_header.SizeOfPartitionEntry * (UINTN) gpt.gpt_header.NumberOfPartitionEntries, 512);
120        entries = xmalloc(size);
121
122        err = block_io->ReadBlocks(
123                        block_io,
124                        block_io->Media->MediaId,
125                        gpt.gpt_header.PartitionEntryLBA,
126                        size, entries);
127        if (err != EFI_SUCCESS)
128                return err;
129
130        /* Calculate CRC of entries array, too */
131        err = BS->CalculateCrc32(entries, size, &crc32);
132        if (err != EFI_SUCCESS || crc32 != gpt.gpt_header.PartitionEntryArrayCRC32)
133                return EFI_CRC_ERROR;
134
135        /* Now we can finally look for xbootloader partitions. */
136        for (UINTN i = 0; i < gpt.gpt_header.NumberOfPartitionEntries; i++) {
137                EFI_PARTITION_ENTRY *entry;
138                EFI_LBA start, end;
139
140                entry = (EFI_PARTITION_ENTRY*) ((uint8_t*) entries + gpt.gpt_header.SizeOfPartitionEntry * i);
141
142                if (memcmp(&entry->PartitionTypeGUID, XBOOTLDR_GUID, sizeof(entry->PartitionTypeGUID)) != 0)
143                        continue;
144
145                /* Let's use memcpy(), in case the structs are not aligned (they really should be though) */
146                memcpy(&start, &entry->StartingLBA, sizeof(start));
147                memcpy(&end, &entry->EndingLBA, sizeof(end));
148
149                if (end < start) /* Bogus? */
150                        continue;
151
152                ret_hd->PartitionNumber = i + 1;
153                ret_hd->PartitionStart = start;
154                ret_hd->PartitionSize = end - start + 1;
155                ret_hd->MBRType = MBR_TYPE_EFI_PARTITION_TABLE_HEADER;
156                ret_hd->SignatureType = SIGNATURE_TYPE_GUID;
157                memcpy(ret_hd->Signature, &entry->UniquePartitionGUID, sizeof(ret_hd->Signature));
158
159                return EFI_SUCCESS;
160        }
161
162        /* This GPT was fully valid, but we didn't find what we are looking for. This
163         * means there's no reason to check the second copy of the GPT header */
164        return EFI_NOT_FOUND;
165}

167static EFI_STATUS find_device(EFI_HANDLE *device, EFI_DEVICE_PATH **ret_device_path) {
168        EFI_STATUS err;
169
170        assert(device);
171        assert(ret_device_path);
172
173        EFI_DEVICE_PATH *partition_path;
174        err = BS->HandleProtocol(device, &DevicePathProtocol, (void **) &partition_path);
175        if (err != EFI_SUCCESS)
176                return err;
177
178        /* Find the (last) partition node itself. */
179        EFI_DEVICE_PATH *part_node = NULL;
180        for (EFI_DEVICE_PATH *node = partition_path; !IsDevicePathEnd(node); node = NextDevicePathNode(node)) {
181                if (DevicePathType(node) != MEDIA_DEVICE_PATH)
182                        continue;
183
184                if (DevicePathSubType(node) != MEDIA_HARDDRIVE_DP)
185                        continue;
186
187                part_node = node;
188        }
189
190        if (!part_node)
191                return EFI_NOT_FOUND;
192
193        /* Chop off the partition part, leaving us with the full path to the disk itself. */
194        _cleanup_free_ EFI_DEVICE_PATH *disk_path = NULL;
195        EFI_DEVICE_PATH *p = disk_path = path_chop(partition_path, part_node);
196
197        EFI_HANDLE disk_handle;
198        EFI_BLOCK_IO_PROTOCOL *block_io;
199        err = BS->LocateDevicePath(&BlockIoProtocol, &p, &disk_handle);
200        if (err != EFI_SUCCESS)
201                return err;
202
203        err = BS->HandleProtocol(disk_handle, &BlockIoProtocol, (void **)&block_io);
204        if (err != EFI_SUCCESS)
205                return err;
206
207        /* Filter out some block devices early. (We only care about block devices that aren't
208         * partitions themselves — we look for GPT partition tables to parse after all —, and only
209         * those which contain a medium and have at least 2 blocks.) */
210        if (
 Recommendations (Experimental)  R1: (block_io || !Media.LogicalPartition || MediaPresent.LastBlock <= 1)
R2: (block_io || !Media || LogicalPartition.MediaPresent == {const})
R3: (block_io || !Media || LogicalPartition == {const})
R4: (block_io.Media || !LogicalPartition || MediaPresent->LastBlock != 1)
R5: (block_io || !Media || Media->LogicalPartition == 1)
 Score  Issue:  1 unfamiliar pattern(s) detected 
Cost: 22.00, Total Score: 1022.00
Class: complex_2
block_io->Media->LogicalPartition ||
211            !block_io->Media->MediaPresent ||
212            block_io->Media->LastBlock <= 1)
213                return EFI_NOT_FOUND;
214
215        /* Try several copies of the GPT header, in case one is corrupted */
216        EFI_LBA backup_lba = 0;
217        HARDDRIVE_DEVICE_PATH hd = *((HARDDRIVE_DEVICE_PATH *) part_node);
218        for (UINTN nr = 0; nr < 3; nr++) {
219                EFI_LBA lba;
220
221                /* Read the first copy at LBA 1 and then try the backup GPT header pointed
222                 * to by the first header if that one was corrupted. As a last resort,
223                 * try the very last LBA of this block device. */
224                if (nr == 0)
225                        lba = 1;
226                else if (nr == 1 && backup_lba != 0)
227                        lba = backup_lba;
228                else if (nr == 2 && backup_lba != block_io->Media->LastBlock)
229                        lba = block_io->Media->LastBlock;
230                else
231                        continue;
232
233                err = try_gpt(
234                        block_io, lba,
235                        nr == 0 ? &backup_lba : NULL, /* Only get backup LBA location from first GPT header. */
236                        &hd);
237                if (err != EFI_SUCCESS) {
238                        /* GPT was valid but no XBOOT loader partition found. */
239                        if (err == EFI_NOT_FOUND)
240                                break;
241                        /* Bad GPT, try next one. */
242                        continue;
243                }
244
245                /* Patch in the data we found */
246                EFI_DEVICE_PATH *xboot_path = path_dup(partition_path);
247                memcpy((uint8_t *) xboot_path + ((uint8_t *) part_node - (uint8_t *) partition_path), &hd, sizeof(hd));
248                *ret_device_path = xboot_path;
249                return EFI_SUCCESS;
250        }
251
252        /* No xbootloader partition found */
253        return EFI_NOT_FOUND;
254}

256EFI_STATUS xbootldr_open(EFI_HANDLE *device, EFI_HANDLE *ret_device, EFI_FILE **ret_root_dir) {
257        _cleanup_free_ EFI_DEVICE_PATH *partition_path = NULL;
258        EFI_HANDLE new_device;
259        EFI_FILE *root_dir;
260        EFI_STATUS err;
261
262        assert(device);
263        assert(ret_device);
264        assert(ret_root_dir);
265
266        err = find_device(device, &partition_path);
267        if (err != EFI_SUCCESS)
268                return err;
269
270        EFI_DEVICE_PATH *dp = partition_path;
271        err = BS->LocateDevicePath(&BlockIoProtocol, &dp, &new_device);
272        if (err != EFI_SUCCESS)
273                return err;
274
275        err = open_volume(new_device, &root_dir);
276        if (err != EFI_SUCCESS)
277                return err;
278
279        *ret_device = new_device;
280        *ret_root_dir = root_dir;
281        return EFI_SUCCESS;
282}
