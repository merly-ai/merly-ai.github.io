50static usec_t get_startup_time(Context *c) {
51        _cleanup_(sd_bus_error_free) sd_bus_error error = SD_BUS_ERROR_NULL;
52        usec_t t = 0;
53        int r;
54
55        assert(c);
56
57        r = sd_bus_get_property_trivial(
58                        c->bus,
59                        "org.freedesktop.systemd1",
60                        "/org/freedesktop/systemd1",
61                        "org.freedesktop.systemd1.Manager",
62                        "UserspaceTimestamp",
63                        &error,
64                        't', &t);
65        if (r < 0) {
66                log_error_errno(r, "Failed to get timestamp: %s", bus_error_message(&error, r));
67                return 0;
68        }
69
70        return t;
71}

73static int get_current_runlevel(Context *c) {
74        static const struct {
75                const int runlevel;
76                const char *special;
77        } table[] = {
78                /* The first target of this list that is active or has
79                 * a job scheduled wins. We prefer runlevels 5 and 3
80                 * here over the others, since these are the main
81                 * runlevels used on Fedora. It might make sense to
82                 * change the order on some distributions. */
83                { '5', SPECIAL_GRAPHICAL_TARGET  },
84                { '3', SPECIAL_MULTI_USER_TARGET },
85                { '1', SPECIAL_RESCUE_TARGET     },
86        };
87
88        _cleanup_(sd_bus_error_free) sd_bus_error error = SD_BUS_ERROR_NULL;
89        int r;
90        unsigned i;
91
92        assert(c);
93
94        for (i = 0; i < ELEMENTSOF(table); i++) {
95                _cleanup_free_ char *state = NULL, *path = NULL;
96
97                path = unit_dbus_path_from_name(table[i].special);
98                if (!path)
99                        return log_oom();
100
101                r = sd_bus_get_property_string(
102                                c->bus,
103                                "org.freedesktop.systemd1",
104                                path,
105                                "org.freedesktop.systemd1.Unit",
106                                "ActiveState",
107                                &error,
108                                &state);
109                if (r < 0)
110                        return log_warning_errno(r, "Failed to get state: %s", bus_error_message(&error, r));
111
112                if (STR_IN_SET(state, "active", "reloading"))
113                        return table[i].runlevel;
114        }
115
116        return 0;
117}

119static int on_reboot(Context *c) {
120        int r = 0, q;
121        usec_t t;
122
123        assert(c);
124
125        /* We finished start-up, so let's write the utmp
126         * record and send the audit msg */
127
128#ifdef HAVE_AUDIT
129        if (c->audit_fd >= 0)
130                if (audit_log_user_comm_message(c->audit_fd, AUDIT_SYSTEM_BOOT, "", "systemd-update-utmp", NULL, NULL, NULL, 1) < 0 &&
131                    errno != EPERM) {
132                        r = log_error_errno(errno, "Failed to send audit message: %m");
133                }
134#endif
135
136        /* If this call fails it will return 0, which
137         * utmp_put_reboot() will then fix to the current time */
138        t = get_startup_time(c);
139
140        q = utmp_put_reboot(t);
141        if (q < 0) {
142                log_error_errno(q, "Failed to write utmp record: %m");
143                r = q;
144        }
145
146        return r;
147}

174static int on_runlevel(Context *c) {
175        int r = 0, q, previous, runlevel;
176
177        assert(c);
178
179        /* We finished changing runlevel, so let's write the
180         * utmp record and send the audit msg */
181
182        /* First, get last runlevel */
183        q = utmp_get_runlevel(&previous, NULL);
184
185        if (q < 0) {
186                if (q != -ESRCH && q != -ENOENT)
187                        return log_error_errno(q, "Failed to get current runlevel: %m");
188
189                previous = 0;
190        }
191
192        /* Secondly, get new runlevel */
193        runlevel = get_current_runlevel(c);
194
195        if (runlevel < 0)
196                return runlevel;
197
198        if (previous == runlevel)
199                return 0;
200
201#ifdef HAVE_AUDIT
202        if (c->audit_fd >= 0) {
203                _cleanup_free_ char *s = NULL;
204
205                if (asprintf(&s, "old-level=%c new-level=%c",
206                             previous > 0 ? previous : 'N',
207                             runlevel > 0 ? runlevel : 'N') < 0)
208                        return log_oom();
209
210                if (audit_log_user_comm_message(c->audit_fd, AUDIT_SYSTEM_RUNLEVEL, s, "systemd-update-utmp", NULL, NULL, NULL, 1) < 0 && errno != EPERM)
211                        r = log_error_errno(errno, "Failed to send audit message: %m");
212        }
213#endif
214
215        q = utmp_put_runlevel(runlevel, previous);
216        if (q < 0 && q != -ESRCH && q != -ENOENT) {
217                log_error_errno(q, "Failed to write utmp record: %m");
218                r = q;
219        }
220
221        return r;
222}
