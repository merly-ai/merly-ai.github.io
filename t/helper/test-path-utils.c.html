57  const char *funcname)

212/*
213 * A fast approximation of the square root, without requiring math.h.
214 *
215 * It uses Newton's method to approximate the solution of 0 = x^2 - value.
216 */
217static double my_sqrt(double value)
218{
219  const double epsilon = 1e-6;
220  double x = value;
221
222  if (value == 0)
223    return 0;
224
225  for (;;) {
226    double delta = (value / x - x) / 2;
227    if (delta < epsilon && delta > -epsilon)
228      return x + delta;
229    x += delta;
230  }
231}

485      else
486        fprintf(stderr,
487          "'%s' is%s a valid path\n",
488          argv[i], expect ? "" : " not");
489
490    return !!res;
491  }
492
493  fprintf(stderr, "%s: unknown function name: %s\n", argv[0],
494    argv[1] ? argv[1] : "(there was none)");
495  return 1;
496}
