11static void create_directories(const char *path, int path_len,
12             const struct checkout *state)
13{
14  char *buf = xmallocz(path_len);
15  int len = 0;
16
17  while (len < path_len) {
18    do {
19      buf[len] = path[len];
20      len++;
21    } while (len < path_len && path[len] != '/');
22    if (len >= path_len)
23      break;
24    buf[len] = 0;
25
26    /*
27     * For 'checkout-index --prefix=<dir>', <dir> is
28     * allowed to be a symlink to an existing directory,
29     * and we set 'state->base_dir_len' below, such that
30     * we test the path components of the prefix with the
31     * stat() function instead of the lstat() function.
32     */
33    if (has_dirs_only_path(buf, len, state->base_dir_len))
34      continue; /* ok, it is already a directory. */
35
36    /*
37     * If this mkdir() would fail, it could be that there
38     * is already a symlink or something else exists
39     * there, therefore we then try to unlink it and try
40     * one more time to create the directory.
41     */
42    if (mkdir(buf, 0777)) {
43      if (errno == EEXIST && state->force &&
44          !unlink_or_warn(buf) && !mkdir(buf, 0777))
45        continue;
46      die_errno("cannot create directory at '%s'", buf);
47    }
48  }
49  free(buf);
50}

52static void remove_subtree(struct strbuf *path)
53{
54  DIR *dir = opendir(path->buf);
55  struct dirent *de;
56  int origlen = path->len;
57
58  if (!dir)
59    die_errno("cannot opendir '%s'", path->buf);
60  while ((de = readdir_skip_dot_and_dotdot(dir)) != NULL) {
61    struct stat st;
62
63    strbuf_addch(path, '/');
64    strbuf_addstr(path, de->d_name);
65    if (lstat(path->buf, &st))
66      die_errno("cannot lstat '%s'", path->buf);
67    if (S_ISDIR(st.st_mode))
68      remove_subtree(path);
69    else if (unlink(path->buf))
70      die_errno("cannot unlink '%s'", path->buf);
71    strbuf_setlen(path, origlen);
72  }
73  closedir(dir);
74  if (rmdir(path->buf))
75    die_errno("cannot rmdir '%s'", path->buf);
76}

163int finish_delayed_checkout(struct checkout *state, int *nr_checkouts,
164          int show_progress)
165{
166  int errs = 0;
167  unsigned processed_paths = 0;
168  off_t filtered_bytes = 0;
169  struct string_list_item *filter, *path;
170  struct progress *progress = NULL;
171  struct delayed_checkout *dco = state->delayed_checkout;
172
173  if (!state->delayed_checkout)
174    return errs;
175
176  dco->state = CE_RETRY;
177  if (show_progress)
178    progress = start_delayed_progress(_("Filtering content"), dco->paths.nr);
179  while (dco->filters.nr > 0) {
180    for_each_string_list_item(filter, &dco->filters) {
181      struct string_list available_paths = STRING_LIST_INIT_NODUP;
182
183      if (!async_query_available_blobs(filter->string, &available_paths)) {
184        /* Filter reported an error */
185        errs = 1;
186        filter->string = "";
187        continue;
188      }
189      if (available_paths.nr <= 0) {
190        /*
191         * Filter responded with no entries. That means
192         * the filter is done and we can remove the
193         * filter from the list (see
194         * "string_list_remove_empty_items" call below).
195         */
196        filter->string = "";
197        continue;
198      }
199
200      /*
201       * In dco->paths we store a list of all delayed paths.
202       * The filter just send us a list of available paths.
203       * Remove them from the list.
204       */
205      filter_string_list(&dco->paths, 0,
206        &remove_available_paths, &available_paths);
207
208      for_each_string_list_item(path, &available_paths) {
209        struct cache_entry* ce;
210
211        if (!path->util) {
212          error("external filter '%s' signaled that '%s' "
213                "is now available although it has not been "
214                "delayed earlier",
215                filter->string, path->string);
216          errs |= 1;
217
218          /*
219           * Do not ask the filter for available blobs,
220           * again, as the filter is likely buggy.
221           */
222          filter->string = "";
223          continue;
224        }
225        ce = index_file_exists(state->istate, path->string,
226                   strlen(path->string), 0);
227        if (ce) {
228          display_progress(progress, ++processed_paths);
229          errs |= checkout_entry(ce, state, NULL, nr_checkouts);
230          filtered_bytes += ce->ce_stat_data.sd_size;
231          display_throughput(progress, filtered_bytes);
232        } else
233          errs = 1;
234      }
235    }
236    string_list_remove_empty_items(&dco->filters, 0);
237  }
238  stop_progress(&progress);
239  string_list_clear(&dco->filters, 0);
240
241  /* At this point we should not have any delayed paths anymore. */
242  errs |= dco->paths.nr;
243  for_each_string_list_item(path, &dco->paths) {
244    error("'%s' was not filtered properly", path->string);
245  }
246  string_list_clear(&dco->paths, 0);
247
248  free(dco);
249  state->delayed_checkout = NULL;
250
251  return errs;
252}

266/* Note: ca is used (and required) iff the entry refers to a regular file. */
267static int write_entry(struct cache_entry *ce, char *path, struct conv_attrs *ca,
268           const struct checkout *state, int to_tempfile)
269{
270  unsigned int ce_mode_s_ifmt = ce->ce_mode & S_IFMT;
271  struct delayed_checkout *dco = state->delayed_checkout;
272  int fd, ret, fstat_done = 0;
273  char *new_blob;
274  struct strbuf buf = STRBUF_INIT;
275  size_t size;
276  ssize_t wrote;
277  size_t newsize = 0;
278  struct stat st;
279  const struct submodule *sub;
280  struct checkout_metadata meta;
281
282  clone_checkout_metadata(&meta, &state->meta, &ce->oid);
283
284  if (ce_mode_s_ifmt == S_IFREG) {
285    struct stream_filter *filter = get_stream_filter_ca(ca, &ce->oid);
286    if (filter &&
287        !streaming_write_entry(ce, path, filter,
288             state, to_tempfile,
289             &fstat_done, &st))
290      goto finish;
291  }
292
293  switch (ce_mode_s_ifmt) {
294  case S_IFLNK:
295    new_blob = read_blob_entry(ce, &size);
296    if (!new_blob)
297      return error("unable to read sha1 file of %s (%s)",
298             ce->name, oid_to_hex(&ce->oid));
299
300    /*
301     * We can't make a real symlink; write out a regular file entry
302     * with the symlink destination as its contents.
303     */
304    if (!has_symlinks || to_tempfile)
305      goto write_file_entry;
306
307    ret = symlink(new_blob, path);
308    free(new_blob);
309    if (ret)
310      return error_errno("unable to create symlink %s", path);
311    break;
312
313  case S_IFREG:
314    /*
315     * We do not send the blob in case of a retry, so do not
316     * bother reading it at all.
317     */
318    if (dco && dco->state == CE_RETRY) {
319      new_blob = NULL;
320      size = 0;
321    } else {
322      new_blob = read_blob_entry(ce, &size);
323      if (!new_blob)
324        return error("unable to read sha1 file of %s (%s)",
325               ce->name, oid_to_hex(&ce->oid));
326    }
327
328    /*
329     * Convert from git internal format to working tree format
330     */
331    if (dco && dco->state != CE_NO_DELAY) {
332      ret = async_convert_to_working_tree_ca(ca, ce->name,
333                     new_blob, size,
334                     &buf, &meta, dco);
335      if (ret && string_list_has_string(&dco->paths, ce->name)) {
336        free(new_blob);
337        goto delayed;
338      }
339    } else {
340      ret = convert_to_working_tree_ca(ca, ce->name, new_blob,
341               size, &buf, &meta);
342    }
343
344    if (ret) {
345      free(new_blob);
346      new_blob = strbuf_detach(&buf, &newsize);
347      size = newsize;
348    }
349    /*
350     * No "else" here as errors from convert are OK at this
351     * point. If the error would have been fatal (e.g.
352     * filter is required), then we would have died already.
353     */
354
355  write_file_entry:
356    fd = open_output_fd(path, ce, to_tempfile);
357    if (fd < 0) {
358      free(new_blob);
359      return error_errno("unable to create file %s", path);
360    }
361
362    wrote = write_in_full(fd, new_blob, size);
363    if (!to_tempfile)
364      fstat_done = fstat_checkout_output(fd, state, &st);
365    close(fd);
366    free(new_blob);
367    if (wrote < 0)
368      return error("unable to write file %s", path);
369    break;
370
371  case S_IFGITLINK:
372    if (to_tempfile)
373      return error("cannot create temporary submodule %s", ce->name);
374    if (mkdir(path, 0777) < 0)
375      return error("cannot create submodule directory %s", path);
376    sub = submodule_from_ce(ce);
377    if (sub)
378      return submodule_move_head(ce->name,
379        NULL, oid_to_hex(&ce->oid),
380        state->force ? SUBMODULE_MOVE_HEAD_FORCE : 0);
381    break;
382
383  default:
384    return error("unknown file mode for %s in index", ce->name);
385  }
386
387finish:
388  if (state->refresh_cache) {
389    if (!fstat_done && lstat(ce->name, &st) < 0)
390      return error_errno("unable to stat just-written file %s",
391             ce->name);
392    update_ce_after_write(state, ce , &st);
393  }
394delayed:
395  return 0;
396}

398/*
399 * This is like 'lstat()', except it refuses to follow symlinks
400 * in the path, after skipping "skiplen".
401 */
402static int check_path(const char *path, int len, struct stat *st, int skiplen)
403{
404  const char *slash = path + len;
405
406  while (path < slash && *slash != '/')
407    slash--;
408  if (!has_dirs_only_path(path, slash - path, skiplen)) {
409    errno = ENOENT;
410    return -1;
411  }
412  return lstat(path, st);
413}

415static void mark_colliding_entries(const struct checkout *state,
416           struct cache_entry *ce, struct stat *st)
417{
418  int i, trust_ino = check_stat;
419
420#if defined(GIT_WINDOWS_NATIVE) || defined(__CYGWIN__)
421  trust_ino = 0;
422#endif
423
424  ce->ce_flags |= CE_MATCHED;
425
426  /* TODO: audit for interaction with sparse-index. */
427  ensure_full_index(state->istate);
428  for (i = 0; i < state->istate->cache_nr; i++) {
429    struct cache_entry *dup = state->istate->cache[i];
430
431    if (dup == ce) {
432      /*
433       * Parallel checkout doesn't create the files in index
434       * order. So the other side of the collision may appear
435       * after the given cache_entry in the array.
436       */
437      if (parallel_checkout_status() == PC_RUNNING)
438        continue;
439      else
440        break;
441    }
442
443    if (dup->ce_flags & (CE_MATCHED | CE_VALID | CE_SKIP_WORKTREE))
444      continue;
445
446    if ((trust_ino && !match_stat_data(&dup->ce_stat_data, st)) ||
447        (!trust_ino && !fspathcmp(ce->name, dup->name))) {
448      dup->ce_flags |= CE_MATCHED;
449      break;
450    }
451  }
452}

454int checkout_entry_ca(struct cache_entry *ce, struct conv_attrs *ca,
455          const struct checkout *state, char *topath,
456          int *nr_checkouts)
457{
458  static struct strbuf path = STRBUF_INIT;
459  struct stat st;
460  struct conv_attrs ca_buf;
461
462  if (ce->ce_flags & CE_WT_REMOVE) {
463    if (topath)
464      /*
465       * No content and thus no path to create, so we have
466       * no pathname to return.
467       */
468      BUG("Can't remove entry to a path");
469    unlink_entry(ce);
470    return 0;
471  }
472
473  if (topath) {
474    if (S_ISREG(ce->ce_mode) && !ca) {
475      convert_attrs(state->istate, &ca_buf, ce->name);
476      ca = &ca_buf;
477    }
478    return write_entry(ce, topath, ca, state, 1);
479  }
480
481  strbuf_reset(&path);
482  strbuf_add(&path, state->base_dir, state->base_dir_len);
483  strbuf_add(&path, ce->name, ce_namelen(ce));
484
485  if (!check_path(path.buf, path.len, &st, state->base_dir_len)) {
486    const struct submodule *sub;
487    unsigned changed = ie_match_stat(state->istate, ce, &st,
488             CE_MATCH_IGNORE_VALID | CE_MATCH_IGNORE_SKIP_WORKTREE);
489    /*
490     * Needs to be checked before !changed returns early,
491     * as the possibly empty directory was not changed
492     */
493    sub = submodule_from_ce(ce);
494    if (sub) {
495      int err;
496      if (!is_submodule_populated_gently(ce->name, &err)) {
497        struct stat sb;
498        if (lstat(ce->name, &sb))
499          die(_("could not stat file '%s'"), ce->name);
500        if (!(st.st_mode & S_IFDIR))
501          unlink_or_warn(ce->name);
502
503        return submodule_move_head(ce->name,
504          NULL, oid_to_hex(&ce->oid), 0);
505      } else
506        return submodule_move_head(ce->name,
507          "HEAD", oid_to_hex(&ce->oid),
508          state->force ? SUBMODULE_MOVE_HEAD_FORCE : 0);
509    }
510
511    if (!changed)
512      return 0;
513    if (!state->force) {
514      if (!state->quiet)
515        fprintf(stderr,
516          "%s already exists, no checkout\n",
517          path.buf);
518      return -1;
519    }
520
521    if (state->clone)
522      mark_colliding_entries(state, ce, &st);
523
524    /*
525     * We unlink the old file, to get the new one with the
526     * right permissions (including umask, which is nasty
527     * to emulate by hand - much easier to let the system
528     * just do the right thing)
529     */
530    if (S_ISDIR(st.st_mode)) {
531      /* If it is a gitlink, leave it alone! */
532      if (S_ISGITLINK(ce->ce_mode))
533        return 0;
534      remove_subtree(&path);
535    } else if (unlink(path.buf))
536      return error_errno("unable to unlink old '%s'", path.buf);
537  } else if (state->not_new)
538    return 0;
539
540  create_directories(path.buf, path.len, state);
541
542  if (nr_checkouts)
543    (*nr_checkouts)++;
544
545  if (S_ISREG(ce->ce_mode) && !ca) {
546    convert_attrs(state->istate, &ca_buf, ce->name);
547    ca = &ca_buf;
548  }
549
550  if (!enqueue_checkout(ce, ca))
551    return 0;
552
553  return write_entry(ce, path.buf, ca, state, 0);
554}
555
556void unlink_entry(const struct cache_entry *ce)
557{
558  const struct submodule *sub = submodule_from_ce(ce);
559  if (sub) {
560    /* state.force is set at the caller. */
561    submodule_move_head(ce->name, "HEAD", NULL,
562            SUBMODULE_MOVE_HEAD_FORCE);
563  }
564  if (check_leading_path(ce->name, ce_namelen(ce), 1) >= 0)
565    return;
566  if (remove_or_warn(ce->ce_mode, ce->name))
567    return;
568  schedule_dir_for_removal(ce->name, ce_namelen(ce));
569}
