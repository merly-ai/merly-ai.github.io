123int is_executable(const char *name)
124{
125  struct stat st;
126
127  if (stat(name, &st) || /* stat, not lstat */
128      !S_ISREG(st.st_mode))
129    return 0;
130
131#if defined(GIT_WINDOWS_NATIVE)
132  /*
133   * On Windows there is no executable bit. The file extension
134   * indicates whether it can be run as an executable, and Git
135   * has special-handling to detect scripts and launch them
136   * through the indicated script interpreter. We test for the
137   * file extension first because virus scanners may make
138   * it quite expensive to open many files.
139   */
140  if (ends_with(name, ".exe"))
141    return S_IXUSR;

143{

146   * peek into the file instead.
147   */

156      if (!strcmp(buf, "#!"))

165/*
166 * Search $PATH for a command.  This emulates the path search that
167 * execvp would perform, without actually executing the command so it
168 * can be used before fork() to prepare to run a command using
169 * execve() or after execvp() to diagnose why it failed.
170 *
171 * The caller should ensure that file contains no directory
172 * separators.
173 *
174 * Returns the path to the command, as found in $PATH or NULL if the
175 * command could not be found.  The caller inherits ownership of the memory
176 * used to store the resultant path.
177 *
178 * This should not be used on Windows, where the $PATH search rules
179 * are more complicated (e.g., a search for "foo" should find
180 * "foo.exe").
181 */
182static char *locate_in_PATH(const char *file)
183{
184  const char *p = getenv("PATH");
185  struct strbuf buf = STRBUF_INIT;
186
187  if (!p || !*p)
188    return NULL;
189
190  while (1) {
191    const char *end = strchrnul(p, ':');
192
193    strbuf_reset(&buf);
194
195    /* POSIX specifies an empty entry as the current directory. */
196    if (end != p) {
197      strbuf_add(&buf, p, end - p);
198      strbuf_addch(&buf, '/');
199    }
200    strbuf_addstr(&buf, file);
201
202    if (is_executable(buf.buf))
203      return strbuf_detach(&buf, NULL);
204
205    if (!*end)
206      break;
207    p = end + 1;
208  }
209
210  strbuf_release(&buf);
211  return NULL;
212}

222int sane_execvp(const char *file, char * const argv[])
223{
224#ifndef GIT_WINDOWS_NATIVE
225  /*
226   * execvp() doesn't return, so we all we can do is tell trace2
227   * what we are about to do and let it leave a hint in the log
228   * (unless of course the execvp() fails).
229   *
230   * we skip this for Windows because the compat layer already
231   * has to emulate the execvp() call anyway.
232   */
233  int exec_id = trace2_exec(file, (const char **)argv);
234#endif
235
236  if (!execvp(file, argv))
237    return 0; /* cannot happen ;-) */
238
239#ifndef GIT_WINDOWS_NATIVE
240  {
241    int ec = errno;
242    trace2_exec_result(exec_id, ec);
243    errno = ec;
244  }
245#endif
246
247  /*
248   * When a command can't be found because one of the directories
249   * listed in $PATH is unsearchable, execvp reports EACCES, but
250   * careful usability testing (read: analysis of occasional bug
251   * reports) reveals that "No such file or directory" is more
252   * intuitive.
253   *
254   * We avoid commands with "/", because execvp will not do $PATH
255   * lookups in that case.
256   *
257   * The reassignment of EACCES to errno looks like a no-op below,
258   * but we need to protect against exists_in_PATH overwriting errno.
259   */
260  if (errno == EACCES && !strchr(file, '/'))
261    errno = exists_in_PATH(file) ? EACCES : ENOENT;
262  else if (errno == ENOTDIR && !strchr(file, '/'))
263    errno = ENOENT;
264  return -1;
265}

397static int prepare_cmd(struct strvec *out, const struct child_process *cmd)
398{
399  if (!cmd->args.v[0])
400    BUG("command is empty");
401
402  /*
403   * Add SHELL_PATH so in the event exec fails with ENOEXEC we can
404   * attempt to interpret the command with 'sh'.
405   */
406  strvec_push(out, SHELL_PATH);
407
408  if (cmd->git_cmd) {
409    prepare_git_cmd(out, cmd->args.v);
410  } else if (cmd->use_shell) {
411    prepare_shell_cmd(out, cmd->args.v);
412  } else {
413    strvec_pushv(out, cmd->args.v);
414  }
415
416  /*
417   * If there are no dir separator characters in the command then perform
418   * a path lookup and use the resolved path as the command to exec. If
419   * there are dir separator characters, we have exec attempt to invoke
420   * the command directly.
421   */
422  if (!has_dir_sep(out->v[1])) {
423    char *program = locate_in_PATH(out->v[1]);
424    if (program) {
425      free((char *)out->v[1]);
426      out->v[1] = program;
427    } else {
428      strvec_clear(out);
429      errno = ENOENT;
430      return -1;
431    }
432  }
433
434  return 0;
435}

437static char **prep_childenv(const char *const *deltaenv)
438{
439  extern char **environ;
440  char **childenv;
441  struct string_list env = STRING_LIST_INIT_DUP;
442  struct strbuf key = STRBUF_INIT;
443  const char *const *p;
444  int i;
445
446  /* Construct a sorted string list consisting of the current environ */
447  for (p = (const char *const *) environ; p && *p; p++) {
448    const char *equals = strchr(*p, '=');
449
450    if (equals) {
451      strbuf_reset(&key);
452      strbuf_add(&key, *p, equals - *p);
453      string_list_append(&env, key.buf)->util = (void *) *p;
454    } else {
455      string_list_append(&env, *p)->util = (void *) *p;
456    }
457  }
458  string_list_sort(&env);
459
460  /* Merge in 'deltaenv' with the current environ */
461  for (p = deltaenv; p && *p; p++) {
462    const char *equals = strchr(*p, '=');
463
464    if (equals) {
465      /* ('key=value'), insert or replace entry */
466      strbuf_reset(&key);
467      strbuf_add(&key, *p, equals - *p);
468      string_list_insert(&env, key.buf)->util = (void *) *p;
469    } else {
470      /* otherwise ('key') remove existing entry */
471      string_list_remove(&env, *p, 0);
472    }
473  }
474
475  /* Create an array of 'char *' to be used as the childenv */
476  ALLOC_ARRAY(childenv, env.nr + 1);
477  for (i = 0; i < env.nr; i++)
478    childenv[i] = env.items[i].util;
479  childenv[env.nr] = NULL;
480
481  string_list_clear(&env, 0);
482  strbuf_release(&key);
483  return childenv;
484}

538static int wait_or_whine(pid_t pid, const char *argv0, int in_signal)
539{
540  int status, code = -1;
541  pid_t waiting;
542  int failed_errno = 0;
543
544  while ((waiting = waitpid(pid, &status, 0)) < 0 && errno == EINTR)
545    ;  /* nothing */
546
547  if (waiting < 0) {
548    failed_errno = errno;
549    if (!in_signal)
550      error_errno("waitpid for %s failed", argv0);
551  } else if (waiting != pid) {
552    if (!in_signal)
553      error("waitpid is confused (%s)", argv0);
554  } else if (WIFSIGNALED(status)) {
555    code = WTERMSIG(status);
556    if (!in_signal && code != SIGINT && code != SIGQUIT && code != SIGPIPE)
557      error("%s died of signal %d", argv0, code);
558    /*
559     * This return value is chosen so that code & 0xff
560     * mimics the exit code that a POSIX shell would report for
561     * a program that died from this signal.
562     */
563    code += 128;
564  } else if (WIFEXITED(status)) {
565    code = WEXITSTATUS(status);
566  } else {
567    if (!in_signal)
568      error("waitpid is confused (%s)", argv0);
569  }
570
571  if (!in_signal)
572    clear_child_for_cleanup(pid);
573
574  errno = failed_errno;
575  return code;
576}

588    char *equals = strchr(*e, '=');
589
590    if (equals) {
591      strbuf_add(&key, *e, equals - *e);
592      string_list_insert(&envs, key.buf)->util = equals + 1;
593    } else {
594      string_list_insert(&envs, *e)->util = NULL;
595    }
596    strbuf_release(&key);
597  }
598
599  /* "unset X Y...;" */
600  for (i = 0; i < envs.nr; i++) {
601    const char *var = envs.items[i].string;
602    const char *val = envs.items[i].util;
603
604    if (val || !getenv(var))
605      continue;
606
607    if (!printed_unset) {
608      strbuf_addstr(dst, " unset");
609      printed_unset = 1;
610    }
611    strbuf_addf(dst, " %s", var);
612  }
613  if (printed_unset)
614    strbuf_addch(dst, ';');
615
616  /* ... followed by "A=B C=D ..." */
617  for (i = 0; i < envs.nr; i++) {
618    const char *var = envs.items[i].string;
619    const char *val = envs.items[i].util;
620    const char *oldval;
621
622    if (!val)
623      continue;
624
625    oldval = getenv(var);
626    if (oldval && !strcmp(val, oldval))
627      continue;
628
629    strbuf_addf(dst, " %s=", var);
630    sq_quote_buf_pretty(dst, val);
631  }
632  string_list_clear(&envs, 0);
633}

657int start_command(struct child_process *cmd)
658{
659  int need_in, need_out, need_err;
660  int fdin[2], fdout[2], fderr[2];
661  int failed_errno;
662  char *str;
663
664  /*
665   * In case of errors we must keep the promise to close FDs
666   * that have been passed in via ->in and ->out.
667   */
668
669  need_in = !cmd->no_stdin && cmd->in < 0;
670  if (need_in) {
671    if (pipe(fdin) < 0) {
672      failed_errno = errno;
673      if (cmd->out > 0)
674        close(cmd->out);
675      str = "standard input";
676      goto fail_pipe;
677    }
678    cmd->in = fdin[1];
679  }
680
681  need_out = !cmd->no_stdout
682    && !cmd->stdout_to_stderr
683    && cmd->out < 0;
684  if (need_out) {
685    if (pipe(fdout) < 0) {
686      failed_errno = errno;
687      if (need_in)
688        close_pair(fdin);
689      else if (cmd->in)
690        close(cmd->in);
691      str = "standard output";
692      goto fail_pipe;
693    }
694    cmd->out = fdout[0];
695  }
696
697  need_err = !cmd->no_stderr && cmd->err < 0;
698  if (need_err) {
699    if (pipe(fderr) < 0) {
700      failed_errno = errno;
701      if (need_in)
702        close_pair(fdin);
703      else if (cmd->in)
704        close(cmd->in);
705      if (need_out)
706        close_pair(fdout);
707      else if (cmd->out)
708        close(cmd->out);
709      str = "standard error";
710fail_pipe:
711      error("cannot create %s pipe for %s: %s",
712        str, cmd->args.v[0], strerror(failed_errno));
713      child_process_clear(cmd);
714      errno = failed_errno;
715      return -1;
716    }
717    cmd->err = fderr[0];
718  }
719
720  trace2_child_start(cmd);
721  trace_run_command(cmd);
722
723  fflush(NULL);
724
725  if (cmd->close_object_store)
726    close_object_store(the_repository->objects);
727
728#ifndef GIT_WINDOWS_NATIVE
729{
730  int notify_pipe[2];
731  int null_fd = -1;
732  char **childenv;
733  struct strvec argv = STRVEC_INIT;
734  struct child_err cerr;
735  struct atfork_state as;
736
737  if (prepare_cmd(&argv, cmd) < 0) {
738    failed_errno = errno;
739    cmd->pid = -1;
740    if (!cmd->silent_exec_failure)
741      error_errno("cannot run %s", cmd->args.v[0]);
742    goto end_of_spawn;
743  }
744
745  if (pipe(notify_pipe))
746    notify_pipe[0] = notify_pipe[1] = -1;
747
748  if (cmd->no_stdin || cmd->no_stdout || cmd->no_stderr) {
749    null_fd = xopen("/dev/null", O_RDWR | O_CLOEXEC);
750    set_cloexec(null_fd);
751  }
752
753  childenv = prep_childenv(cmd->env.v);
754  atfork_prepare(&as);
755
756  /*
757   * NOTE: In order to prevent deadlocking when using threads special
758   * care should be taken with the function calls made in between the
759   * fork() and exec() calls.  No calls should be made to functions which
760   * require acquiring a lock (e.g. malloc) as the lock could have been
761   * held by another thread at the time of forking, causing the lock to
762   * never be released in the child process.  This means only
763   * Async-Signal-Safe functions are permitted in the child.
764   */
765  cmd->pid = fork();
766  failed_errno = errno;
767  if (!cmd->pid) {
768    int sig;
769    /*
770     * Ensure the default die/error/warn routines do not get
771     * called, they can take stdio locks and malloc.
772     */
773    set_die_routine(child_die_fn);
774    set_error_routine(child_error_fn);
775    set_warn_routine(child_warn_fn);
776
777    close(notify_pipe[0]);
778    set_cloexec(notify_pipe[1]);
779    child_notifier = notify_pipe[1];
780
781    if (cmd->no_stdin)
782      child_dup2(null_fd, 0);
783    else if (need_in) {
784      child_dup2(fdin[0], 0);
785      child_close_pair(fdin);
786    } else if (cmd->in) {
787      child_dup2(cmd->in, 0);
788      child_close(cmd->in);
789    }
790
791    if (cmd->no_stderr)
792      child_dup2(null_fd, 2);
793    else if (need_err) {
794      child_dup2(fderr[1], 2);
795      child_close_pair(fderr);
796    } else if (cmd->err > 1) {
797      child_dup2(cmd->err, 2);
798      child_close(cmd->err);
799    }
800
801    if (cmd->no_stdout)
802      child_dup2(null_fd, 1);
803    else if (cmd->stdout_to_stderr)
804      child_dup2(2, 1);
805    else if (need_out) {
806      child_dup2(fdout[1], 1);
807      child_close_pair(fdout);
808    } else if (cmd->out > 1) {
809      child_dup2(cmd->out, 1);
810      child_close(cmd->out);
811    }
812
813    if (cmd->dir && chdir(cmd->dir))
814      child_die(CHILD_ERR_CHDIR);
815
816    /*
817     * restore default signal handlers here, in case
818     * we catch a signal right before execve below
819     */
820    for (sig = 1; sig < NSIG; sig++) {
821      /* ignored signals get reset to SIG_DFL on execve */
822      if (signal(sig, SIG_DFL) == SIG_IGN)
823        signal(sig, SIG_IGN);
824    }
825
826    if (sigprocmask(SIG_SETMASK, &as.old, NULL) != 0)
827      child_die(CHILD_ERR_SIGPROCMASK);
828
829    /*
830     * Attempt to exec using the command and arguments starting at
831     * argv.argv[1].  argv.argv[0] contains SHELL_PATH which will
832     * be used in the event exec failed with ENOEXEC at which point
833     * we will try to interpret the command using 'sh'.
834     */
835    execve(argv.v[1], (char *const *) argv.v + 1,
836           (char *const *) childenv);
837    if (errno == ENOEXEC)
838      execve(argv.v[0], (char *const *) argv.v,
839             (char *const *) childenv);
840
841    if (errno == ENOENT) {
842      if (cmd->silent_exec_failure)
843        child_die(CHILD_ERR_SILENT);
844      child_die(CHILD_ERR_ENOENT);
845    } else {
846      child_die(CHILD_ERR_ERRNO);
847    }
848  }
849  atfork_parent(&as);
850  if (cmd->pid < 0)
851    error_errno("cannot fork() for %s", cmd->args.v[0]);
852  else if (cmd->clean_on_exit)
853    mark_child_for_cleanup(cmd->pid, cmd);
854
855  /*
856   * Wait for child's exec. If the exec succeeds (or if fork()
857   * failed), EOF is seen immediately by the parent. Otherwise, the
858   * child process sends a child_err struct.
859   * Note that use of this infrastructure is completely advisory,
860   * therefore, we keep error checks minimal.
861   */
862  close(notify_pipe[1]);
863  if (xread(notify_pipe[0], &cerr, sizeof(cerr)) == sizeof(cerr)) {
864    /*
865     * At this point we know that fork() succeeded, but exec()
866     * failed. Errors have been reported to our stderr.
867     */
868    wait_or_whine(cmd->pid, cmd->args.v[0], 0);
869    child_err_spew(cmd, &cerr);
870    failed_errno = errno;
871    cmd->pid = -1;
872  }
873  close(notify_pipe[0]);
874
875  if (null_fd >= 0)
876    close(null_fd);
877  strvec_clear(&argv);
878  free(childenv);
879}
880end_of_spawn:
881
882#else
883{
884  int fhin = 0, fhout = 1, fherr = 2;
885  const char **sargv = cmd->args.v;
886  struct strvec nargv = STRVEC_INIT;
887
888  if (cmd->no_stdin)
889    fhin = open("/dev/null", O_RDWR);
890  else if (need_in)
891    fhin = dup(fdin[0]);
892  else if (cmd->in)
893    fhin = dup(cmd->in);
894
895  if (cmd->no_stderr)
896    fherr = open("/dev/null", O_RDWR);
897  else if (need_err)
898    fherr = dup(fderr[1]);
899  else if (cmd->err > 2)
900    fherr = dup(cmd->err);
901
902  if (cmd->no_stdout)
903    fhout = open("/dev/null", O_RDWR);
904  else if (cmd->stdout_to_stderr)
905    fhout = dup(fherr);
906  else if (need_out)
907    fhout = dup(fdout[1]);
908  else if (cmd->out > 1)
909    fhout = dup(cmd->out);
910
911  if (cmd->git_cmd)
912    cmd->args.v = prepare_git_cmd(&nargv, sargv);
913  else if (cmd->use_shell)
914    cmd->args.v = prepare_shell_cmd(&nargv, sargv);
915
916  cmd->pid = mingw_spawnvpe(cmd->args.v[0], cmd->args.v,
917          (char**) cmd->env.v,
918          cmd->dir, fhin, fhout, fherr);
919  failed_errno = errno;
920  if (cmd->pid < 0 && (!cmd->silent_exec_failure || errno != ENOENT))
921    error_errno("cannot spawn %s", cmd->args.v[0]);
922  if (cmd->clean_on_exit && cmd->pid >= 0)
923    mark_child_for_cleanup(cmd->pid, cmd);
924
925  strvec_clear(&nargv);
926  cmd->args.v = sargv;
927  if (fhin != 0)
928    close(fhin);
929  if (fhout != 1)
930    close(fhout);
931  if (fherr != 2)
932    close(fherr);
933}
934#endif
935
936  if (cmd->pid < 0) {
937    trace2_child_exit(cmd, -1);
938
939    if (need_in)
940      close_pair(fdin);
941    else if (cmd->in)
942      close(cmd->in);
943    if (need_out)
944      close_pair(fdout);
945    else if (cmd->out)
946      close(cmd->out);
947    if (need_err)
948      close_pair(fderr);
949    else if (cmd->err)
950      close(cmd->err);
951    child_process_clear(cmd);
952    errno = failed_errno;
953    return -1;
954  }
955
956  if (need_in)
957    close(fdin[0]);
958  else if (cmd->in)
959    close(cmd->in);
960
961  if (need_out)
962    close(fdout[1]);
963  else if (cmd->out)
964    close(cmd->out);
965
966  if (need_err)
967    close(fderr[1]);
968  else if (cmd->err)
969    close(cmd->err);
970
971  return 0;
972}

1731
1732    if (pp->ungroup) {
1733      ; /* no strbuf_*() work to do here */
1734    } else if (i != pp->output_owner) {
1735      strbuf_addbuf(&pp->buffered_output, &pp->children[i].err);
1736      strbuf_reset(&pp->children[i].err);
1737    } else {
1738      strbuf_write(&pp->children[i].err, stderr);
1739      strbuf_reset(&pp->children[i].err);
1740
1741      /* Output all other finished child processes */
1742      strbuf_write(&pp->buffered_output, stderr);
1743      strbuf_reset(&pp->buffered_output);
1744
1745      /*
1746       * Pick next process to output live.
1747       * NEEDSWORK:
1748       * For now we pick it randomly by doing a round
1749       * robin. Later we may want to pick the one with
1750       * the most output or the longest or shortest
1751       * running process time.
1752       */
1753      for (i = 0; i < n; i++)
1754        if (pp->children[(pp->output_owner + i) % n].state == GIT_CP_WORKING)
1755          break;
1756      pp->output_owner = (pp->output_owner + i) % n;
1757    }
1758  }
1759  return result;
1760}

1762int run_processes_parallel(int n,
1763         get_next_task_fn get_next_task,
1764         start_failure_fn start_failure,
1765         task_finished_fn task_finished,
1766         void *pp_cb)
1767{
1768  int i, code;
1769  int output_timeout = 100;
1770  int spawn_cap = 4;
1771  int ungroup = run_processes_parallel_ungroup;
1772  struct parallel_processes pp;
1773
1774  /* unset for the next API user */
1775  run_processes_parallel_ungroup = 0;
1776
1777  pp_init(&pp, n, get_next_task, start_failure, task_finished, pp_cb,
1778    ungroup);
1779  while (1) {
1780    for (i = 0;
1781        i < spawn_cap && !pp.shutdown &&
1782        pp.nr_processes < pp.max_processes;
1783        i++) {
1784      code = pp_start_one(&pp);
1785      if (!code)
1786        continue;
1787      if (code < 0) {
1788        pp.shutdown = 1;
1789        kill_children(&pp, -code);
1790      }
1791      break;
1792    }
1793    if (!pp.nr_processes)
1794      break;
1795    if (ungroup) {
1796      int i;
1797
1798      for (i = 0; i < pp.max_processes; i++)
1799        pp.children[i].state = GIT_CP_WAIT_CLEANUP;
1800    } else {
1801      pp_buffer_stderr(&pp, output_timeout);
1802      pp_output(&pp);
1803    }
1804    code = pp_collect_finished(&pp);
1805    if (code) {
1806      pp.shutdown = 1;
1807      if (code < 0)
1808        kill_children(&pp, -code);
1809    }
1810  }
1811
1812  pp_cleanup(&pp);
1813  return 0;
1814}

1863enum start_bg_result start_bg_command(struct child_process *cmd,
1864              start_bg_wait_cb *wait_cb,
1865              void *cb_data,
1866              unsigned int timeout_sec)
1867{
1868  enum start_bg_result sbgr = SBGR_ERROR;
1869  int ret;
1870  int wait_status;
1871  pid_t pid_seen;
1872  time_t time_limit;
1873
1874  /*
1875   * We do not allow clean-on-exit because the child process
1876   * should persist in the background and possibly/probably
1877   * after this process exits.  So we don't want to kill the
1878   * child during our atexit routine.
1879   */
1880  if (cmd->clean_on_exit)
1881    BUG("start_bg_command() does not allow non-zero clean_on_exit");
1882
1883  if (!cmd->trace2_child_class)
1884    cmd->trace2_child_class = "background";
1885
1886  ret = start_command(cmd);
1887  if (ret) {
1888    /*
1889     * We assume that if `start_command()` fails, we
1890     * either get a complete `trace2_child_start() /
1891     * trace2_child_exit()` pair or it fails before the
1892     * `trace2_child_start()` is emitted, so we do not
1893     * need to worry about it here.
1894     *
1895     * We also assume that `start_command()` does not add
1896     * us to the cleanup list.  And that it calls
1897     * calls `child_process_clear()`.
1898     */
1899    sbgr = SBGR_ERROR;
1900    goto done;
1901  }
1902
1903  time(&time_limit);
1904  time_limit += timeout_sec;
1905
1906wait:
1907  pid_seen = waitpid(cmd->pid, &wait_status, WNOHANG);
1908
1909  if (!pid_seen) {
1910    /*
1911     * The child is currently running.  Ask the callback
1912     * if the child is ready to do work or whether we
1913     * should keep waiting for it to boot up.
1914     */
1915    ret = (*wait_cb)(cmd, cb_data);
1916    if (!ret) {
1917      /*
1918       * The child is running and "ready".
1919       */
1920      trace2_child_ready(cmd, "ready");
1921      sbgr = SBGR_READY;
1922      goto done;
1923    } else if (ret > 0) {
1924      /*
1925       * The callback said to give it more time to boot up
1926       * (subject to our timeout limit).
1927       */
1928      time_t now;
1929
1930      time(&now);
1931      if (now < time_limit)
1932        goto wait;
1933
1934      /*
1935       * Our timeout has expired.  We don't try to
1936       * kill the child, but rather let it continue
1937       * (hopefully) trying to startup.
1938       */
1939      trace2_child_ready(cmd, "timeout");
1940      sbgr = SBGR_TIMEOUT;
1941      goto done;
1942    } else {
1943      /*
1944       * The cb gave up on this child.  It is still running,
1945       * but our cb got an error trying to probe it.
1946       */
1947      trace2_child_ready(cmd, "error");
1948      sbgr = SBGR_CB_ERROR;
1949      goto done;
1950    }
1951  }
1952
1953  else if (pid_seen == cmd->pid) {
1954    int child_code = -1;
1955
1956    /*
1957     * The child started, but exited or was terminated
1958     * before becoming "ready".
1959     *
1960     * We try to match the behavior of `wait_or_whine()`
1961     * WRT the handling of WIFSIGNALED() and WIFEXITED()
1962     * and convert the child's status to a return code for
1963     * tracing purposes and emit the `trace2_child_exit()`
1964     * event.
1965     *
1966     * We do not want the wait_or_whine() error message
1967     * because we will be called by client-side library
1968     * routines.
1969     */
1970    if (WIFEXITED(wait_status))
1971      child_code = WEXITSTATUS(wait_status);
1972    else if (WIFSIGNALED(wait_status))
1973      child_code = WTERMSIG(wait_status) + 128;
1974    trace2_child_exit(cmd, child_code);
1975
1976    sbgr = SBGR_DIED;
1977    goto done;
1978  }
1979
1980  else if (pid_seen < 0 && errno == EINTR)
1981    goto wait;
1982
1983  trace2_child_exit(cmd, -1);
1984  sbgr = SBGR_ERROR;
1985
1986done:
1987  child_process_clear(cmd);
1988  invalidate_lstat_cache();
1989  return sbgr;
1990}
