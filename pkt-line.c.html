33static void packet_trace(const char *buf, unsigned int len, int write)
34{
35  int i;
36  struct strbuf out;
37  static int in_pack, sideband;
38
39  if (!trace_want(&trace_packet) && !trace_want(&trace_pack))
40    return;
41
42  if (in_pack) {
43    if (packet_trace_pack(buf, len, sideband))
44      return;
45  } else if (starts_with(buf, "PACK") || starts_with(buf, "\1PACK")) {
46    in_pack = 1;
47    sideband = *buf == '\1';
48    packet_trace_pack(buf, len, sideband);
49
50    /*
51     * Make a note in the human-readable trace that the pack data
52     * started.
53     */
54    buf = "PACK ...";
55    len = strlen(buf);
56  }
57
58  if (!trace_want(&trace_packet))
59    return;
60
61  /* +32 is just a guess for header + quoting */
62  strbuf_init(&out, len+32);
63
64  strbuf_addf(&out, "packet: %12s%c ",
65        get_trace_prefix(), write ? '>' : '<');
66
67  /* XXX we should really handle printable utf8 */
68  for (i = 0; i < len; i++) {
69    /* suppress newlines */
70    if (buf[i] == '\n')
71      continue;
72    if (buf[i] >= 0x20 && buf[i] <= 0x7e)
73      strbuf_addch(&out, buf[i]);
74    else
75      strbuf_addf(&out, "\\%o", buf[i]);
76  }
77
78  strbuf_addch(&out, '\n');
79  trace_strbuf(&trace_packet, &out);
80  strbuf_release(&out);
81}

197         struct strbuf *err)

200  size_t packet_size;
201
202  if (size > LARGE_PACKET_DATA_MAX) {
203    strbuf_addstr(err, _("packet write failed - data exceeds max packet size"));
204    return -1;

330static int get_packet_data(int fd, char **src_buf, size_t *src_size,
331         void *dst, unsigned size, int options)
332{
333  ssize_t ret;
334
335  if (fd >= 0 && src_buf && *src_buf)
336    BUG("multiple sources given to packet_read");
337
338  /* Read up to "size" bytes from our source, whatever it is. */
339  if (src_buf && *src_buf) {
340    ret = size < *src_size ? size : *src_size;
341    memcpy(dst, *src_buf, ret);
342    *src_buf += ret;
343    *src_size -= ret;
344  } else {
345    ret = read_in_full(fd, dst, size);
346    if (ret < 0) {
347      if (options & PACKET_READ_GENTLE_ON_READ_ERROR)
348        return error_errno(_("read error"));
349      die_errno(_("read error"));
350    }
351  }
352
353  /* And complain if we didn't get enough bytes to satisfy the read. */
354  if (ret != size) {
355    if (options & PACKET_READ_GENTLE_ON_EOF)
356      return -1;
357
358    if (options & PACKET_READ_GENTLE_ON_READ_ERROR)
359      return error(_("the remote end hung up unexpectedly"));
360    die(_("the remote end hung up unexpectedly"));
361  }
362
363  return ret;
364}

372static char *find_packfile_uri_path(const char *buffer)
373{
374  const char *URI_MARK = "://";
375  char *path;
376  int len;
377
378  /* First char is sideband mark */
379  buffer += 1;
380
381  len = strspn(buffer, "0123456789abcdefABCDEF");
382  /* size of SHA1 and SHA256 hash */
383  if (!(len == 40 || len == 64) || buffer[len] != ' ')
384    return NULL; /* required "<hash>SP" not seen */
385
386  path = strstr(buffer + len + 1, URI_MARK);
387  if (!path)
388    return NULL;
389
390  path = strchr(path + strlen(URI_MARK), '/');
391  if (!path || !*(path + 1))
392    return NULL;
393
394  /* position after '/' */
395  return ++path;
396}

398enum packet_read_status packet_read_with_status(int fd, char **src_buffer,
399            size_t *src_len, char *buffer,
400            unsigned size, int *pktlen,
401            int options)
402{
403  int len;
404  char linelen[4];
405  char *uri_path_start;
406
407  if (get_packet_data(fd, src_buffer, src_len, linelen, 4, options) < 0) {
408    *pktlen = -1;
409    return PACKET_READ_EOF;
410  }
411
412  len = packet_length(linelen);
413
414  if (len < 0) {
415    if (options & PACKET_READ_GENTLE_ON_READ_ERROR)
416      return error(_("protocol error: bad line length "
417               "character: %.4s"), linelen);
418    die(_("protocol error: bad line length character: %.4s"), linelen);
419  } else if (!len) {
420    packet_trace("0000", 4, 0);
421    *pktlen = 0;
422    return PACKET_READ_FLUSH;
423  } else if (len == 1) {
424    packet_trace("0001", 4, 0);
425    *pktlen = 0;
426    return PACKET_READ_DELIM;
427  } else if (len == 2) {
428    packet_trace("0002", 4, 0);
429    *pktlen = 0;
430    return PACKET_READ_RESPONSE_END;
431  } else if (len < 4) {
432    if (options & PACKET_READ_GENTLE_ON_READ_ERROR)
433      return error(_("protocol error: bad line length %d"),
434             len);
435    die(_("protocol error: bad line length %d"), len);
436  }
437
438  len -= 4;
439  if ((unsigned)len >= size) {
440    if (options & PACKET_READ_GENTLE_ON_READ_ERROR)
441      return error(_("protocol error: bad line length %d"),
442             len);
443    die(_("protocol error: bad line length %d"), len);
444  }
445
446  if (get_packet_data(fd, src_buffer, src_len, buffer, len, options) < 0) {
447    *pktlen = -1;
448    return PACKET_READ_EOF;
449  }
450
451  if ((options & PACKET_READ_CHOMP_NEWLINE) &&
452      len && buffer[len-1] == '\n')
453    len--;
454
455  buffer[len] = 0;
456  if (options & PACKET_READ_REDACT_URI_PATH &&
457      (uri_path_start = find_packfile_uri_path(buffer))) {
458    const char *redacted = "<redacted>";
459    struct strbuf tracebuf = STRBUF_INIT;
460    strbuf_insert(&tracebuf, 0, buffer, len);
461    strbuf_splice(&tracebuf, uri_path_start - buffer,
462            strlen(uri_path_start), redacted, strlen(redacted));
463    packet_trace(tracebuf.buf, tracebuf.len, 0);
464    strbuf_release(&tracebuf);
465  } else {
466    packet_trace(buffer, len, 0);
467  }
468
469  if ((options & PACKET_READ_DIE_ON_ERR_PACKET) &&
470      starts_with(buffer, "ERR "))
471    die(_("remote error: %s"), buffer + 4);
472
473  *pktlen = len;
474  return PACKET_READ_NORMAL;
475}

507ssize_t read_packetized_to_strbuf(int fd_in, struct strbuf *sb_out, int options)
508{
509  int packet_len;
510
511  size_t orig_len = sb_out->len;
512  size_t orig_alloc = sb_out->alloc;
513
514  for (;;) {
515    strbuf_grow(sb_out, LARGE_PACKET_DATA_MAX);
516    packet_len = packet_read(fd_in,
517      /* strbuf_grow() above always allocates one extra byte to
518       * store a '\0' at the end of the string. packet_read()
519       * writes a '\0' extra byte at the end, too. Let it know
520       * that there is already room for the extra byte.
521       */
522      sb_out->buf + sb_out->len, LARGE_PACKET_DATA_MAX+1,
523      options);
524    if (packet_len <= 0)
525      break;
526    sb_out->len += packet_len;
527  }
528
529  if (packet_len < 0) {
530    if (orig_alloc == 0)
531      strbuf_release(sb_out);
532    else
533      strbuf_setlen(sb_out, orig_len);
534    return packet_len;
535  }
536  return sb_out->len - orig_len;
537}

539int recv_sideband(const char *me, int in_stream, int out)
540{
541  char buf[LARGE_PACKET_MAX + 1];
542  int len;
543  struct strbuf scratch = STRBUF_INIT;
544  enum sideband_type sideband_type;
545
546  while (1) {
547    int status = packet_read_with_status(in_stream, NULL, NULL,
548                 buf, LARGE_PACKET_MAX,
549                 &len,
550                 PACKET_READ_GENTLE_ON_EOF);
551    if (!demultiplex_sideband(me, status, buf, len, 0, &scratch,
552            &sideband_type))
553      continue;
554    switch (sideband_type) {
555    case SIDEBAND_PRIMARY:
556      write_or_die(out, buf + 1, len - 1);
557      break;
558    default: /* errors: message already written */
559      if (scratch.len > 0)
560        BUG("unhandled incomplete sideband: '%s'",
561            scratch.buf);
562      return sideband_type;
563    }
564  }
565}

572  memset(reader, 0, sizeof(*reader));
573
574  reader->fd = fd;
575  reader->src_buffer = src_buffer;
576  reader->src_len = src_len;
577  reader->buffer = packet_buffer;
578  reader->buffer_size = sizeof(packet_buffer);
579  reader->options = options;
580  reader->me = "git";
581  reader->hash_algo = &hash_algos[GIT_HASH_SHA1];
582}
583
584enum packet_read_status packet_reader_read(struct packet_reader *reader)
585{
586  struct strbuf scratch = STRBUF_INIT;
587
588  if (reader->line_peeked) {
589    reader->line_peeked = 0;
590    return reader->status;
591  }
592
593  /*
594   * Consume all progress packets until a primary payload packet is
595   * received
596   */
597  while (1) {
598    enum sideband_type sideband_type;
599    reader->status = packet_read_with_status(reader->fd,
600               &reader->src_buffer,
601               &reader->src_len,
602               reader->buffer,
603               reader->buffer_size,
604               &reader->pktlen,
605               reader->options);
606    if (!reader->use_sideband)
607      break;
608    if (demultiplex_sideband(reader->me, reader->status,
609           reader->buffer, reader->pktlen, 1,
610           &scratch, &sideband_type))
611      break;
612  }
613
614  if (reader->status == PACKET_READ_NORMAL)
615    /* Skip the sideband designator if sideband is used */
616    reader->line = reader->use_sideband ?
617      reader->buffer + 1 : reader->buffer;
618  else
619    reader->line = NULL;
620
621  return reader->status;
622}
623
624enum packet_read_status packet_reader_peek(struct packet_reader *reader)
625{
626  /* Only allow peeking a single line */
627  if (reader->line_peeked)
628    return reader->status;
629
630  /* Peek a line by reading it and setting peeked flag */
631  packet_reader_read(reader);
632  reader->line_peeked = 1;
633  return reader->status;
634}
