17/*
18 * The input parameter must contain an absolute path, and it must already be
19 * normalized.
20 *
21 * Find the part of an absolute path that lies inside the work tree by
22 * dereferencing symlinks outside the work tree, for example:
23 * /dir1/repo/dir2/file   (work tree is /dir1/repo)      -> dir2/file
24 * /dir/file              (work tree is /)               -> dir/file
25 * /dir/symlink1/symlink2 (symlink1 points to work tree) -> symlink2
26 * /dir/repolink/file     (repolink points to /dir/repo) -> file
27 * /dir/repo              (exactly equal to work tree)   -> (empty string)
28 */
29static int abspath_part_inside_repo(char *path)
30{
31  size_t len;
32  size_t wtlen;
33  char *path0;
34  int off;
35  const char *work_tree = get_git_work_tree();
36  struct strbuf realpath = STRBUF_INIT;
37
38  if (!work_tree)
39    return -1;
40  wtlen = strlen(work_tree);
41  len = strlen(path);
42  off = offset_1st_component(path);
43
44  /* check if work tree is already the prefix */
45  if (wtlen <= len && !fspathncmp(path, work_tree, wtlen)) {
46    if (path[wtlen] == '/') {
47      memmove(path, path + wtlen + 1, len - wtlen);
48      return 0;
49    } else if (path[wtlen - 1] == '/' || path[wtlen] == '\0') {
50      /* work tree is the root, or the whole path */
51      memmove(path, path + wtlen, len - wtlen + 1);
52      return 0;
53    }
54    /* work tree might match beginning of a symlink to work tree */
55    off = wtlen;
56  }
57  path0 = path;
58  path += off;
59
60  /* check each '/'-terminated level */
61  while (*path) {
62    path++;
63    if (*path == '/') {
64      *path = '\0';
65      strbuf_realpath(&realpath, path0, 1);
66      if (fspathcmp(realpath.buf, work_tree) == 0) {
67        memmove(path0, path + 1, len - (path - path0));
68        strbuf_release(&realpath);
69        return 0;
70      }
71      *path = '/';
72    }
73  }
74
75  /* check whole path */
76  strbuf_realpath(&realpath, path0, 1);
77  if (fspathcmp(realpath.buf, work_tree) == 0) {
78    *path0 = '\0';
79    strbuf_release(&realpath);
80    return 0;
81  }
82
83  strbuf_release(&realpath);
84  return -1;
85}

87/*
88 * Normalize "path", prepending the "prefix" for relative paths. If
89 * remaining_prefix is not NULL, return the actual prefix still
90 * remains in the path. For example, prefix = sub1/sub2/ and path is
91 *
92 *  foo          -> sub1/sub2/foo  (full prefix)
93 *  ../foo       -> sub1/foo       (remaining prefix is sub1/)
94 *  ../../bar    -> bar            (no remaining prefix)
95 *  ../../sub1/sub2/foo -> sub1/sub2/foo (but no remaining prefix)
96 *  `pwd`/../bar -> sub1/bar       (no remaining prefix)
97 */
98char *prefix_path_gently(const char *prefix, int len,
99       int *remaining_prefix, const char *path)
100{
101  const char *orig = path;
102  char *sanitized;
103  if (is_absolute_path(orig)) {
104    sanitized = xmallocz(strlen(path));
105    if (remaining_prefix)
106      *remaining_prefix = 0;
107    if (normalize_path_copy_len(sanitized, path, remaining_prefix)) {
108      free(sanitized);
109      return NULL;
110    }
111    if (abspath_part_inside_repo(sanitized)) {
112      free(sanitized);
113      return NULL;
114    }
115  } else {
116    sanitized = xstrfmt("%.*s%s", len, len ? prefix : "", path);
117    if (remaining_prefix)
118      *remaining_prefix = len;
119    if (normalize_path_copy_len(sanitized, sanitized, remaining_prefix)) {
120      free(sanitized);
121      return NULL;
122    }
123  }
124  return sanitized;
125}

151int check_filename(const char *prefix, const char *arg)
152{
153  char *to_free = NULL;
154  struct stat st;
155
156  if (skip_prefix(arg, ":/", &arg)) {
157    if (!*arg) /* ":/" is root dir, always exists */
158      return 1;
159    prefix = NULL;
160  } else if (skip_prefix(arg, ":!", &arg) ||
161       skip_prefix(arg, ":^", &arg)) {
162    if (!*arg) /* excluding everything is silly, but allowed */
163      return 1;
164  }
165
166  if (prefix)
167    arg = to_free = prefix_filename(prefix, arg);
168
169  if (!lstat(arg, &st)) {
170    free(to_free);
171    return 1; /* file exists */
172  }
173  if (is_missing_file_error(errno)) {
174    free(to_free);
175    return 0; /* file does not exist */
176  }
177  die_errno(_("failed to stat '%s'"), arg);
178}

180static void NORETURN die_verify_filename(struct repository *r,
181           const char *prefix,
182           const char *arg,
183           int diagnose_misspelt_rev)
184{
185  if (!diagnose_misspelt_rev)
186    die(_("%s: no such path in the working tree.\n"
187          "Use 'git <command> -- <path>...' to specify paths that do not exist locally."),
188        arg);
189  /*
190   * Saying "'(icase)foo' does not exist in the index" when the
191   * user gave us ":(icase)foo" is just stupid.  A magic pathspec
192   * begins with a colon and is followed by a non-alnum; do not
193   * let maybe_die_on_misspelt_object_name() even trigger.
194   */

199  die(_("ambiguous argument '%s': unknown revision or path not in the working tree.\n"
200        "Use '--' to separate paths from revisions, like this:\n"
201        "'git <command> [<revision>...] -- [<file>...]'"), arg);

205/*
206 * Check for arguments that don't resolve as actual files,
207 * but which look sufficiently like pathspecs that we'll consider
208 * them such for the purposes of rev/pathspec DWIM parsing.
209 */
210static int looks_like_pathspec(const char *arg)
211{
212  const char *p;
213  int escaped = 0;
214
215  /*
216   * Wildcard characters imply the user is looking to match pathspecs
217   * that aren't in the filesystem. Note that this doesn't include
218   * backslash even though it's a glob special; by itself it doesn't
219   * cause any increase in the match. Likewise ignore backslash-escaped
220   * wildcard characters.
221   */
222  for (p = arg; *p; p++) {
223    if (escaped) {
224      escaped = 0;
225    } else if (is_glob_special(*p)) {
226      if (*p == '\\')
227        escaped = 1;
228      else
229        return 1;
230    }
231  }
232
233  /* long-form pathspec magic */
234  if (starts_with(arg, ":("))
235    return 1;
236
237  return 0;
238}

240/*
241 * Verify a filename that we got as an argument for a pathspec
242 * entry. Note that a filename that begins with "-" never verifies
243 * as true, because even if such a filename were to exist, we want
244 * it to be preceded by the "--" marker (or we want the user to
245 * use a format like "./-filename")
246 *
247 * The "diagnose_misspelt_rev" is used to provide a user-friendly
248 * diagnosis when dying upon finding that "name" is not a pathname.
249 * If set to 1, the diagnosis will try to diagnose "name" as an
250 * invalid object name (e.g. HEAD:foo). If set to 0, the diagnosis
251 * will only complain about an inexisting file.
252 *
253 * This function is typically called to check that a "file or rev"
254 * argument is unambiguous. In this case, the caller will want
255 * diagnose_misspelt_rev == 1 when verifying the first non-rev
256 * argument (which could have been a revision), and
257 * diagnose_misspelt_rev == 0 for the next ones (because we already
258 * saw a filename, there's not ambiguity anymore).
259 */
260void verify_filename(const char *prefix,
261         const char *arg,
262         int diagnose_misspelt_rev)
263{
264  if (*arg == '-')
265    die(_("option '%s' must come before non-option arguments"), arg);
266  if (looks_like_pathspec(arg) || check_filename(prefix, arg))
267    return;
268  die_verify_filename(the_repository, prefix, arg, diagnose_misspelt_rev);
269}

329/*
330 * Test if it looks like we're at a git directory.
331 * We want to see:
332 *
333 *  - either an objects/ directory _or_ the proper
334 *    GIT_OBJECT_DIRECTORY environment variable
335 *  - a refs/ directory
336 *  - either a HEAD symlink or a HEAD file that is formatted as
337 *    a proper "ref:", or a regular file HEAD that has a properly
338 *    formatted sha1 object name.
339 */
340int is_git_directory(const char *suspect)
341{
342  struct strbuf path = STRBUF_INIT;
343  int ret = 0;
344  size_t len;
345
346  /* Check worktree-related signatures */
347  strbuf_addstr(&path, suspect);
348  strbuf_complete(&path, '/');
349  strbuf_addstr(&path, "HEAD");
350  if (validate_headref(path.buf))
351    goto done;
352
353  strbuf_reset(&path);
354  get_common_dir(&path, suspect);
355  len = path.len;
356
357  /* Check non-worktree-related signatures */
358  if (getenv(DB_ENVIRONMENT)) {
359    if (access(getenv(DB_ENVIRONMENT), X_OK))
360      goto done;
361  }
362  else {
363    strbuf_setlen(&path, len);
364    strbuf_addstr(&path, "/objects");
365    if (access(path.buf, X_OK))
366      goto done;
367  }
368
369  strbuf_setlen(&path, len);
370  strbuf_addstr(&path, "/refs");
371  if (access(path.buf, X_OK))
372    goto done;
373
374  ret = 1;
375done:
376  strbuf_release(&path);
377  return ret;
378}

436static void setup_original_cwd(void)
437{
438  struct strbuf tmp = STRBUF_INIT;
439  const char *worktree = NULL;
440  int offset = -1;
441
442  if (!tmp_original_cwd)
443    return;
444
445  /*
446   * startup_info->original_cwd points to the current working
447   * directory we inherited from our parent process, which is a
448   * directory we want to avoid removing.
449   *
450   * For convience, we would like to have the path relative to the
451   * worktree instead of an absolute path.
452   *
453   * Yes, startup_info->original_cwd is usually the same as 'prefix',
454   * but differs in two ways:
455   *   - prefix has a trailing '/'
456   *   - if the user passes '-C' to git, that modifies the prefix but
457   *     not startup_info->original_cwd.
458   */
459
460  /* Normalize the directory */
461  if (!strbuf_realpath(&tmp, tmp_original_cwd, 0)) {
462    trace2_data_string("setup", the_repository,
463           "realpath-path", tmp_original_cwd);
464    trace2_data_string("setup", the_repository,
465           "realpath-failure", strerror(errno));
466    free((char*)tmp_original_cwd);
467    tmp_original_cwd = NULL;
468    return;
469  }
470
471  free((char*)tmp_original_cwd);
472  tmp_original_cwd = NULL;
473  startup_info->original_cwd = strbuf_detach(&tmp, NULL);
474
475  /*
476   * Get our worktree; we only protect the current working directory
477   * if it's in the worktree.
478   */
479  worktree = get_git_work_tree();
480  if (!worktree)
481    goto no_prevention_needed;
482
483  offset = dir_inside_of(startup_info->original_cwd, worktree);
484  if (offset >= 0) {
485    /*
486     * If startup_info->original_cwd == worktree, that is already
487     * protected and we don't need original_cwd as a secondary
488     * protection measure.
489     */
490    if (!*(startup_info->original_cwd + offset))
491      goto no_prevention_needed;
492
493    /*
494     * original_cwd was inside worktree; precompose it just as
495     * we do prefix so that built up paths will match
496     */
497    startup_info->original_cwd = \
498      precompose_string_if_needed(startup_info->original_cwd
499                + offset);
500    return;
501  }
502
503no_prevention_needed:
504  free((char*)startup_info->original_cwd);
505  startup_info->original_cwd = NULL;
506}

524  EXTENSION_ERROR = -1, /* compatible with error(), etc */
525  EXTENSION_UNKNOWN = 0,
526  EXTENSION_OK = 1
527};
528
529/*
530 * Do not add new extensions to this function. It handles extensions which are
531 * respected even in v0-format repositories for historical compatibility.
532 */
533static enum extension_result handle_extension_v0(const char *var,
534             const char *value,
535             const char *ext,
536             struct repository_format *data)
537{
538    if (!strcmp(ext, "noop")) {
539      return EXTENSION_OK;
540    } else if (!strcmp(ext, "preciousobjects")) {
541      data->precious_objects = git_config_bool(var, value);
542      return EXTENSION_OK;
543    } else if (!strcmp(ext, "partialclone")) {
544      data->partial_clone = xstrdup(value);
545      return EXTENSION_OK;
546    } else if (!strcmp(ext, "worktreeconfig")) {
547      data->worktree_config = git_config_bool(var, value);
548      return EXTENSION_OK;
549    }
550
551    return EXTENSION_UNKNOWN;
552}
553
554/*
555 * Record any new extensions in this function.
556 */
557static enum extension_result handle_extension(const char *var,
558                const char *value,
559                const char *ext,
560                struct repository_format *data)
561{
562  if (!strcmp(ext, "noop-v1")) {
563    return EXTENSION_OK;
564  } else if (!strcmp(ext, "objectformat")) {
565    int format;
566
567    if (!value)
568      return config_error_nonbool(var);
569    format = hash_algo_by_name(value);
570    if (format == GIT_HASH_UNKNOWN)
571      return error(_("invalid value for '%s': '%s'"),
572             "extensions.objectformat", value);
573    data->hash_algo = format;
574    return EXTENSION_OK;
575  }
576  return EXTENSION_UNKNOWN;
577}
578
579static int check_repo_format(const char *var, const char *value, void *vdata)
580{
581  struct repository_format *data = vdata;
582  const char *ext;
583
584  if (strcmp(var, "core.repositoryformatversion") == 0)
585    data->version = git_config_int(var, value);
586  else if (skip_prefix(var, "extensions.", &ext)) {
587    switch (handle_extension_v0(var, value, ext, data)) {
588    case EXTENSION_ERROR:
589      return -1;
590    case EXTENSION_OK:
591      return 0;
592    case EXTENSION_UNKNOWN:
593      break;
594    }
595
596    switch (handle_extension(var, value, ext, data)) {
597    case EXTENSION_ERROR:
598      return -1;
599    case EXTENSION_OK:
600      string_list_append(&data->v1_only_extensions, ext);
601      return 0;
602    case EXTENSION_UNKNOWN:
603      string_list_append(&data->unknown_extensions, ext);
604      return 0;
605    }
606  }
607
608  return read_worktree_config(var, value, vdata);
609}

611static int check_repository_format_gently(const char *gitdir, struct repository_format *candidate, int *nongit_ok)
612{
613  struct strbuf sb = STRBUF_INIT;
614  struct strbuf err = STRBUF_INIT;
615  int has_common;
616
617  has_common = get_common_dir(&sb, gitdir);
618  strbuf_addstr(&sb, "/config");
619  read_repository_format(candidate, sb.buf);
620  strbuf_release(&sb);
621
622  /*
623   * For historical use of check_repository_format() in git-init,
624   * we treat a missing config as a silent "ok", even when nongit_ok
625   * is unset.
626   */
627  if (candidate->version < 0)
628    return 0;
629
630  if (verify_repository_format(candidate, &err) < 0) {
631    if (nongit_ok) {
632      warning("%s", err.buf);
633      strbuf_release(&err);
634      *nongit_ok = -1;
635      return -1;
636    }
637    die("%s", err.buf);
638  }
639
640  repository_format_precious_objects = candidate->precious_objects;
641  repository_format_worktree_config = candidate->worktree_config;
642  string_list_clear(&candidate->unknown_extensions, 0);
643  string_list_clear(&candidate->v1_only_extensions, 0);
644
645  if (repository_format_worktree_config) {
646    /*
647     * pick up core.bare and core.worktree from per-worktree
648     * config if present
649     */
650    strbuf_addf(&sb, "%s/config.worktree", gitdir);
651    git_config_from_file(read_worktree_config, sb.buf, candidate);
652    strbuf_release(&sb);
653    has_common = 0;
654  }
655
656  if (!has_common) {
657    if (candidate->is_bare != -1) {
658      is_bare_repository_cfg = candidate->is_bare;
659      if (is_bare_repository_cfg == 1)
660        inside_work_tree = -1;
661    }
662    if (candidate->work_tree) {
663      free(git_work_tree_cfg);
664      git_work_tree_cfg = xstrdup(candidate->work_tree);
665      inside_work_tree = -1;
666    }
667  }
668
669  return 0;
670}

672int upgrade_repository_format(int target_version)
673{
674  struct strbuf sb = STRBUF_INIT;
675  struct strbuf err = STRBUF_INIT;
676  struct strbuf repo_version = STRBUF_INIT;
677  struct repository_format repo_fmt = REPOSITORY_FORMAT_INIT;
678
679  strbuf_git_common_path(&sb, the_repository, "config");
680  read_repository_format(&repo_fmt, sb.buf);
681  strbuf_release(&sb);
682
683  if (repo_fmt.version >= target_version)
684    return 0;
685
686  if (verify_repository_format(&repo_fmt, &err) < 0) {
687    error("cannot upgrade repository format from %d to %d: %s",
688          repo_fmt.version, target_version, err.buf);
689    strbuf_release(&err);
690    return -1;
691  }
692  if (!repo_fmt.version && repo_fmt.unknown_extensions.nr)
693    return error("cannot upgrade repository format: "
694           "unknown extension %s",
695           repo_fmt.unknown_extensions.items[0].string);
696
697  strbuf_addf(&repo_version, "%d", target_version);
698  git_config_set("core.repositoryformatversion", repo_version.buf);
699  strbuf_release(&repo_version);
700  return 1;
701}

728int verify_repository_format(const struct repository_format *format,
729           struct strbuf *err)
730{
731  if (GIT_REPO_VERSION_READ < format->version) {
732    strbuf_addf(err, _("Expected git repo version <= %d, found %d"),
733          GIT_REPO_VERSION_READ, format->version);
734    return -1;
735  }
736
737  if (format->version >= 1 && format->unknown_extensions.nr) {
738    int i;
739
740    strbuf_addstr(err, Q_("unknown repository extension found:",
741              "unknown repository extensions found:",
742              format->unknown_extensions.nr));
743
744    for (i = 0; i < format->unknown_extensions.nr; i++)
745      strbuf_addf(err, "\n\t%s",
746            format->unknown_extensions.items[i].string);
747    return -1;
748  }
749
750  if (format->version == 0 && format->v1_only_extensions.nr) {
751    int i;
752
753    strbuf_addstr(err,
754            Q_("repo version is 0, but v1-only extension found:",
755         "repo version is 0, but v1-only extensions found:",
756         format->v1_only_extensions.nr));
757
758    for (i = 0; i < format->v1_only_extensions.nr; i++)
759      strbuf_addf(err, "\n\t%s",
760            format->v1_only_extensions.items[i].string);
761    return -1;
762  }
763
764  return 0;
765}

767void read_gitfile_error_die(int error_code, const char *path, const char *dir)
768{
769  switch (error_code) {
770  case READ_GITFILE_ERR_STAT_FAILED:
771  case READ_GITFILE_ERR_NOT_A_FILE:
772    /* non-fatal; follow return path */
773    break;
774  case READ_GITFILE_ERR_OPEN_FAILED:
775    die_errno(_("error opening '%s'"), path);
776  case READ_GITFILE_ERR_TOO_LARGE:
777    die(_("too large to be a .git file: '%s'"), path);
778  case READ_GITFILE_ERR_READ_FAILED:
779    die(_("error reading %s"), path);
780  case READ_GITFILE_ERR_INVALID_FORMAT:
781    die(_("invalid gitfile format: %s"), path);
782  case READ_GITFILE_ERR_NO_PATH:
783    die(_("no path in gitfile: %s"), path);
784  case READ_GITFILE_ERR_NOT_A_REPO:
785    die(_("not a git repository: %s"), dir);
786  default:
787    BUG("unknown error code");
788  }
789}
790
791/*
792 * Try to read the location of the git directory from the .git file,
793 * return path to git directory if found. The return value comes from
794 * a shared buffer.
795 *
796 * On failure, if return_error_code is not NULL, return_error_code
797 * will be set to an error code and NULL will be returned. If
798 * return_error_code is NULL the function will die instead (for most
799 * cases).
800 */
801const char *read_gitfile_gently(const char *path, int *return_error_code)
802{
803  const int max_file_size = 1 << 20;  /* 1MB */
804  int error_code = 0;
805  char *buf = NULL;
806  char *dir = NULL;
807  const char *slash;
808  struct stat st;
809  int fd;
810  ssize_t len;
811  static struct strbuf realpath = STRBUF_INIT;
812
813  if (stat(path, &st)) {
814    /* NEEDSWORK: discern between ENOENT vs other errors */
815    error_code = READ_GITFILE_ERR_STAT_FAILED;
816    goto cleanup_return;
817  }
818  if (!S_ISREG(st.st_mode)) {
819    error_code = READ_GITFILE_ERR_NOT_A_FILE;
820    goto cleanup_return;
821  }
822  if (st.st_size > max_file_size) {
823    error_code = READ_GITFILE_ERR_TOO_LARGE;
824    goto cleanup_return;
825  }
826  fd = open(path, O_RDONLY);
827  if (fd < 0) {
828    error_code = READ_GITFILE_ERR_OPEN_FAILED;
829    goto cleanup_return;
830  }
831  buf = xmallocz(st.st_size);
832  len = read_in_full(fd, buf, st.st_size);
833  close(fd);
834  if (len != st.st_size) {
835    error_code = READ_GITFILE_ERR_READ_FAILED;
836    goto cleanup_return;
837  }
838  if (!starts_with(buf, "gitdir: ")) {
839    error_code = READ_GITFILE_ERR_INVALID_FORMAT;
840    goto cleanup_return;
841  }
842  while (buf[len - 1] == '\n' || buf[len - 1] == '\r')
843    len--;
844  if (len < 9) {
845    error_code = READ_GITFILE_ERR_NO_PATH;
846    goto cleanup_return;
847  }
848  buf[len] = '\0';
849  dir = buf + 8;
850
851  if (!is_absolute_path(dir) && (slash = strrchr(path, '/'))) {
852    size_t pathlen = slash+1 - path;
853    dir = xstrfmt("%.*s%.*s", (int)pathlen, path,
854            (int)(len - 8), buf + 8);
855    free(buf);
856    buf = dir;
857  }
858  if (!is_git_directory(dir)) {
859    error_code = READ_GITFILE_ERR_NOT_A_REPO;
860    goto cleanup_return;
861  }
862
863  strbuf_realpath(&realpath, dir, 1);
864  path = realpath.buf;
865
866cleanup_return:
867  if (return_error_code)
868    *return_error_code = error_code;
869  else if (error_code)
870    read_gitfile_error_die(error_code, path, dir);
871
872  free(buf);
873  return error_code ? NULL : path;
874}

876static const char *setup_explicit_git_dir(const char *gitdirenv,
877            struct strbuf *cwd,
878            struct repository_format *repo_fmt,
879            int *nongit_ok)
880{
881  const char *work_tree_env = getenv(GIT_WORK_TREE_ENVIRONMENT);
882  const char *worktree;
883  char *gitfile;
884  int offset;
885
886  if (PATH_MAX - 40 < strlen(gitdirenv))
887    die(_("'$%s' too big"), GIT_DIR_ENVIRONMENT);
888
889  gitfile = (char*)read_gitfile(gitdirenv);
890  if (gitfile) {
891    gitfile = xstrdup(gitfile);
892    gitdirenv = gitfile;
893  }
894
895  if (!is_git_directory(gitdirenv)) {
896    if (nongit_ok) {
897      *nongit_ok = 1;
898      free(gitfile);
899      return NULL;
900    }
901    die(_("not a git repository: '%s'"), gitdirenv);
902  }
903
904  if (check_repository_format_gently(gitdirenv, repo_fmt, nongit_ok)) {
905    free(gitfile);
906    return NULL;
907  }
908
909  /* #3, #7, #11, #15, #19, #23, #27, #31 (see t1510) */
910  if (work_tree_env)
911    set_git_work_tree(work_tree_env);
912  else if (is_bare_repository_cfg > 0) {
913    if (git_work_tree_cfg) {
914      /* #22.2, #30 */
915      warning("core.bare and core.worktree do not make sense");
916      work_tree_config_is_bogus = 1;
917    }
918
919    /* #18, #26 */
920    set_git_dir(gitdirenv, 0);
921    free(gitfile);
922    return NULL;
923  }
924  else if (git_work_tree_cfg) { /* #6, #14 */
925    if (is_absolute_path(git_work_tree_cfg))
926      set_git_work_tree(git_work_tree_cfg);
927    else {
928      char *core_worktree;
929      if (chdir(gitdirenv))
930        die_errno(_("cannot chdir to '%s'"), gitdirenv);
931      if (chdir(git_work_tree_cfg))
932        die_errno(_("cannot chdir to '%s'"), git_work_tree_cfg);
933      core_worktree = xgetcwd();
934      if (chdir(cwd->buf))
935        die_errno(_("cannot come back to cwd"));
936      set_git_work_tree(core_worktree);
937      free(core_worktree);
938    }
939  }
940  else if (!git_env_bool(GIT_IMPLICIT_WORK_TREE_ENVIRONMENT, 1)) {
941    /* #16d */
942    set_git_dir(gitdirenv, 0);
943    free(gitfile);
944    return NULL;
945  }
946  else /* #2, #10 */
947    set_git_work_tree(".");
948
949  /* set_git_work_tree() must have been called by now */
950  worktree = get_git_work_tree();
951
952  /* both get_git_work_tree() and cwd are already normalized */
953  if (!strcmp(cwd->buf, worktree)) { /* cwd == worktree */
954    set_git_dir(gitdirenv, 0);
955    free(gitfile);
956    return NULL;
957  }
958
959  offset = dir_inside_of(cwd->buf, worktree);
960  if (offset >= 0) {  /* cwd inside worktree? */
961    set_git_dir(gitdirenv, 1);
962    if (chdir(worktree))
963      die_errno(_("cannot chdir to '%s'"), worktree);
964    strbuf_addch(cwd, '/');
965    free(gitfile);
966    return cwd->buf + offset;
967  }
968
969  /* cwd outside worktree */
970  set_git_dir(gitdirenv, 0);
971  free(gitfile);
972  return NULL;
973}

975static const char *setup_discovered_git_dir(const char *gitdir,
976              struct strbuf *cwd, int offset,
977              struct repository_format *repo_fmt,
978              int *nongit_ok)
979{
980  if (check_repository_format_gently(gitdir, repo_fmt, nongit_ok))
981    return NULL;
982
983  /* --work-tree is set without --git-dir; use discovered one */
984  if (getenv(GIT_WORK_TREE_ENVIRONMENT) || git_work_tree_cfg) {
985    char *to_free = NULL;
986    const char *ret;
987
988    if (offset != cwd->len && !is_absolute_path(gitdir))
989      gitdir = to_free = real_pathdup(gitdir, 1);
990    if (chdir(cwd->buf))
991      die_errno(_("cannot come back to cwd"));
992    ret = setup_explicit_git_dir(gitdir, cwd, repo_fmt, nongit_ok);
993    free(to_free);
994    return ret;
995  }
996
997  /* #16.2, #17.2, #20.2, #21.2, #24, #25, #28, #29 (see t1510) */
998  if (is_bare_repository_cfg > 0) {
999    set_git_dir(gitdir, (offset != cwd->len));
1000    if (chdir(cwd->buf))
1001      die_errno(_("cannot come back to cwd"));
1002    return NULL;
1003  }
1004
1005  /* #0, #1, #5, #8, #9, #12, #13 */
1006  set_git_work_tree(".");
1007  if (strcmp(gitdir, DEFAULT_GIT_DIR_ENVIRONMENT))
1008    set_git_dir(gitdir, 0);
1009  inside_git_dir = 0;
1010  inside_work_tree = 1;
1011  if (offset >= cwd->len)
1012    return NULL;
1013
1014  /* Make "offset" point past the '/' (already the case for root dirs) */
1015  if (offset != offset_1st_component(cwd->buf))
1016    offset++;
1017  /* Add a '/' at the end */
1018  strbuf_addch(cwd, '/');
1019  return cwd->buf + offset;
1020}

1022/* #16.1, #17.1, #20.1, #21.1, #22.1 (see t1510) */
1023static const char *setup_bare_git_dir(struct strbuf *cwd, int offset,
1024              struct repository_format *repo_fmt,
1025              int *nongit_ok)
1026{
1027  int root_len;
1028
1029  if (check_repository_format_gently(".", repo_fmt, nongit_ok))
1030    return NULL;
1031
1032  setenv(GIT_IMPLICIT_WORK_TREE_ENVIRONMENT, "0", 1);
1033
1034  /* --work-tree is set without --git-dir; use discovered one */
1035  if (getenv(GIT_WORK_TREE_ENVIRONMENT) || git_work_tree_cfg) {
1036    static const char *gitdir;
1037
1038    gitdir = offset == cwd->len ? "." : xmemdupz(cwd->buf, offset);
1039    if (chdir(cwd->buf))
1040      die_errno(_("cannot come back to cwd"));
1041    return setup_explicit_git_dir(gitdir, cwd, repo_fmt, nongit_ok);
1042  }
1043
1044  inside_git_dir = 1;
1045  inside_work_tree = 0;

1072 * from GIT_CEILING_DIRECTORIES using real_pathdup(), or
1073 * discards it if unusable.  The presence of an empty entry in
1074 * GIT_CEILING_DIRECTORIES turns off canonicalization for all
1075 * subsequent entries.

1078              void *cb_data)
1079{

1084    *empty_entry_found = 1;
1085    return 0;
1086  } else if (!is_absolute_path(ceil)) {
1087    return 0;
1088  } else if (*empty_entry_found) {
1089    /* Keep entry but do not canonicalize it */

1091  } else {

1093    if (!real_path) {

1095    }

1097    item->string = real_path;

1162enum discovery_result {
1163  GIT_DIR_NONE = 0,
1164  GIT_DIR_EXPLICIT,
1165  GIT_DIR_DISCOVERED,
1166  GIT_DIR_BARE,
1167  /* these are errors */
1168  GIT_DIR_HIT_CEILING = -1,
1169  GIT_DIR_HIT_MOUNT_POINT = -2,
1170  GIT_DIR_INVALID_GITFILE = -3,
1171  GIT_DIR_INVALID_OWNERSHIP = -4
1172};
1173
1174/*
1175 * We cannot decide in this function whether we are in the work tree or
1176 * not, since the config can only be read _after_ this function was called.
1177 *
1178 * Also, we avoid changing any global state (such as the current working
1179 * directory) to allow early callers.
1180 *
1181 * The directory where the search should start needs to be passed in via the
1182 * `dir` parameter; upon return, the `dir` buffer will contain the path of
1183 * the directory where the search ended, and `gitdir` will contain the path of
1184 * the discovered .git/ directory, if any. If `gitdir` is not absolute, it
1185 * is relative to `dir` (i.e. *not* necessarily the cwd).
1186 */
1187static enum discovery_result setup_git_directory_gently_1(struct strbuf *dir,
1188                struct strbuf *gitdir,
1189                int die_on_error)
1190{
1191  const char *env_ceiling_dirs = getenv(CEILING_DIRECTORIES_ENVIRONMENT);
1192  struct string_list ceiling_dirs = STRING_LIST_INIT_DUP;
1193  const char *gitdirenv;
1194  int ceil_offset = -1, min_offset = offset_1st_component(dir->buf);
1195  dev_t current_device = 0;
1196  int one_filesystem = 1;
1197
1198  /*
1199   * If GIT_DIR is set explicitly, we're not going
1200   * to do any discovery, but we still do repository
1201   * validation.
1202   */
1203  gitdirenv = getenv(GIT_DIR_ENVIRONMENT);
1204  if (gitdirenv) {
1205    strbuf_addstr(gitdir, gitdirenv);
1206    return GIT_DIR_EXPLICIT;
1207  }
1208
1209  if (env_ceiling_dirs) {
1210    int empty_entry_found = 0;
1211
1212    string_list_split(&ceiling_dirs, env_ceiling_dirs, PATH_SEP, -1);
1213    filter_string_list(&ceiling_dirs, 0,
1214           canonicalize_ceiling_entry, &empty_entry_found);
1215    ceil_offset = longest_ancestor_length(dir->buf, &ceiling_dirs);
1216    string_list_clear(&ceiling_dirs, 0);
1217  }
1218
1219  if (ceil_offset < 0)
1220    ceil_offset = min_offset - 2;
1221
1222  if (min_offset && min_offset == dir->len &&
1223      !is_dir_sep(dir->buf[min_offset - 1])) {
1224    strbuf_addch(dir, '/');
1225    min_offset++;
1226  }
1227
1228  /*
1229   * Test in the following order (relative to the dir):
1230   * - .git (file containing "gitdir: <path>")
1231   * - .git/
1232   * - ./ (bare)
1233   * - ../.git
1234   * - ../.git/
1235   * - ../ (bare)
1236   * - ../../.git
1237   *   etc.
1238   */
1239  one_filesystem = !git_env_bool("GIT_DISCOVERY_ACROSS_FILESYSTEM", 0);
1240  if (one_filesystem)
1241    current_device = get_device_or_die(dir->buf, NULL, 0);
1242  for (;;) {
1243    int offset = dir->len, error_code = 0;
1244    char *gitdir_path = NULL;
1245    char *gitfile = NULL;
1246
1247    if (offset > min_offset)
1248      strbuf_addch(dir, '/');
1249    strbuf_addstr(dir, DEFAULT_GIT_DIR_ENVIRONMENT);
1250    gitdirenv = read_gitfile_gently(dir->buf, die_on_error ?
1251            NULL : &error_code);
1252    if (!gitdirenv) {
1253      if (die_on_error ||
1254          error_code == READ_GITFILE_ERR_NOT_A_FILE) {
1255        /* NEEDSWORK: fail if .git is not file nor dir */
1256        if (is_git_directory(dir->buf)) {
1257          gitdirenv = DEFAULT_GIT_DIR_ENVIRONMENT;
1258          gitdir_path = xstrdup(dir->buf);
1259        }
1260      } else if (error_code != READ_GITFILE_ERR_STAT_FAILED)
1261        return GIT_DIR_INVALID_GITFILE;
1262    } else
1263      gitfile = xstrdup(dir->buf);
1264    /*
1265     * Earlier, we tentatively added DEFAULT_GIT_DIR_ENVIRONMENT
1266     * to check that directory for a repository.
1267     * Now trim that tentative addition away, because we want to
1268     * focus on the real directory we are in.
1269     */
1270    strbuf_setlen(dir, offset);
1271    if (gitdirenv) {
1272      enum discovery_result ret;
1273
1274      if (ensure_valid_ownership(gitfile,
1275             dir->buf,
1276         (gitdir_path ? gitdir_path : gitdirenv))) {
1277        strbuf_addstr(gitdir, gitdirenv);
1278        ret = GIT_DIR_DISCOVERED;
1279      } else
1280        ret = GIT_DIR_INVALID_OWNERSHIP;
1281
1282      /*
1283       * Earlier, during discovery, we might have allocated
1284       * string copies for gitdir_path or gitfile so make
1285       * sure we don't leak by freeing them now, before
1286       * leaving the loop and function.
1287       *
1288       * Note: gitdirenv will be non-NULL whenever these are
1289       * allocated, therefore we need not take care of releasing
1290       * them outside of this conditional block.
1291       */
1292      free(gitdir_path);
1293      free(gitfile);
1294
1295      return ret;
1296    }
1297
1298    if (is_git_directory(dir->buf)) {
1299      if (!ensure_valid_ownership(NULL, NULL, dir->buf))
1300        return GIT_DIR_INVALID_OWNERSHIP;
1301      strbuf_addstr(gitdir, ".");
1302      return GIT_DIR_BARE;
1303    }
1304
1305    if (offset <= min_offset)
1306      return GIT_DIR_HIT_CEILING;
1307
1308    while (--offset > ceil_offset && !is_dir_sep(dir->buf[offset]))
1309      ; /* continue */
1310    if (offset <= ceil_offset)
1311      return GIT_DIR_HIT_CEILING;
1312
1313    strbuf_setlen(dir, offset > min_offset ?  offset : min_offset);
1314    if (one_filesystem &&
1315        current_device != get_device_or_die(dir->buf, NULL, offset))
1316      return GIT_DIR_HIT_MOUNT_POINT;
1317  }
1318}

1320int discover_git_directory(struct strbuf *commondir,
1321         struct strbuf *gitdir)
1322{
1323  struct strbuf dir = STRBUF_INIT, err = STRBUF_INIT;
1324  size_t gitdir_offset = gitdir->len, cwd_len;
1325  size_t commondir_offset = commondir->len;
1326  struct repository_format candidate = REPOSITORY_FORMAT_INIT;
1327
1328  if (strbuf_getcwd(&dir))
1329    return -1;
1330
1331  cwd_len = dir.len;
1332  if (setup_git_directory_gently_1(&dir, gitdir, 0) <= 0) {
1333    strbuf_release(&dir);
1334    return -1;
1335  }
1336
1337  /*
1338   * The returned gitdir is relative to dir, and if dir does not reflect
1339   * the current working directory, we simply make the gitdir absolute.
1340   */
1341  if (dir.len < cwd_len && !is_absolute_path(gitdir->buf + gitdir_offset)) {
1342    /* Avoid a trailing "/." */
1343    if (!strcmp(".", gitdir->buf + gitdir_offset))
1344      strbuf_setlen(gitdir, gitdir_offset);
1345    else
1346      strbuf_addch(&dir, '/');
1347    strbuf_insert(gitdir, gitdir_offset, dir.buf, dir.len);
1348  }
1349
1350  get_common_dir(commondir, gitdir->buf + gitdir_offset);
1351
1352  strbuf_reset(&dir);
1353  strbuf_addf(&dir, "%s/config", commondir->buf + commondir_offset);
1354  read_repository_format(&candidate, dir.buf);
1355  strbuf_release(&dir);
1356
1357  if (verify_repository_format(&candidate, &err) < 0) {
1358    warning("ignoring git dir '%s': %s",
1359      gitdir->buf + gitdir_offset, err.buf);
1360    strbuf_release(&err);
1361    strbuf_setlen(commondir, commondir_offset);
1362    strbuf_setlen(gitdir, gitdir_offset);
1363    clear_repository_format(&candidate);
1364    return -1;
1365  }
1366
1367  /* take ownership of candidate.partial_clone */
1368  the_repository->repository_format_partial_clone =
1369    candidate.partial_clone;
1370  candidate.partial_clone = NULL;
1371
1372  clear_repository_format(&candidate);
1373  return 0;
1374}

1376const char *setup_git_directory_gently(int *nongit_ok)
1377{
1378  static struct strbuf cwd = STRBUF_INIT;
1379  struct strbuf dir = STRBUF_INIT, gitdir = STRBUF_INIT;
1380  const char *prefix = NULL;
1381  struct repository_format repo_fmt = REPOSITORY_FORMAT_INIT;
1382
1383  /*
1384   * We may have read an incomplete configuration before
1385   * setting-up the git directory. If so, clear the cache so
1386   * that the next queries to the configuration reload complete
1387   * configuration (including the per-repo config file that we
1388   * ignored previously).
1389   */
1390  git_config_clear();
1391
1392  /*
1393   * Let's assume that we are in a git repository.
1394   * If it turns out later that we are somewhere else, the value will be
1395   * updated accordingly.
1396   */
1397  if (nongit_ok)
1398    *nongit_ok = 0;
1399
1400  if (strbuf_getcwd(&cwd))
1401    die_errno(_("Unable to read current working directory"));
1402  strbuf_addbuf(&dir, &cwd);
1403
1404  switch (setup_git_directory_gently_1(&dir, &gitdir, 1)) {
1405  case GIT_DIR_EXPLICIT:
1406    prefix = setup_explicit_git_dir(gitdir.buf, &cwd, &repo_fmt, nongit_ok);
1407    break;
1408  case GIT_DIR_DISCOVERED:
1409    if (dir.len < cwd.len && chdir(dir.buf))
1410      die(_("cannot change to '%s'"), dir.buf);
1411    prefix = setup_discovered_git_dir(gitdir.buf, &cwd, dir.len,
1412              &repo_fmt, nongit_ok);
1413    break;
1414  case GIT_DIR_BARE:
1415    if (dir.len < cwd.len && chdir(dir.buf))
1416      die(_("cannot change to '%s'"), dir.buf);
1417    prefix = setup_bare_git_dir(&cwd, dir.len, &repo_fmt, nongit_ok);
1418    break;
1419  case GIT_DIR_HIT_CEILING:
1420    if (!nongit_ok)
1421      die(_("not a git repository (or any of the parent directories): %s"),
1422          DEFAULT_GIT_DIR_ENVIRONMENT);
1423    *nongit_ok = 1;
1424    break;
1425  case GIT_DIR_HIT_MOUNT_POINT:
1426    if (!nongit_ok)
1427      die(_("not a git repository (or any parent up to mount point %s)\n"
1428            "Stopping at filesystem boundary (GIT_DISCOVERY_ACROSS_FILESYSTEM not set)."),
1429          dir.buf);
1430    *nongit_ok = 1;
1431    break;
1432  case GIT_DIR_INVALID_OWNERSHIP:
1433    if (!nongit_ok) {
1434      struct strbuf quoted = STRBUF_INIT;
1435
1436      sq_quote_buf_pretty(&quoted, dir.buf);
1437      die(_("detected dubious ownership in repository at '%s'\n"
1438            "To add an exception for this directory, call:\n"
1439            "\n"
1440            "\tgit config --global --add safe.directory %s"),
1441          dir.buf, quoted.buf);
1442    }
1443    *nongit_ok = 1;
1444    break;
1445  case GIT_DIR_NONE:
1446    /*
1447     * As a safeguard against setup_git_directory_gently_1 returning
1448     * this value, fallthrough to BUG. Otherwise it is possible to
1449     * set startup_info->have_repository to 1 when we did nothing to
1450     * find a repository.
1451     */
1452  default:
1453    BUG("unhandled setup_git_directory_gently_1() result");
1454  }
1455
1456  /*
1457   * At this point, nongit_ok is stable. If it is non-NULL and points
1458   * to a non-zero value, then this means that we haven't found a
1459   * repository and that the caller expects startup_info to reflect
1460   * this.
1461   *
1462   * Regardless of the state of nongit_ok, startup_info->prefix and
1463   * the GIT_PREFIX environment variable must always match. For details
1464   * see Documentation/config/alias.txt.
1465   */
1466  if (nongit_ok && *nongit_ok)
1467    startup_info->have_repository = 0;
1468  else
1469    startup_info->have_repository = 1;
1470
1471  /*
1472   * Not all paths through the setup code will call 'set_git_dir()' (which
1473   * directly sets up the environment) so in order to guarantee that the
1474   * environment is in a consistent state after setup, explicitly setup
1475   * the environment if we have a repository.
1476   *
1477   * NEEDSWORK: currently we allow bogus GIT_DIR values to be set in some
1478   * code paths so we also need to explicitly setup the environment if
1479   * the user has set GIT_DIR.  It may be beneficial to disallow bogus
1480   * GIT_DIR values at some point in the future.
1481   */
1482  if (/* GIT_DIR_EXPLICIT, GIT_DIR_DISCOVERED, GIT_DIR_BARE */
1483      startup_info->have_repository ||
1484      /* GIT_DIR_EXPLICIT */
1485      getenv(GIT_DIR_ENVIRONMENT)) {
1486    if (!the_repository->gitdir) {
1487      const char *gitdir = getenv(GIT_DIR_ENVIRONMENT);
1488      if (!gitdir)
1489        gitdir = DEFAULT_GIT_DIR_ENVIRONMENT;
1490      setup_git_env(gitdir);
1491    }
1492    if (startup_info->have_repository) {
1493      repo_set_hash_algo(the_repository, repo_fmt.hash_algo);
1494      /* take ownership of repo_fmt.partial_clone */
1495      the_repository->repository_format_partial_clone =
1496        repo_fmt.partial_clone;
1497      repo_fmt.partial_clone = NULL;
1498    }
1499  }
1500  /*
1501   * Since precompose_string_if_needed() needs to look at
1502   * the core.precomposeunicode configuration, this
1503   * has to happen after the above block that finds
1504   * out where the repository is, i.e. a preparation
1505   * for calling git_config_get_bool().
1506   */
1507  if (prefix) {
1508    prefix = precompose_string_if_needed(prefix);
1509    startup_info->prefix = prefix;
1510    setenv(GIT_PREFIX_ENVIRONMENT, prefix, 1);
1511  } else {
1512    startup_info->prefix = NULL;
1513    setenv(GIT_PREFIX_ENVIRONMENT, "", 1);
1514  }
1515
1516  setup_original_cwd();
1517
1518  strbuf_release(&dir);
1519  strbuf_release(&gitdir);
1520  clear_repository_format(&repo_fmt);
1521
1522  return prefix;
1523}
1524
1525int git_config_perm(const char *var, const char *value)
1526{
1527  int i;
1528  char *endptr;
1529
1530  if (!value)
1531    return PERM_GROUP;
1532
1533  if (!strcmp(value, "umask"))
1534    return PERM_UMASK;
1535  if (!strcmp(value, "group"))
1536    return PERM_GROUP;
1537  if (!strcmp(value, "all") ||
1538      !strcmp(value, "world") ||
1539      !strcmp(value, "everybody"))
1540    return PERM_EVERYBODY;
1541
1542  /* Parse octal numbers */
1543  i = strtol(value, &endptr, 8);
1544
1545  /* If not an octal number, maybe true/false? */
1546  if (*endptr != 0)
1547    return git_config_bool(var, value) ? PERM_GROUP : PERM_UMASK;
1548
1549  /*
1550   * Treat values 0, 1 and 2 as compatibility cases, otherwise it is
1551   * a chmod value to restrict to.
1552   */
1553  switch (i) {
1554  case PERM_UMASK:               /* 0 */
1555    return PERM_UMASK;
1556  case OLD_PERM_GROUP:           /* 1 */
1557    return PERM_GROUP;
1558  case OLD_PERM_EVERYBODY:       /* 2 */
1559    return PERM_EVERYBODY;
1560  }
1561
1562  /* A filemode value was given: 0xxx */
1563
1564  if ((i & 0600) != 0600)
1565    die(_("problem with core.sharedRepository filemode value "
1566        "(0%.3o).\nThe owner of files must always have "
1567        "read and write permissions."), i);
1568
1569  /*
1570   * Mask filemode value. Others can not get write permission.
1571   * x flags for directories are handled separately.
1572   */
1573  return -(i & 0666);
1574}

1617int daemonize(void)
1618{
1619#ifdef NO_POSIX_GOODIES
1620  errno = ENOSYS;
1621  return -1;
1622#else
1623  switch (fork()) {
1624    case 0:
1625      break;
1626    case -1:
1627      die_errno(_("fork failed"));
1628    default:
1629      exit(0);
1630  }
1631  if (setsid() == -1)
1632    die_errno(_("setsid failed"));
1633  close(0);
1634  close(1);
1635  close(2);
1636  sanitize_stdfds();
1637  return 0;
1638#endif
1639}
