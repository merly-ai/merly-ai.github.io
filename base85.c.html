39int decode_85(char *dst, const char *buffer, int len)
40{
41  prep_base85();
42
43  say2("decode 85 <%.*s>", len / 4 * 5, buffer);
44  while (len) {
45    unsigned acc = 0;
46    int de, cnt = 4;
47    unsigned char ch;
48    do {
49      ch = *buffer++;
50      de = de85[ch];
51      if (--de < 0)
52        return error("invalid base85 alphabet %c", ch);
53      acc = acc * 85 + de;
54    } while (--cnt);
55    ch = *buffer++;
56    de = de85[ch];
57    if (--de < 0)
58      return error("invalid base85 alphabet %c", ch);
59    /* Detect overflow. */
60    if (0xffffffff / 85 < acc ||
61        0xffffffff - de < (acc *= 85))
62      return error("invalid base85 sequence %.5s", buffer-5);
63    acc += de;
64    say1(" %08x", acc);
65
66    cnt = (len < 4) ? len : 4;
67    len -= cnt;
68    do {
69      acc = (acc << 8) | (acc >> 24);
70      *dst++ = acc;
71    } while (--cnt);
72  }
73  say("\n");
74
75  return 0;
76}

78void encode_85(char *buf, const unsigned char *data, int bytes)
79{
80  say("encode 85");
81  while (bytes) {
82    unsigned acc = 0;
83    int cnt;
84    for (cnt = 24; cnt >= 0; cnt -= 8) {
85      unsigned ch = *data++;
86      acc |= ch << cnt;
87      if (--bytes == 0)
88        break;
89    }
90    say1(" %08x", acc);
91    for (cnt = 4; cnt >= 0; cnt--) {
92      int val = acc % 85;
93      acc /= 85;
94      buf[cnt] = en85[val];
95    }
96    buf += 5;
97  }
98  say("\n");
99
100  *buf = 0;
101}
