194static zend_always_inline int zend_ssa_next_use(const zend_ssa_op *ssa_op, int var, int use)
195{
196  ssa_op += use;
197  if (ssa_op->op1_use == var) {
198    return ssa_op->op1_use_chain;
199  } else if (ssa_op->op2_use == var) {
200    return ssa_op->op2_use_chain;
201  } else {
202    return ssa_op->res_use_chain;
203  }
204}

206static zend_always_inline zend_ssa_phi* zend_ssa_next_use_phi(const zend_ssa *ssa, int var, const zend_ssa_phi *p)
207{
208  if (p->pi >= 0) {
209    return p->use_chains[0];
210  } else {
211    int j;
212    for (j = 0; j < ssa->cfg.blocks[p->block].predecessors_count; j++) {
213      if (p->sources[j] == var) {
214        return p->use_chains[j];
215      }
216    }
217  }
218  return NULL;
219}
