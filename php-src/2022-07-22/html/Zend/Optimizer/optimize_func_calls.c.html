39static void zend_delete_call_instructions(zend_op_array *op_array, zend_op *opline)
40{
41  int call = 0;
42
43  while (1) {
44    switch (opline->opcode) {
45      case ZEND_INIT_FCALL_BY_NAME:
46      case ZEND_INIT_NS_FCALL_BY_NAME:
47      case ZEND_INIT_STATIC_METHOD_CALL:
48      case ZEND_INIT_METHOD_CALL:
49      case ZEND_INIT_FCALL:
50        if (call == 0) {
51          MAKE_NOP(opline);
52          return;
53        }
54        ZEND_FALLTHROUGH;
55      case ZEND_NEW:
56      case ZEND_INIT_DYNAMIC_CALL:
57      case ZEND_INIT_USER_CALL:
58        call--;
59        break;
60      case ZEND_DO_FCALL:
61      case ZEND_DO_ICALL:
62      case ZEND_DO_UCALL:
63      case ZEND_DO_FCALL_BY_NAME:
64        call++;
65        break;
66      case ZEND_SEND_VAL:
67      case ZEND_SEND_VAR:
68        if (call == 0) {
69          zend_optimizer_convert_to_free_op1(op_array, opline);
70        }
71        break;
72    }
73    opline--;
74  }
75}

77static void zend_try_inline_call(zend_op_array *op_array, zend_op *fcall, zend_op *opline, zend_function *func)
78{
79  if (func->type == ZEND_USER_FUNCTION
80   && !(func->op_array.fn_flags & (ZEND_ACC_ABSTRACT|ZEND_ACC_HAS_TYPE_HINTS))
81    /* TODO: function copied from trait may be inconsistent ??? */
82   && !(func->op_array.fn_flags & (ZEND_ACC_TRAIT_CLONE))
83   && fcall->extended_value >= func->op_array.required_num_args
84   && func->op_array.opcodes[func->op_array.num_args].opcode == ZEND_RETURN) {
85
86    zend_op *ret_opline = func->op_array.opcodes + func->op_array.num_args;
87
88    if (ret_opline->op1_type == IS_CONST) {
89      uint32_t i, num_args = func->op_array.num_args;
90      num_args += (func->op_array.fn_flags & ZEND_ACC_VARIADIC) != 0;
91
92      if (fcall->opcode == ZEND_INIT_STATIC_METHOD_CALL
93          && !(func->op_array.fn_flags & ZEND_ACC_STATIC)) {
94        /* Don't inline static call to instance method. */
95        return;
96      }
97
98      for (i = 0; i < num_args; i++) {
99        /* Don't inline functions with by-reference arguments. This would require
100         * correct handling of INDIRECT arguments. */
101        if (ZEND_ARG_SEND_MODE(&func->op_array.arg_info[i])) {
102          return;
103        }
104      }
105
106      if (fcall->extended_value < func->op_array.num_args) {
107        /* don't inline functions with named constants in default arguments */
108        i = fcall->extended_value;
109
110        do {
111          if (Z_TYPE_P(CRT_CONSTANT_EX(&func->op_array, &func->op_array.opcodes[i], func->op_array.opcodes[i].op2)) == IS_CONSTANT_AST) {
112            return;
113          }
114          i++;
115        } while (i < func->op_array.num_args);
116      }
117
118      if (RETURN_VALUE_USED(opline)) {
119        zval zv;
120
121        ZVAL_COPY(&zv, CRT_CONSTANT_EX(&func->op_array, ret_opline, ret_opline->op1));
122        opline->opcode = ZEND_QM_ASSIGN;
123        opline->op1_type = IS_CONST;
124        opline->op1.constant = zend_optimizer_add_literal(op_array, &zv);
125        SET_UNUSED(opline->op2);
126      } else {
127        MAKE_NOP(opline);
128      }
129
130      zend_delete_call_instructions(op_array, opline-1);
131    }
132  }
133}

135/* arg_num is 1-based here, to match SEND encoding. */
136static bool has_known_send_mode(const optimizer_call_info *info, uint32_t arg_num)
137{
138  if (!info->func) {
139    return false;
140  }
141
142  /* For prototype functions we should not make assumptions about arguments that are not part of
143   * the signature: And inheriting method can add an optional by-ref argument. */
144  return !info->is_prototype
145    || arg_num <= info->func->common.num_args
146    || (info->func->common.fn_flags & ZEND_ACC_VARIADIC);
147}

202            }
203          } else if (fcall->opcode == ZEND_INIT_NS_FCALL_BY_NAME) {
204            fcall->opcode = ZEND_INIT_FCALL;
205            fcall->op1.num = zend_vm_calc_used_stack(fcall->extended_value, call_stack[call].func);
206            literal_dtor(&op_array->literals[fcall->op2.constant]);
207            literal_dtor(&op_array->literals[fcall->op2.constant + 2]);
208            fcall->op2.constant = fcall->op2.constant + 1;
209            if (opline->opcode != ZEND_CALLABLE_CONVERT) {
210              opline->opcode = zend_get_call_op(fcall, call_stack[call].func);
211            }
212          } else if (fcall->opcode == ZEND_INIT_STATIC_METHOD_CALL
213              || fcall->opcode == ZEND_INIT_METHOD_CALL
214              || fcall->opcode == ZEND_NEW) {
