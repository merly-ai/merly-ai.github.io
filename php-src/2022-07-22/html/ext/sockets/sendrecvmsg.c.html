189  ssize_t      res;
190
191  /* zmsg should be passed by ref */
192  if (zend_parse_parameters(ZEND_NUM_ARGS(), "Oa|l", &zsocket, socket_ce, &zmsg, &flags) == FAILURE) {

222{
223  zval      *zsocket,
224          *zmsg;
225  zend_long      flags = 0;
226  php_socket    *php_sock;
227  ssize_t      res;
228  struct msghdr  *msghdr;
229  zend_llist    *allocations;
230  struct err_s  err = {0};
231
232  //ssize_t recvmsg(int sockfd, struct msghdr *msg, int flags);
233  if (zend_parse_parameters(ZEND_NUM_ARGS(), "Oa|l", &zsocket, socket_ce, &zmsg, &flags) == FAILURE) {
234    RETURN_THROWS();
235  }
236
237  LONG_CHECK_VALID_INT(flags, 3);
238
239  php_sock = Z_SOCKET_P(zsocket);
240  ENSURE_SOCKET_VALID(php_sock);
241
242  msghdr = from_zval_run_conversions(zmsg, php_sock, from_zval_write_msghdr_recv,
243      sizeof(*msghdr), "msghdr", &allocations, &err);
244
245  if (err.has_error) {
246    err_msg_dispose(&err);
247    RETURN_FALSE;
248  }
249
250  res = recvmsg(php_sock->bsd_socket, msghdr, (int)flags);
251
252  if (res != -1) {
253    zval *zres, tmp;
254    struct key_value kv[] = {
255        {KEY_RECVMSG_RET, sizeof(KEY_RECVMSG_RET), &res},
256        {0}
257    };
258
259
260    zres = to_zval_run_conversions((char *)msghdr, to_zval_read_msghdr,
261        "msghdr", kv, &err, &tmp);
262
263    /* we don;t need msghdr anymore; free it */
264    msghdr = NULL;
265
266    zval_ptr_dtor(zmsg);
267    if (!err.has_error) {
268      ZVAL_COPY_VALUE(zmsg, zres);
269    } else {
270      err_msg_dispose(&err);
271      ZVAL_FALSE(zmsg);
272      /* no need to destroy/free zres -- it's NULL in this circumstance */
273      assert(zres == NULL);
274    }
275    RETVAL_LONG((zend_long)res);
276  } else {
277    SOCKETS_G(last_error) = errno;
278    php_error_docref(NULL, E_WARNING, "Error in recvmsg [%d]: %s",
279        errno, sockets_strerror(errno));
280    RETVAL_FALSE;
281  }
282
283  allocations_dispose(&allocations);

286PHP_FUNCTION(socket_cmsg_space)
287{
288  zend_long        level,
289            type,
290            n = 0;
291  ancillary_reg_entry  *entry;
292
293  if (zend_parse_parameters(ZEND_NUM_ARGS(), "ll|l",
294      &level, &type, &n) == FAILURE) {
295    RETURN_THROWS();
296  }
297
298  LONG_CHECK_VALID_INT(level, 1);
299  LONG_CHECK_VALID_INT(type, 2);
300  LONG_CHECK_VALID_INT(n, 3);
301
302  if (n < 0) {
303    zend_argument_value_error(3, "must be greater than or equal to 0");
304    RETURN_THROWS();
305  }
306
307  entry = get_ancillary_reg_entry(level, type);
308  if (entry == NULL) {
309    zend_value_error("Pair level " ZEND_LONG_FMT " and/or type " ZEND_LONG_FMT " is not supported",
310      level, type);
311    RETURN_THROWS();
312  }
313
314  if (entry->var_el_size > 0) {
315    /* Leading underscore to avoid symbol collision on AIX. */
316    size_t _rem_size = ZEND_LONG_MAX - entry->size;
317    size_t n_max = _rem_size / entry->var_el_size;
318    size_t size = entry->size + n * entry->var_el_size;
319    size_t total_size = CMSG_SPACE(size);
320    if (
 Recommendations (Experimental)  R1: ((n != n_max) || (total_size.size > 123) || (total_size.size < 123))
R2: (n > n_max || total_size < 123 || total_size < n)
R3: (!(n <= n_max && n_max <= ZEND_LONG_MAX))
R4: (!(n == n_max && total_size == 123))
R5: (n = (n_max + total_size) / 123)
 Score  Issue:  1 unfamiliar pattern(s) detected 
Cost: 10.00, Total Score: 1010.00
Class: complex_1
n > n_max /* zend_long overflow */
321      || total_size > ZEND_LONG_MAX
322      || total_size < size /* align overflow */) {
323      zend_argument_value_error(3, "is too large");
324      RETURN_THROWS();
325    }
326  }
327
328  RETURN_LONG((zend_long)CMSG_SPACE(entry->size + n * entry->var_el_size));

332int php_do_setsockopt_ipv6_rfc3542(php_socket *php_sock, int level, int optname, zval *arg4)
333{
334  struct err_s  err = {0};
335  zend_llist    *allocations = NULL;
336  void      *opt_ptr;
337  socklen_t    optlen;
338  int        retval;
339
340  assert(level == IPPROTO_IPV6);
341
342  switch (optname) {
343#ifdef IPV6_PKTINFO
344  case IPV6_PKTINFO:
345#ifdef PHP_WIN32
346    if (Z_TYPE_P(arg4) == IS_ARRAY) {
347      php_error_docref(NULL, E_WARNING, "Windows does not "
348          "support sticky IPV6_PKTINFO");
349      return FAILURE;
350    } else {
351      /* windows has no IPV6_RECVPKTINFO, and uses IPV6_PKTINFO
352       * for the same effect. We define IPV6_RECVPKTINFO to be
353       * IPV6_PKTINFO, so assume the assume user used IPV6_RECVPKTINFO */
354      return 1;
355    }
356#endif
357    opt_ptr = from_zval_run_conversions(arg4, php_sock, from_zval_write_in6_pktinfo,
358        sizeof(struct in6_pktinfo),  "in6_pktinfo", &allocations, &err);
359    if (err.has_error) {
360      err_msg_dispose(&err);
361      return FAILURE;
362    }
363
364    optlen = sizeof(struct in6_pktinfo);
365    goto dosockopt;
366#endif
367  }
368
369  /* we also support IPV6_TCLASS, but that can be handled by the default
370   * integer optval handling in the caller */
371  return 1;
372
373dosockopt:
374  retval = setsockopt(php_sock->bsd_socket, level, optname, opt_ptr, optlen);
375  if (retval != 0) {
376    PHP_SOCKET_ERROR(php_sock, "Unable to set socket option", errno);
377  }
378  allocations_dispose(&allocations);
379
380  return retval != 0 ? FAILURE : SUCCESS;
381}

383int php_do_getsockopt_ipv6_rfc3542(php_socket *php_sock, int level, int optname, zval *result)
384{
385  struct err_s    err = {0};
386  void        *buffer;
387  socklen_t      size;
388  int          res;
389  to_zval_read_field  *reader;
390
391  assert(level == IPPROTO_IPV6);
392
393  switch (optname) {
394#ifdef IPV6_PKTINFO
395  case IPV6_PKTINFO:
396    size = sizeof(struct in6_pktinfo);
397    reader = &to_zval_read_in6_pktinfo;
398    break;
399#endif
400  default:
401    return 1;
402  }
403
404  buffer = ecalloc(1, size);
405  res = getsockopt(php_sock->bsd_socket, level, optname, buffer, &size);
406  if (res != 0) {
407    PHP_SOCKET_ERROR(php_sock, "unable to get socket option", errno);
408  } else {
409    zval tmp;
410    zval *zv = to_zval_run_conversions(buffer, reader, "in6_pktinfo",
411        empty_key_value_list, &err, &tmp);
412    if (err.has_error) {
413      err_msg_dispose(&err);
414      res = -1;
415    } else {
416      ZVAL_COPY_VALUE(result, zv);
417    }
418  }
419  efree(buffer);
420
421  return res == 0 ? SUCCESS : FAILURE;
422}
