60static bool php_mail_build_headers_check_field_value(zval *val)
61{
62  size_t len = 0;
63  zend_string *value = Z_STR_P(val);
64
65  /* https://tools.ietf.org/html/rfc2822#section-2.2.1 */
66  /* https://tools.ietf.org/html/rfc2822#section-2.2.3 */
67  while (len < value->len) {
68    if (*(value->val+len) == '\r') {
69      if (value->len - len >= 3
70        &&  *(value->val+len+1) == '\n'
71        && (*(value->val+len+2) == ' '  || *(value->val+len+2) == '\t')) {
72        len += 3;
73        continue;
74      }
75      return FAILURE;
76    }
77    if (*(value->val+len) == '\0') {
78      return FAILURE;
79    }
80    len++;
81  }
82  return SUCCESS;
83}

335static int php_mail_detect_multiple_crlf(const char *hdr) {
336  /* This function detects multiple/malformed multiple newlines. */
337
338  if (!hdr || !strlen(hdr)) {
339    return 0;
340  }
341
342  /* Should not have any newlines at the beginning. */
343  /* RFC 2822 2.2. Header Fields */
344  if (*hdr < 33 || *hdr > 126 || *hdr == ':') {
345    return 1;
346  }
347
348  while(*hdr) {
349    if (*hdr == '\r') {
350      if (*(hdr+1) == '\0' || *(hdr+1) == '\r' || (*(hdr+1) == '\n' && (*(hdr+2) == '\0' || *(hdr+2) == '\n' || *(hdr+2) == '\r'))) {
351        /* Malformed or multiple newlines. */
352        return 1;
353      } else {
354        hdr += 2;
355      }
356    } else if (*hdr == '\n') {
357      if (*(hdr+1) == '\0' || *(hdr+1) == '\r' || *(hdr+1) == '\n') {
358        /* Malformed or multiple newlines. */
359        return 1;
360      } else {
361        hdr += 2;
362      }
363    } else {
364      hdr++;
365    }
366  }
367
368  return 0;
369}

479  sig_handler = (void *)signal(SIGCHLD, SIG_DFL);
480  if (sig_handler == SIG_ERR) {
481    sig_handler = NULL;
482  }
483#endif
484
485#ifdef PHP_WIN32
486  sendmail = popen_ex(sendmail_cmd, "wb", NULL, NULL);
487#else
488  /* Since popen() doesn't indicate if the internal fork() doesn't work
489   * (e.g. the shell can't be executed) we explicitly set it to 0 to be
490   * sure we don't catch any older errno value. */
491  errno = 0;
492  sendmail = popen(sendmail_cmd, "w");
493#endif
494  if (extra_cmd != NULL) {
495    efree (sendmail_cmd);
496  }
497
498  if (sendmail) {
499#ifndef PHP_WIN32
500    if (EACCES == errno) {
501      php_error_docref(NULL, E_WARNING, "Permission denied: unable to execute shell to run mail delivery binary '%s'", sendmail_path);
502      pclose(sendmail);
503#if PHP_SIGCHILD
504      /* Restore handler in case of error on Windows
505         Not sure if this applicable on Win but just in case. */
506      if (sig_handler) {
507        signal(SIGCHLD, sig_handler);
508      }
509#endif
510      MAIL_RET(0);
511    }
512#endif
513    fprintf(sendmail, "To: %s\r\n", to);
514    fprintf(sendmail, "Subject: %s\r\n", subject);
515    if (hdr != NULL) {
516      fprintf(sendmail, "%s\r\n", hdr);
517    }
518    fprintf(sendmail, "\r\n%s\r\n", message);
519    ret = pclose(sendmail);
520
521#if PHP_SIGCHILD
522    if (sig_handler) {
523      signal(SIGCHLD, sig_handler);
524    }
525#endif
526
527#ifdef PHP_WIN32
528    if (ret == -1)
529#else
530#if defined(EX_TEMPFAIL)
531    if ((ret != EX_OK)&&(ret != EX_TEMPFAIL))
532#elif defined(EX_OK)
533    if (ret != EX_OK)
534#else
535    if (ret != 0)
536#endif
537#endif
538    {
539      MAIL_RET(0);
540    } else {
541      MAIL_RET(1);
542    }
543  } else {
544    php_error_docref(NULL, E_WARNING, "Could not execute mail delivery program '%s'", sendmail_path);
545#if PHP_SIGCHILD
546    if (sig_handler) {
547      signal(SIGCHLD, sig_handler);
548    }
549#endif
550    MAIL_RET(0);
551  }
552
553  MAIL_RET(1); /* never reached */
554}
