451      ZEND_ASSERT(p->key != NULL);
452      ADD_INTERNED_STRING(p->key);
453      zend_persist_class_constant_calc(&p->val);
454    } ZEND_HASH_FOREACH_END();
455
456    zend_hash_persist_calc(&ce->properties_info);
457    ZEND_HASH_MAP_FOREACH_BUCKET(&ce->properties_info, p) {
458      zend_property_info *prop = Z_PTR(p->val);
459      ZEND_ASSERT(p->key != NULL);
460      ADD_INTERNED_STRING(p->key);
461      if (prop->ce == ce) {
462        zend_persist_property_info_calc(prop);
463      }
464    } ZEND_HASH_FOREACH_END();
465
466    if (ce->properties_info_table) {
467      ADD_SIZE(sizeof(zend_property_info *) * ce->default_properties_count);
468    }
469
470    if (ce->num_interfaces && (ce->ce_flags & ZEND_ACC_LINKED)) {
471      ADD_SIZE(sizeof(zend_class_entry*) * ce->num_interfaces);
472    }
473
474    if (ce->iterator_funcs_ptr) {
475      ADD_SIZE(sizeof(zend_class_iterator_funcs));
476    }
477    if (ce->arrayaccess_funcs_ptr) {
478      ADD_SIZE(sizeof(zend_class_arrayaccess_funcs));
479    }
480
481    if (ce->ce_flags & ZEND_ACC_CACHED) {
482      return;
483    }
484
485    if (ce->info.user.filename) {
486      ADD_STRING(ce->info.user.filename);
487    }
488
489    if (ZCG(accel_directives).save_comments && ce->info.user.doc_comment) {
490      ADD_STRING(ce->info.user.doc_comment);
491    }
492
493    if (ce->attributes) {
494      zend_persist_attributes_calc(ce->attributes);
495    }
496
497    if (ce->num_interfaces) {
498      uint32_t i;
499
500      if (!(ce->ce_flags & ZEND_ACC_LINKED)) {
501        for (i = 0; i < ce->num_interfaces; i++) {
502          ADD_INTERNED_STRING(ce->interface_names[i].name);
503          ADD_INTERNED_STRING(ce->interface_names[i].lc_name);
504        }
505        ADD_SIZE(sizeof(zend_class_name) * ce->num_interfaces);
506      }
507    }
508
509    if (ce->num_traits) {
510      uint32_t i;
511
512      for (i = 0; i < ce->num_traits; i++) {
513        ADD_INTERNED_STRING(ce->trait_names[i].name);
514        ADD_INTERNED_STRING(ce->trait_names[i].lc_name);
515      }
516      ADD_SIZE(sizeof(zend_class_name) * ce->num_traits);
517
518      if (ce->trait_aliases) {
519        i = 0;
520        while (ce->trait_aliases[i]) {
521          if (ce->trait_aliases[i]->trait_method.method_name) {
522            ADD_INTERNED_STRING(ce->trait_aliases[i]->trait_method.method_name);
523          }
524          if (ce->trait_aliases[i]->trait_method.class_name) {
525            ADD_INTERNED_STRING(ce->trait_aliases[i]->trait_method.class_name);
526          }
527
528          if (ce->trait_aliases[i]->alias) {
529            ADD_INTERNED_STRING(ce->trait_aliases[i]->alias);
530          }
531          ADD_SIZE(sizeof(zend_trait_alias));
532          i++;
533        }
534        ADD_SIZE(sizeof(zend_trait_alias*) * (i + 1));
535      }
536
537      if (ce->trait_precedences) {
538        int j;
539
540        i = 0;
541        while (ce->trait_precedences[i]) {
542          ADD_INTERNED_STRING(ce->trait_precedences[i]->trait_method.method_name);
543          ADD_INTERNED_STRING(ce->trait_precedences[i]->trait_method.class_name);
544
545          for (j = 0; j < ce->trait_precedences[i]->num_excludes; j++) {
546            ADD_INTERNED_STRING(ce->trait_precedences[i]->exclude_class_names[j]);
547          }
548          ADD_SIZE(sizeof(zend_trait_precedence) + (ce->trait_precedences[i]->num_excludes - 1) * sizeof(zend_string*));
549          i++;
550        }
551        ADD_SIZE(sizeof(zend_trait_precedence*) * (i + 1));
552      }
553    }
554  }
555}

557static void zend_accel_persist_class_table_calc(HashTable *class_table)
558{
559  Bucket *p;
560
561  zend_hash_persist_calc(class_table);
562  ZEND_HASH_MAP_FOREACH_BUCKET(class_table, p) {
563    ZEND_ASSERT(p->key != NULL);
564    ADD_INTERNED_STRING(p->key);
565    zend_persist_class_entry_calc(Z_CE(p->val));
566  } ZEND_HASH_FOREACH_END();
567}

569void zend_persist_warnings_calc(uint32_t num_warnings, zend_error_info **warnings) {
570  ADD_SIZE(num_warnings * sizeof(zend_error_info *));
571  for (uint32_t i = 0; i < num_warnings; i++) {
572    ADD_SIZE(sizeof(zend_error_info));
573    ADD_STRING(warnings[i]->filename);
574    ADD_STRING(warnings[i]->message);
575  }
576}

579  uint32_t num_early_bindings, zend_early_binding *early_bindings)

584    ADD_INTERNED_STRING(early_binding->lcname);

587  }

590uint32_t zend_accel_script_persist_calc(zend_persistent_script *new_persistent_script, int for_shm)
591{
592  Bucket *p;
593
594  new_persistent_script->mem = NULL;
595  new_persistent_script->size = 0;
596  new_persistent_script->corrupted = false;
597  ZCG(current_persistent_script) = new_persistent_script;
598
599  if (!for_shm) {
600    /* script is not going to be saved in SHM */
601    new_persistent_script->corrupted = true;
602  }
603
604  ADD_SIZE(sizeof(zend_persistent_script));
605  ADD_INTERNED_STRING(new_persistent_script->script.filename);
606
607#if defined(__AVX__) || defined(__SSE2__)
608  /* Align size to 64-byte boundary */
609  new_persistent_script->size = (new_persistent_script->size + 63) & ~63;
610#endif
611
612  if (new_persistent_script->script.class_table.nNumUsed != new_persistent_script->script.class_table.nNumOfElements) {
613    zend_hash_rehash(&new_persistent_script->script.class_table);
614  }
615  zend_accel_persist_class_table_calc(&new_persistent_script->script.class_table);
616  if (new_persistent_script->script.function_table.nNumUsed != new_persistent_script->script.function_table.nNumOfElements) {
617    zend_hash_rehash(&new_persistent_script->script.function_table);
618  }
619  zend_hash_persist_calc(&new_persistent_script->script.function_table);
620  ZEND_HASH_MAP_FOREACH_BUCKET(&new_persistent_script->script.function_table, p) {
621    ZEND_ASSERT(p->key != NULL);
622    ADD_INTERNED_STRING(p->key);
623    zend_persist_op_array_calc(&p->val);
624  } ZEND_HASH_FOREACH_END();
625  zend_persist_op_array_calc_ex(&new_persistent_script->script.main_op_array);
626  zend_persist_warnings_calc(
627    new_persistent_script->num_warnings, new_persistent_script->warnings);
628  zend_persist_early_bindings_calc(
629    new_persistent_script->num_early_bindings, new_persistent_script->early_bindings);
630
631  new_persistent_script->corrupted = false;
632
633  ZCG(current_persistent_script) = NULL;
634
635  return new_persistent_script->size;
636}
