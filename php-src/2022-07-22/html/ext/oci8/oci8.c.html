186/* {{{  php_oci_init_global_handles()
187 *
188 * Initialize global handles only when they are needed
189 */
190static void php_oci_init_global_handles(void)
191{
192  sword errstatus;
193  sb4   ora_error_code = 0;
194  text  tmp_buf[PHP_OCI_ERRBUF_LEN];  /* Use traditional smaller size: non-PL/SQL errors should fit and it keeps the stack smaller */
195
196  errstatus = OCIEnvNlsCreate(&OCI_G(env), PHP_OCI_INIT_MODE, 0, NULL, NULL, NULL, 0, NULL, 0, 0);
197
198  if (errstatus == OCI_ERROR) {
199#ifdef HAVE_OCI_INSTANT_CLIENT
200    php_error_docref(NULL, E_WARNING, "OCIEnvNlsCreate() failed. There is something wrong with your system - please check that " PHP_OCI8_LIB_PATH_MSG " includes the directory with Oracle Instant Client libraries");
201#else
202    php_error_docref(NULL, E_WARNING, "OCIEnvNlsCreate() failed. There is something wrong with your system - please check that ORACLE_HOME and " PHP_OCI8_LIB_PATH_MSG " are set and point to the right directories");
203#endif
204    if (OCI_G(env)
205      && OCIErrorGet(OCI_G(env), (ub4)1, NULL, &ora_error_code, tmp_buf, (ub4)PHP_OCI_ERRBUF_LEN, (ub4)OCI_HTYPE_ENV) == OCI_SUCCESS
206      && *tmp_buf) {
207      php_error_docref(NULL, E_WARNING, "%s", tmp_buf);
208    }
209
210    OCI_G(env) = NULL;
211    OCI_G(err) = NULL;
212    return;
213  }
214
215  errstatus = OCIHandleAlloc (OCI_G(env), (dvoid **)&OCI_G(err), OCI_HTYPE_ERROR, 0, NULL);
216
217  if (errstatus != OCI_SUCCESS) {
218    OCIErrorGet(OCI_G(env), (ub4)1, NULL, &ora_error_code, tmp_buf, (ub4)PHP_OCI_ERRBUF_LEN, (ub4)OCI_HTYPE_ERROR);
219
220    if (ora_error_code) {
221      int tmp_buf_len = (int) strlen((char *)tmp_buf);
222
223      if (tmp_buf_len > 0 && tmp_buf[tmp_buf_len - 1] == '\n') {
224        tmp_buf[tmp_buf_len - 1] = '\0';
225      }
226
227      if (errstatus == OCI_SUCCESS_WITH_INFO) {
228        php_error_docref(NULL, E_WARNING, "Initialization error: OCI_SUCCESS_WITH_INFO: %s", tmp_buf);
229      } else {
230        php_error_docref(NULL, E_WARNING, "Initialization error: OCI_ERROR: %s", tmp_buf);
231
232        OCIHandleFree((dvoid *) OCI_G(env), OCI_HTYPE_ENV);
233
234        OCI_G(env) = NULL;
235        OCI_G(err) = NULL;
236      }
237    }
238  }
239}

252/* {{{ PHP_GSHUTDOWN_FUNCTION
253 *
254 * Called for thread shutdown in ZTS, after module shutdown for non-ZTS
255 * Free global handles (if they were initialized before)
256 */
257static PHP_GSHUTDOWN_FUNCTION(oci)
258{
259  if (oci_globals->err) {
260    oci_globals->in_call = 1;
261    OCIHandleFree((dvoid *) oci_globals->err, OCI_HTYPE_ERROR);
262    oci_globals->in_call = 0;
263    oci_globals->err = NULL;
264  }
265
266  if (oci_globals->env) {
267    oci_globals->in_call = 1;
268    OCIHandleFree((dvoid *) oci_globals->env, OCI_HTYPE_ENV);
269    oci_globals->in_call = 0;
270    oci_globals->env = NULL;
271  }
272}

2609  /* Cache the pointer so we do not have to do OCIContextGetValue repeatedly */

2612  return OCI_SUCCESS;
