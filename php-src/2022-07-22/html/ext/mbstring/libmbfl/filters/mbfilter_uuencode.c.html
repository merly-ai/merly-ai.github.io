75int mbfl_filt_conv_uudec(int c, mbfl_convert_filter *filter)
76{
77  int n;
78
79  switch(filter->status)  {
80    case uudec_state_ground:
81      /* looking for "begin 0666 filename\n" line */
82      if (filter->cache == 0 && c == 'b')
83      {
84        filter->status = uudec_state_inbegin;
85        filter->cache = 1; /* move to 'e' */
86      }
87      else if (c == '\n')
88        filter->cache = 0;
89      else
90        filter->cache++;
91      break;
92    case uudec_state_inbegin:
93      if (uuenc_begin_text[filter->cache++] != c)  {
94        /* doesn't match pattern */
95        filter->status = uudec_state_ground;
96        break;
97      }
98      if (filter->cache == 5)
99      {
100        /* that's good enough - wait for a newline */
101        filter->status = uudec_state_until_newline;
102        filter->cache = 0;
103      }
104      break;
105    case uudec_state_until_newline:
106      if (c == '\n')
107        filter->status = uudec_state_size;
108      break;
109    case uudec_state_size:
110      /* get "size" byte */
111      n = UUDEC(c);
112      filter->cache = n << 24;
113      filter->status = uudec_state_a;
114      break;
115    case uudec_state_a:
116      /* get "a" byte */
117      n = UUDEC(c);
118      filter->cache |= (n << 16);
119      filter->status = uudec_state_b;
120      break;
121    case uudec_state_b:
122      /* get "b" byte */
123      n = UUDEC(c);
124      filter->cache |= (n << 8);
125      filter->status = uudec_state_c;
126      break;
127    case uudec_state_c:
128      /* get "c" byte */
129      n = UUDEC(c);
130      filter->cache |= n;
131      filter->status = uudec_state_d;
132      break;
133    case uudec_state_d:
134      /* get "d" byte */
135      {
136        int A, B, C, D = UUDEC(c);
137        A = (filter->cache >> 16) & 0xff;
138        B = (filter->cache >> 8) & 0xff;
139        C = (filter->cache) & 0xff;
140        n = (filter->cache >> 24) & 0xff;
141        if (n-- > 0)
142          CK((*filter->output_function)( (A << 2) | (B >> 4), filter->data));
143        if (n-- > 0)
144          CK((*filter->output_function)( (B << 4) | (C >> 2), filter->data));
145        if (n-- > 0)
146          CK((*filter->output_function)( (C << 6) | D, filter->data));
147        if (n < 0)
148          n = 0;
149        filter->cache = n << 24;
150
151        if (n == 0)
152          filter->status = uudec_state_skip_newline;  /* skip next byte (newline) */
153        else
154          filter->status = uudec_state_a; /* go back to fetch "A" byte */
155      }
156      break;
157    case uudec_state_skip_newline:
158      /* skip newline */
159      filter->status = uudec_state_size;
160  }
161  return 0;
162}

246static void mb_wchar_to_uuencode(uint32_t *in, size_t len, mb_convert_buf *buf, bool end)
247{
248  unsigned char *out, *limit;
249  MB_CONVERT_BUF_LOAD(buf, out, limit);
250  /* Every 3 bytes of input gets encoded as 4 bytes of output
251   * Additionally, we have a 'length' byte and a newline for each line of output
252   * (Maximum 45 input bytes can be encoded on a single output line)
253   * Make space for two more bytes in case we start close to where a line must end,
254   * and another two if there are cached bits remaining from the previous call */
255  MB_CONVERT_BUF_ENSURE(buf, out, limit, ((len + 2) * 4 / 3) + (((len + 44) / 45) * 2) + (buf->state ? 0 : sizeof("begin 0644 filename\n")) + 4);
256
257  unsigned int bytes_encoded = (buf->state >> 1) & 0x7F;
258  /* UUEncode naturally wants to process input bytes in groups of 3, but
259   * our buffer size may not be a multiple of 3
260   * So there may be data from the previous call which we need to flush out */
261  unsigned int n_cached_bits = (buf->state >> 8) & 0xFF;
262  unsigned int cached_bits = buf->state >> 16;
263
264  if (!buf->state) {
265    for (char *s = "begin 0644 filename\n"; *s; s++) {
266      out = mb_convert_buf_add(out, *s);
267    }
268    out = mb_convert_buf_add(out, MIN(len, 45) + 32);
269    buf->state |= 1;
270  } else if (!len && end && !bytes_encoded && !n_cached_bits) {
271    /* Corner case: under EXTREMELY rare circumstances, it's possible that the
272     * final call to this conversion function will happen with an empty input
273     * buffer, leaving an unwanted trailing len byte in the output buffer. */
274    buf->out--;
275    return;
276  } else {
277    /* UUEncode starts each line with a byte which indicates how many bytes
278     * are encoded on the line
279     * This can create a problem, since we receive the incoming data one buffer
280     * at a time, and there is no requirement that the buffers should be aligned
281     * with line boundaries
282     * So if a previous line was cut off, we need to go back and fix up
283     * the preceding len byte */
284    unsigned char *len_byte = out - (bytes_encoded * 4 / 3) - 1;
285    if (n_cached_bits) {
286      len_byte -= (n_cached_bits == 2) ? 1 : 2;
287    }
288    *len_byte = MIN(bytes_encoded + len + (n_cached_bits ? (n_cached_bits == 2 ? 1 : 2) : 0), 45) + 32;
289
290    if (n_cached_bits) {
291      /* Flush out bits which remained from previous call */
292      if (n_cached_bits == 2) {
293        uint32_t w = cached_bits;
294        uint32_t w2 = 0, w3 = 0;
295        if (len) {
296          w2 = *in++;
297          len--;
298        }

305        uint32_t w2 = cached_bits;
306        uint32_t w3 = 0;
307        if (len) {
308          w3 = *in++;
309          len--;

311        out = mb_convert_buf_add2(out, uuencode_six_bits((w2 << 2) + ((w3 >> 6) & 0x3)), uuencode_six_bits(w3 & 0x3F));

313      n_cached_bits = cached_bits = 0;

339        cached_bits = w2 & 0xF;

360  }
361
