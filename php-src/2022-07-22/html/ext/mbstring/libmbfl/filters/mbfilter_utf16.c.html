143int mbfl_filt_conv_utf16_wchar(int c, mbfl_convert_filter *filter)
144{
145  /* Start with the assumption that the string is big-endian;
146   * If we find a little-endian BOM, then we will change that assumption */
147  if (filter->status == 0) {
148    filter->cache = c & 0xFF;
149    filter->status = 1;
150  } else {
151    int n = (filter->cache << 8) | (c & 0xFF);
152    if (n == 0xFFFE) {
153      /* Switch to little-endian mode */
154      filter->filter_function = mbfl_filt_conv_utf16le_wchar;
155      filter->cache = filter->status = 0;
156    } else {
157      filter->filter_function = mbfl_filt_conv_utf16be_wchar;
158      if (n >= 0xD800 && n <= 0xDBFF) {
159        filter->cache = n & 0x3FF; /* Pick out 10 data bits */
160        filter->status = 2;
161        return 0;
162      } else if (n >= 0xDC00 && n <= 0xDFFF) {
163        /* This is wrong; second part of surrogate pair has come first */
164        CK((*filter->output_function)(MBFL_BAD_INPUT, filter->data));
165      } else if (n != 0xFEFF) {
166        CK((*filter->output_function)(n, filter->data));
167      }
168      filter->cache = filter->status = 0;
169    }
170  }
171
172  return 0;
173}

175int mbfl_filt_conv_utf16be_wchar(int c, mbfl_convert_filter *filter)
176{
177  int n;
178
179  switch (filter->status) {
180  case 0: /* First byte */
181    filter->cache = c & 0xFF;
182    filter->status = 1;
183    break;
184
185  case 1: /* Second byte */
186    n = (filter->cache << 8) | (c & 0xFF);
187    if (n >= 0xD800 && n <= 0xDBFF) {
188      filter->cache = n & 0x3FF; /* Pick out 10 data bits */
189      filter->status = 2;
190    } else if (n >= 0xDC00 && n <= 0xDFFF) {
191      /* This is wrong; second part of surrogate pair has come first */
192      CK((*filter->output_function)(MBFL_BAD_INPUT, filter->data));
193      filter->status = 0;
194    } else {
195      CK((*filter->output_function)(n, filter->data));
196      filter->status = 0;
197    }
198    break;
199
200  case 2: /* Second part of surrogate, first byte */
201    filter->cache = (filter->cache << 8) | (c & 0xFF);
202    filter->status = 3;
203    break;
204
205  case 3: /* Second part of surrogate, second byte */
206    n = ((filter->cache & 0xFF) << 8) | (c & 0xFF);
207    if (n >= 0xD800 && n <= 0xDBFF) {
208      /* Wrong; that's the first half of a surrogate pair, not the second */
209      CK((*filter->output_function)(MBFL_BAD_INPUT, filter->data));
210      filter->cache = n & 0x3FF;
211      filter->status = 2;
212    } else if (n >= 0xDC00 && n <= 0xDFFF) {
213      n = ((filter->cache & 0x3FF00) << 2) + (n & 0x3FF) + 0x10000;
214      CK((*filter->output_function)(n, filter->data));
215      filter->status = 0;
216    } else {
217      CK((*filter->output_function)(MBFL_BAD_INPUT, filter->data));
218      CK((*filter->output_function)(n, filter->data));
219      filter->status = 0;
220    }
221  }
222
223  return 0;
224}

229

233  } else if (c >= MBFL_WCSPLANE_SUPMIN && c < MBFL_WCSPLANE_SUPMAX) {

247int mbfl_filt_conv_utf16le_wchar(int c, mbfl_convert_filter *filter)
248{
249  int n;
250
251  switch (filter->status) {
252  case 0:
253    filter->cache = c & 0xff;
254    filter->status = 1;
255    break;
256
257  case 1:
258    if ((c & 0xfc) == 0xd8) {
259      /* Looks like we have a surrogate pair here */
260      filter->cache += ((c & 0x3) << 8);
261      filter->status = 2;
262    } else if ((c & 0xfc) == 0xdc) {
263      /* This is wrong; the second part of the surrogate pair has come first */
264      CK((*filter->output_function)(MBFL_BAD_INPUT, filter->data));
265      filter->status = 0;
266    } else {
267      CK((*filter->output_function)(filter->cache + ((c & 0xff) << 8), filter->data));
268      filter->status = 0;
269    }
270    break;
271
272  case 2:
273    filter->cache = (filter->cache << 10) + (c & 0xff);
274    filter->status = 3;
275    break;
276
277  case 3:
278    n = (filter->cache & 0xFF) | ((c & 0xFF) << 8);
279    if (n >= 0xD800 && n <= 0xDBFF) {
280      /* We previously saw the first part of a surrogate pair and were
281       * expecting the second part; this is another first part */
282      CK((*filter->output_function)(MBFL_BAD_INPUT, filter->data));
283      filter->cache = n & 0x3FF;
284      filter->status = 2;
285    } else if (n >= 0xDC00 && n <= 0xDFFF) {
286      n = filter->cache + ((c & 0x3) << 8) + 0x10000;
287      CK((*filter->output_function)(n, filter->data));
288      filter->status = 0;
289    } else {
290      /* The first part of a surrogate pair was followed by some other codepoint
291       * which is not part of a surrogate pair at all */
292      CK((*filter->output_function)(MBFL_BAD_INPUT, filter->data));
293      CK((*filter->output_function)(n, filter->data));
294      filter->status = 0;
295    }
296    break;
297  }
298
299  return 0;
300}

340static size_t mb_utf16_to_wchar(unsigned char **in, size_t *in_len, uint32_t *buf, size_t bufsize, unsigned int *state)
341{
342  if (*state == DETECTED_BE) {
343    return mb_utf16be_to_wchar(in, in_len, buf, bufsize, NULL);
344  } else if (*state == DETECTED_LE) {
345    return mb_utf16le_to_wchar(in, in_len, buf, bufsize, NULL);
346  } else if (*in_len >= 2) {
347    unsigned char *p = *in;
348    unsigned char c1 = *p++;
349    unsigned char c2 = *p++;
350    uint16_t n = (c1 << 8) | c2;
351
352    if (n == 0xFFFE) {
353      /* Little-endian BOM */
354      *in = p;
355      *in_len -= 2;
356      *state = DETECTED_LE;
357      return mb_utf16le_to_wchar(in, in_len, buf, bufsize, NULL);
358    } if (n == 0xFEFF) {
359      /* Big-endian BOM; don't send to output */
360      *in = p;
361      *in_len -= 2;
362    }
363  }
364
365  *state = DETECTED_BE;
366  return mb_utf16be_to_wchar(in, in_len, buf, bufsize, NULL);
367}

369static size_t mb_utf16be_to_wchar(unsigned char **in, size_t *in_len, uint32_t *buf, size_t bufsize, unsigned int *state)
370{
371  /* We only want to read 16-bit words out of `str`; any trailing byte will be handled at the end */
372  unsigned char *p = *in, *e = p + (*in_len & ~1);
373  /* Set `limit` to one less than the actual amount of space in the buffer; this is because
374   * on some iterations of the below loop, we might produce two output words */
375  uint32_t *out = buf, *limit = buf + bufsize - 1;
376
377  while (p < e && out < limit) {
378    unsigned char c1 = *p++;
379    unsigned char c2 = *p++;
380    uint16_t n = (c1 << 8) | c2;
381
382    if (n >= 0xD800 && n <= 0xDBFF) {
383      /* Handle surrogate */
384      if (p < e) {
385        unsigned char c3 = *p++;
386        unsigned char c4 = *p++;
387        uint16_t n2 = (c3 << 8) | c4;
388
389        if (n2 >= 0xD800 && n2 <= 0xDBFF) {
390          /* Wrong; that's the first half of a surrogate pair, when we were expecting the second */
391          *out++ = MBFL_BAD_INPUT;
392          p -= 2;
393        } else if (n2 >= 0xDC00 && n2 <= 0xDFFF) {
394          *out++ = (((n & 0x3FF) << 10) | (n2 & 0x3FF)) + 0x10000;
395        } else {
396          /* The first half of a surrogate pair was followed by a 'normal' codepoint */
397          *out++ = MBFL_BAD_INPUT;
398          *out++ = n2;
399        }
400      } else {
401        *out++ = MBFL_BAD_INPUT;
402      }
403    } else if (n >= 0xDC00 && n <= 0xDFFF) {
404      /* This is wrong; second part of surrogate pair has come first */
405      *out++ = MBFL_BAD_INPUT;
406    } else {
407      *out++ = n;
408    }
409  }
410
411  if (
 Recommendations (Experimental)  R1: (!(p <= e && e <= {const}))
R2: (!(p == e && in_len == 0x1))
R3: (p = ((e + in_len) / 0x1))
R4: (p = (e + in_len >> 0x1))
R5: (p = ((e + in_len) >> 0x1))
 Score  Issue:  1 unfamiliar pattern(s) detected 
Cost: 16.25, Total Score: 1016.25
Class: complex_2
p == e && (*in_len & 0x1) && out < limit) {
412    /* There is an extra trailing byte (which shouldn't be there) */
413    *out++ = MBFL_BAD_INPUT;
414    p++;
415  }
416
417  *in_len -= (p - *in);
418  *in = p;
419  return out - buf;
420}

422static void mb_wchar_to_utf16be(uint32_t *in, size_t len, mb_convert_buf *buf, bool end)
423{
424  unsigned char *out, *limit;
425  MB_CONVERT_BUF_LOAD(buf, out, limit);
426  MB_CONVERT_BUF_ENSURE(buf, out, limit, len * 2);
427
428  while (len--) {
429    uint32_t w = *in++;
430
431    if (w < MBFL_WCSPLANE_UCS2MAX) {
432      out = mb_convert_buf_add2(out, (w >> 8) & 0xFF, w & 0xFF);
433    } else if (w < MBFL_WCSPLANE_UTF32MAX) {
434      uint16_t n1 = ((w >> 10) - 0x40) | 0xD800;
435      uint16_t n2 = (w & 0x3FF) | 0xDC00;
436      MB_CONVERT_BUF_ENSURE(buf, out, limit, (len * 2) + 4);
437      out = mb_convert_buf_add4(out, (n1 >> 8) & 0xFF, n1 & 0xFF, (n2 >> 8) & 0xFF, n2 & 0xFF);
438    } else {
439      MB_CONVERT_ERROR(buf, out, limit, w, mb_wchar_to_utf16be);
440      MB_CONVERT_BUF_ENSURE(buf, out, limit, len * 2);
441    }
442  }
443
444  MB_CONVERT_BUF_STORE(buf, out, limit);

447static size_t mb_utf16le_to_wchar(unsigned char **in, size_t *in_len, uint32_t *buf, size_t bufsize, unsigned int *state)
448{
449  /* We only want to read 16-bit words out of `str`; any trailing byte will be handled at the end */
450  unsigned char *p = *in, *e = p + (*in_len & ~1);
451  /* Set `limit` to one less than the actual amount of space in the buffer; this is because
452   * on some iterations of the below loop, we might produce two output words */
453  uint32_t *out = buf, *limit = buf + bufsize - 1;
454
455  while (p < e && out < limit) {
456    unsigned char c1 = *p++;
457    unsigned char c2 = *p++;
458    uint16_t n = (c2 << 8) | c1;
459
460    if (n >= 0xD800 && n <= 0xDBFF) {
461      /* Handle surrogate */
462      if (p < e) {
463        unsigned char c3 = *p++;
464        unsigned char c4 = *p++;
465        uint16_t n2 = (c4 << 8) | c3;

470          p -= 2;
471        } else if (n2 >= 0xDC00 && n2 <= 0xDFFF) {
472          *out++ = (((n & 0x3FF) << 10) | (n2 & 0x3FF)) + 0x10000;
473        } else {
474          /* The first half of a surrogate pair was followed by a 'normal' codepoint */
475          *out++ = MBFL_BAD_INPUT;
476          *out++ = n2;
477        }
478      } else {
479        *out++ = MBFL_BAD_INPUT;
480      }
481    } else if (n >= 0xDC00 && n <= 0xDFFF) {
482      /* This is wrong; second part of surrogate pair has come first */
483      *out++ = MBFL_BAD_INPUT;
484    } else {
485      *out++ = n;
486    }
487  }
488
489  if (

R2: (!(p == e && in_len == 0x1))
R3: (p = ((e + in_len) / 0x1))

490    /* There is an extra trailing byte (which shouldn't be there) */
491    *out++ = MBFL_BAD_INPUT;
492    p++;
493  }

495  *in_len -= (p - *in);
496  *in = p;
497  return out - buf;

500static void mb_wchar_to_utf16le(uint32_t *in, size_t len, mb_convert_buf *buf, bool end)
501{
502  unsigned char *out, *limit;
503  MB_CONVERT_BUF_LOAD(buf, out, limit);
504  MB_CONVERT_BUF_ENSURE(buf, out, limit, len * 2);
505
506  while (len--) {
507    uint32_t w = *in++;
508
509    if (w < MBFL_WCSPLANE_UCS2MAX) {
510      out = mb_convert_buf_add2(out, w & 0xFF, (w >> 8) & 0xFF);
511    } else if (w < MBFL_WCSPLANE_UTF32MAX) {
512      uint16_t n1 = ((w >> 10) - 0x40) | 0xD800;
513      uint16_t n2 = (w & 0x3FF) | 0xDC00;
514      MB_CONVERT_BUF_ENSURE(buf, out, limit, (len * 2) + 4);
515      out = mb_convert_buf_add4(out, n1 & 0xFF, (n1 >> 8) & 0xFF, n2 & 0xFF, (n2 >> 8) & 0xFF);
516    } else {
517      MB_CONVERT_ERROR(buf, out, limit, w, mb_wchar_to_utf16le);
518      MB_CONVERT_BUF_ENSURE(buf, out, limit, len * 2);
519    }
520  }
521
522  MB_CONVERT_BUF_STORE(buf, out, limit);
523}
