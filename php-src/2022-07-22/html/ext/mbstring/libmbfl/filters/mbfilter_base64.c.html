89int mbfl_filt_conv_base64enc(int c, mbfl_convert_filter *filter)
90{
91  int n;
92
93  n = (filter->status & 0xff);
94  if (n == 0) {
95    filter->status++;
96    filter->cache = (c & 0xff) << 16;
97  } else if (n == 1) {
98    filter->status++;
99    filter->cache |= (c & 0xff) << 8;
100  } else {
101    filter->status &= ~0xff;
102    if ((filter->status & MBFL_BASE64_STS_MIME_HEADER) == 0) {
103      n = (filter->status & 0xff00) >> 8;
104      if (n > 72) {
105        CK((*filter->output_function)(0x0d, filter->data));    /* CR */
106        CK((*filter->output_function)(0x0a, filter->data));    /* LF */
107        filter->status &= ~0xff00;
108      }
109      filter->status += 0x400;
110    }
111    n = filter->cache | (c & 0xff);
112    CK((*filter->output_function)(mbfl_base64_table[(n >> 18) & 0x3f], filter->data));
113    CK((*filter->output_function)(mbfl_base64_table[(n >> 12) & 0x3f], filter->data));
114    CK((*filter->output_function)(mbfl_base64_table[(n >> 6) & 0x3f], filter->data));
115    CK((*filter->output_function)(mbfl_base64_table[n & 0x3f], filter->data));
116  }
117
118  return 0;
119}

121int mbfl_filt_conv_base64enc_flush(mbfl_convert_filter *filter)
122{
123  int status, cache, len;
124
125  status = filter->status & 0xff;
126  cache = filter->cache;
127  len = (filter->status & 0xff00) >> 8;
128  filter->status &= ~0xffff;
129  filter->cache = 0;
130  /* flush fragments */
131  if (status >= 1) {
132    if ((filter->status & MBFL_BASE64_STS_MIME_HEADER) == 0) {
133      if (len > 72){
134        CK((*filter->output_function)(0x0d, filter->data));    /* CR */
135        CK((*filter->output_function)(0x0a, filter->data));    /* LF */
136      }
137    }
138    CK((*filter->output_function)(mbfl_base64_table[(cache >> 18) & 0x3f], filter->data));
139    CK((*filter->output_function)(mbfl_base64_table[(cache >> 12) & 0x3f], filter->data));
140    if (status == 1) {
141      CK((*filter->output_function)(0x3d, filter->data));    /* '=' */
142      CK((*filter->output_function)(0x3d, filter->data));    /* '=' */
143    } else {
144      CK((*filter->output_function)(mbfl_base64_table[(cache >> 6) & 0x3f], filter->data));
145      CK((*filter->output_function)(0x3d, filter->data));    /* '=' */
146    }
147  }
148
149  if (filter->flush_function) {
150    (*filter->flush_function)(filter->data);
151  }
152
153  return 0;

159int mbfl_filt_conv_base64dec(int c, mbfl_convert_filter *filter)
160{
161  int n;
162
163  if (c == 0x0d || c == 0x0a || c == 0x20 || c == 0x09 || c == 0x3d) {  /* CR or LF or SPACE or HTAB or '=' */
164    return 0;
165  }
166
167  n = 0;
168  if (c >= 0x41 && c <= 0x5a) {    /* A - Z */
169    n = c - 65;
170  } else if (c >= 0x61 && c <= 0x7a) {  /* a - z */
171    n = c - 71;
172  } else if (c >= 0x30 && c <= 0x39) {  /* 0 - 9 */
173    n = c + 4;
174  } else if (c == 0x2b) {      /* '+' */
175    n = 62;
176  } else if (c == 0x2f) {      /* '/' */
177    n = 63;
178  } else {
179    CK((*filter->output_function)(MBFL_BAD_INPUT, filter->data));
180    return 0;
181  }
182  n &= 0x3f;
183
184  switch (filter->status) {
185  case 0:
186    filter->status = 1;
187    filter->cache = n << 18;
188    break;
189  case 1:
190    filter->status = 2;
191    filter->cache |= n << 12;
192    break;
193  case 2:
194    filter->status = 3;
195    filter->cache |= n << 6;
196    break;
197  default:
198    filter->status = 0;
199    n |= filter->cache;
200    CK((*filter->output_function)((n >> 16) & 0xff, filter->data));
201    CK((*filter->output_function)((n >> 8) & 0xff, filter->data));
202    CK((*filter->output_function)(n & 0xff, filter->data));
203    break;
204  }
205
206  return 0;
207}

232static int decode_base64(char c)
233{
234  if (c >= 'A' && c <= 'Z') {
235    return c - 'A';
236  } else if (c >= 'a' && c <= 'z') {  /* a - z */
237    return c - 'a' + 26;
238  } else if (c >= '0' && c <= '9') {  /* 0 - 9 */
239    return c - '0' + 52;
240  } else if (c == '+') {
241    return 62;
242  } else if (c == '/') {
243    return 63;
244  }
245  return -1;
246}

299static void mb_wchar_to_base64(uint32_t *in, size_t len, mb_convert_buf *buf, bool end)
300{
301  unsigned int bits = (buf->state & 0x3) * 8;
302  unsigned int chars_output = ((buf->state >> 2) & 0x3F) * 4;
303  unsigned int cache = buf->state >> 8;
304
305  unsigned char *out, *limit;
306  MB_CONVERT_BUF_LOAD(buf, out, limit);
307  /* Every 3 bytes of input converts to 4 bytes of output... but if the number of input
308   * bytes is not a multiple of 3, we still pad the output out to a multiple of 4
309   * That's `(len + 2) * 4 / 3`, to calculate the amount of space needed in the output buffer
310   *
311   * But also, we add a CR+LF line ending (2 bytes) for every 76 bytes of output
312   * That means we must multiply the above number by 78/76
313   * Use `zend_safe_address_guarded` to check that the multiplication doesn't overflow
314   *
315   * And since we may enter this function multiple times when converting a large string, and
316   * we might already be close to where a CR+LF needs to be emitted, make space for an extra
317   * CR+LF pair in the output buffer */
318  MB_CONVERT_BUF_ENSURE(buf, out, limit, (zend_safe_address_guarded(len + (bits / 8), 26, 52) / 19) + 2);
319
320  while (len--) {
321    uint32_t w = *in++;
322    cache = (cache << 8) | (w & 0xFF);
323    bits += 8;
324    if (bits == 24) {
325      if (chars_output > 72) {
326        out = mb_convert_buf_add2(out, '\r', '\n');
327        chars_output = 0;
328      }
329      out = mb_convert_buf_add4(out,
330        mbfl_base64_table[(cache >> 18) & 0x3F],
331        mbfl_base64_table[(cache >> 12) & 0x3F],
332        mbfl_base64_table[(cache >> 6) & 0x3F],
333        mbfl_base64_table[cache & 0x3F]);
334      chars_output += 4;
335      bits = cache = 0;
336    }
337  }
338
339  if (end && bits) {
340    if (bits == 8) {
341      out = mb_convert_buf_add4(out, mbfl_base64_table[(cache >> 2) & 0x3F], mbfl_base64_table[(cache & 0x3) << 4], '=', '=');
342    } else {
343      out = mb_convert_buf_add4(out, mbfl_base64_table[(cache >> 10) & 0x3F], mbfl_base64_table[(cache >> 4) & 0x3F], mbfl_base64_table[(cache & 0xF) << 2], '=');
344    }
345  } else {
346    buf->state = (cache << 8) | (((chars_output / 4) & 0x3F) << 2) | ((bits / 8) & 0x3);
347  }
348
349  MB_CONVERT_BUF_STORE(buf, out, limit);
350}
