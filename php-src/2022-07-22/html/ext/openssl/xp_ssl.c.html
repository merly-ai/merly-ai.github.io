203static int php_openssl_handle_ssl_error(php_stream *stream, int nr_bytes, bool is_init) /* {{{ */
204{
205  php_openssl_netstream_data_t *sslsock = (php_openssl_netstream_data_t*)stream->abstract;
206  int err = SSL_get_error(sslsock->ssl_handle, nr_bytes);
207  char esbuf[512];
208  smart_str ebuf = {0};
209  unsigned long ecode;
210  int retry = 1;
211
212  switch(err) {
213    case SSL_ERROR_ZERO_RETURN:
214      /* SSL terminated (but socket may still be active) */
215      retry = 0;
216      break;
217    case SSL_ERROR_WANT_READ:
218    case SSL_ERROR_WANT_WRITE:
219      /* re-negotiation, or perhaps the SSL layer needs more
220       * packets: retry in next iteration */
221      errno = EAGAIN;
222      retry = is_init ? 1 : sslsock->s.is_blocked;
223      break;
224    case SSL_ERROR_SYSCALL:
225      if (ERR_peek_error() == 0) {
226        if (nr_bytes == 0) {
227          if (!php_openssl_is_http_stream_talking_to_iis(stream) && ERR_get_error() != 0) {
228            php_error_docref(NULL, E_WARNING, "SSL: fatal protocol error");
229          }
230          SSL_set_shutdown(sslsock->ssl_handle, SSL_SENT_SHUTDOWN|SSL_RECEIVED_SHUTDOWN);
231          stream->eof = 1;
232          retry = 0;
233        } else {
234          char *estr = php_socket_strerror(php_socket_errno(), NULL, 0);
235
236          php_error_docref(NULL, E_WARNING,
237              "SSL: %s", estr);
238
239          efree(estr);
240          retry = 0;
241        }
242        break;
243      }
244
245      ZEND_FALLTHROUGH;
246    default:
247      /* some other error */
248      ecode = ERR_get_error();
249
250      switch (ERR_GET_REASON(ecode)) {
251        case SSL_R_NO_SHARED_CIPHER:
252          php_error_docref(NULL, E_WARNING,
253              "SSL_R_NO_SHARED_CIPHER: no suitable shared cipher could be used.  "
254              "This could be because the server is missing an SSL certificate "
255              "(local_cert context option)");
256          retry = 0;
257          break;
258
259        default:
260          do {
261            /* NULL is automatically added */
262            ERR_error_string_n(ecode, esbuf, sizeof(esbuf));
263            if (ebuf.s) {
264              smart_str_appendc(&ebuf, '\n');
265            }
266            smart_str_appends(&ebuf, esbuf);
267          } while ((ecode = ERR_get_error()) != 0);
268
269          smart_str_0(&ebuf);
270
271          php_error_docref(NULL, E_WARNING,
272              "SSL operation failed with code %d. %s%s",
273              err,
274              ebuf.s ? "OpenSSL Error messages:\n" : "",
275              ebuf.s ? ZSTR_VAL(ebuf.s) : "");
276          if (ebuf.s) {
277            smart_str_free(&ebuf);
278          }
279      }
280
281      retry = 0;
282      errno = 0;
283  }
284  return retry;
285}

288static int verify_callback(int preverify_ok, X509_STORE_CTX *ctx) /* {{{ */
289{
290  php_stream *stream;
291  SSL *ssl;
292  int err, depth, ret;
293  zval *val;
294  zend_ulong allowed_depth = OPENSSL_DEFAULT_STREAM_VERIFY_DEPTH;
295
296
297  ret = preverify_ok;
298
299  /* determine the status for the current cert */
300  err = X509_STORE_CTX_get_error(ctx);
301  depth = X509_STORE_CTX_get_error_depth(ctx);
302
303  /* conjure the stream & context to use */
304  ssl = X509_STORE_CTX_get_ex_data(ctx, SSL_get_ex_data_X509_STORE_CTX_idx());
305  stream = (php_stream*)SSL_get_ex_data(ssl, php_openssl_get_ssl_stream_data_index());
306
307  /* if allow_self_signed is set, make sure that verification succeeds */
308  if (err == X509_V_ERR_DEPTH_ZERO_SELF_SIGNED_CERT &&
309    GET_VER_OPT("allow_self_signed") &&
310    zend_is_true(val)
311  ) {
312    ret = 1;
313  }
314
315  /* check the depth */
316  GET_VER_OPT_LONG("verify_depth", allowed_depth);
317  if ((zend_ulong)depth > allowed_depth) {
318    ret = 0;
319    X509_STORE_CTX_set_error(ctx, X509_V_ERR_CERT_CHAIN_TOO_LONG);
320  }
321
322  return ret;
323}

340static bool php_openssl_x509_fingerprint_match(X509 *peer, zval *val)
341{
342  if (Z_TYPE_P(val) == IS_STRING) {
343    const char *method = NULL;
344
345    switch (Z_STRLEN_P(val)) {
346      case 32:
347        method = "md5";
348        break;
349
350      case 40:
351        method = "sha1";
352        break;
353    }
354
355    return method && php_openssl_x509_fingerprint_cmp(peer, method, Z_STRVAL_P(val)) == 0;
356  } else if (Z_TYPE_P(val) == IS_ARRAY) {
357    zval *current;
358    zend_string *key;
359
360    if (!zend_hash_num_elements(Z_ARRVAL_P(val))) {
361      php_error_docref(NULL, E_WARNING, "Invalid peer_fingerprint array; [algo => fingerprint] form required");
362      return 0;
363    }
364
365    ZEND_HASH_FOREACH_STR_KEY_VAL(Z_ARRVAL_P(val), key, current) {
366      if (key == NULL || Z_TYPE_P(current) != IS_STRING) {
367        php_error_docref(NULL, E_WARNING, "Invalid peer_fingerprint array; [algo => fingerprint] form required");
368        return 0;
369      }
370      if (php_openssl_x509_fingerprint_cmp(peer, ZSTR_VAL(key), Z_STRVAL_P(current)) != 0) {
371        return 0;
372      }
373    } ZEND_HASH_FOREACH_END();
374
375    return 1;
376  } else {
377    php_error_docref(NULL, E_WARNING,
378      "Invalid peer_fingerprint value; fingerprint string or array of the form [algo => fingerprint] required");
379  }
380
381  return 0;
382}

384static bool php_openssl_matches_wildcard_name(const char *subjectname, const char *certname) /* {{{ */
385{
386  char *wildcard = NULL;
387  ptrdiff_t prefix_len;
388  size_t suffix_len, subject_len;
389
390  if (strcasecmp(subjectname, certname) == 0) {
391    return 1;
392  }
393
394  /* wildcard, if present, must only be present in the left-most component */
395  if (!(wildcard = strchr(certname, '*')) || memchr(certname, '.', wildcard - certname)) {
396    return 0;
397  }
398
399  /* 1) prefix, if not empty, must match subject */
400  prefix_len = wildcard - certname;
401  if (prefix_len && strncasecmp(subjectname, certname, prefix_len) != 0) {
402    return 0;
403  }
404
405  suffix_len = strlen(wildcard + 1);
406  subject_len = strlen(subjectname);
407  if (suffix_len <= subject_len) {
408    /* 2) suffix must match
409     * 3) no . between prefix and suffix
410     **/
411    return strcasecmp(wildcard + 1, subjectname + subject_len - suffix_len) == 0 &&
412      memchr(subjectname + prefix_len, '.', subject_len - suffix_len - prefix_len) == NULL;
413  }
414
415  return 0;
416}

419static bool php_openssl_matches_san_list(X509 *peer, const char *subject_name) /* {{{ */
420{
421  int i, len;
422  unsigned char *cert_name = NULL;
423  char ipbuffer[64];
424
425  GENERAL_NAMES *alt_names = X509_get_ext_d2i(peer, NID_subject_alt_name, 0, 0);
426  int alt_name_count = sk_GENERAL_NAME_num(alt_names);
427
428  for (i = 0; i < alt_name_count; i++) {
429    GENERAL_NAME *san = sk_GENERAL_NAME_value(alt_names, i);
430
431    if (san->type == GEN_DNS) {
432      ASN1_STRING_to_UTF8(&cert_name, san->d.dNSName);
433      if ((size_t)ASN1_STRING_length(san->d.dNSName) != strlen((const char*)cert_name)) {
434        OPENSSL_free(cert_name);
435        /* prevent null-byte poisoning*/
436        continue;
437      }
438
439      /* accommodate valid FQDN entries ending in "." */
440      len = strlen((const char*)cert_name);
441      if (len && strcmp((const char *)&cert_name[len-1], ".") == 0) {
442        cert_name[len-1] = '\0';
443      }
444
445      if (php_openssl_matches_wildcard_name(subject_name, (const char *)cert_name)) {
446        OPENSSL_free(cert_name);
447        sk_GENERAL_NAME_pop_free(alt_names, GENERAL_NAME_free);
448
449        return 1;
450      }
451      OPENSSL_free(cert_name);
452    } else if (san->type == GEN_IPADD) {
453      if (san->d.iPAddress->length == 4) {
454        sprintf(ipbuffer, "%d.%d.%d.%d",
455          san->d.iPAddress->data[0],
456          san->d.iPAddress->data[1],
457          san->d.iPAddress->data[2],
458          san->d.iPAddress->data[3]
459        );
460        if (strcasecmp(subject_name, (const char*)ipbuffer) == 0) {
461          sk_GENERAL_NAME_pop_free(alt_names, GENERAL_NAME_free);
462
463          return 1;
464        }
465      }
466      /* No, we aren't bothering to check IPv6 addresses. Why?
467       * Because IP SAN names are officially deprecated and are
468       * not allowed by CAs starting in 2015. Deal with it.
469       */
470    }
471  }
472
473  sk_GENERAL_NAME_pop_free(alt_names, GENERAL_NAME_free);
474
475  return 0;
476}

480{

491  } else if ((size_t)cert_name_len != strlen(buf)) {

498      cert_name_len, buf, subject_name);
499  }

1059static void php_openssl_limit_handshake_reneg(const SSL *ssl) /* {{{ */
1060{
1061  php_stream *stream;
1062  php_openssl_netstream_data_t *sslsock;
1063  struct timeval now;
1064  zend_long elapsed_time;
1065
1066  stream = php_openssl_get_stream_from_ssl_handle(ssl);
1067  sslsock = (php_openssl_netstream_data_t*)stream->abstract;
1068  gettimeofday(&now, NULL);
1069
1070  /* The initial handshake is never rate-limited */
1071  if (sslsock->reneg->prev_handshake == 0) {
1072    sslsock->reneg->prev_handshake = now.tv_sec;
1073    return;
1074  }
1075
1076  elapsed_time = (now.tv_sec - sslsock->reneg->prev_handshake);
1077  sslsock->reneg->prev_handshake = now.tv_sec;
1078  sslsock->reneg->tokens -= (elapsed_time * (sslsock->reneg->limit / sslsock->reneg->window));
1079
1080  if (sslsock->reneg->tokens < 0) {
1081    sslsock->reneg->tokens = 0;
1082  }
1083  ++sslsock->reneg->tokens;
1084
1085  /* The token level exceeds our allowed limit */
1086  if (sslsock->reneg->tokens > sslsock->reneg->limit) {
1087    zval *val;
1088
1089
1090    sslsock->reneg->should_close = 1;
1091
1092    if (PHP_STREAM_CONTEXT(stream) && (val = php_stream_context_get_option(PHP_STREAM_CONTEXT(stream),
1093        "ssl", "reneg_limit_callback")) != NULL
1094    ) {
1095      zval param, retval;
1096
1097      php_stream_to_zval(stream, &param);
1098
1099      /* Closing the stream inside this callback would segfault! */
1100      stream->flags |= PHP_STREAM_FLAG_NO_FCLOSE;
1101      if (FAILURE == call_user_function(NULL, NULL, val, &retval, 1, &param)) {
1102        php_error(E_WARNING, "SSL: failed invoking reneg limit notification callback");
1103      }
1104      stream->flags ^= PHP_STREAM_FLAG_NO_FCLOSE;
1105
1106      /* If the reneg_limit_callback returned true don't auto-close */
1107      if (Z_TYPE(retval) == IS_TRUE) {
1108        sslsock->reneg->should_close = 0;
1109      }
1110
1111      zval_ptr_dtor(&retval);
1112    } else {
1113      php_error_docref(NULL, E_WARNING,
1114        "SSL: client-initiated handshake rate limit exceeded by peer");
1115    }
1116  }

2421              }
2422            }
2423          } else if (0 == recv(sslsock->s.socket, &buf, sizeof(buf), MSG_PEEK|MSG_DONTWAIT) && php_socket_errno() != EAGAIN) {
2424            alive = 0;
2425          }
2426        }
2427        return alive ? PHP_STREAM_OPTION_RETURN_OK : PHP_STREAM_OPTION_RETURN_ERR;
2428      }
2429
2430    case PHP_STREAM_OPTION_CRYPTO_API:
2431
2432      switch(cparam->op) {
2433
2434        case STREAM_XPORT_CRYPTO_OP_SETUP:

2437          break;
2438        case STREAM_XPORT_CRYPTO_OP_ENABLE:
2439          cparam->outputs.returncode = php_openssl_enable_crypto(stream, sslsock, cparam);
2440          return PHP_STREAM_OPTION_RETURN_OK;
2441          break;
2442        default:
2443          /* fall through */
2444          break;
2445      }
2446
2447      break;
2448
2449    case PHP_STREAM_OPTION_XPORT_API:
2450      switch(xparam->op) {
2451
2452        case STREAM_XPORT_OP_CONNECT:
2453        case STREAM_XPORT_OP_CONNECT_ASYNC:
2454          /* TODO: Async connects need to check the enable_on_connect option when
2455           * we notice that the connect has actually been established */
2456          php_stream_socket_ops.set_option(stream, option, value, ptrparam);
2457
2458          if ((sslsock->enable_on_connect) &&
2459            ((xparam->outputs.returncode == 0) ||
2460            (xparam->op == STREAM_XPORT_OP_CONNECT_ASYNC &&
2461            xparam->outputs.returncode == 1 && xparam->outputs.error_code == EINPROGRESS)))
2462          {
2463            if (php_stream_xport_crypto_setup(stream, sslsock->method, NULL) < 0 ||
2464                php_stream_xport_crypto_enable(stream, 1) < 0) {
2465              php_error_docref(NULL, E_WARNING, "Failed to enable crypto");
2466              xparam->outputs.returncode = -1;
2467            }
2468          }
2469          return PHP_STREAM_OPTION_RETURN_OK;
2470
2471        case STREAM_XPORT_OP_ACCEPT:
2472          /* we need to copy the additional fields that the underlying tcp transport
2473           * doesn't know about */
2474          xparam->outputs.returncode = php_openssl_tcp_sockop_accept(stream, sslsock, xparam STREAMS_CC);
2475
2476
2477          return PHP_STREAM_OPTION_RETURN_OK;
2478
2479        default:
2480          /* fall through */
2481          break;
2482      }
2483  }
2484
2485  return php_stream_socket_ops.set_option(stream, option, value, ptrparam);
