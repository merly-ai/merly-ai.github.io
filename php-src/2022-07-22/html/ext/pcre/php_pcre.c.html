104static void pcre_handle_exec_error(int pcre_code) /* {{{ */
105{
106  int preg_code = 0;
107
108  switch (pcre_code) {
109    case PCRE2_ERROR_MATCHLIMIT:
110      preg_code = PHP_PCRE_BACKTRACK_LIMIT_ERROR;
111      break;
112
113    case PCRE2_ERROR_RECURSIONLIMIT:
114      preg_code = PHP_PCRE_RECURSION_LIMIT_ERROR;
115      break;
116
117    case PCRE2_ERROR_BADUTFOFFSET:
118      preg_code = PHP_PCRE_BAD_UTF8_OFFSET_ERROR;
119      break;
120
121#ifdef HAVE_PCRE_JIT_SUPPORT
122    case PCRE2_ERROR_JIT_STACKLIMIT:
123      preg_code = PHP_PCRE_JIT_STACKLIMIT_ERROR;
124      break;
125#endif
126
127    default:
128      if (pcre_code <= PCRE2_ERROR_UTF8_ERR1 && pcre_code >= PCRE2_ERROR_UTF8_ERR21) {
129        preg_code = PHP_PCRE_BAD_UTF8_ERROR;
130      } else  {
131        preg_code = PHP_PCRE_INTERNAL_ERROR;
132      }
133      break;
134  }
135
136  PCRE_G(error_code) = preg_code;
137}

214static void php_pcre_init_pcre2(uint8_t jit)
215{/*{{{*/
216  if (!gctx) {
217    gctx = pcre2_general_context_create(php_pcre_malloc, php_pcre_free, NULL);
218    if (!gctx) {
219      pcre2_init_ok = 0;
220      return;
221    }
222  }
223
224  if (!cctx) {
225    cctx = pcre2_compile_context_create(gctx);
226    if (!cctx) {
227      pcre2_init_ok = 0;
228      return;
229    }
230  }
231
232  pcre2_set_compile_extra_options(cctx, PHP_PCRE_DEFAULT_EXTRA_COPTIONS);
233
234  if (!mctx) {
235    mctx = pcre2_match_context_create(gctx);
236    if (!mctx) {
237      pcre2_init_ok = 0;
238      return;
239    }
240  }
241
242#ifdef HAVE_PCRE_JIT_SUPPORT
243  if (jit && !jit_stack) {
244    jit_stack = pcre2_jit_stack_create(PCRE_JIT_STACK_MIN_SIZE, PCRE_JIT_STACK_MAX_SIZE, gctx);
245    if (!jit_stack) {
246      pcre2_init_ok = 0;
247      return;
248    }
249  }
250#endif
251
252  if (!mdata) {
253    mdata = pcre2_match_data_create(PHP_PCRE_PREALLOC_MDATA_SIZE, gctx);
254    if (!mdata) {
255      pcre2_init_ok = 0;
256      return;
257    }
258  }
259
260  pcre2_init_ok = 1;
261}/*}}}*/

615    key = zend_string_concat2(
616      ZSTR_VAL(BG(ctype_string)), ZSTR_LEN(BG(ctype_string)),
617      ZSTR_VAL(regex), ZSTR_LEN(regex));
618  } else {
619    key = regex;
620  }
621
622  /* Try to lookup the cached regex entry, and if successful, just pass
623     back the compiled pattern, otherwise go on and compile it. */
624  zv = zend_hash_find(&PCRE_G(pcre_cache), key);
625  if (zv) {
626    if (key != regex) {
627      zend_string_release_ex(key, 0);
628    }
629    return (pcre_cache_entry*)Z_PTR_P(zv);
630  }
631
632  p = ZSTR_VAL(regex);
633  const char* end_p = ZSTR_VAL(regex) + ZSTR_LEN(regex);

636     get to the end without encountering a delimiter. */
637  while (isspace((int)*(unsigned char *)p)) p++;
638  if (p >= end_p) {
639    if (key != regex) {
640      zend_string_release_ex(key, 0);
641    }
642    php_error_docref(NULL, E_WARNING, "Empty regular expression");

651    if (key != regex) {

653    }

655    pcre_handle_exec_error(PCRE2_ERROR_INTERNAL);

657  }

660  if ((pp = strchr("([{< )]}> )]}>", delimiter)))

668       but skipping the backslashed delimiters.  If the ending delimiter is not
669       found, display a warning. */

672      else if (*pp == delimiter)
673        break;
674      pp++;
675    }
676  } else {
677    /* We iterate through the pattern, searching for the matching ending
678     * delimiter. For each matching starting delimiter, we increment nesting
679     * level, and decrement it for each matching ending delimiter. If we
680     * reach the end of the pattern without matching, display a warning.
681     */
682    int brackets = 1;   /* brackets nesting level */
683    while (pp < end_p) {
684      if (*pp == '\\' && pp + 1 < end_p) pp++;
685      else if (*pp == end_delimiter && --brackets <= 0)
686        break;
687      else if (*pp == start_delimiter)
688        brackets++;
689      pp++;
690    }
691  }
692
693  if (pp >= end_p) {
694    if (key != regex) {
695      zend_string_release_ex(key, 0);
696    }
697    if (start_delimiter == end_delimiter) {
698      php_error_docref(NULL,E_WARNING, "No ending delimiter '%c' found", delimiter);
699    } else {

701    }
702    pcre_handle_exec_error(PCRE2_ERROR_INTERNAL);
703    return NULL;
704  }

706  /* Make a copy of the actual pattern. */

710  /* Move on to the options */

722      case 'x':  coptions |= PCRE2_EXTENDED;    break;

726      case 'D':  coptions |= PCRE2_DOLLAR_ENDONLY;break;

728      case 'X':  /* Pass. */          break;
729      case 'U':  coptions |= PCRE2_UNGREEDY;    break;
730      case 'u':  coptions |= PCRE2_UTF;

732     characters, even in UTF-8 mode. However, this can be changed by setting

753        }

760    }

768      zend_string_release_ex(key, 0);
769    }
770    return NULL;
771  }

869   * but all the interned strings created during HTTP request are removed
870   * at end of request. However PCRE_G(pcre_cache) must be consistent

2867PHPAPI void  php_pcre_grep_impl(pcre_cache_entry *pce, zval *input, zval *return_value, zend_long flags) /* {{{ */
2868{
2869  zval            *entry;             /* An entry in the input array */
2870  uint32_t     num_subpats;    /* Number of captured subpatterns */
2871  int         count;        /* Count of matched subpatterns */
2872  uint32_t     options;      /* Execution options */
2873  zend_string    *string_key;
2874  zend_ulong     num_key;
2875  bool     invert;      /* Whether to return non-matching
2876                       entries */
2877  pcre2_match_data *match_data;
2878  invert = flags & PREG_GREP_INVERT ? 1 : 0;
2879
2880  /* Calculate the size of the offsets array, and allocate memory for it. */
2881  num_subpats = pce->capture_count + 1;
2882
2883  /* Initialize return array */
2884  array_init(return_value);
2885
2886  PCRE_G(error_code) = PHP_PCRE_NO_ERROR;
2887
2888  if (!mdata_used && num_subpats <= PHP_PCRE_PREALLOC_MDATA_SIZE) {
2889    match_data = mdata;
2890  } else {
2891    match_data = pcre2_match_data_create_from_pattern(pce->re, PCRE_G(gctx_zmm));
2892    if (!match_data) {
2893      PCRE_G(error_code) = PHP_PCRE_INTERNAL_ERROR;
2894      return;
2895    }
2896  }
2897
2898  options = (pce->compile_options & PCRE2_UTF) ? 0 : PCRE2_NO_UTF_CHECK;
2899
2900  /* Go through the input array */
2901  ZEND_HASH_FOREACH_KEY_VAL(Z_ARRVAL_P(input), num_key, string_key, entry) {
2902    zend_string *tmp_subject_str;
2903    zend_string *subject_str = zval_get_tmp_string(entry, &tmp_subject_str);
2904
2905    /* Perform the match */
2906#ifdef HAVE_PCRE_JIT_SUPPORT
2907    if ((pce->preg_options & PREG_JIT) && options) {
2908      count = pcre2_jit_match(pce->re, (PCRE2_SPTR)ZSTR_VAL(subject_str), ZSTR_LEN(subject_str), 0,
2909          PCRE2_NO_UTF_CHECK, match_data, mctx);
2910    } else
2911#endif
2912    count = pcre2_match(pce->re, (PCRE2_SPTR)ZSTR_VAL(subject_str), ZSTR_LEN(subject_str), 0,
2913        options, match_data, mctx);
2914
2915    /* If the entry fits our requirements */
2916    if (count >= 0) {
2917      /* Check for too many substrings condition. */
2918      if (UNEXPECTED(count == 0)) {
2919        php_error_docref(NULL, E_NOTICE, "Matched, but too many substrings");
2920      }
2921      if (!invert) {
2922        Z_TRY_ADDREF_P(entry);
2923
2924        /* Add to return array */
2925        if (string_key) {
2926          zend_hash_update(Z_ARRVAL_P(return_value), string_key, entry);
2927        } else {
2928          zend_hash_index_update(Z_ARRVAL_P(return_value), num_key, entry);
2929        }
2930      }
2931    } else if (count == PCRE2_ERROR_NOMATCH) {
2932      if (invert) {
2933        Z_TRY_ADDREF_P(entry);
2934
2935        /* Add to return array */
2936        if (string_key) {
2937          zend_hash_update(Z_ARRVAL_P(return_value), string_key, entry);
2938        } else {
2939          zend_hash_index_update(Z_ARRVAL_P(return_value), num_key, entry);
2940        }
2941      }
2942    } else {
2943      pcre_handle_exec_error(count);
2944      zend_tmp_string_release(tmp_subject_str);
2945      break;
2946    }
2947
2948    zend_tmp_string_release(tmp_subject_str);
2949  } ZEND_HASH_FOREACH_END();
2950  if (match_data != mdata) {
2951    pcre2_match_data_free(match_data);
2952  }
2953}
