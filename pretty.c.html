299static int is_rfc2047_special(char ch, enum rfc2047_type type)
300{
301  /*
302   * rfc2047, section 4.2:
303   *
304   *    8-bit values which correspond to printable ASCII characters other
305   *    than "=", "?", and "_" (underscore), MAY be represented as those
306   *    characters.  (But see section 5 for restrictions.)  In
307   *    particular, SPACE and TAB MUST NOT be represented as themselves
308   *    within encoded words.
309   */
310
311  /*
312   * rule out non-ASCII characters and non-printable characters (the
313   * non-ASCII check should be redundant as isprint() is not localized
314   * and only knows about ASCII, but be defensive about that)
315   */
316  if (non_ascii(ch) || !isprint(ch))
317    return 1;
318
319  /*
320   * rule out special printable characters (' ' should be the only
321   * whitespace character considered printable, but be defensive and use
322   * isspace())
323   */
324  if (isspace(ch) || ch == '=' || ch == '?' || ch == '_')
325    return 1;
326
327  /*
328   * rfc2047, section 5.3:
329   *
330   *    As a replacement for a 'word' entity within a 'phrase', for example,
331   *    one that precedes an address in a From, To, or Cc header.  The ABNF
332   *    definition for 'phrase' from RFC 822 thus becomes:
333   *
334   *    phrase = 1*( encoded-word / word )
335   *
336   *    In this case the set of characters that may be used in a "Q"-encoded
337   *    'encoded-word' is restricted to: <upper and lower case ASCII
338   *    letters, decimal digits, "!", "*", "+", "-", "/", "=", and "_"
339   *    (underscore, ASCII 95.)>.  An 'encoded-word' that appears within a
340   *    'phrase' MUST be separated from any adjacent 'word', 'text' or
341   *    'special' by 'linear-white-space'.
342   */
343
344  if (type != RFC2047_ADDRESS)
345    return 0;
346
347  /* '=' and '_' are special cases and have been checked above */
348  return !(isalnum(ch) || ch == '!' || ch == '*' || ch == '+' || ch == '-' || ch == '/');
349}

366static void add_rfc2047(struct strbuf *sb, const char *line, size_t len,
367           const char *encoding, enum rfc2047_type type)
368{
369  static const int max_encoded_length = 76; /* per rfc2047 */
370  int i;
371  int line_len = last_line_length(sb);
372
373  strbuf_grow(sb, len * 3 + strlen(encoding) + 100);
374  strbuf_addf(sb, "=?%s?q?", encoding);
375  line_len += strlen(encoding) + 5; /* 5 for =??q? */
376
377  while (len) {
378    /*
379     * RFC 2047, section 5 (3):
380     *
381     * Each 'encoded-word' MUST represent an integral number of
382     * characters.  A multi-octet character may not be split across
383     * adjacent 'encoded- word's.
384     */
385    const unsigned char *p = (const unsigned char *)line;
386    int chrlen = mbs_chrlen(&line, &len, encoding);
387    int is_special = (chrlen > 1) || is_rfc2047_special(*p, type);
388
389    /* "=%02X" * chrlen, or the byte itself */
390    const char *encoded_fmt = is_special ? "=%02X"    : "%c";
391    int      encoded_len = is_special ? 3 * chrlen : 1;
392
393    /*
394     * According to RFC 2047, we could encode the special character
395     * ' ' (space) with '_' (underscore) for readability. But many
396     * programs do not understand this and just leave the
397     * underscore in place. Thus, we do nothing special here, which
398     * causes ' ' to be encoded as '=20', avoiding this problem.
399     */
400
401    if (line_len + encoded_len + 2 > max_encoded_length) {
402      /* It won't fit with trailing "?=" --- break the line */
403      strbuf_addf(sb, "?=\n =?%s?q?", encoding);
404      line_len = strlen(encoding) + 5 + 1; /* =??q? plus SP */
405    }
406
407    for (i = 0; i < chrlen; i++)
408      strbuf_addf(sb, encoded_fmt, p[i]);
409    line_len += encoded_len;
410  }
411  strbuf_addstr(sb, "?=");
412}

443           const char *line, size_t linelen,

479void pp_user_info(struct pretty_print_context *pp,
480      const char *what, struct strbuf *sb,
481      const char *line, const char *encoding)
482{
483  struct ident_split ident;
484  char *line_end;
485  const char *mailbuf, *namebuf;
486  size_t namelen, maillen;
487  int max_length = 78; /* per rfc2822 */
488
489  if (pp->fmt == CMIT_FMT_ONELINE)
490    return;
491
492  line_end = strchrnul(line, '\n');
493  if (split_ident_line(&ident, line, line_end - line))
494    return;
495
496  mailbuf = ident.mail_begin;
497  maillen = ident.mail_end - ident.mail_begin;
498  namebuf = ident.name_begin;
499  namelen = ident.name_end - ident.name_begin;
500
501  if (pp->mailmap)
502    map_user(pp->mailmap, &mailbuf, &maillen, &namebuf, &namelen);
503
504  if (cmit_fmt_is_mail(pp->fmt)) {
505    if (pp->from_ident && ident_cmp(pp->from_ident, &ident)) {
506      struct strbuf buf = STRBUF_INIT;
507
508      strbuf_addstr(&buf, "From: ");
509      strbuf_add(&buf, namebuf, namelen);
510      strbuf_addstr(&buf, " <");
511      strbuf_add(&buf, mailbuf, maillen);
512      strbuf_addstr(&buf, ">\n");
513      string_list_append(&pp->in_body_headers,
514             strbuf_detach(&buf, NULL));
515
516      mailbuf = pp->from_ident->mail_begin;
517      maillen = pp->from_ident->mail_end - mailbuf;
518      namebuf = pp->from_ident->name_begin;
519      namelen = pp->from_ident->name_end - namebuf;
520    }
521
522    strbuf_addstr(sb, "From: ");
523    if (pp->encode_email_headers &&
524        needs_rfc2047_encoding(namebuf, namelen)) {
525      add_rfc2047(sb, namebuf, namelen,
526            encoding, RFC2047_ADDRESS);
527      max_length = 76; /* per rfc2047 */
528    } else if (needs_rfc822_quoting(namebuf, namelen)) {
529      struct strbuf quoted = STRBUF_INIT;
530      add_rfc822_quoted(&quoted, namebuf, namelen);
531      strbuf_add_wrapped_bytes(sb, quoted.buf, quoted.len,
532              -6, 1, max_length);
533      strbuf_release(&quoted);
534    } else {

639static char *replace_encoding_header(char *buf, const char *encoding)
640{
641  struct strbuf tmp = STRBUF_INIT;
642  size_t start, len;
643  char *cp = buf;
644
645  /* guess if there is an encoding header before a \n\n */
646  while (!starts_with(cp, "encoding ")) {
647    cp = strchr(cp, '\n');
648    if (!cp || *++cp == '\n')
649      return buf;
650  }
651  start = cp - buf;
652  cp = strchr(cp, '\n');
653  if (!cp)
654    return buf; /* should not happen but be defensive */
655  len = cp + 1 - (buf + start);
656
657  strbuf_attach(&tmp, buf, strlen(buf), strlen(buf) + 1);
658  if (is_encoding_utf8(encoding)) {
659    /* we have re-coded to UTF-8; drop the header */
660    strbuf_remove(&tmp, start, len);
661  } else {
662    /* just replaces XXXX in 'encoding XXXX\n' */
663    strbuf_splice(&tmp, start + strlen("encoding "),
664            len - strlen("encoding \n"),
665            encoding, strlen(encoding));
666  }
667  return strbuf_detach(&tmp, NULL);
668}

670const char *repo_logmsg_reencode(struct repository *r,
671         const struct commit *commit,
672         char **commit_encoding,
673         const char *output_encoding)
674{
675  static const char *utf8 = "UTF-8";
676  const char *use_encoding;
677  char *encoding;
678  const char *msg = repo_get_commit_buffer(r, commit, NULL);
679  char *out;
680
681  if (!output_encoding || !*output_encoding) {
682    if (commit_encoding)
683      *commit_encoding = get_header(msg, "encoding");
684    return msg;
685  }
686  encoding = get_header(msg, "encoding");
687  if (commit_encoding)
688    *commit_encoding = encoding;
689  use_encoding = encoding ? encoding : utf8;
690  if (same_encoding(use_encoding, output_encoding)) {
691    /*
692     * No encoding work to be done. If we have no encoding header
693     * at all, then there's nothing to do, and we can return the
694     * message verbatim (whether newly allocated or not).
695     */
696    if (!encoding)
697      return msg;
698
699    /*
700     * Otherwise, we still want to munge the encoding header in the
701     * result, which will be done by modifying the buffer. If we
702     * are using a fresh copy, we can reuse it. But if we are using
703     * the cached copy from get_commit_buffer, we need to duplicate it
704     * to avoid munging the cached copy.
705     */
706    if (msg == get_cached_commit_buffer(r, commit, NULL))
707      out = xstrdup(msg);
708    else
709      out = (char *)msg;
710  }
711  else {
712    /*
713     * There's actual encoding work to do. Do the reencoding, which
714     * still leaves the header to be replaced in the next step. At
715     * this point, we are done with msg. If we allocated a fresh
716     * copy, we can free it.
717     */
718    out = reencode_string(msg, output_encoding, use_encoding);
719    if (out)
720      repo_unuse_commit_buffer(r, commit, msg);
721  }
722
723  /*
724   * This replacement actually consumes the buffer we hand it, so we do
725   * not have to worry about freeing the old "out" here.
726   */
727  if (out)
728    out = replace_encoding_header(out, output_encoding);
729
730  if (!commit_encoding)
731    free(encoding);
732  /*
733   * If the re-encoding failed, out might be NULL here; in that
734   * case we just return the commit message verbatim.
735   */
736  return out ? out : msg;
737}

750static size_t format_person_part(struct strbuf *sb, char part,
751         const char *msg, int len,
752         const struct date_mode *dmode)
753{
754  /* currently all placeholders have same length */
755  const int placeholder_len = 2;
756  struct ident_split s;
757  const char *name, *mail;
758  size_t maillen, namelen;
759
760  if (split_ident_line(&s, msg, len) < 0)
761    goto skip;
762
763  name = s.name_begin;
764  namelen = s.name_end - s.name_begin;
765  mail = s.mail_begin;
766  maillen = s.mail_end - s.mail_begin;
767
768  if (part == 'N' || part == 'E' || part == 'L') /* mailmap lookup */
769    mailmap_name(&mail, &maillen, &name, &namelen);
770  if (part == 'n' || part == 'N') {  /* name */
771    strbuf_add(sb, name, namelen);
772    return placeholder_len;
773  }
774  if (part == 'e' || part == 'E') {  /* email */
775    strbuf_add(sb, mail, maillen);
776    return placeholder_len;
777  }
778  if (part == 'l' || part == 'L') {  /* local-part */
779    const char *at = memchr(mail, '@', maillen);
780    if (at)
781      maillen = at - mail;
782    strbuf_add(sb, mail, maillen);
783    return placeholder_len;
784  }
785
786  if (!s.date_begin)
787    goto skip;
788
789  if (part == 't') {  /* date, UNIX timestamp */
790    strbuf_add(sb, s.date_begin, s.date_end - s.date_begin);
791    return placeholder_len;
792  }
793
794  switch (part) {
795  case 'd':  /* date */
796    strbuf_addstr(sb, show_ident_date(&s, dmode));
797    return placeholder_len;
798  case 'D':  /* date, RFC2822 style */
799    strbuf_addstr(sb, show_ident_date(&s, DATE_MODE(RFC2822)));
800    return placeholder_len;
801  case 'r':  /* date, relative */
802    strbuf_addstr(sb, show_ident_date(&s, DATE_MODE(RELATIVE)));
803    return placeholder_len;
804  case 'i':  /* date, ISO 8601-like */
805    strbuf_addstr(sb, show_ident_date(&s, DATE_MODE(ISO8601)));
806    return placeholder_len;
807  case 'I':  /* date, ISO 8601 strict */
808    strbuf_addstr(sb, show_ident_date(&s, DATE_MODE(ISO8601_STRICT)));
809    return placeholder_len;
810  case 'h':  /* date, human */
811    strbuf_addstr(sb, show_ident_date(&s, DATE_MODE(HUMAN)));
812    return placeholder_len;
813  case 's':
814    strbuf_addstr(sb, show_ident_date(&s, DATE_MODE(SHORT)));
815    return placeholder_len;
816  }
817
818skip:
819  /*
820   * reading from either a bogus commit, or a reflog entry with
821   * %gn, %ge, etc.; 'sb' cannot be updated, but we still need
822   * to compute a valid return value.
823   */
824  if (part == 'n' || part == 'e' || part == 't' || part == 'd'
825      || part == 'D' || part == 'r' || part == 'i')
826    return placeholder_len;
827
828  return 0; /* unknown placeholder */
829}

877static void parse_commit_header(struct format_commit_context *context)
878{
879  const char *msg = context->message;
880  int i;
881
882  for (i = 0; msg[i]; i++) {
883    const char *name;

900  context->commit_header_parsed = 1;
901}

1023static size_t parse_color(struct strbuf *sb, /* in UTF-8 */
1024        const char *placeholder,
1025        struct format_commit_context *c)
1026{
1027  const char *rest = placeholder;
1028  const char *basic_color = NULL;
1029
1030  if (placeholder[1] == '(') {
1031    const char *begin = placeholder + 2;
1032    const char *end = strchr(begin, ')');
1033    char color[COLOR_MAXLEN];
1034
1035    if (!end)
1036      return 0;
1037
1038    if (skip_prefix(begin, "auto,", &begin)) {
1039      if (!want_color(c->pretty_ctx->color))
1040        return end - placeholder + 1;
1041    } else if (skip_prefix(begin, "always,", &begin)) {
1042      /* nothing to do; we do not respect want_color at all */
1043    } else {
1044      /* the default is the same as "auto" */
1045      if (!want_color(c->pretty_ctx->color))
1046        return end - placeholder + 1;
1047    }
1048
1049    if (color_parse_mem(begin, end - begin, color) < 0)
1050      die(_("unable to parse --pretty format"));
1051    strbuf_addstr(sb, color);
1052    return end - placeholder + 1;
1053  }
1054
1055  /*
1056   * We handle things like "%C(red)" above; for historical reasons, there
1057   * are a few colors that can be specified without parentheses (and
1058   * they cannot support things like "auto" or "always" at all).
1059   */
1060  if (skip_prefix(placeholder + 1, "red", &rest))

1672static size_t format_and_pad_commit(struct strbuf *sb, /* in UTF-8 */
1673            const char *placeholder,
1674            struct format_commit_context *c)
1675{
1676  struct strbuf local_sb = STRBUF_INIT;
1677  int total_consumed = 0, len, padding = c->padding;
1678  if (padding < 0) {
1679    const char *start = strrchr(sb->buf, '\n');
1680    int occupied;
1681    if (!start)
1682      start = sb->buf;
1683    occupied = utf8_strnwidth(start, -1, 1);
1684    occupied += c->pretty_ctx->graph_width;
1685    padding = (-padding) - occupied;
1686  }
1687  while (1) {
1688    int modifier = *placeholder == 'C';
1689    int consumed = format_commit_one(&local_sb, placeholder, c);
1690    total_consumed += consumed;
1691
1692    if (!modifier)
1693      break;
1694
1695    placeholder += consumed;
1696    if (*placeholder != '%')
1697      break;
1698    placeholder++;
1699    total_consumed++;
1700  }
1701  len = utf8_strnwidth(local_sb.buf, -1, 1);
1702
1703  if (c->flush_type == flush_left_and_steal) {
1704    const char *ch = sb->buf + sb->len - 1;
1705    while (len > padding && ch > sb->buf) {
1706      const char *p;
1707      if (*ch == ' ') {
1708        ch--;
1709        padding++;
1710        continue;
1711      }
1712      /* check for trailing ansi sequences */
1713      if (*ch != 'm')
1714        break;
1715      p = ch - 1;
1716      while (ch - p < 10 && *p != '\033')
1717        p--;
1718      if (*p != '\033' ||
1719          ch + 1 - p != display_mode_esc_sequence_len(p))
1720        break;
1721      /*
1722       * got a good ansi sequence, put it back to
1723       * local_sb as we're cutting sb
1724       */
1725      strbuf_insert(&local_sb, 0, p, ch + 1 - p);
1726      ch = p - 1;
1727    }
1728    strbuf_setlen(sb, ch + 1 - sb->buf);
1729    c->flush_type = flush_left;
1730  }
1731
1732  if (len > padding) {
1733    switch (c->truncate) {
1734    case trunc_left:
1735      strbuf_utf8_replace(&local_sb,
1736              0, len - (padding - 2),
1737              "..");
1738      break;
1739    case trunc_middle:
1740      strbuf_utf8_replace(&local_sb,
1741              padding / 2 - 1,
1742              len - (padding - 2),
1743              "..");
1744      break;
1745    case trunc_right:
1746      strbuf_utf8_replace(&local_sb,
1747              padding - 2, len - (padding - 2),
1748              "..");
1749      break;
1750    case trunc_none:
1751      break;
1752    }
1753    strbuf_addbuf(sb, &local_sb);
1754  } else {
1755    int sb_len = sb->len, offset = 0;
1756    if (c->flush_type == flush_left)
1757      offset = padding - len;
1758    else if (c->flush_type == flush_both)
1759      offset = (padding - len) / 2;
1760    /*
1761     * we calculate padding in columns, now
1762     * convert it back to chars
1763     */
1764    padding = padding - len + local_sb.len;
1765    strbuf_addchars(sb, ' ', padding);
1766    memcpy(sb->buf + sb_len + offset, local_sb.buf,
1767           local_sb.len);
1768  }
1769  strbuf_release(&local_sb);
1770  c->flush_type = no_flush;
1771  return total_consumed;
1772}

1774static size_t format_commit_item(struct strbuf *sb, /* in UTF-8 */
1775         const char *placeholder,
1776         void *context)
1777{
1778  int consumed;
1779  size_t orig_len;
1780  enum {
1781    NO_MAGIC,
1782    ADD_LF_BEFORE_NON_EMPTY,
1783    DEL_LF_BEFORE_EMPTY,
1784    ADD_SP_BEFORE_NON_EMPTY
1785  } magic = NO_MAGIC;
1786
1787  switch (placeholder[0]) {
1788  case '-':
1789    magic = DEL_LF_BEFORE_EMPTY;
1790    break;
1791  case '+':
1792    magic = ADD_LF_BEFORE_NON_EMPTY;
1793    break;
1794  case ' ':
1795    magic = ADD_SP_BEFORE_NON_EMPTY;
1796    break;
1797  default:
1798    break;
1799  }
1800  if (magic != NO_MAGIC)
1801    placeholder++;
1802
1803  orig_len = sb->len;
1804  if (((struct format_commit_context *)context)->flush_type != no_flush)
1805    consumed = format_and_pad_commit(sb, placeholder, context);
1806  else
1807    consumed = format_commit_one(sb, placeholder, context);
1808  if (magic == NO_MAGIC)
1809    return consumed;
1810
1811  if ((orig_len == sb->len) && magic == DEL_LF_BEFORE_EMPTY) {
1812    while (sb->len && sb->buf[sb->len - 1] == '\n')
1813      strbuf_setlen(sb, sb->len - 1);
1814  } else if (orig_len != sb->len) {
1815    if (magic == ADD_LF_BEFORE_NON_EMPTY)
1816      strbuf_insertstr(sb, orig_len, "\n");
1817    else if (magic == ADD_SP_BEFORE_NON_EMPTY)
1818      strbuf_insertstr(sb, orig_len, " ");
1819  }
1820  return consumed + 1;
1821}
